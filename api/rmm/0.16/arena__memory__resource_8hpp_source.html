154  {

176  void deallocate_from_other_arena(void* p, std::size_t bytes, cudaStream_t stream)
177  {
178  RMM_ASSERT_CUDA_SUCCESS(cudaStreamSynchronize(stream));
179 
180  read_lock lock(mtx_);
181 
182  if (use_per_thread_arena(stream)) {
183  auto const id = std::this_thread::get_id();
184  for (auto& kv : thread_arenas_) {
185  // Check the per-thread arena if it does not belong to the current thread, and return if the
186  // pointer is found.
187  if (kv.first != id && kv.second->deallocate(p, bytes)) return;
188  }
189  } else {
190  for (auto& kv : stream_arenas_) {
191  // Check the per-stream arena if it does not belong to the current stream, and return if the
192  // pointer is found.
193  if (kv.first != stream && kv.second.deallocate(p, bytes)) return;
194  }
195  }
196 
197  // The thread that originally allocated the block has terminated, and the allocation has been
198  // transferred to the global arena.
199  global_arena_.deallocate(p, bytes);
200  }

222  arena& get_thread_arena()
223  {
224  auto const id = std::this_thread::get_id();
225  {
226  read_lock lock(mtx_);
227  auto const it = thread_arenas_.find(id);
228  if (it != thread_arenas_.end()) { return *it->second; }
229  }
230  {

232  auto a = std::make_shared<arena>(global_arena_);
233  thread_arenas_.emplace(id, a);
234  thread_local detail::arena::arena_cleaner<Upstream> cleaner{a};
235  return *a;
236  }

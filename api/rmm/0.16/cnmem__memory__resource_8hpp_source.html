124  cnmem_memory_resource(std::size_t initial_pool_size,

128  std::vector<cnmemDevice_t> cnmem_devices;
129 
130  // If no devices were specified, use the current one
131  if (devices.empty()) {
132  int current_device{};
133  RMM_CUDA_TRY(cudaGetDevice(&current_device));
134  cnmemDevice_t dev{};
135  dev.device = current_device;
136  dev.size = initial_pool_size;
137  cnmem_devices.push_back(dev);
138  } else {
139  for (auto const& d : devices) {
140  cnmemDevice_t dev{};
141  dev.device = d;

143  cnmem_devices.push_back(dev);
144  }

45 explicit cnmem_memory_resource (std::size_t initial_pool_size = 0) {
46 cnmemDevice_t dev;
47 // TODO Update exception
48 if (cudaSuccess != cudaGetDevice(&(dev. device ))) {
49 throw std::runtime_error{ "Failed to get CUDA device" };
50 }
51 // Note: cnmem defaults to half GPU memory
52 dev. size = initial_pool_size;
53 dev. numStreams = 1;
54 cudaStream_t streams[1];
55 streams[0] = 0;
56 dev. streams = streams;
57 dev. streamSizes = 0;
58 unsigned flags = 0;
59 // TODO Update exception
60 auto status = cnmemInit(1, &dev, flags);
61 if (CNMEM_STATUS_SUCCESS != status) {
62 std::string msg = cnmemGetErrorString(status);
63 throw std::runtime_error{ "Failed to intialize cnmem: " + msg};
64 }
65 }

90 void * do_allocate(std::size_t bytes, cudaStream_t stream) override {
91 register_stream(stream);
92 void * p{ nullptr };
93 auto status = cnmemMalloc(&p, bytes, stream);
94 if (CNMEM_STATUS_SUCCESS != status) {
95 #ifndef NDEBUG

114 std::cerr << "cnmemFree failed \n" ;
115 #endif
116 }

119 void register_stream(cudaStream_t stream) {
120 // Don't register null stream with CNMEM
121 if (stream != 0) {
122 // TODO Probably don't want to have to take a lock for every memory
123 // allocation
124 std::lock_guard<std::mutex> lock(streams_mutex);
125 auto result = registered_streams.insert(stream);
126
127 if (result.second == true ) {
128 auto status = cnmemRegisterStream(stream);
129 if (CNMEM_STATUS_SUCCESS != status) {
130 throw std::runtime_error{ "Falied to register stream with cnmem" };
131 }
132 }
133 }
134 }

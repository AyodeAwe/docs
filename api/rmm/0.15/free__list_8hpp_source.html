199  void insert(block const& b)
200  {
201  if (is_empty()) {
202  insert(blocks.end(), b);
203  return;
204  }
205 
206  // Find the right place (in ascending ptr order) to insert the block
207  // Can't use binary_search because it's a linked list and will be quadratic
208  auto const next =
209  std::find_if(blocks.begin(), blocks.end(), [b](block const& i) { return b < i; });
210  auto const previous = (next == blocks.begin()) ? next : std::prev(next);
211 
212  // Coalesce with neighboring blocks or insert the new block if it can't be coalesced
213  bool const merge_prev = previous->is_contiguous_before(b);
214  bool const merge_next = (next != blocks.end()) && b.is_contiguous_before(*next);
215 
216  if (merge_prev && merge_next) {
217  *previous = previous->merge(b).merge(*next);
218  erase(next);
219  } else if (merge_prev) {
220  *previous = previous->merge(b);
221  } else if (merge_next) {
222  *next = b.merge(*next);
223  } else {
224  insert(next, b); // cannot be coalesced, just insert
225  }
226  }

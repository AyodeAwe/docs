202  virtual void* do_allocate(std::size_t bytes, cudaStream_t stream) override
203  {
204  if (bytes <= 0) return nullptr;

206  lock_guard lock(mtx_);
207

209  bytes = rmm::detail::align_up(bytes, allocation_alignment);

271  stream_event_pair get_event(cudaStream_t stream)
272  {
273 #ifdef CUDA_API_PER_THREAD_DEFAULT_STREAM
274  if (cudaStreamDefault == stream || cudaStreamPerThread == stream) {
275  // Create a thread-local shared event wrapper. Shared pointers in the thread and in each MR
276  // instance ensures it is destroyed cleaned up only after all are finished with it.
277  thread_local auto event_tls = std::make_shared<event_wrapper>();
278  default_stream_events.insert(event_tls);
279  return stream_event_pair{stream, event_tls.get()->event};
280  }
281 #else
282  // We use cudaStreamLegacy as the event map key for the default stream for consistency between
283  // PTDS and non-PTDS mode. In PTDS mode, the cudaStreamLegacy map key will only exist if the
284  // user explicitly passes it, so it is used as the default location for the free list
285  // at construction. For consistency, the same key is used for null stream free lists in non-PTDS
286  // mode.
287  if (cudaStreamDefault == stream) { stream = cudaStreamLegacy; }
288 #endif

291  return (iter != stream_events_.end()) ? iter->second : [&]() {
292  stream_event_pair stream_event{stream};
293  RMM_ASSERT_CUDA_SUCCESS(
294  cudaEventCreateWithFlags(&stream_event.event, cudaEventDisableTiming));
295  stream_events_[stream] = stream_event;
296  return stream_event;

307  block_type get_block(size_t size, stream_event_pair stream_event)
308  {
309  // Try to find a satisfactory block in free list for the same stream (no sync required)
310  auto iter = stream_free_blocks_.find(stream_event);
311  if (iter != stream_free_blocks_.end()) {
312  block_type b = iter->second.get_block(size);
313  if (b.is_valid()) return b;
314  }
315 
316  // Otherwise try to find a block associated with another stream
317  block_type b = get_block_from_other_stream(size, stream_event);
318  if (b.is_valid()) return b;
319 
320  // no larger blocks available on other streams, so grow the pool and create a block
321 
322  // avoid searching for this stream's list again
323  free_list& blocks =
324  (iter != stream_free_blocks_.end()) ? iter->second : stream_free_blocks_[stream_event];
325 
326  return this->underlying().expand_pool(size, blocks, stream_event.stream);
327  }

346  block_type get_block_from_other_stream(size_t size, stream_event_pair stream_event)
347  {
348  // nothing in this stream's free list, look for one on another stream
349  for (auto s = stream_free_blocks_.begin(); s != stream_free_blocks_.end(); ++s) {
350  auto blocks_event = s->first;
351  if (blocks_event.event != stream_event.event) {
352  auto blocks = s->second;
353 
354  block_type const b = blocks.get_block(size); // get the best fit block
355 
356  if (b.is_valid()) {
357  // Since we found a block associated with a different stream, we have to insert a wait
358  // on the stream's associated event into the allocating stream.
359  // TODO: could eliminate this ifdef and have the same behavior for PTDS and non-PTDS
360  // But the cudaEventRecord() on every free_block reduces performance significantly
361 #ifdef CUDA_API_PER_THREAD_DEFAULT_STREAM
362  RMM_CUDA_TRY(cudaStreamWaitEvent(stream_event.stream, blocks_event.event, 0));
363 
364 #else
365  RMM_CUDA_TRY(cudaStreamSynchronize(blocks_event.stream));
366 #endif
367  // Move all the blocks to the requesting stream, since it has waited on them
368  stream_free_blocks_[stream_event].insert(std::move(blocks));
369  stream_free_blocks_.erase(s);
370 
371  return b;
372  }
373  }
374  }
375  return block_type{};
376  }

386

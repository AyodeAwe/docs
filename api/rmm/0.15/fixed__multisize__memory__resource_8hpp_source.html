44 // Integer pow function
45 constexpr std::size_t ipow(std::size_t base, std::size_t exp)
46 {
47  std::size_t ret = 1;
48  while (exp > 0) {
49  if (exp & 1) ret *= base; // multiply the result by the current base
50  base *= base; // square the base
51  exp = exp >> 1; // divide the exponent in half
52  }
53  return ret;
54 }
55 } // namespace detail

101  max_size_bytes_{detail::ipow(size_base, max_size_exponent)}

107  fixed_size_mr_.emplace_back(new fixed_size_memory_resource<Upstream>(
108  upstream_resource, detail::ipow(size_base, i), initial_blocks_per_size));

171  device_memory_resource* get_resource(std::size_t bytes)
172  {
173  assert(bytes <= get_max_size());
174 
175  auto exponentiate = [this](std::size_t const& k) { return detail::ipow(size_base_, k); };
176  auto min_exp = thrust::make_transform_iterator(
177  thrust::make_counting_iterator(min_size_exponent_), exponentiate);
178 
179  auto iter = std::upper_bound(min_exp,
180  min_exp + (max_size_exponent_ - min_size_exponent_),
181  bytes,
182  [](std::size_t const& a, std::size_t const& b) { return a <= b; });
183 
184  return fixed_size_mr_[std::distance(min_exp, iter)].get();
185  }

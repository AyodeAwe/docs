158  block get_block_from_other_stream(size_t size, stream_event_pair stream_event)
159  {
160  // nothing in this stream's free list, look for one on another stream
161  for (auto s = stream_free_blocks_.begin(); s != stream_free_blocks_.end(); ++s) {
162  auto blocks_event = s->first;
163  if (blocks_event.event != stream_event.event) {
164  auto blocks = s->second;
165 
166  block const b = blocks.best_fit(size); // get the best fit block
167 
168  if (b.is_valid()) {
169  // Since we found a block associated with a different stream, we have to insert a wait on
170  // the stream's associated event into the allocating stream.
171  // TODO: could eliminate this ifdef and have the same behavior for PTDS and non-PTDS
172  // But the cudaEventRecord() on every free_block reduces performance significantly
173 #ifdef CUDA_API_PER_THREAD_DEFAULT_STREAM
174  RMM_CUDA_TRY(cudaStreamWaitEvent(stream_event.stream, blocks_event.event, 0));
175 #else
176  RMM_CUDA_TRY(cudaStreamSynchronize(blocks_event.stream));
177 #endif
178  // Move all the blocks to the requesting stream, since it has waited on them
179  stream_free_blocks_[stream_event].insert(blocks.begin(), blocks.end());
180  stream_free_blocks_.erase(s);
181 
182  return b;
183  }
184  }
185  }
186  return block{};
187  }

203  block available_larger_block(size_t size, stream_event_pair stream_event)
204  {
205  // Try to find a larger block in free list for the same stream (no sync required)
206  auto iter = stream_free_blocks_.find(stream_event);
207  if (iter != stream_free_blocks_.end()) {
208  block b = iter->second.best_fit(size);
209  if (b.is_valid()) return b;
210  }
211 
212  block b = get_block_from_other_stream(size, stream_event);
213  if (b.is_valid()) return b;
214 
215  // no larger blocks available on other streams, so grow the pool and create a block
216  size_t grow_size = size_to_grow(size);
217  RMM_EXPECTS(grow_size > 0, rmm::bad_alloc, "Maximum pool size exceeded");
218  return block_from_upstream(grow_size, stream_event.stream);
219  }

231  void* allocate_from_block(block const& b, size_t size, stream_event_pair stream_event)
232  {
233  block const alloc{b.pointer(), size, b.is_head()};
234 
235  if (b.size() > size) {

251  void free_block(void* p, size_t size, cudaStream_t stream)
252  {
253  if (p == nullptr) return;
254 
255  stream_event_pair stream_event = get_event(stream);
256 
257  auto const i = allocated_blocks_.find(static_cast<char*>(p));
258  assert(i != allocated_blocks_.end());
259  assert(i->size() == rmm::detail::align_up(size, allocation_alignment));
260 
261  // TODO: cudaEventRecord has significant overhead on deallocations, however it could mean less
262  // synchronization So we need to test in real non-PTDS applications that have multiple streams
263  // whether or not the overhead is worth it
264 #ifdef CUDA_API_PER_THREAD_DEFAULT_STREAM
265  auto result = cudaEventRecord(stream_event.event, stream);
266  assert(cudaSuccess == result);
267 #endif
268 
269  stream_free_blocks_[stream_event].insert(*i);
270  allocated_blocks_.erase(i);
271  }

282  size_t size_to_grow(size_t size) const
283  {
284  auto const remaining =
285  rmm::detail::align_up(maximum_pool_size_ - pool_size(), allocation_alignment);
286  auto const aligned_size = rmm::detail::align_up(size, allocation_alignment);
287  if (aligned_size <= remaining / 2) {
288  return remaining / 2;
289  } else if (aligned_size <= remaining) {
290  return remaining;
291  } else {
292  return 0;
293  }

475  stream_event_pair get_event(cudaStream_t stream)
476  {
477 #ifdef CUDA_API_PER_THREAD_DEFAULT_STREAM
478  if (cudaStreamDefault == stream || cudaStreamPerThread == stream) {
479  static thread_local default_stream_event e{this};
480  return stream_event_pair{stream, e.event};
481  }
482 #else
483  // We use cudaStreamLegacy as the event map key for the default stream for consistency between
484  // PTDS and non-PTDS mode. In PTDS mode, the cudaStreamLegacy map key will only exist if the
485  // user explicitly passes it, so it is used as the default location for the free list
486  // at construction, and for merging free lists when a thread exits (see destroy_event()).
487  // For consistency, the same key is used for null stream free lists in non-PTDS mode.
488  if (cudaStreamDefault == stream) { stream = cudaStreamLegacy; }
489 #endif
490 
491  auto iter = stream_events_.find(stream);
492  if (iter == stream_events_.end()) {
493  stream_event_pair stream_event{stream};
494  auto result = cudaEventCreateWithFlags(&stream_event.event, cudaEventDisableTiming);
495  assert(cudaSuccess == result);
496  stream_events_[stream] = stream_event;
497  return stream_event;
498  } else {
499  return iter->second;
500  }
501  }

509  void destroy_event(stream_event_pair stream_event)
510  {
511  // If we are destroying an event with associated free list, we need to synchronize that event
512  // and then merge its free list into the (legacy) default stream's list
513  auto free_list_iter = stream_free_blocks_.find(stream_event);
514  if (free_list_iter != stream_free_blocks_.end()) {
515  auto blocks = free_list_iter->second;
516  stream_free_blocks_[get_event(cudaStreamLegacy)].insert(blocks.begin(), blocks.end());
517  stream_free_blocks_.erase(free_list_iter);
518 
519  auto result = cudaEventSynchronize(stream_event.event);
520  assert(cudaSuccess == result);
521  }
522  auto result = cudaEventDestroy(stream_event.event);
523  assert(cudaSuccess == result);
524  }

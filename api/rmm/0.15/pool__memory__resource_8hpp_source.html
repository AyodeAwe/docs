219  block_type free_block(void* p, size_t size) noexcept
220  {
221  if (p == nullptr) return block_type{};
222 
223  auto const i = allocated_blocks_.find(static_cast<char*>(p));
224  assert(i != allocated_blocks_.end());
225 
226  auto block = *i;
227  assert(block.size() == rmm::detail::align_up(size, allocation_alignment));
228  allocated_blocks_.erase(i);

242  size_t size_to_grow(size_t size) const
243  {
244  auto const remaining =
245  rmm::detail::align_up(maximum_pool_size_ - pool_size(), allocation_alignment);
246  auto const aligned_size = rmm::detail::align_up(size, allocation_alignment);
247  if (aligned_size <= remaining / 2) {
248  return remaining / 2;
249  } else if (aligned_size <= remaining) {

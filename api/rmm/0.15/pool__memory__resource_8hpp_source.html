198  split_block allocate_from_block(block_type const& b, size_t size)

201  allocated_blocks_.insert(alloc);

217  block_type free_block(void* p, size_t size) noexcept
218  {
219  if (p == nullptr) return block_type{};
220 
221  auto const i = allocated_blocks_.find(static_cast<char*>(p));
222  assert(i != allocated_blocks_.end());
223 
224  auto block = *i;
225  assert(block.size() == rmm::detail::align_up(size, allocation_alignment));
226  allocated_blocks_.erase(i);
227 
228  return block;
229  }

240  size_t size_to_grow(size_t size) const
241  {

243  rmm::detail::align_up(maximum_pool_size_ - pool_size(), allocation_alignment);
244  auto const aligned_size = rmm::detail::align_up(size, allocation_alignment);
245  if (aligned_size <= remaining / 2) {
246  return remaining / 2;
247  } else if (aligned_size <= remaining) {

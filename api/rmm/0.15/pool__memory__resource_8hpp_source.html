125  block block_from_stream(free_list& blocks,
126  cudaStream_t blocks_stream,
127  size_t size,
128  cudaStream_t stream)
129  {
130  block const b = blocks.best_fit(size); // get the best fit block
131 
132  // If we found a block associated with a different stream,
133  // we have to synchronize the stream in order to use it
134  if ((blocks_stream != stream) && (b.ptr != nullptr)) {
135  cudaError_t result = cudaStreamSynchronize(blocks_stream);
136 
137  RMM_EXPECTS((result == cudaSuccess || // stream synced
138  result == cudaErrorInvalidResourceHandle), // stream deleted
139  rmm::bad_alloc,
140  "cudaStreamSynchronize failure");
141 
142  // Now that this stream is synced, insert all other blocks into this stream's list
143  // Note: This could cause thrashing between two streams. On the other hand, it reduces
144  // fragmentation by coalescing.
145  stream_free_blocks_[stream].insert(blocks.begin(), blocks.end());
146 
147  // remove this stream from the freelist
148  stream_free_blocks_.erase(blocks_stream);
149  }
150  return b;
151  }

165  block available_larger_block(size_t size, cudaStream_t stream)
166  {
167  // Try to find a larger block in free list for the same stream
168  auto iter = stream_free_blocks_.find(stream);
169  if (iter != stream_free_blocks_.end()) {
170  block b = block_from_stream(iter->second, stream, size, stream);
171  if (b.ptr != nullptr) return b;
172  }
173 
174  // nothing in this stream's free list, look for one on another stream
175  auto s = stream_free_blocks_.begin();
176  while (s != stream_free_blocks_.end()) {
177  if (s->first != stream) {
178  block b = block_from_stream(s->second, s->first, size, stream);
179  if (b.ptr != nullptr) return b;
180  }
181  ++s;
182  }
183 
184  // no larger blocks available on other streams, so grow the pool and create a block
185  size_t grow_size = size_to_grow(size);
186  RMM_EXPECTS(grow_size > 0, rmm::bad_alloc, "Maximum pool size exceeded");
187  return block_from_upstream(grow_size, stream);
188  }

200  void* allocate_from_block(block const& b, size_t size, cudaStream_t stream)
201  {
202  block const alloc{b.ptr, size, b.is_head};
203 
204  if (b.size > size) {
205  block rest{b.ptr + size, b.size - size, false};
206  stream_free_blocks_[stream].insert(rest);
207  }
208 
209  allocated_blocks_.insert(alloc);
210  return reinterpret_cast<void*>(alloc.ptr);
211  }

220  void free_block(void* p, size_t size, cudaStream_t stream)
221  {
222  if (p == nullptr) return;
223 
224  auto const i = allocated_blocks_.find(block{static_cast<char*>(p)});
225  assert(i != allocated_blocks_.end());
226  assert(i->size == rmm::detail::align_up(size, allocation_alignment));
227 
228  stream_free_blocks_[stream].insert(*i);
229  allocated_blocks_.erase(i);
230  }

241  size_t size_to_grow(size_t size) const
242  {
243  auto const remaining =
244  rmm::detail::align_up(maximum_pool_size_ - pool_size(), allocation_alignment);
245  auto const aligned_size = rmm::detail::align_up(size, allocation_alignment);
246  if (aligned_size <= remaining / 2) {
247  return remaining / 2;
248  } else if (aligned_size <= remaining) {
249  return remaining;
250  } else {
251  return 0;
252  }

262  block block_from_upstream(size_t size, cudaStream_t stream)
263  {
264  void* p = upstream_mr_->allocate(size, stream);

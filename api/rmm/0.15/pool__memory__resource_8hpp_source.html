166  block get_block_from_other_stream(size_t size, stream_event_pair stream_event)
167  {
168  // nothing in this stream's free list, look for one on another stream
169  for (auto s = stream_free_blocks_.begin(); s != stream_free_blocks_.end(); ++s) {
170  auto blocks_event = s->first;
171  if (blocks_event.event != stream_event.event) {
172  auto blocks = s->second;
173 
174  block const b = blocks.best_fit(size); // get the best fit block
175 
176  if (b.is_valid()) {
177  // Since we found a block associated with a different stream, we have to insert a wait on
178  // the stream's associated event into the allocating stream.
179  // TODO: could eliminate this ifdef and have the same behavior for PTDS and non-PTDS
180  // But the cudaEventRecord() on every free_block reduces performance significantly
181 #ifdef CUDA_API_PER_THREAD_DEFAULT_STREAM
182  RMM_CUDA_TRY(cudaStreamWaitEvent(stream_event.stream, blocks_event.event, 0));
183 #else
184  RMM_CUDA_TRY(cudaStreamSynchronize(blocks_event.stream));
185 #endif
186  // Move all the blocks to the requesting stream, since it has waited on them
187  stream_free_blocks_[stream_event].insert(blocks.begin(), blocks.end());
188  stream_free_blocks_.erase(s);
189 
190  return b;
191  }
192  }
193  }
194  return block{};
195  }

211  block available_larger_block(size_t size, stream_event_pair stream_event)
212  {
213  // Try to find a larger block in free list for the same stream (no sync required)
214  auto iter = stream_free_blocks_.find(stream_event);
215  if (iter != stream_free_blocks_.end()) {
216  block b = iter->second.best_fit(size);
217  if (b.is_valid()) return b;
218  }
219 
220  block b = get_block_from_other_stream(size, stream_event);
221  if (b.is_valid()) return b;
222 
223  // no larger blocks available on other streams, so grow the pool and create a block
224  size_t grow_size = size_to_grow(size);
225  RMM_EXPECTS(grow_size > 0, rmm::bad_alloc, "Maximum pool size exceeded");
226  return block_from_upstream(grow_size, stream_event.stream);
227  }

259  void free_block(void* p, size_t size, cudaStream_t stream)
260  {
261  if (p == nullptr) return;
262 
263  stream_event_pair stream_event = get_event(stream);
264 
265  auto const i = allocated_blocks_.find(static_cast<char*>(p));
266  assert(i != allocated_blocks_.end());
267  assert(i->size() == rmm::detail::align_up(size, allocation_alignment));
268 
269  // TODO: cudaEventRecord has significant overhead on deallocations, however it could mean less
270  // synchronization So we need to test in real non-PTDS applications that have multiple streams
271  // whether or not the overhead is worth it
272 #ifdef CUDA_API_PER_THREAD_DEFAULT_STREAM
273  RMM_ASSERT_CUDA_SUCCESS(cudaEventRecord(stream_event.event, stream));
274 #endif
275 
276  stream_free_blocks_[stream_event].insert(*i);
277  allocated_blocks_.erase(i);
278  }

289  size_t size_to_grow(size_t size) const
290  {
291  auto const remaining =
292  rmm::detail::align_up(maximum_pool_size_ - pool_size(), allocation_alignment);
293  auto const aligned_size = rmm::detail::align_up(size, allocation_alignment);
294  if (aligned_size <= remaining / 2) {
295  return remaining / 2;
296  } else if (aligned_size <= remaining) {
297  return remaining;
298  } else {
299  return 0;

482  stream_event_pair get_event(cudaStream_t stream)
483  {
484 #ifdef CUDA_API_PER_THREAD_DEFAULT_STREAM
485  if (cudaStreamDefault == stream || cudaStreamPerThread == stream) {
486  static thread_local default_stream_event e{this};
487  return stream_event_pair{stream, e.event};
488  }
489 #else
490  // We use cudaStreamLegacy as the event map key for the default stream for consistency between
491  // PTDS and non-PTDS mode. In PTDS mode, the cudaStreamLegacy map key will only exist if the
492  // user explicitly passes it, so it is used as the default location for the free list
493  // at construction, and for merging free lists when a thread exits (see destroy_event()).
494  // For consistency, the same key is used for null stream free lists in non-PTDS mode.
495  if (cudaStreamDefault == stream) { stream = cudaStreamLegacy; }
496 #endif
497 
498  auto iter = stream_events_.find(stream);
499  if (iter == stream_events_.end()) {
500  stream_event_pair stream_event{stream};
501  auto result = cudaEventCreateWithFlags(&stream_event.event, cudaEventDisableTiming);
502  assert(cudaSuccess == result);
503  stream_events_[stream] = stream_event;
504  return stream_event;
505  } else {
506  return iter->second;
507  }
508  }

516  void destroy_event(stream_event_pair stream_event)
517  {
518  // If we are destroying an event with associated free list, we need to synchronize that event
519  // and then merge its free list into the (legacy) default stream's list
520  auto free_list_iter = stream_free_blocks_.find(stream_event);
521  if (free_list_iter != stream_free_blocks_.end()) {
522  auto blocks = free_list_iter->second;
523  stream_free_blocks_[get_event(cudaStreamLegacy)].insert(blocks.begin(), blocks.end());
524  stream_free_blocks_.erase(free_list_iter);
525 
526  auto result = cudaEventSynchronize(stream_event.event);
527  assert(cudaSuccess == result);
528  }
529  auto result = cudaEventDestroy(stream_event.event);
530  assert(cudaSuccess == result);
531  }

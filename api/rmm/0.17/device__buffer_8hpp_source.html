198  device_buffer& operator=(device_buffer const& other)
199  {
200  if (&other != this) {
201  // If the current capacity is large enough and the resources are
202  // compatible, just reuse the existing memory
203  if ((capacity() > other.size()) and _mr->is_equal(*other._mr)) {
204  resize(other.size(), other.stream());
205  copy(other.data(), other.size());

288  void resize(std::size_t new_size, cudaStream_t stream = 0)
289  {
290  set_stream(stream);
291  // If the requested size is smaller than the current capacity, just update
292  // the size without any allocations

318  void shrink_to_fit(cudaStream_t stream = 0)
319  {
320  set_stream(stream);
321  if (size() != capacity()) {
322  // Invoke copy ctor on self which only copies `[0, size())` and swap it
323  // with self. The temporary `device_buffer` will hold the old contents

161  free_list blocks_from_upstream(cudaStream_t stream)
162  {
163  void* p = upstream_mr_->allocate(upstream_chunk_size_, stream);
164  block_type b{p};
165  upstream_blocks_.push_back(b);
166 
167  auto num_blocks = upstream_chunk_size_ / block_size_;
168 
169  auto g = [p, this](int i) { return block_type{static_cast<char*>(p) + i * block_size_}; };
170  auto first = thrust::make_transform_iterator(thrust::make_counting_iterator(std::size_t{0}), g);
171  return free_list(first, first + num_blocks);

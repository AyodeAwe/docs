119 template < typename T>
120 inline rmmError_t alloc(T** ptr, size_t size, cudaStream_t stream, const char * file,
121 unsigned int line) {
122 if (!rmmIsInitialized( nullptr )) {
123 if (ptr)
124 *ptr = nullptr ;
125 return RMM_ERROR_NOT_INITIALIZED;
126 }
127
128 rmm::LogIt log(rmm::Logger::Alloc, 0, size, stream, file, line);
129
130 if (!ptr && !size) {
131 return RMM_SUCCESS;
132 }
133
134 if (!ptr)
135 return RMM_ERROR_INVALID_ARGUMENT;
136
137 try {
138 *ptr = static_cast<T*>(
139 rmm::mr::get_default_resource()->allocate(size,stream));
140
141 } catch (std::exception e) {
142 *ptr = nullptr ;
143 return RMM_ERROR_OUT_OF_MEMORY;
144 }
145
146 log.setPointer(*ptr);
147 return RMM_SUCCESS;
148 }

34 template < class T>
35 class rmm_allocator : public thrust::device_malloc_allocator<T>
36 {
37 public :
38 using value_type = T;
39
40 rmm_allocator (cudaStream_t stream = 0) : stream(stream) {}
41 ~ rmm_allocator () {
42 }
43
44 typedef thrust::device_ptr<value_type> pointer;
45 inline pointer allocate( size_t n)
46 {
47 value_type* result = nullptr ;
48
49 rmmError_t error = RMM_ALLOC(( void **)&result, n* sizeof (value_type), stream);
50
51 if (error != RMM_SUCCESS)
52 {
53 throw thrust::system_error(error, thrust::cuda_category(), "rmm_allocator::allocate(): RMM_ALLOC" );
54 }
55
56 return thrust::device_pointer_cast(result);
57 }
58
59 inline void deallocate(pointer ptr, size_t )
60 {
61 rmmError_t error = RMM_FREE(thrust::raw_pointer_cast(ptr), stream);
62
63 if (error != RMM_SUCCESS)
64 {
65 throw thrust::system_error(error, thrust::cuda_category(), "rmm_allocator::deallocate(): RMM_FREE" );
66 }
67 }
68
69 private :
70 cudaStream_t stream;
71 };

80 template < typename T>

83 using par_t = decltype(thrust::cuda::par(*( new rmm_allocator<char> (0))));
84 using deleter_t = std::function<void(par_t *)>;
85 using exec_policy_t = std::unique_ptr<par_t, deleter_t>;
86
87 /* --------------------------------------------------------------------------*/
97 /* --------------------------------------------------------------------------*/
98 inline exec_policy_t exec_policy(cudaStream_t stream = 0) {
99 rmm_allocator<char> *alloc{ nullptr };
100 alloc = new rmm_allocator<char> (stream);
101 auto deleter = [alloc](par_t *pointer) {
102 delete alloc;
103 delete pointer;
104 };
105
106 exec_policy_t policy{ new par_t(*alloc), deleter};
107 return policy;
108 }

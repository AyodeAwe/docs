42 template <typename T>
43 std::future<std::size_t> parallel_io(
44  T op, const void* devPtr, std::size_t size, std::size_t file_offset, std::size_t task_size)
45 {
46  if (task_size == 0) { throw std::invalid_argument("`task_size` cannot be zero"); }
47 
48  CUcontext ctx = get_context_from_device_pointer(devPtr);
49  auto [devPtr_base, base_size, devPtr_offset] = get_alloc_info(devPtr, &ctx);
50 
51  auto task = [op, ctx](void* devPtr_base,
52  std::size_t size,
53  std::size_t file_offset,
54  std::size_t devPtr_offset) -> std::size_t {
55  PushAndPopContext c(ctx);
56  return op(devPtr_base, size, file_offset, devPtr_offset);
57  };
58 
59  std::vector<std::future<std::size_t>> tasks;
60  if (task_size >= size || page_size >= size) { // Single-task guard
61  tasks.push_back(
62  defaults::thread_pool().submit(task, devPtr_base, size, file_offset, devPtr_offset));
63  } else {
64  // We have an upper bound of the total number of tasks
65  tasks.reserve(size / task_size + 2);
66 
67  // 1) Submit a task for the unaligned range from the start to the first page boundary
68  {
69  const std::size_t unaligned_range = file_offset % page_size;
70  assert(unaligned_range < size); // This is true because of the single-task guard above.
71  if (unaligned_range > 0) {
72  tasks.push_back(defaults::thread_pool().submit(
73  task, devPtr_base, unaligned_range, file_offset, devPtr_offset));
74  file_offset += unaligned_range;
75  devPtr_offset += unaligned_range;
76  size -= unaligned_range;
77  }
78  assert(file_offset % page_size == 0); // `file_offset` is now aligned
79  }
80  // 2) Submit tasks for the aligned range from the first page boundary to the last page boundary
81  {
82  while (size >= task_size) {
83  tasks.push_back(
84  defaults::thread_pool().submit(task, devPtr_base, task_size, file_offset, devPtr_offset));
85  file_offset += task_size;
86  devPtr_offset += task_size;
87  size -= task_size;
88  }
89  }
90  // 3) Submit a task for the remainder range from the last page boundary to the end
91  if (size > 0) {
92  tasks.push_back(
93  defaults::thread_pool().submit(task, devPtr_base, size, file_offset, devPtr_offset));
94  }
95  }
96 
97  // Finally, we sum the result of all tasks.
98  auto gather_tasks = [](std::vector<std::future<std::size_t>>&& tasks) -> std::size_t {
99  std::size_t ret = 0;
100  for (auto& task : tasks) {
101  ret += task.get();
102  }
103  return ret;
104  };
105  return std::async(std::launch::deferred, gather_tasks, std::move(tasks));
106 }

84 // Find the base and offset of the memory allocation `devPtr` is in
85 inline std::tuple<void*, std::size_t, std::size_t> get_alloc_info(const void* devPtr,
86  CUcontext* ctx = nullptr)
87 {
88  auto dev = convert_void2deviceptr(devPtr);
89  CUdeviceptr base_ptr{};
90  std::size_t base_size{};
91  CUcontext _ctx{};
92  if (ctx != nullptr) {
93  _ctx = *ctx;
94  } else {
95  _ctx = get_context_from_device_pointer(devPtr);
96  }
97  PushAndPopContext context(_ctx);
98  CUDA_DRIVER_TRY(cudaAPI::instance().MemGetAddressRange(&base_ptr, &base_size, dev));
99  std::size_t offset = dev - base_ptr;
100  // NOLINTNEXTLINE(performance-no-int-to-ptr, cppcoreguidelines-pro-type-reinterpret-cast)
101  return std::make_tuple(reinterpret_cast<void*>(base_ptr), base_size, offset);
102 }

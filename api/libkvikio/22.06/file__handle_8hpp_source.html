47 inline int open_fd_parse_flags(const std::string& flags)
48 {
49  int file_flags = -1;
50  if (flags.empty()) { throw std::invalid_argument("Unknown file open flag"); }
51  switch (flags[0]) {
52  case 'r':
53  file_flags = O_RDONLY;
54  if (flags[1] == '+') { file_flags = O_RDWR; }
55  break;
56  case 'w':
57  file_flags = O_WRONLY;
58  if (flags[1] == '+') { file_flags = O_RDWR; }
59  file_flags |= O_CREAT | O_TRUNC;
60  break;
61  case 'a': throw std::invalid_argument("Open flag 'a' isn't supported");
62  default: throw std::invalid_argument("Unknown file open flag");
63  }
64  file_flags |= O_CLOEXEC;
65  if (!defaults::compat_mode()) { file_flags |= O_DIRECT; }
66  return file_flags;
67 }

41 template <typename T>
42 std::future<std::size_t> parallel_io(
43  T op, const void* devPtr, std::size_t size, std::size_t file_offset, std::size_t ntasks)
44 {
45  CUcontext ctx = get_context_from_device_pointer(devPtr);
46  auto [devPtr_base, base_size, devPtr_offset] = get_alloc_info(devPtr, &ctx);
47 
48  auto task = [op, ctx](void* devPtr_base,
49  std::size_t size,
50  std::size_t file_offset,
51  std::size_t devPtr_offset) -> std::size_t {
52  PushAndPopContext c(ctx);
53  return op(devPtr_base, size, file_offset, devPtr_offset);
54  };
55 
56  const std::size_t tasksize = size / ntasks;
57  std::size_t last_jobsize = tasksize + size % ntasks;
58  std::vector<std::future<std::size_t>> tasks;
59  tasks.reserve(ntasks);
60  for (std::size_t i = 0; i < ntasks; ++i) {
61  const std::size_t cur_size = (i == ntasks - 1) ? last_jobsize : tasksize;
62  const std::size_t offset = i * tasksize;
63  tasks.push_back(default_thread_pool::get().submit(
64  task, devPtr_base, cur_size, file_offset + offset, devPtr_offset + offset));
65  }
66 
67  // Finally, we sum the result of all tasks.
68  auto gather_tasks = [](std::vector<std::future<std::size_t>>&& tasks) -> std::size_t {
69  std::size_t ret = 0;
70  for (auto& task : tasks) {
71  ret += task.get();
72  }
73  return ret;
74  };
75  return std::async(std::launch::deferred, gather_tasks, std::move(tasks));
76 }

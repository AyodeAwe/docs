63  AllocRetain() = default;
64  Alloc get()
65  {
66  const std::lock_guard lock(_mutex);
67  // Check if we have an allocation available
68  if (!_free_allocs.empty()) {
69  void* ret = _free_allocs.top();
70  _free_allocs.pop();
71  return Alloc(this, ret);
72  }
73 
74  // If no available allocation, allocate and register a new one
75  void* alloc{};
76  // Allocate memory
77  int err = ::posix_memalign(&alloc, page_size, chunk_size);
78  if (err != 0) {
79  throw CUfileException{std::string{"POSIX error at: "} + __FILE__ + ":" +
80  KVIKIO_STRINGIFY(__LINE__) + ": " + strerror(err)};
81  }
82  // Register memory
83  CUDA_DRIVER_TRY(cuMemHostRegister(
84  alloc, chunk_size, CU_MEMHOSTREGISTER_PORTABLE | CU_MEMHOSTREGISTER_DEVICEMAP));
85 
86  return Alloc(this, alloc);
87  }

124 inline void pwrite_all(int fd, const void* buf, size_t count, off_t offset)
125 {
126  off_t cur_offset = offset;
127  size_t byte_remaining = count;
128  const char* buffer = static_cast<const char*>(buf);
129  while (byte_remaining > 0) {
130  ssize_t nbytes_written = ::pwrite(fd, buffer, byte_remaining, cur_offset);
131  if (nbytes_written == -1) {
132  if (errno == EBADF) {
133  throw CUfileException{std::string{"POSIX error on pread at: "} + __FILE__ + ":" +
134  KVIKIO_STRINGIFY(__LINE__) + ": unsupported file open flags"};
135  }
136  throw CUfileException{std::string{"POSIX error on pwrite at: "} + __FILE__ + ":" +
137  KVIKIO_STRINGIFY(__LINE__) + ": " + strerror(errno)};
138  }
139  if (nbytes_written == 0) {
140  throw CUfileException{std::string{"POSIX error on pwrite at: "} + __FILE__ + ":" +
141  KVIKIO_STRINGIFY(__LINE__) + ": EOF"};
142  }
143 
144  /*NOLINTNEXTLINE(cppcoreguidelines-pro-bounds-pointer-arithmetic)*/
145  buffer += nbytes_written;
146  cur_offset += nbytes_written;
147  byte_remaining -= nbytes_written;
148  }
149 }

160 inline ssize_t pread_some(int fd, void* buf, size_t count, off_t offset)
161 {
162  ssize_t ret = ::pread(fd, buf, count, offset);
163  if (ret == -1) {
164  if (errno == EBADF) {
165  throw CUfileException{std::string{"POSIX error on pread at: "} + __FILE__ + ":" +
166  KVIKIO_STRINGIFY(__LINE__) + ": unsupported file open flags"};
167  }
168  throw CUfileException{std::string{"POSIX error on pread at: "} + __FILE__ + ":" +
169  KVIKIO_STRINGIFY(__LINE__) + ": " + strerror(errno)};
170  }
171  if (ret == 0) {
172  throw CUfileException{std::string{"POSIX error on pread at: "} + __FILE__ + ":" +
173  KVIKIO_STRINGIFY(__LINE__) + ": EOF"};
174  }
175  return ret;
176 }

189 template <bool IsReadOperation>
190 inline std::size_t posix_io(int fd,
191  const void* devPtr_base,
192  std::size_t size,
193  std::size_t file_offset,
194  std::size_t devPtr_offset)
195 {
196  auto alloc = manager.get();
197  CUdeviceptr devPtr = convert_void2deviceptr(devPtr_base) + devPtr_offset;
198  off_t cur_file_offset = convert_size2off(file_offset);
199  off_t byte_remaining = convert_size2off(size);
200  const off_t chunk_size2 = convert_size2off(chunk_size);
201 
202  while (byte_remaining > 0) {
203  const off_t nbytes_requested = std::min(chunk_size2, byte_remaining);
204  ssize_t nbytes_got = nbytes_requested;
205  if constexpr (IsReadOperation) {
206  nbytes_got = pread_some(fd, alloc.get(), nbytes_requested, cur_file_offset);
207  CUDA_DRIVER_TRY(cuMemcpyHtoD(devPtr, alloc.get(), nbytes_got));
208  } else { // Is a write operation
209  CUDA_DRIVER_TRY(cuMemcpyDtoH(alloc.get(), devPtr, nbytes_requested));
210  pwrite_all(fd, alloc.get(), nbytes_requested, cur_file_offset);
211  }
212  cur_file_offset += nbytes_got;
213  devPtr += nbytes_got;
214  byte_remaining -= nbytes_got;
215  }
216  return size;
217 }

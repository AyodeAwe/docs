30 template <typename T>
31 T getenv_or(std::string_view env_var_name, T default_val)
32 {
33  const auto* env_val = std::getenv(env_var_name.data());
34  if (env_val == nullptr) { return default_val; }
35 
36  std::stringstream sstream(env_val);
37  T converted_val;
38  sstream >> converted_val;
39  return converted_val;
40 }

42 template <>
43 bool getenv_or(std::string_view env_var_name, bool default_val)
44 {
45  const auto* env_val = std::getenv(env_var_name.data());
46  if (env_val == nullptr) { return default_val; }
47  try {
48  // Try parsing `env_var_name` as a integer
49  return static_cast<bool>(std::stoi(env_val));
50  } catch (std::invalid_argument) {
51  }
52  // Convert to lowercase
53  std::string str{env_val};
54  std::transform(str.begin(), str.end(), str.begin(), ::tolower);
55  // Trim whitespaces
56  std::stringstream trimmer;
57  trimmer << str;
58  str.clear();
59  trimmer >> str;
60  // Match value
61  if (str == "true" || str == "on" || str == "yes") { return true; }
62  if (str == "false" || str == "off" || str == "no") { return false; }
63  throw std::invalid_argument("Unknown config value: " + std::string{env_val});
64 }

66 inline bool _get_global_compat_mode()
67 {
68  if (std::getenv("KVIKIO_COMPAT_MODE") != nullptr) {
69  // Setting `KVIKIO_COMPAT_MODE` take precedence
70  return getenv_or("KVIKIO_COMPAT_MODE", false);
71  }
72  // If `KVIKIO_COMPAT_MODE` isn't set, we infer based on runtime environment
73  return !is_cufile_library_available() || is_running_in_wsl() || !run_udev_readable();
74 }

29 inline bool str_to_boolean(std::string str)
30 {
31  try {
32  // Try parsing `str` as a integer
33  return static_cast<bool>(std::stoi(str));
34  } catch (...) {
35  }
36  // Convert to lowercase
37  std::transform(str.begin(), str.end(), str.begin(), ::tolower);
38  // Try parsing `str` using std::boolalpha, which support "true" and "false"
39  bool ret = false;
40  std::istringstream(str) >> std::boolalpha >> ret;
41  return ret;
42 }

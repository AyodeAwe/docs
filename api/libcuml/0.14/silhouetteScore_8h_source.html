49 template < typename DataT, typename LabelT>
50 __global__ void populateAKernel (DataT *sampleToClusterSumOfDistances,
51 DataT *binCountArray, DataT *d_aArray,
52 LabelT *labels, int nRows, int nLabels,
53 const DataT MAX_VAL) {
54 //getting the current index
55 int sampleIndex = threadIdx.x + blockIdx.x * blockDim.x;
56
57 if (sampleIndex >= nRows) return ;
58
59 //sampleDistanceVector is an array that stores that particular row of the distanceMatrix
60 DataT *sampleToClusterSumOfDistancesVector =
61 &sampleToClusterSumOfDistances[sampleIndex * nLabels];
62
63 LabelT sampleCluster = labels[sampleIndex];
64
65 int sampleClusterIndex = (int)sampleCluster;
66
67 if (binCountArray[sampleClusterIndex] - 1 <= 0) {
68 d_aArray[sampleIndex] = -1;
69 return ;
70
71 }
72
73 else {
74 d_aArray[sampleIndex] =
75 (sampleToClusterSumOfDistancesVector[sampleClusterIndex]) /
76 (binCountArray[sampleClusterIndex] - 1);
77
78 //modifying the sampleDistanceVector to give sample average distance
79 sampleToClusterSumOfDistancesVector[sampleClusterIndex] = MAX_VAL;
80 }
81 }

121 template < typename DataT>
122 struct DivOp {
123 HDI DataT operator() (DataT a, int b, int c) {
124 if (b == 0)
125 return ULLONG_MAX;
126 else
127 return a / b;
128 }
129 };

134 template < typename DataT>
135 struct SilOp {
136 HDI DataT operator() (DataT a, DataT b) {
137 if (a == 0 && b == 0 || a == b)
138 return 0;
139 else if (a == -1)
140 return 0;
141 else if (a > b)
142 return (b - a) / a;
143 else
144 return (b - a) / b;
145 }
146 };

151 template < typename DataT>
152 struct MinOp {
153 HDI DataT operator() (DataT a, DataT b) {
154 if (a > b)
155 return b;
156 else
157 return a;
158 }
159 };

175 template < typename DataT, typename LabelT>
176 DataT silhouetteScore (DataT *X_in, int nRows, int nCols, LabelT *labels,
177 int nLabels, DataT *silhouetteScorePerSample,
178 std::shared_ptr<MLCommon::deviceAllocator> allocator,
179 cudaStream_t stream, int metric = 4) {
180 ASSERT (nLabels >= 2 && nLabels <= (nRows - 1),
181 "silhouette Score not defined for the given number of labels!" );
182
183 //compute the distance matrix
184 MLCommon::device_buffer<DataT> distanceMatrix(allocator, stream,
185 nRows * nRows);
186 MLCommon::device_buffer<char> workspace(allocator, stream, 1);
187
188 Distance::pairwiseDistance (
189 X_in, X_in, distanceMatrix. data (), nRows, nRows, nCols, workspace,
190 static_cast< Distance::DistanceType > (metric), stream);
191
192 //deciding on the array of silhouette scores for each dataPoint
193 MLCommon::device_buffer<DataT> silhouetteScoreSamples(allocator, stream, 0);
194 DataT *perSampleSilScore = nullptr ;
195 if (silhouetteScorePerSample == nullptr ) {
196 silhouetteScoreSamples. resize (nRows, stream);
197 perSampleSilScore = silhouetteScoreSamples. data ();
198 } else {
199 perSampleSilScore = silhouetteScorePerSample;
200 }
201 CUDA_CHECK (
202 cudaMemsetAsync(perSampleSilScore, 0, nRows * sizeof (DataT), stream));
203
204 //getting the sample count per cluster
205 MLCommon::device_buffer<DataT> binCountArray(allocator, stream, nLabels);
206 CUDA_CHECK (
207 cudaMemsetAsync(binCountArray. data (), 0, nLabels * sizeof (DataT), stream));
208 countLabels (labels, binCountArray. data (), nRows, nLabels, workspace,
209 allocator, stream);
210
211 //calculating the sample-cluster-distance-sum-array
212 device_buffer<DataT> sampleToClusterSumOfDistances(allocator, stream,
213 nRows * nLabels);
214 CUDA_CHECK (cudaMemsetAsync(sampleToClusterSumOfDistances. data (), 0,
215 nRows * nLabels * sizeof (DataT), stream));
216 LinAlg::reduce_cols_by_key<DataT, LabelT>(
217 distanceMatrix. data (), labels, sampleToClusterSumOfDistances. data (), nRows,
218 nRows, nLabels, stream);
219
220 //creating the a array and b array
221 device_buffer<DataT> d_aArray(allocator, stream, nRows);
222 device_buffer<DataT> d_bArray(allocator, stream, nRows);
223 CUDA_CHECK (
224 cudaMemsetAsync(d_aArray. data (), 0, nRows * sizeof (DataT), stream));
225 CUDA_CHECK (
226 cudaMemsetAsync(d_bArray. data (), 0, nRows * sizeof (DataT), stream));
227
228 //kernel that populates the d_aArray
229 //kernel configuration
230 dim3 numThreadsPerBlock(32, 1, 1);
231 dim3 numBlocks(ceildiv<int>(nRows, numThreadsPerBlock.x), 1, 1);
232
233 //calling the kernel
234 populateAKernel<<<numBlocks, numThreadsPerBlock, 0, stream>>>(
235 sampleToClusterSumOfDistances. data (), binCountArray. data (), d_aArray. data (),
236 labels, nRows, nLabels, std::numeric_limits<DataT>::max ());
237
238 //elementwise dividing by bincounts
239 device_buffer<DataT> averageDistanceBetweenSampleAndCluster(allocator, stream,
240 nRows * nLabels);
241 CUDA_CHECK (cudaMemsetAsync(averageDistanceBetweenSampleAndCluster.data(), 0,
242 nRows * nLabels * sizeof (DataT), stream));
243
244 LinAlg::matrixVectorOp<DataT, DivOp<DataT>>(
245 averageDistanceBetweenSampleAndCluster.data(),
246 sampleToClusterSumOfDistances. data (), binCountArray. data (),
247 binCountArray. data (), nLabels, nRows, true , true , DivOp<DataT> (), stream);
248
249 //calculating row-wise minimum
250 LinAlg::reduce<DataT, DataT, int, Nop<DataT>, MinOp<DataT> >(
251 d_bArray. data (), averageDistanceBetweenSampleAndCluster.data(), nLabels,
252 nRows, std::numeric_limits<DataT>::max (), true , true , stream, false ,
253 Nop<DataT> (), MinOp<DataT> ());
254
255 //calculating the silhouette score per sample using the d_aArray and d_bArray
256 LinAlg::binaryOp<DataT, SilOp<DataT>>(perSampleSilScore, d_aArray. data (),
257 d_bArray. data (), nRows, SilOp<DataT> (),
258 stream);
259
260 //calculating the sum of all the silhouette score
261 device_buffer<DataT> d_avgSilhouetteScore(allocator, stream, 1);
262 CUDA_CHECK (
263 cudaMemsetAsync(d_avgSilhouetteScore.data(), 0, sizeof (DataT), stream));
264
265 DataT avgSilhouetteScore;
266
267 MLCommon::LinAlg::mapThenSumReduce<double, Nop<DataT>>(
268 d_avgSilhouetteScore.data(), nRows, Nop<DataT> (), stream, perSampleSilScore,
269 perSampleSilScore);
270
271 updateHost (&avgSilhouetteScore, d_avgSilhouetteScore.data(), 1, stream);
272
273 CUDA_CHECK (cudaStreamSynchronize(stream));
274
275 avgSilhouetteScore /= nRows;
276
277 return avgSilhouetteScore;
278 }

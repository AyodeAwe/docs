77 template < typename T>
78 __global__ void combined_dot_product ( int rows, int cols, const T *W, T *matrix,
79 int *check) {
80 int m_i = threadIdx.x + blockDim.x * blockIdx.x;
81 int Wi = m_i / cols;
82 if (m_i < cols * rows) {
83 if (W[Wi] >= 0.0)
84 matrix[m_i] = pow(W[Wi], 0.5) * (matrix[m_i]);
85 else
86 check[0] = Wi; // reports Wi'th eigen values is negative.
87 }
88 }

90 template < typename T> // if uplo = 0, lower part of dim x dim matrix set to
91 // value
92 __global__ void fill_uplo ( int dim, Filler uplo, T value, T *A) {
93 int j = threadIdx.x + blockDim.x * blockIdx.x;
94 int i = threadIdx.y + blockDim.y * blockIdx.y;
95 if ( i < dim && j < dim) {
96 // making off-diagonals == value
97 if ( i < j ) {
98 if (uplo == 1) A[ IDX2C ( i , j , dim)] = value;
99 } else if ( i > j ) {
100 if (uplo == 0) A[ IDX2C ( i , j , dim)] = value;
101 }
102 }
103 }

131 size_t give_buffer_size() {
132 // malloc workspace_decomp
133 size_t granuality = 256, offset = 0;
134 workspace_decomp = (T *)offset;
135 offset += alignTo ( sizeof (T) * Lwork, granuality);
136 eig = (T *)offset;

148 size_t init (cublasHandle_t cublasH, cusolverDnHandle_t cusolverH,
149 cudaStream_t stream) {
150 cublasHandle = cublasH;
151 cusolverHandle = cusolverH;
152 cudaStream = stream;
153 CURAND_CHECK (curandCreateGenerator(&gen, CURAND_RNG_PSEUDO_DEFAULT));
154 CURAND_CHECK (curandSetPseudoRandomGeneratorSeed(gen, 28)); // SEED
155 if (method == chol_decomp ) {
156 CUSOLVER_CHECK ( LinAlg::cusolverDnpotrf_bufferSize (cusolverHandle, uplo,
157 dim, P, dim, &Lwork));
158 } else if (method == jacobi ) { // jacobi init
159 CUSOLVER_CHECK (cusolverDnCreateSyevjInfo(&syevj_params));
160 CUSOLVER_CHECK (cusolverDnXsyevjSetTolerance(syevj_params, tol));
161 CUSOLVER_CHECK (cusolverDnXsyevjSetMaxSweeps(syevj_params, max_sweeps));
162 CUSOLVER_CHECK ( LinAlg::cusolverDnsyevj_bufferSize (
163 cusolverHandle, jobz, uplo, dim, P, dim, eig, &Lwork, syevj_params));
164 } else { // method == qr
165 CUSOLVER_CHECK ( LinAlg::cusolverDnsyevd_bufferSize (
166 cusolverHandle, jobz, uplo, dim, P, dim, eig, &Lwork));
167 }
168 return give_buffer_size();
169 }

177 void give_gaussian ( const int nPoints, T *P, T *X, const T *x = 0) {
178 if (method == chol_decomp ) {
179 // lower part will contains chol_decomp
180 CUSOLVER_CHECK ( LinAlg::cusolverDnpotrf (cusolverHandle, uplo, dim, P, dim,
181 workspace_decomp, Lwork, info,
182 cudaStream));
183 } else if (method == jacobi ) {
184 CUSOLVER_CHECK ( LinAlg::cusolverDnsyevj (
185 cusolverHandle, jobz, uplo, dim, P, dim, eig, workspace_decomp, Lwork,
186 info, syevj_params,
187 cudaStream)); // vectors stored as cols. & col major
188 } else { // qr
189 CUSOLVER_CHECK ( LinAlg::cusolverDnsyevd (cusolverHandle, jobz, uplo, dim, P,
190 dim, eig, workspace_decomp, Lwork,
191 info, cudaStream));
192 }
193 updateHost (&info_h, info, 1, cudaStream);
194 CUDA_CHECK (cudaStreamSynchronize(cudaStream));
195 ASSERT (info_h == 0, "mvg: error in syevj/syevd/potrf, info=%d | expected=0" ,
196 info_h);
197 T mean = 0.0, stddv = 1.0;
198 // generate nxN gaussian nums in X
199 CURAND_CHECK ( curandGenerateNormal (
200 gen, X, (nPoints * dim) + (nPoints * dim) % 2, mean , stddv));
201 T alfa = 1.0, beta = 0.0;
202 if (method == chol_decomp ) {
203 // upper part (0) being filled with 0.0
204 dim3 block(32, 32);
205 dim3 grid( ceildiv (dim, ( int )block.x), ceildiv (dim, ( int )block.y));
206 fill_uplo<T><<<grid, block, 0, cudaStream>>>(dim, UPPER , (T)0.0, P);
207 CUDA_CHECK (cudaPeekAtLastError());
208
209 // P is lower triangular chol decomp mtrx
210 CUBLAS_CHECK ( LinAlg::cublasgemm (cublasHandle, CUBLAS_OP_N, CUBLAS_OP_N,
211 dim, nPoints, dim, &alfa, P, dim, X, dim,
212 &beta, X, dim, cudaStream));
213 } else {
214 epsilonToZero (eig, epsilon, dim, cudaStream);
215 dim3 block(64);
216 dim3 grid( ceildiv (dim, ( int )block.x));
217 CUDA_CHECK (cudaMemsetAsync(info, 0, sizeof ( int ), cudaStream));
218 grid.x = ceildiv (dim * dim, ( int )block.x);
219 combined_dot_product<T>
220 <<<grid, block, 0, cudaStream>>>(dim, dim, eig, P, info);
221 CUDA_CHECK (cudaPeekAtLastError());
222
223 // checking if any eigen vals were negative
224 updateHost (&info_h, info, 1, cudaStream);
225 CUDA_CHECK (cudaStreamSynchronize(cudaStream));
226 ASSERT (info_h == 0, "mvg: Cov matrix has %dth Eigenval negative" , info_h);
227
228 // Got Q = eigvect*eigvals.sqrt in P, Q*X in X below
229 CUBLAS_CHECK ( LinAlg::cublasgemm (cublasHandle, CUBLAS_OP_N, CUBLAS_OP_N,
230 dim, nPoints, dim, &alfa, P, dim, X, dim,
231 &beta, X, dim, cudaStream));
232 }
233 // working to make mean not 0
234 // since we are working with column-major, nPoints and dim are swapped
235 if (x != NULL) matVecAdd (X, X, x, T(1.0), nPoints, dim, cudaStream);
236 }

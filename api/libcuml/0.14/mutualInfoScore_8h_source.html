49 template < typename T, int BLOCK_DIM_X, int BLOCK_DIM_Y>
50 __global__ void mutualInfoKernel ( const int *dContingencyMatrix, const int *a,
51 const int *b, int numUniqueClasses, int size,
52 double *d_MI) {
53 //calculating the indices of pairs of datapoints compared by the current thread
54 int j = threadIdx.x + blockIdx.x * blockDim.x;
55 int i = threadIdx.y + blockIdx.y * blockDim.y;
56
57 //thread-local variable to count the mutual info
58 double localMI = 0.0;
59
60 if ( i < numUniqueClasses && j < numUniqueClasses && a[ i ] * b[ j ] != 0 &&
61 dContingencyMatrix[ i * numUniqueClasses + j ] != 0) {
62 localMI += (double(dContingencyMatrix[ i * numUniqueClasses + j ])) *
63 (log( double (size) *
64 double (dContingencyMatrix[ i * numUniqueClasses + j ])) -
65 log( double (a[ i ] * b[ j ])));
66 }
67
68 //specialize blockReduce for a 2D block of 1024 threads of type uint64_t
69 typedef cub::BlockReduce <double, BLOCK_DIM_X,
70 cub::BLOCK_REDUCE_WARP_REDUCTIONS, BLOCK_DIM_Y>
71 BlockReduce ;
72
73 //Allocate shared memory for blockReduce
74 __shared__ typename BlockReduce::TempStorage temp_storage ;
75
76 //summing up thread-local counts specific to a block
77 localMI = BlockReduce ( temp_storage ).Sum(localMI);
78 __syncthreads ();
79
80 //executed once per block
81 if (threadIdx.x == 0 && threadIdx.y == 0) {
82 myAtomicAdd (d_MI, localMI);
83 }
84 }

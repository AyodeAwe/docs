44 template < typename OutType, typename MathType, typename GenType,
45 typename LenType, typename Lambda>
46 __global__ void randKernel (uint64_t seed, uint64_t offset, OutType *ptr,
47 LenType len, Lambda randOp) {
48 LenType tid = (blockIdx.x * blockDim.x) + threadIdx.x;
49 detail::Generator<GenType> gen(seed, (uint64_t)tid, offset);
50 const LenType stride = gridDim.x * blockDim.x;
51 for (LenType idx = tid; idx < len; idx += stride) {
52 MathType val;
53 gen. next (val);
54 ptr[idx] = randOp(val, idx);
55 }
56 }

59 template < typename OutType, typename MathType, typename GenType,
60 typename LenType, typename Lambda2>
61 __global__ void rand2Kernel (uint64_t seed, uint64_t offset, OutType *ptr,
62 LenType len, Lambda2 rand2Op) {
63 LenType tid = (blockIdx.x * blockDim.x) + threadIdx.x;
64 detail::Generator<GenType> gen(seed, (uint64_t)tid, offset);
65 const LenType stride = gridDim.x * blockDim.x;
66 for (LenType idx = tid; idx < len; idx += stride) {
67 MathType val1, val2;
68 gen. next (val1);
69 gen.next(val2);
70 rand2Op(val1, val2, idx, idx + stride);
71 if (idx < len) ptr[idx] = (OutType)val1;
72 idx += stride;
73 if (idx < len) ptr[idx] = (OutType)val2;
74 }
75 }

225 template < typename Type, typename LenType = int >
226 void normalTable (Type *ptr, LenType n_rows, LenType n_cols, const Type *mu,
227 const Type *sigma_vec, Type sigma, cudaStream_t stream) {
228 rand2Impl(
229 offset, ptr, n_rows * n_cols,
230 [=] __device__(Type & val1, Type & val2, LenType idx1, LenType idx2) {
231 // yikes! use fast-int-div
232 auto col1 = idx1 % n_cols;
233 auto col2 = idx2 % n_cols;
234 auto mean1 = mu[col1];
235 auto mean2 = mu[col2];
236 auto sig1 = sigma_vec == nullptr ? sigma : sigma_vec[col1];
237 auto sig2 = sigma_vec == nullptr ? sigma : sigma_vec[col2];
238 constexpr Type twoPi = Type(2.0) * Type(3.141592654);
239 constexpr Type minus2 = -Type(2.0);
240 Type R = mySqrt (minus2 * myLog (val1));
241 Type theta = twoPi * val2;
242 Type s, c;
243 mySinCos (theta, s, c);
244 val1 = R * c * sig1 + mean1;
245 val2 = R * s * sig2 + mean2;
246 },
247 NumThreads, nBlocks, type, stream);
248 }

433 void laplace (Type *ptr, LenType len, Type mu, Type scale,
434 cudaStream_t stream) {
435 randImpl(
436 offset, ptr, len,
437 [=] __device__(Type val, LenType idx) {
438 constexpr Type one = (Type)1.0;
439 constexpr Type two = (Type)2.0;
440 constexpr Type oneHalf = (Type)0.5;
441 Type out;

478 template < typename DataT, typename WeightsT, typename IdxT = int >
479 void sampleWithoutReplacement (DataT *out, IdxT *outIdx, const DataT *in,
480 const WeightsT *wts, IdxT sampledLen, IdxT len,
481 std::shared_ptr<deviceAllocator> allocator,
482 cudaStream_t stream) {
483 ASSERT (sampledLen <= len,
484 "sampleWithoutReplacement: 'sampledLen' cant be more than 'len'." );
485 device_buffer<WeightsT> expWts(allocator, stream, len);
486 device_buffer<WeightsT> sortedWts(allocator, stream, len);
487 device_buffer<IdxT> inIdx(allocator, stream, len);
488 device_buffer<IdxT> outIdxBuff(allocator, stream, len);
489 auto *inIdxPtr = inIdx. data ();
490 // generate modified weights
491 randImpl(
492 offset, expWts. data (), len,
493 [wts, inIdxPtr] __device__(WeightsT val, IdxT idx) {
494 inIdxPtr[idx] = idx;
495 constexpr WeightsT one = (WeightsT)1.0;
496 auto exp = - myLog (one - val);
497 if (wts != nullptr ) {
498 return exp / wts[idx];
499 }
500 return exp;
501 },
502 NumThreads, nBlocks, type, stream);
504 // sort the array and pick the top sampledLen items
505 IdxT *outIdxPtr = outIdxBuff. data ();
506 device_buffer<char> workspace(allocator, stream);
507 sortPairs (workspace, expWts. data (), sortedWts. data (), inIdxPtr, outIdxPtr,
508 ( int )len, stream);
509 if (outIdx != nullptr ) {
510 CUDA_CHECK (cudaMemcpyAsync(outIdx, outIdxPtr, sizeof (IdxT) * sampledLen,
511 cudaMemcpyDeviceToDevice, stream));
512 }
513 scatter<DataT, IdxT>(out, in, outIdxPtr, sampledLen, stream);
514 }

532 template < bool IsNormal, typename Type, typename LenType>
533 uint64_t _setupSeeds(uint64_t & seed , uint64_t &offset, LenType len,
534 int nThreads, int nBlocks) {
535 LenType itemsPerThread = ceildiv (len, LenType(nBlocks * nThreads));
536 if (IsNormal && itemsPerThread % 2 == 1) {
537 ++itemsPerThread;
538 }
539 // curand uses 2 32b uint's to generate one double
540 uint64_t factor = sizeof (Type) / sizeof ( float );
541 if (factor == 0) ++factor;
542 // Check if there are enough random numbers left in sequence
543 // If not, then generate new seed and start from zero offset
544 uint64_t newOffset = offset + LenType(itemsPerThread) * factor;
545 if (newOffset < offset) {
546 offset = 0;
547 seed = gen();
548 newOffset = itemsPerThread * factor;
549 }
550 return newOffset;
551 }

553 template < typename OutType, typename MathType = OutType,
554 typename LenType = int, typename Lambda>
555 void randImpl(uint64_t &offset, OutType *ptr, LenType len, Lambda randOp,
556 int nThreads, int nBlocks, GeneratorType type,
557 cudaStream_t stream) {
558 if (len <= 0) return ;
559 uint64_t seed = gen();
560 auto newOffset = _setupSeeds<false, MathType, LenType>( seed , offset, len,
561 nThreads, nBlocks);
562 switch (type) {
563 case GenPhilox :
564 randKernel<OutType, MathType, detail::PhiloxGenerator, LenType, Lambda>
565 <<<nBlocks, nThreads, 0, stream>>>( seed , offset, ptr, len, randOp);
566 break ;
567 case GenTaps :
568 randKernel<OutType, MathType, detail::TapsGenerator, LenType, Lambda>
569 <<<nBlocks, nThreads, 0, stream>>>( seed , offset, ptr, len, randOp);
570 break ;
571 case GenKiss99 :
572 randKernel<OutType, MathType, detail::Kiss99Generator, LenType, Lambda>
573 <<<nBlocks, nThreads, 0, stream>>>( seed , offset, ptr, len, randOp);
574 break ;
575 default :
576 ASSERT ( false , "randImpl: Incorrect generator type! %d" , type);
577 };
578 CUDA_CHECK (cudaPeekAtLastError());
579 offset = newOffset;
580 }

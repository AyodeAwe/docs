25 template < typename InType, int VecLen, typename Lambda, typename IdxType,
26 typename OutType>
27 __global__ void binaryOpKernel (OutType *out, const InType *in1,
28 const InType *in2, IdxType len, Lambda op) {
29 typedef TxN_t<InType, VecLen> InVecType;
30 typedef TxN_t<OutType, VecLen> OutVecType;
31 InVecType a, b;
32 OutVecType c;
33 IdxType idx = threadIdx.x + ((IdxType)blockIdx.x * blockDim.x);
34 idx *= InVecType::Ratio;
35 if (idx >= len) return ;
36 a. load (in1, idx);
37 b.load(in2, idx);
38 #pragma unroll
39 for ( int i = 0; i < InVecType::Ratio; ++ i ) {
40 c.val.data[ i ] = op(a.val.data[ i ], b.val.data[ i ]);
41 }
42 c.store(out, idx);
43 }

45 template < typename InType, int VecLen, typename Lambda, typename IdxType,
46 typename OutType, int TPB>
47 void binaryOpImpl (OutType *out, const InType *in1, const InType *in2,
48 IdxType len, Lambda op, cudaStream_t stream) {
49 const IdxType nblks = ceildiv (VecLen ? len / VecLen : len, (IdxType)TPB);
50 binaryOpKernel<InType, VecLen, Lambda, IdxType, OutType>
51 <<<nblks, TPB, 0, stream>>>(out, in1, in2, len, op);
52 CUDA_CHECK (cudaPeekAtLastError());
53 }

71 template < typename InType, typename Lambda, typename OutType = InType,
72 typename IdxType = int, int TPB = 256>
73 void binaryOp (OutType *out, const InType *in1, const InType *in2, IdxType len,
74 Lambda op, cudaStream_t stream) {
75 constexpr auto maxSize =
76 sizeof (InType) > sizeof (OutType) ? sizeof (InType) : sizeof (OutType);
77 size_t bytes = len * maxSize;
78 if (16 / maxSize && bytes % 16 == 0) {
79 binaryOpImpl <InType, 16 / maxSize, Lambda, IdxType, OutType, TPB>(
80 out, in1, in2, len, op, stream);
81 } else if (8 / maxSize && bytes % 8 == 0) {
82 binaryOpImpl <InType, 8 / maxSize, Lambda, IdxType, OutType, TPB>(
83 out, in1, in2, len, op, stream);
84 } else if (4 / maxSize && bytes % 4 == 0) {
85 binaryOpImpl <InType, 4 / maxSize, Lambda, IdxType, OutType, TPB>(
86 out, in1, in2, len, op, stream);
87 } else if (2 / maxSize && bytes % 2 == 0) {
88 binaryOpImpl <InType, 2 / maxSize, Lambda, IdxType, OutType, TPB>(
89 out, in1, in2, len, op, stream);
90 } else if (1 / maxSize) {
91 binaryOpImpl <InType, 1 / maxSize, Lambda, IdxType, OutType, TPB>(
92 out, in1, in2, len, op, stream);
93 } else {
94 binaryOpImpl<InType, 1, Lambda, IdxType, OutType, TPB>(out, in1, in2, len,
95 op, stream);
96 }

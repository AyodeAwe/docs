62 template < typename T, int BLOCK_DIM_X, int BLOCK_DIM_Y>
63 __global__ void computeTheNumerator ( const T* firstClusterArray,
64 const T* secondClusterArray, uint64_t size,
65 uint64_t* a, uint64_t* b) {
66 //calculating the indices of pairs of datapoints compared by the current thread
67 uint64_t j = threadIdx.x + blockIdx.x * blockDim.x;
68 uint64_t i = threadIdx.y + blockIdx.y * blockDim.y;
69
70 //thread-local variables to count a and b
71 uint64_t myA = 0, myB = 0;
72
73 if ( i < size && j < size && j < i ) {
74 //checking if the pair have been classified the same by both the clusters
75 if (firstClusterArray[ i ] == firstClusterArray[ j ] &&
76 secondClusterArray[ i ] == secondClusterArray[ j ]) {
77 ++myA;
78 }
79
80 //checking if the pair have been classified differently by both the clusters
81 else if (firstClusterArray[ i ] != firstClusterArray[ j ] &&
82 secondClusterArray[ i ] != secondClusterArray[ j ]) {
83 ++myB;
84 }
85 }
86
87 //specialize blockReduce for a 2D block of 1024 threads of type uint64_t
88 typedef cub::BlockReduce <uint64_t, BLOCK_DIM_X,
89 cub::BLOCK_REDUCE_WARP_REDUCTIONS, BLOCK_DIM_Y>
90 BlockReduce ;
91
92 //Allocate shared memory for blockReduce
93 __shared__ typename BlockReduce::TempStorage temp_storage ;
94
95 //summing up thread-local counts specific to a block
96 myA = BlockReduce ( temp_storage ).Sum(myA);
97 __syncthreads ();
98 myB = BlockReduce ( temp_storage ).Sum(myB);
99 __syncthreads ();
100
101 //executed once per block
102 if (threadIdx.x == 0 && threadIdx.y == 0) {
103 atomicAdd(( unsigned long long int *)a, myA);
104 atomicAdd(( unsigned long long int *)b, myB);
105 }
106 }

117 template < typename T>
118 double computeRandIndex (T* firstClusterArray, T* secondClusterArray,
119 uint64_t size,
120 std::shared_ptr<MLCommon::deviceAllocator> allocator,
121 cudaStream_t stream) {
122 //rand index for size less than 2 is not defined
123 ASSERT (size >= 2, "Rand Index for size less than 2 not defined!" );
124
125 //allocating and initializing memory for a and b in the GPU
126 MLCommon::device_buffer<uint64_t> arr_buf(allocator, stream, 2);
127 CUDA_CHECK (cudaMemsetAsync(arr_buf. data (), 0, 2 * sizeof (uint64_t), stream));
128
129 //kernel configuration
130 static const int BLOCK_DIM_Y = 16, BLOCK_DIM_X = 16;
131 dim3 numThreadsPerBlock(BLOCK_DIM_X, BLOCK_DIM_Y);
132 dim3 numBlocks(ceildiv<int>(size, numThreadsPerBlock.x),
133 ceildiv<int>(size, numThreadsPerBlock.y));
134
135 //calling the kernel
136 computeTheNumerator<T, BLOCK_DIM_X, BLOCK_DIM_Y>
137 <<<numBlocks, numThreadsPerBlock, 0, stream>>>(
138 firstClusterArray, secondClusterArray, size, arr_buf. data (),
139 arr_buf. data () + 1);
140
141 //synchronizing and updating the calculated values of a and b from device to host
142 uint64_t ab_host[2] = {0};
143 MLCommon::updateHost (ab_host, arr_buf. data (), 2, stream);
144 CUDA_CHECK (cudaStreamSynchronize(stream));
145
146 //error handling
147 CUDA_CHECK (cudaGetLastError());
148
149 //denominator
150 uint64_t nChooseTwo = size * (size - 1) / 2;
151
152 //calculating the randIndex
153 return ( double )(((double)(ab_host[0] + ab_host[1])) / ( double )nChooseTwo);
154 }

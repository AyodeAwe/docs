25 template < typename Type, int veclen_, typename Lambda, typename IdxType>
26 __global__ void matrixVectorOpKernel (Type *out, const Type *matrix,
27 const Type *vector, IdxType D, IdxType N ,
28 bool rowMajor, bool bcastAlongRows,
29 Lambda op) {
30 typedef TxN_t<Type, veclen_> VecType;
31 IdxType len = N * D;
32 IdxType idx = threadIdx.x;
33 idx += (IdxType)blockIdx.x * (IdxType)blockDim.x;
34 idx *= VecType::Ratio;
35 if (idx >= len) return ;
36 IdxType vIdx;
37 VecType mat, vec;
40 if (rowMajor && bcastAlongRows) {
41 vIdx = idx % D;
42 vec.load(vector, vIdx);
43 } else if (!rowMajor && !bcastAlongRows) {
44 vIdx = idx % N ;
45 vec.load(vector, vIdx);
46 } else if (rowMajor && !bcastAlongRows) {
47 vIdx = idx / D;
48 vec.fill(vector[vIdx]);
49 } else {
50 vIdx = idx / N ;
51 vec.fill(vector[vIdx]);
52 }
53 mat.load(matrix, idx);
54 #pragma unroll
55 for ( int i = 0; i < VecType::Ratio; ++ i )
56 mat.val.data[ i ] = op(mat.val.data[ i ], vec.val.data[ i ]);
57 mat.store(out, idx);
58 }

119 template < typename Type, int veclen_, typename Lambda, typename IdxType>
120 __global__ void matrixVectorOpKernel (Type *out, const Type *matrix,
121 const Type *vector1, const Type *vector2,
122 IdxType D, IdxType N , bool rowMajor,
123 bool bcastAlongRows, Lambda op) {
124 typedef TxN_t<Type, veclen_> VecType;
125 IdxType len = N * D;
126 IdxType idx = (threadIdx.x + (blockIdx.x * blockDim.x)) * VecType::Ratio;
127 if (idx >= len) return ;
128 IdxType vIdx;
129 VecType mat, vec1, vec2;
132 if (rowMajor && bcastAlongRows) {
133 vIdx = idx % D;
134 vec1.load(vector1, vIdx);
135 vec2.load(vector2, vIdx);
136 } else if (!rowMajor && !bcastAlongRows) {
137 vIdx = idx % N ;
138 vec1.load(vector1, vIdx);
139 vec2.load(vector2, vIdx);
140 } else if (rowMajor && !bcastAlongRows) {
141 vIdx = idx / D;
142 vec1.fill(vector1[vIdx]);
143 vec2.fill(vector2[vIdx]);
144 } else {
145 vIdx = idx / N ;
146 vec1.fill(vector1[vIdx]);
147 vec2.fill(vector2[vIdx]);
148 }
149 mat.load(matrix, idx);
150 #pragma unroll
151 for ( int i = 0; i < VecType::Ratio; ++ i )
152 mat.val.data[ i ] = op(mat.val.data[ i ], vec1.val.data[ i ], vec2.val.data[ i ]);
153 mat.store(out, idx);
154 }

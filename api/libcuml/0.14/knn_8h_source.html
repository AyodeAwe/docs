58 template < int warp_q, int thread_q, int tpb>
59 __global__ void knn_merge_parts_kernel ( float *inK, int64_t *inV, float *outK,
60 int64_t *outV, size_t n_samples,
61 int n_parts, float initK, int64_t initV,
62 int k , int64_t *translations) {
63 constexpr int kNumWarps = tpb / faiss::gpu::kWarpSize;
64
65 __shared__ float smemK[kNumWarps * warp_q];
66 __shared__ int64_t smemV[kNumWarps * warp_q];
67
71 faiss::gpu::BlockSelect<float, int64_t, false, faiss::gpu::Comparator<float>,
72 warp_q, thread_q, tpb>
73 heap(initK, initV, smemK, smemV, k );
74
75 // Grid is exactly sized to rows available
76 int row = blockIdx.x;
77 int total_k = k * n_parts;
78
79 int i = threadIdx.x;
80
81 // Get starting pointers for cols in current thread
82 int part = i / k ;
83 size_t row_idx = (row * k ) + (part * n_samples * k );
84
85 int col = i % k ;
86
87 float *inKStart = inK + (row_idx + col);
88 int64_t *inVStart = inV + (row_idx + col);
89
90 int limit = faiss::gpu::utils::roundDown(total_k, faiss::gpu::kWarpSize);
91 int64_t translation = 0;
92
93 for (; i < limit; i += tpb) {
94 translation = translations[part];
95 heap.add(*inKStart, (*inVStart) + translation);
96
97 part = ( i + tpb) / k ;
98 row_idx = (row * k ) + (part * n_samples * k );
99
100 col = ( i + tpb) % k ;
101
102 inKStart = inK + (row_idx + col);
103 inVStart = inV + (row_idx + col);
104 }
105
106 // Handle last remainder fraction of a warp of elements
107 if ( i < total_k) {
108 translation = translations[part];
109 heap.addThreadQ(*inKStart, (*inVStart) + translation);
110 }
111
112 heap.reduce();
113
114 for ( int i = threadIdx.x; i < k ; i += tpb) {
115 outK[row * k + i ] = smemK[ i ];
116 outV[row * k + i ] = smemV[ i ];
117 }
118 }

201 template < typename IntType = int,
202 Distance::DistanceType DistanceType = Distance::EucUnexpandedL2 >
203 void brute_force_knn (std::vector<float *> &input, std::vector<int> &sizes,
204 IntType D, float *search_items, IntType n , int64_t *res_I,
205 float *res_D, IntType k ,
206 std::shared_ptr<deviceAllocator> allocator,
207 cudaStream_t userStream,
208 cudaStream_t *internalStreams = nullptr ,
209 int n_int_streams = 0, bool rowMajorIndex = true ,
210 bool rowMajorQuery = true ,
211 std::vector<int64_t> *translations = nullptr ) {
212 ASSERT ( DistanceType == Distance::EucUnexpandedL2 ||
213 DistanceType == Distance::EucUnexpandedL2Sqrt ,
214 "Only EucUnexpandedL2Sqrt and EucUnexpandedL2 metrics are supported "
215 "currently." );
216
217 ASSERT (input.size() == sizes.size(),
218 "input and sizes vectors should be the same size" );
219
220 std::vector<int64_t> *id_ranges;
221 if (translations == nullptr ) {
222 // If we don't have explicit translations
223 // for offsets of the indices, build them
224 // from the local partitions
225 id_ranges = new std::vector<int64_t>();
226 int64_t total_n = 0;
227 for ( int i = 0; i < input.size(); i ++) {
228 id_ranges->push_back(total_n);
229 total_n += sizes[ i ];
230 }
231 } else {
232 // otherwise, use the given translations
233 id_ranges = translations;
234 }
235
236 int device;
237 CUDA_CHECK (cudaGetDevice(&device));
238
239 device_buffer<int64_t> trans(allocator, userStream, id_ranges->size());
240 updateDevice (trans. data (), id_ranges->data(), id_ranges->size(), userStream);
241
242 device_buffer<float> all_D(allocator, userStream, input.size() * k * n );
243 device_buffer<int64_t> all_I(allocator, userStream, input.size() * k * n );
244
245 // Sync user stream only if using other streams to parallelize query
246 if (n_int_streams > 0) CUDA_CHECK (cudaStreamSynchronize(userStream));
247
248 for ( int i = 0; i < input.size(); i ++) {
249 faiss::gpu::StandardGpuResources gpu_res;
250
251 cudaStream_t stream =
252 select_stream (userStream, internalStreams, n_int_streams, i );
253
254 gpu_res.noTempMemory();
255 gpu_res.setCudaMallocWarning( false );
256 gpu_res.setDefaultStream(device, stream);
257
258 faiss::gpu::bruteForceKnn(
259 &gpu_res, faiss::METRIC_L2, input[ i ], rowMajorIndex, sizes[ i ],
260 search_items, rowMajorQuery, n , D, k , all_D. data () + ( i * k * n ),
261 all_I. data () + ( i * k * n ));
262
263 CUDA_CHECK (cudaPeekAtLastError());
264 }
265
266 // Sync internal streams if used. We don't need to
267 // sync the user stream because we'll already have
268 // fully serial execution.
269 for ( int i = 0; i < n_int_streams; i ++) {
270 CUDA_CHECK (cudaStreamSynchronize(internalStreams[ i ]));
271 }
272
273 knn_merge_parts (all_D. data (), all_I. data (), res_D, res_I, n , input.size(), k ,
274 userStream, trans. data ());
275
276 MLCommon::LinAlg::unaryOp<float>(
277 res_D, res_D, n * k , [] __device__( float input) { return sqrt (input); },
278 userStream);
279
280 if (translations == nullptr ) delete id_ranges;
281 };

308 template < typename OutType = float >
309 __global__ void class_probs_kernel (OutType *out, const int64_t *knn_indices,
310 const int *labels, int n_uniq_labels,
311 size_t n_labels, size_t n_samples,
312 int n_neighbors) {
313 int row = (blockIdx.x * blockDim.x) + threadIdx.x;
314 int i = row * n_neighbors;
315
316 float n_neigh_inv = 1.0f / n_neighbors;
317
318 if (row >= n_samples) return ;
319
320 for ( int j = 0; j < n_neighbors; j ++) {
321 int64_t neighbor_idx = knn_indices[ i + j ];
322 int out_label = labels[neighbor_idx];
323 int out_idx = row * n_uniq_labels + out_label;
324 out[out_idx] += n_neigh_inv;
325 }
326 }

328 template < typename OutType = int >
329 __global__ void class_vote_kernel (OutType *out, const float *class_proba,
330 int *unique_labels, int n_uniq_labels,
331 size_t n_labels, size_t n_samples,
332 int n_outputs, int output_offset) {
333 int row = (blockIdx.x * blockDim.x) + threadIdx.x;
334 int i = row * n_uniq_labels;
335
336 extern __shared__ int label_cache[];
337 for ( int j = threadIdx.x; j < n_uniq_labels; j += blockDim.x) {
338 label_cache[ j ] = unique_labels[ j ];
339 }
340
341 __syncthreads ();
342
343 if (row >= n_samples) return ;
344 float cur_max = -1.0;
345 int cur_label = -1;
346 for ( int j = 0; j < n_uniq_labels; j ++) {
347 float cur_proba = class_proba[ i + j ];
348 if (cur_proba > cur_max) {
349 cur_max = cur_proba;
350 cur_label = j ;
351 }
352 }
353 out[row * n_outputs + output_offset] = label_cache[cur_label];
354 }

356 template < typename LabelType>
357 __global__ void regress_avg_kernel (LabelType *out, const int64_t *knn_indices,
358 const LabelType *labels, size_t n_labels,
359 size_t n_samples, int n_neighbors,
360 int n_outputs, int output_offset) {
361 int row = (blockIdx.x * blockDim.x) + threadIdx.x;
362 int i = row * n_neighbors;
363
364 if (row >= n_samples) return ;
365
366 // should work for moderately small number of classes
367 LabelType pred = 0;
368 for ( int j = 0; j < n_neighbors; j ++) {
369 int64_t neighbor_idx = knn_indices[ i + j ];
370 pred += labels[neighbor_idx];
371 }
372
373 out[row * n_outputs + output_offset] = pred / (LabelType)n_neighbors;
374 }

398 template < int TPB_X = 32>
399 void class_probs (std::vector<float *> &out, const int64_t *knn_indices,
400 std::vector<int *> & y , size_t n_labels, size_t n_rows, int k ,
401 std::vector<int *> &uniq_labels, std::vector<int> &n_unique,
402 std::shared_ptr<deviceAllocator> allocator,

404 int n_int_streams = 0) {
405 for ( int i = 0; i < y .size(); i ++) {
406 cudaStream_t stream =
407 select_stream (user_stream, int_streams, n_int_streams, i );
408
409 int n_unique_labels = n_unique[ i ];

423 stream);

425 y_normalized. data (), y_normalized. data (), n_labels,
426 [] __device__( int input) { return input - 1; }, stream);

456 template < int TPB_X = 32>
457 void knn_classify ( int *out, const int64_t *knn_indices, std::vector<int *> & y ,
458 size_t n_labels, size_t n_rows, int k ,
459 std::vector<int *> &uniq_labels, std::vector<int> &n_unique,
460 std::shared_ptr<deviceAllocator> &allocator,
461 cudaStream_t user_stream, cudaStream_t *int_streams = nullptr ,
462 int n_int_streams = 0) {
463 std::vector<float *> probs;
464 std::vector<device_buffer<float> *> tmp_probs;
465
466 // allocate temporary memory
467 for ( int i = 0; i < n_unique.size(); i ++) {
468 int size = n_unique[ i ];
469
470 cudaStream_t stream =
471 select_stream (user_stream, int_streams, n_int_streams, i );
472
473 device_buffer<float> *probs_buff =
474 new device_buffer<float> (allocator, stream, n_rows * size);
475
476 tmp_probs.push_back(probs_buff);
477 probs.push_back(probs_buff-> data ());
478 }
479
487 class_probs (probs, knn_indices, y , n_labels, n_rows, k , uniq_labels, n_unique,
488 allocator, user_stream, int_streams, n_int_streams);
489
490 dim3 grid( MLCommon::ceildiv (n_rows, ( size_t ) TPB_X ), 1, 1);
491 dim3 blk( TPB_X , 1, 1);
492
493 for ( int i = 0; i < y .size(); i ++) {
494 cudaStream_t stream =
495 select_stream (user_stream, int_streams, n_int_streams, i );
496
497 int n_unique_labels = n_unique[ i ];
498
502 int smem = sizeof (int) * n_labels;
503 class_vote_kernel<<<grid, blk, smem, stream>>>(
504 out, probs[ i ], uniq_labels[ i ], n_unique_labels, n_labels, n_rows,
505 y .size(), i );
506 CUDA_CHECK (cudaPeekAtLastError());
507
508 delete tmp_probs[ i ];
509 }

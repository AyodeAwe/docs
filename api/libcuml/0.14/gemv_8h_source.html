30 
31  // Unfortunately there is a clash of terminology
32  // in BLAS https://docs.nvidia.com/cuda/cublas/index.html is opposite to Machine Learning
33  // In blas:
34  // m - number of rows in input matrix
35  // n - number of columns in input matrix
36  // lda - purpose of it to have ability to operate on submatrices of matrix without copying.
37  // If you're not think about it it's always should be equal to m
38  // lda has deal with memory layout, but has nothing with the requirement for cuBLAS perform transpose
39 
40  // In Machine Learning:
41  // m - nunmber of columns in design matrix(number of features)
42  // n - number of rows in designed matrix (number of train examples)
43 
44  int m = n_rows;
45  int n = n_cols;
46  int lda = trans_a ? m : n;
47 
48  CUBLAS_CHECK(cublasgemv(cublas_h, op_a, m, n, &alpha, a, lda, x, incx, &beta,
49  y, incy, stream));
50 }

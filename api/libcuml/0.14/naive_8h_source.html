79 template < int TPB_X, typename T>
80 __global__ void smooth_knn_dist_kernel (
81 const T *knn_dists, int n , float mean_dist, T *sigmas,
82 T *rhos, // Size of n, iniitalized to zeros
83 int n_neighbors, float local_connectivity = 1.0, int n_iter = 64,
84 float bandwidth = 1.0) {
85 // row-based matrix 1 thread per row
86 int row = (blockIdx.x * TPB_X ) + threadIdx.x;
87 int i =
88 row * n_neighbors; // each thread processes one row of the dist matrix
89
90 if (row < n ) {
91 float target = __log2f(n_neighbors) * bandwidth;
92
93 float lo = 0.0;
94 float hi = MAX_FLOAT;
95 float mid = 1.0;
96
97 int total_nonzero = 0;
98 int max_nonzero = -1;
99
100 int start_nonzero = -1;
101 float sum = 0.0;
102
103 for ( int idx = 0; idx < n_neighbors; idx++) {
104 float cur_dist = knn_dists[ i + idx];
105 sum += cur_dist;
106
107 if (cur_dist > 0.0) {
108 if (start_nonzero == -1) start_nonzero = idx;
109 total_nonzero++;
110 }
111
112 if (cur_dist > max_nonzero) max_nonzero = cur_dist;
113 }
114
115 float ith_distances_mean = sum / float(n_neighbors);
116 if (total_nonzero >= local_connectivity) {
117 int index = int(floor(local_connectivity));
118 float interpolation = local_connectivity - index;
119
120 if (index > 0) {
121 rhos[row] = knn_dists[ i + start_nonzero + (index - 1)];
122
123 if (interpolation > SMOOTH_K_TOLERANCE) {
124 rhos[row] +=
125 interpolation * (knn_dists[ i + start_nonzero + index] -
126 knn_dists[ i + start_nonzero + (index - 1)]);
127 }
128 } else
129 rhos[row] = interpolation * knn_dists[ i + start_nonzero];
130 } else if (total_nonzero > 0)
131 rhos[row] = max_nonzero;
132
133 for ( int iter = 0; iter < n_iter; iter++) {
134 float psum = 0.0;
135
136 for ( int j = 1; j < n_neighbors; j ++) {
137 float d = knn_dists[ i + j ] - rhos[row];
138 if (d > 0)
139 psum += exp(-(d / mid));
140 else
141 psum += 1.0;
142 }
143
144 if (fabsf(psum - target) < SMOOTH_K_TOLERANCE) {
145 break ;
146 }
147
148 if (psum > target) {
149 hi = mid;
150 mid = (lo + hi) / 2.0;
151 } else {
152 lo = mid;
153 if (hi == MAX_FLOAT)
154 mid *= 2;
155 else
156 mid = (lo + hi) / 2.0;
157 }
158 }
159
160 sigmas[row] = mid;
161
162 if (rhos[row] > 0.0) {
163 if (sigmas[row] < MIN_K_DIST_SCALE * ith_distances_mean)
164 sigmas[row] = MIN_K_DIST_SCALE * ith_distances_mean;
165 } else {
166 if (sigmas[row] < MIN_K_DIST_SCALE * mean_dist)
167 sigmas[row] = MIN_K_DIST_SCALE * mean_dist;
168 }
169 }
170 }

190 template < int TPB_X, typename T>
191 __global__ void compute_membership_strength_kernel (
192 const int64_t *knn_indices,
193 const float *knn_dists, // nn outputs
194 const T *sigmas, const T *rhos, // continuous dists to nearest neighbors
195 T *vals, int *rows, int *cols, // result coo
196 int n , int n_neighbors) { // model params
197
198 // row-based matrix is best
199 int idx = (blockIdx.x * TPB_X ) + threadIdx.x;
200
201 if (idx < n * n_neighbors) {
202 int row = idx / n_neighbors; // one neighbor per thread
203
204 double cur_rho = rhos[row];
205 double cur_sigma = sigmas[row];
206
207 int64_t cur_knn_ind = knn_indices[idx];
208 double cur_knn_dist = knn_dists[idx];
209
210 if (cur_knn_ind != -1) {
211 double val = 0.0;
212 if (cur_knn_ind == row)
213 val = 0.0;
214 else if (cur_knn_dist - cur_rho <= 0.0 || cur_sigma == 0.0)
215 val = 1.0;
216 else {
217 val = exp(-((cur_knn_dist - cur_rho) / (cur_sigma)));
218
219 if (val < MIN_FLOAT) val = MIN_FLOAT;
220 }
221
222 rows[idx] = row;
223 cols[idx] = cur_knn_ind;
224 vals[idx] = val;
225 }
226 }
227 }

232 template < int TPB_X, typename T>
233 void smooth_knn_dist ( int n , const int64_t *knn_indices, const float *knn_dists,
234 T *rhos, T *sigmas, UMAPParams * params , int n_neighbors,
235 float local_connectivity,
236 std::shared_ptr<deviceAllocator> d_alloc,
237 cudaStream_t stream) {
238 dim3 grid( MLCommon::ceildiv ( n , TPB_X ), 1, 1);
239 dim3 blk( TPB_X , 1, 1);
240
241 MLCommon::device_buffer<T> dist_means_dev(d_alloc, stream, n_neighbors);
242
243 MLCommon::Stats::mean (dist_means_dev. data (), knn_dists, 1, n_neighbors * n ,
244 false , false , stream);
245 CUDA_CHECK (cudaPeekAtLastError());
246
247 T mean_dist = 0.0;
248 MLCommon::updateHost (&mean_dist, dist_means_dev. data (), 1, stream);
249 CUDA_CHECK (cudaStreamSynchronize(stream));
250
254 smooth_knn_dist_kernel<TPB_X><<<grid, blk, 0, stream>>>(
255 knn_dists, n , mean_dist, sigmas, rhos, n_neighbors, local_connectivity);
256 CUDA_CHECK (cudaPeekAtLastError());
257 }

72 template < typename IdxT, int VecLen>
73 dim3 computeGridDim (IdxT nrows , IdxT ncols , const void * kernel) {
74 int occupancy;
75 CUDA_CHECK (cudaOccupancyMaxActiveBlocksPerMultiprocessor(&occupancy, kernel,
76 ThreadsPerBlock, 0));
77 const auto maxBlks = occupancy * getMultiProcessorCount ();
78 int nblksx = ceildiv<int>(VecLen ? nrows / VecLen : nrows , ThreadsPerBlock);
79 // for cases when there aren't a lot of blocks for computing one histogram
80 nblksx = std::min(nblksx, maxBlks);
81 return dim3(nblksx, ncols );
82 }

84 template < typename DataT, typename BinnerOp, typename IdxT, int VecLen,
85 typename CoreOp>
86 DI void histCoreOp ( const DataT* data, IdxT nrows , IdxT nbins, BinnerOp binner,
87 CoreOp op, IdxT col) {
88 IdxT offset = col * nrows ;
89 auto bdim = IdxT(blockDim.x);
90 IdxT tid = threadIdx.x + bdim * blockIdx.x;
91 tid *= VecLen;
92 IdxT stride = bdim * gridDim.x * VecLen;
93 int nCeil = alignTo<int>( nrows , stride);
94 typedef TxN_t<DataT, VecLen> VecType;
95 VecType a;
96 for ( auto i = tid; i < nCeil; i += stride) {
97 if ( i < nrows ) {
98 a. load (data, offset + i );
99 }
100 #pragma unroll
101 for ( int j = 0; j < VecLen; ++ j ) {
102 int binId = binner(a.val.data[ j ], i + j , col);
103 op(binId, i + j , col);
104 }
105 }
106 }

108 template < typename DataT, typename BinnerOp, typename IdxT, int VecLen>
109 __global__ void gmemHistKernel ( int * bins, const DataT* data, IdxT nrows ,
110 IdxT nbins, BinnerOp binner) {
111 auto op = [=] __device__( int binId, IdxT row, IdxT col) {
112 if (row >= nrows ) return ;
113 auto binOffset = col * nbins;
114 #if __CUDA_ARCH__ < 700
115 atomicAdd(bins + binOffset + binId, 1);
116 #else
117 auto amask = __activemask();
118 auto mask = __match_any_sync(amask, binId);
119 auto leader = __ffs(mask) - 1;
120 if ( laneId () == leader) {
121 atomicAdd(bins + binOffset + binId, __popc(mask));
122 }
123 #endif // __CUDA_ARCH__
124 };
125 histCoreOp<DataT, BinnerOp, IdxT, VecLen>(data, nrows , nbins, binner, op,
126 blockIdx.y);
127 }

138 template < typename DataT, typename BinnerOp, typename IdxT, int VecLen,
139 bool UseMatchAny>
140 __global__ void smemHistKernel ( int * bins, const DataT* data, IdxT nrows ,
141 IdxT nbins, BinnerOp binner) {
142 extern __shared__ unsigned sbins[];
143 for ( auto i = threadIdx.x; i < nbins; i += blockDim.x) {
144 sbins[ i ] = 0;
145 }
146 __syncthreads ();
147 auto op = [=] __device__( int binId, IdxT row, IdxT col) {
148 if (row >= nrows ) return ;
149 #if __CUDA_ARCH__ < 700
150 atomicAdd(sbins + binId, 1);
151 #else
152 if (UseMatchAny) {
153 auto amask = __activemask();
154 auto mask = __match_any_sync(amask, binId);
155 auto leader = __ffs(mask) - 1;
156 if ( laneId () == leader) {
157 atomicAdd(sbins + binId, __popc(mask));
158 }
159 } else {
160 atomicAdd(sbins + binId, 1);
161 }
162 #endif // __CUDA_ARCH__
163 };
164 IdxT col = blockIdx.y;
165 histCoreOp<DataT, BinnerOp, IdxT, VecLen>(data, nrows , nbins, binner, op,
166 col);
167 __syncthreads ();
168 auto binOffset = col * nbins;
169 for ( auto i = threadIdx.x; i < nbins; i += blockDim.x) {
170 auto val = sbins[ i ];
171 if (val > 0) {
172 atomicAdd(bins + binOffset + i , val);
173 }
174 }
175 }

190 template < unsigned _BIN_BITS>
191 struct BitsInfo {
192 static unsigned const BIN_BITS = _BIN_BITS;
193 static unsigned const WORD_BITS = sizeof (unsigned) * 8;
194 static unsigned const WORD_BINS = WORD_BITS / BIN_BITS ;
195 static unsigned const BIN_MASK = (1 << BIN_BITS ) - 1;
196 };
197
198 template < unsigned BIN_BITS>
199 DI void incrementBin ( unsigned * sbins, int * bins, int nbins, int binId) {
200 typedef BitsInfo<BIN_BITS> Bits;
201 auto iword = binId / Bits::WORD_BINS;
202 auto ibin = binId % Bits::WORD_BINS;
203 auto sh = ibin * Bits::BIN_BITS;
204 auto old_word = atomicAdd(sbins + iword, unsigned (1 << sh));
205 auto new_word = old_word + unsigned(1 << sh);
206 if ((new_word >> sh & Bits::BIN_MASK) != 0) return ;
207 // overflow
208 atomicAdd(bins + binId, Bits::BIN_MASK + 1);
209 for ( int dbin = 1; ibin + dbin < Bits::WORD_BINS && binId + dbin < nbins;
210 ++dbin) {
211 auto sh1 = (ibin + dbin) * Bits::BIN_BITS;
212 if ((new_word >> sh1 & Bits::BIN_MASK) == 0) {
213 // overflow
214 atomicAdd(bins + binId + dbin, Bits::BIN_MASK);
215 } else {
216 // correction
217 atomicAdd(bins + binId + dbin, -1);
218 break ;
219 }
220 }
221 }
222
223 template <>
224 DI void incrementBin<1> ( unsigned * sbins, int * bins, int nbins, int binId) {
225 typedef BitsInfo<1> Bits;
226 auto iword = binId / Bits::WORD_BITS;
227 auto sh = binId % Bits::WORD_BITS;
228 auto old_word = atomicXor(sbins + iword, unsigned (1 << sh));
229 if ((old_word >> sh & 1) != 0) atomicAdd(bins + binId, 2);
230 }

232 template < typename DataT, typename BinnerOp, typename IdxT, int BIN_BITS,
233 int VecLen>
234 __global__ void smemBitsHistKernel ( int * bins, const DataT* data, IdxT nrows ,
235 IdxT nbins, BinnerOp binner) {
236 extern __shared__ unsigned sbins[];
237 typedef BitsInfo<BIN_BITS> Bits;
238 auto nwords = ceildiv<int>(nbins, Bits::WORD_BINS);
239 for ( auto j = threadIdx.x; j < nwords; j += blockDim.x) {
240 sbins[ j ] = 0;
241 }
242 __syncthreads ();
243 IdxT col = blockIdx.y;
244 IdxT binOffset = col * nbins;
245 auto op = [=] __device__( int binId, IdxT row, IdxT col) {
246 if (row >= nrows ) return ;
247 incrementBin<Bits::BIN_BITS>(sbins, bins + binOffset, ( int )nbins, binId);
248 };
249 histCoreOp<DataT, BinnerOp, IdxT, VecLen>(data, nrows , nbins, binner, op,
250 col);
251 __syncthreads ();
252 for ( auto j = threadIdx.x; j < ( int )nbins; j += blockDim.x) {
253 auto shift = j % Bits::WORD_BINS * Bits::BIN_BITS;
254 int count = sbins[ j / Bits::WORD_BINS] >> shift & Bits::BIN_MASK;
255 if (count > 0) atomicAdd(bins + binOffset + j , count);
256 }
257 }

275 #define INVALID_KEY -1
276 DI void clearHashTable (int2* ht, int hashSize) {
277 for ( auto i = threadIdx.x; i < hashSize; i += blockDim.x) {
278 ht[ i ] = { INVALID_KEY , 0};
279 }

281
282 DI int findEntry (int2* ht, int hashSize, int binId, int threshold) {
283 int idx = binId % hashSize;
284 int t;
285 int count = 0;
286 while ((t = atomicCAS(&(ht[idx]. x ), INVALID_KEY , binId)) != INVALID_KEY &&
287 t != binId) {
288 ++count;
289 if (count >= threshold) {
290 idx = INVALID_KEY ;
291 break ;
292 }
293 ++idx;
294 if (idx >= hashSize) {
295 idx = 0;
296 }
297 }
298 return idx;
299 }
300
301 DI void flushHashTable (int2* ht, int hashSize, int * bins, int nbins, int col) {
302 int binOffset = col * nbins;
303 for ( auto i = threadIdx.x; i < hashSize; i += blockDim.x) {
304 if (ht[ i ]. x != INVALID_KEY && ht[ i ]. y > 0) {
305 atomicAdd(bins + binOffset + ht[ i ]. x , ht[ i ]. y );
306 }
307 ht[ i ] = { INVALID_KEY , 0};
308 }
309 }

313 template < typename DataT, typename BinnerOp, typename IdxT, int VecLen>
314 __global__ void smemHashHistKernel ( int * bins, const DataT* data, IdxT nrows ,
315 IdxT nbins, BinnerOp binner, int hashSize,
316 int threshold) {
317 extern __shared__ int2 ht[];
318 int * needFlush = ( int *)&(ht[hashSize]);
319 if (threadIdx.x == 0) {
320 needFlush[0] = 0;
321 }
322 clearHashTable (ht, hashSize);
323 __syncthreads ();
324 auto op = [=] __device__( int binId, IdxT row, IdxT col) {
325 bool iNeedFlush = false ;
326 if (row < nrows ) {
327 int hidx = findEntry (ht, hashSize, binId, threshold);
328 if (hidx >= 0) {
329 atomicAdd(&(ht[hidx]. y ), 1);
330 } else {
331 needFlush[0] = 1;
332 iNeedFlush = true ;
333 }
334 }
335 __syncthreads ();
336 if (needFlush[0]) {
337 flushHashTable (ht, hashSize, bins, nbins, col);
338 __syncthreads ();
339 if (threadIdx.x == 0) {
340 needFlush[0] = 0;
341 }
342 __syncthreads ();
343 }
344 if (iNeedFlush) {
345 int hidx = findEntry (ht, hashSize, binId, threshold);
346 // all threads are bound to get one valid entry as all threads in this
347 // block will make forward progress due to the __syncthreads call in the
348 // subsequent iteration
349 atomicAdd(&(ht[hidx]. y ), 1);
350 }
351 };
352 IdxT col = blockIdx.y;
353 histCoreOp<DataT, BinnerOp, IdxT, VecLen>(data, nrows , nbins, binner, op,
354 col);
355 __syncthreads ();
356 flushHashTable (ht, hashSize, bins, nbins, col);
357 }

359 inline int computeHashTableSize () {
360 // we shouldn't have this much of shared memory available anytime soon!
361 static const unsigned maxBinsEverPossible = 256 * 1024;
362 static Seive primes(maxBinsEverPossible);
363 unsigned smem = getSharedMemPerBlock ();
364 // divide-by-2 because hash table entry stores 2 elements: idx and count
365 auto binsPossible = smem / sizeof (unsigned) / 2;
366 for (; binsPossible > 1; --binsPossible) {
367 if (primes.isPrime(binsPossible)) return ( int )binsPossible;
368 }
369 return 1; // should not happen!
370 }

385 template < typename DataT, typename BinnerOp, typename IdxT, int VecLen>
386 void histogramVecLen ( HistType type, int * bins, IdxT nbins, const DataT* data,
387 IdxT nrows , IdxT ncols , cudaStream_t stream,
388 BinnerOp binner) {
389 CUDA_CHECK (cudaMemsetAsync(bins, 0, ncols * nbins * sizeof ( int ), stream));
390 switch (type) {
391 case HistTypeGmem :
392 gmemHist<DataT, BinnerOp, IdxT, VecLen>(bins, nbins, data, nrows , ncols ,
393 binner, stream);
394 break ;
395 case HistTypeSmem :
396 smemHist<DataT, BinnerOp, IdxT, VecLen, false>(bins, nbins, data, nrows ,
397 ncols , binner, stream);
398 break ;
399 case HistTypeSmemMatchAny :
400 smemHist<DataT, BinnerOp, IdxT, VecLen, true>(bins, nbins, data, nrows ,
401 ncols , binner, stream);
402 break ;
403 case HistTypeSmemBits16 :
404 smemBitsHist<DataT, BinnerOp, IdxT, 16, VecLen>(bins, nbins, data, nrows ,
405 ncols , binner, stream);
406 break ;
407 case HistTypeSmemBits8 :
408 smemBitsHist<DataT, BinnerOp, IdxT, 8, VecLen>(bins, nbins, data, nrows ,
409 ncols , binner, stream);
410 break ;
411 case HistTypeSmemBits4 :
412 smemBitsHist<DataT, BinnerOp, IdxT, 4, VecLen>(bins, nbins, data, nrows ,
413 ncols , binner, stream);
414 break ;
415 case HistTypeSmemBits2 :
416 smemBitsHist<DataT, BinnerOp, IdxT, 2, VecLen>(bins, nbins, data, nrows ,
417 ncols , binner, stream);
418 break ;
419 case HistTypeSmemBits1 :
420 smemBitsHist<DataT, BinnerOp, IdxT, 1, VecLen>(bins, nbins, data, nrows ,
421 ncols , binner, stream);
422 break ;
423 case HistTypeSmemHash :
424 smemHashHist<DataT, BinnerOp, IdxT, VecLen>(bins, nbins, data, nrows ,
425 ncols , binner, stream);
426 break ;
427 default :
428 ASSERT ( false , "histogram: Invalid type passed '%d'!" , type);
429 };
430 CUDA_CHECK (cudaGetLastError());
431 }

433 template < typename DataT, typename BinnerOp, typename IdxT>
434 void histogramImpl ( HistType type, int * bins, IdxT nbins, const DataT* data,
435 IdxT nrows , IdxT ncols , cudaStream_t stream,
436 BinnerOp binner) {
437 size_t bytes = nrows * sizeof (DataT);
438 if ( nrows <= 0) return ;
439 if (16 % sizeof (DataT) == 0 && bytes % 16 == 0) {
440 histogramVecLen <DataT, BinnerOp, IdxT, 16 / sizeof (DataT)>(
441 type, bins, nbins, data, nrows , ncols , stream, binner);
442 } else if (8 % sizeof (DataT) == 0 && bytes % 8 == 0) {
443 histogramVecLen <DataT, BinnerOp, IdxT, 8 / sizeof (DataT)>(
444 type, bins, nbins, data, nrows , ncols , stream, binner);
445 } else if (4 % sizeof (DataT) == 0 && bytes % 4 == 0) {
446 histogramVecLen <DataT, BinnerOp, IdxT, 4 / sizeof (DataT)>(
447 type, bins, nbins, data, nrows , ncols , stream, binner);
448 } else if (2 % sizeof (DataT) == 0 && bytes % 2 == 0) {
449 histogramVecLen <DataT, BinnerOp, IdxT, 2 / sizeof (DataT)>(
450 type, bins, nbins, data, nrows , ncols , stream, binner);
451 } else {
452 histogramVecLen<DataT, BinnerOp, IdxT, 1>(type, bins, nbins, data, nrows ,
453 ncols , stream, binner);
454 }

457 template < typename IdxT>
458 HistType selectBestHistAlgo (IdxT nbins) {
459 size_t smem = getSharedMemPerBlock ();
460 size_t requiredSize = nbins * sizeof (unsigned);
461 if (requiredSize <= smem) {
462 return HistTypeSmem ;
463 }
464 for ( int bits = 16; bits >= 1; bits >>= 1) {
465 auto nBytesForBins = ceildiv<size_t>(bits * nbins, 8);
466 requiredSize = alignTo<size_t>(nBytesForBins, sizeof ( unsigned ));
467 if (requiredSize <= smem) {
468 return static_cast< HistType > (bits);
469 }
470 }
471 return HistTypeGmem ;
472 }

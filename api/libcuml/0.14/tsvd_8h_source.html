44 template < typename math_t>
45 void calCompExpVarsSvd ( const cumlHandle_impl &handle, math_t *in,
46 math_t *components, math_t *singular_vals,
47 math_t *explained_vars, math_t *explained_var_ratio,
48 const paramsTSVD &prms, cudaStream_t stream) {
49 auto cusolver_handle = handle. getcusolverDnHandle ();
50 auto cublas_handle = handle. getCublasHandle ();
51 auto allocator = handle. getDeviceAllocator ();
52
53 int diff = prms. n_cols - prms. n_components ;
54 math_t ratio = math_t( diff ) / math_t(prms. n_cols );
55 ASSERT ( ratio >= math_t(0.2),
56 "Number of components should be less than at least 80 percent of the "
57 "number of features" );
58
59 int p = int(math_t(0.1) * math_t(prms. n_cols ));
60 // int p = int(math_t(prms.n_cols) / math_t(4));
61 ASSERT (p >= 5,
62 "RSVD should be used where the number of columns are at least 50" );
63
64 int total_random_vecs = prms. n_components + p;
65 ASSERT (total_random_vecs < prms. n_cols ,
66 "RSVD should be used where the number of columns are at least 50" );
67
68 device_buffer<math_t> components_temp(allocator, stream,
69 prms. n_cols * prms. n_components );
70 math_t *left_eigvec = nullptr ;
71 LinAlg::rsvdFixedRank (in, prms. n_rows , prms. n_cols , singular_vals,
72 left_eigvec, components_temp. data (), prms. n_components ,
73 p, true , false , true , false , (math_t)prms. tol ,
74 prms. n_iterations , cusolver_handle, cublas_handle,
75 stream);
76
77 LinAlg::transpose (components_temp. data (), components, prms. n_cols ,
78 prms. n_components , cublas_handle, stream);
79 Matrix::power (singular_vals, explained_vars, math_t(1), prms. n_components ,
80 stream);
81 Matrix::ratio (explained_vars, explained_var_ratio, prms. n_components , stream,
82 allocator);
83 }

85 template < typename math_t>
86 void calEig ( const cumlHandle_impl &handle, math_t *in, math_t *components,
87 math_t *explained_var, const paramsTSVD &prms,
88 cudaStream_t stream) {
89 auto cusolver_handle = handle. getcusolverDnHandle ();
90 auto allocator = handle. getDeviceAllocator ();
91
92 if (prms. algorithm == solver::COV_EIG_JACOBI ) {
93 LinAlg::eigJacobi (in, prms. n_cols , prms. n_cols , components, explained_var,
94 cusolver_handle, stream, allocator, (math_t)prms. tol ,
95 prms. n_iterations );
96 } else {
97 LinAlg::eigDC (in, prms. n_cols , prms. n_cols , components, explained_var,
98 cusolver_handle, stream, allocator);
99 }
100
101 Matrix::colReverse (components, prms. n_cols , prms. n_cols , stream);
102 LinAlg::transpose (components, prms. n_cols , stream);
103
104 Matrix::rowReverse (explained_var, prms. n_cols , 1, stream);
105 }

118 template < typename math_t>
119 void signFlip (math_t *input, int n_rows, int n_cols, math_t *components,
120 int n_cols_comp, std::shared_ptr<deviceAllocator> allocator,
121 cudaStream_t stream) {
122 auto counting = thrust::make_counting_iterator(0);
123 auto m = n_rows;
124
125 ML::thrustAllocatorAdapter alloc(allocator, stream);
126 auto execution_policy = thrust::cuda::par(alloc).on(stream);
127 thrust::for_each(execution_policy, counting, counting + n_cols,
128 [=] __device__( int idx) {
129 int d_i = idx * m ;
130 int end = d_i + m ;
131
132 math_t max = 0.0;
133 int max_index = 0;
134 for ( int i = d_i; i < end ; i ++) {
135 math_t val = input[ i ];
136 if (val < 0.0) {
137 val = -val;
138 }
139 if (val > max ) {
140 max = val;
141 max_index = i ;
142 }
143 }
144
145 if (input[max_index] < 0.0) {
146 for ( int i = d_i; i < end ; i ++) {
147 input[ i ] = -input[ i ];
148 }
149
150 int len = n_cols * n_cols_comp;
151 for ( int i = idx; i < len; i = i + n_cols) {
152 components[ i ] = -components[ i ];
153 }
154 }
155 });
156 }

167 template < typename math_t>
168 void tsvdFit ( const cumlHandle_impl &handle, math_t *input, math_t *components,
169 math_t *singular_vals, const paramsTSVD &prms,
170 cudaStream_t stream) {
171 auto cublas_handle = handle. getCublasHandle ();
172 auto allocator = handle. getDeviceAllocator ();
173
174 ASSERT (prms. n_cols > 1,
175 "Parameter n_cols: number of columns cannot be less than two" );
176 ASSERT (prms. n_rows > 1,
177 "Parameter n_rows: number of rows cannot be less than two" );
178 ASSERT (
179 prms. n_components > 0,
180 "Parameter n_components: number of components cannot be less than one" );
181
182 int n_components = prms. n_components ;
183 if (prms. n_components > prms. n_cols ) n_components = prms. n_cols ;
184
185 int len = prms. n_cols * prms. n_cols ;
186 device_buffer<math_t> input_cross_mult(allocator, stream, len);
187
188 math_t alpha = math_t(1);
189 math_t beta = math_t(0);
190 LinAlg::gemm (input, prms. n_rows , prms. n_cols , input, input_cross_mult. data (),
191 prms. n_cols , prms. n_cols , CUBLAS_OP_T, CUBLAS_OP_N, alpha, beta,
192 cublas_handle, stream);
193
194 device_buffer<math_t> components_all(allocator, stream, len);
195 device_buffer<math_t> explained_var_all(allocator, stream, prms. n_cols );
196
197 calEig (handle, input_cross_mult. data (), components_all. data (),
198 explained_var_all. data (), prms, stream);
199
200 Matrix::truncZeroOrigin (components_all. data (), prms. n_cols , components,
201 n_components, prms. n_cols , stream);
202
203 math_t scalar = math_t(1);
204 Matrix::seqRoot (explained_var_all. data (), singular_vals, scalar, n_components,
205 stream);
206 }

220 template < typename math_t>
221 void tsvdFitTransform ( const cumlHandle_impl &handle, math_t *input,
222 math_t *trans_input, math_t *components,
223 math_t *explained_var, math_t *explained_var_ratio,
224 math_t *singular_vals, const paramsTSVD &prms,
225 cudaStream_t stream) {
226 auto allocator = handle. getDeviceAllocator ();
227
228 tsvdFit (handle, input, components, singular_vals, prms, stream);
229 tsvdTransform (handle, input, components, trans_input, prms, stream);
230
231 signFlip (trans_input, prms. n_rows , prms. n_components , components, prms. n_cols ,
232 allocator, stream);
233
234 device_buffer<math_t> mu_trans(allocator, stream, prms. n_components );
235 Stats::mean (mu_trans. data (), trans_input, prms. n_components , prms. n_rows ,
236 true , false , stream);
237 Stats::vars (explained_var, trans_input, mu_trans. data (), prms. n_components ,
238 prms. n_rows , true , false , stream);
239
240 device_buffer<math_t> mu(allocator, stream, prms. n_cols );
241 device_buffer<math_t> vars (allocator, stream, prms. n_cols );
242
243 Stats::mean (mu. data (), input, prms. n_cols , prms. n_rows , true , false , stream);
244 Stats::vars ( vars .data(), input, mu. data (), prms. n_cols , prms. n_rows , true ,
245 false , stream);
246
247 device_buffer<math_t> total_vars(allocator, stream, 1);
248 Stats::sum (total_vars. data (), vars .data(), 1, prms. n_cols , false , stream);
249
250 math_t total_vars_h;
251 updateHost (&total_vars_h, total_vars. data (), 1, stream);
252 CUDA_CHECK (cudaStreamSynchronize(stream));
253 math_t scalar = math_t(1) / total_vars_h;
254
255 LinAlg::scalarMultiply (explained_var_ratio, explained_var, scalar,
256 prms. n_components , stream);
257 }

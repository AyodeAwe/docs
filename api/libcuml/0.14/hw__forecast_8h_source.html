20 template < typename Dtype>
21 __global__ void holtwinters_seasonal_forecast_kernel (
22 Dtype * forecast , int h, int batch_size, int frequency,
23 const Dtype *level_coef, const Dtype *trend_coef, const Dtype *season_coef,
24 bool additive) {
25 int tid = GET_TID ;
26 if (tid < batch_size) {
27 const Dtype level = (level_coef) ? level_coef[tid] : 0.;
28 const Dtype trend = (trend_coef) ? trend_coef[tid] : 0.;
29 for ( int i = 0; i < h; ++ i ) {
30 const Dtype season = season_coef[tid + ( i % frequency) * batch_size];
31 if (additive)
32 forecast [tid + i * batch_size] = level + trend * ( i + 1) + season;
33 else
34 forecast [tid + i * batch_size] = (level + trend * ( i + 1)) * season;
35 }
36 }
37 }

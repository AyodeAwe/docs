43 template < typename T, int TPB_X, typename Lambda>
44 __global__ void map_kernel (T *output, T *X, int n_rows, T *coef, Lambda grad) {
45 int row = (blockIdx.x * TPB_X ) + threadIdx.x;
46 if (row < n_rows) {
47 T x = X[row];
48 T a = coef[0];
49 T b = coef[1];
50 output[row] = grad( x , a, b);
51 if (isnan(output[row])) output[row] = 0.0;
52 }
53 }

59 template < typename T, int TPB_X>
60 void f (T *input, int n_rows, T *coef, T *preds) {
61 dim3 grid( MLCommon::ceildiv (n_rows, TPB_X ), 1, 1);
62 dim3 blk( TPB_X , 1, 1);
63
64 // Function: 1/1+ax^(2b)
65 map_kernel<T, TPB_X><<<grid, blk>>>(
66 preds, input, n_rows, coef,
67 [] __device__(T x , T a, T b) { return 1.0 / (1 + a * pow( x , 2.0 * b)); });
68 }

74 template < typename T, int TPB_X>
75 void abLossGrads (T *input, int n_rows, const T *labels, T *coef, T *grads,
76 UMAPParams * params , std::shared_ptr<deviceAllocator> d_alloc,
77 cudaStream_t stream) {
78 dim3 grid( MLCommon::ceildiv (n_rows, TPB_X ), 1, 1);
79 dim3 blk( TPB_X , 1, 1);
80
84 MLCommon::device_buffer<T> residuals(d_alloc, stream, n_rows);
85
86 f<T, TPB_X>(input, n_rows, coef, residuals. data ());
87 MLCommon::LinAlg::eltwiseSub (residuals. data (), residuals. data (), labels,
88 n_rows, stream);
89 CUDA_CHECK (cudaPeekAtLastError());
90
94 MLCommon::device_buffer<T> a_deriv(d_alloc, stream, n_rows);
95 MLCommon::copy (a_deriv. data (), input, n_rows, stream);
96 map_kernel<T, TPB_X><<<grid, blk, 0, stream>>>(
97 a_deriv. data (), a_deriv. data (), n_rows, coef,
98 [] __device__ __host__(T x , T a, T b) {
99 return -(pow( x , 2.0 * b)) / pow((1.0 + a * pow( x , 2.0 * b)), 2.0);
100 });
101
102 MLCommon::LinAlg::eltwiseMultiply (a_deriv. data (), a_deriv. data (),
103 residuals. data (), n_rows, stream);
104 CUDA_CHECK (cudaPeekAtLastError());
105
109 MLCommon::device_buffer<T> b_deriv(d_alloc, stream, n_rows);
110 MLCommon::copy (b_deriv. data (), input, n_rows, stream);
111 map_kernel<T, TPB_X>
112 <<<grid, blk, 0, stream>>>(b_deriv. data (), b_deriv. data (), n_rows, coef,
113 [] __device__ __host__(T x , T a, T b) {
114 return -(2.0 * a * pow( x , 2.0 * b) * log( x )) /
115 pow(1 + a * pow( x , 2.0 * b), 2.0);
116 });
117
121 MLCommon::LinAlg::eltwiseMultiply (b_deriv. data (), b_deriv. data (),
122 residuals. data (), n_rows, stream);
123 CUDA_CHECK (cudaPeekAtLastError());
124
128 MLCommon::Stats::mean (grads, a_deriv. data (), 1, n_rows, false , false , stream);
129 MLCommon::Stats::mean (grads + 1, b_deriv. data (), 1, n_rows, false , false ,
130 stream);
131
132 CUDA_CHECK (cudaPeekAtLastError());
133 }

138 template < typename T, int TPB_X>
139 void optimize_params (T *input, int n_rows, const T *labels, T *coef,
140 UMAPParams * params ,
141 std::shared_ptr<deviceAllocator> d_alloc,
142 cudaStream_t stream, float tolerance = 1e-6,
143 int max_epochs = 25000) {
144 // Don't really need a learning rate since
145 // we aren't using stochastic GD
146 float learning_rate = 1.0;
147
148 int num_iters = 0;
149 int tol_grads = 0;
150 do {
151 tol_grads = 0;
152 MLCommon::device_buffer<T> grads(d_alloc, stream, 2);
153 CUDA_CHECK (cudaMemsetAsync(grads. data (), 0, 2 * sizeof (T), stream));
154
155 abLossGrads<T, TPB_X>(input, n_rows, labels, coef, grads. data (), params ,
156 d_alloc, stream);
157
158 MLCommon::LinAlg::multiplyScalar (grads. data (), grads. data (), learning_rate,
159 2, stream);
160 MLCommon::LinAlg::eltwiseSub (coef, coef, grads. data (), 2, stream);
161
162 T *grads_h = (T *)malloc(2 * sizeof (T));
163 MLCommon::updateHost (grads_h, grads. data (), 2, stream);
164
165 CUDA_CHECK (cudaStreamSynchronize(stream));
166
167 for ( int i = 0; i < 2; i ++) {
168 if (abs(grads_h[ i ]) - tolerance <= 0) tol_grads += 1;
169 }
170
171 num_iters += 1;
172
173 delete grads_h;
174
175 } while (tol_grads < 2 && num_iters < max_epochs);
176 }

178 void find_params_ab ( UMAPParams * params ,
179 std::shared_ptr<deviceAllocator> d_alloc,
180 cudaStream_t stream) {
181 float spread = params ->spread;
182 float min_dist = params ->min_dist;
183
184 float step = (spread * 3.0) / 300.0;
185
186 float *X = ( float *)malloc(300 * sizeof ( float ));
187 float * y = ( float *)malloc(300 * sizeof ( float ));
188
189 for ( int i = 0; i < 300; i ++) {
190 X[ i ] = i * step;
191 y [ i ] = 0.0;
192 if (X[ i ] >= min_dist)
193 y [ i ] = exp(-(X[ i ] - min_dist) / spread);
194 else if (X[ i ] < min_dist)
195 y [ i ] = 1.0;
196 }
197
198 MLCommon::device_buffer<float> X_d(d_alloc, stream, 300);
199 MLCommon::updateDevice (X_d. data (), X, 300, stream);
200
201 MLCommon::device_buffer<float> y_d(d_alloc, stream, 300);
202 MLCommon::updateDevice (y_d. data (), y , 300, stream);
203 float *coeffs_h = ( float *)malloc(2 * sizeof ( float ));
204 coeffs_h[0] = 1.0;
205 coeffs_h[1] = 1.0;
206
207 MLCommon::device_buffer<float> coeffs(d_alloc, stream, 2);
208 CUDA_CHECK (cudaMemsetAsync(coeffs. data (), 0, 2 * sizeof ( float ), stream));
209
210 MLCommon::updateDevice (coeffs. data (), coeffs_h, 2, stream);
211
212 optimize_params<float, 256>(X_d. data (), 300, y_d. data (), coeffs. data (),
213 params , d_alloc, stream);
214
215 MLCommon::updateHost (&( params ->a), coeffs. data (), 1, stream);
216 MLCommon::updateHost (&( params ->b), coeffs. data () + 1, 1, stream);
217
218 CUDA_CHECK (cudaStreamSynchronize(stream));
219
220 CUML_LOG_DEBUG ( "a=%f, b=%f" , params ->a, params ->b);
221
222 delete coeffs_h;
223 }

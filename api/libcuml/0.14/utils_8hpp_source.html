56  void collectCallStack() throw() {
57 #ifdef __GNUC__
58  const int MaxStackDepth = 64;
59  void* stack[MaxStackDepth];
60  auto depth = backtrace(stack, MaxStackDepth);
61  std::ostringstream oss;
62  oss << std::endl << "Obtained " << depth << " stack frames" << std::endl;
63  char** strings = backtrace_symbols(stack, depth);
64  if (strings == nullptr) {
65  oss << "But no stack trace could be found!" << std::endl;
66  msg += oss.str();
67  return;
68  }
70  for (int i = 0; i < depth; ++i) {
71  oss << "#" << i << " in " << strings[i] << std::endl;
72  }
73  free(strings);
74  msg += oss.str();
75 #endif // __GNUC__
76  }

57 template < typename DataT, typename IdxT, int VALUE>
58 inline __device__ void transform (DataT* tmp, DataT* myNewParams, bool isAr) {
59 //do the ar transformation
60 PAC<DataT> pac;
61 for ( int i = 0; i < VALUE; ++ i ) {
62 tmp[ i ] = pac(tmp[ i ]);
63 myNewParams[ i ] = tmp[ i ];
64 }
65 if (isAr) {
66 for ( int j = 1; j < VALUE; ++ j ) {
67 DataT a = myNewParams[ j ];
68
69 for ( int k = 0; k < j ; ++ k ) {
70 tmp[ k ] -= a * myNewParams[ j - k - 1];
71 }
72
73 for ( int iter = 0; iter < j ; ++iter) {
74 myNewParams[iter] = tmp[iter];
75 }
76 }
77 } else { //do the ma transformation
78 for ( int j = 1; j < VALUE; ++ j ) {
79 DataT a = myNewParams[ j ];
80
81 for ( int k = 0; k < j ; ++ k ) {
82 tmp[ k ] += a * myNewParams[ j - k - 1];
83 }
84
85 for ( int iter = 0; iter < j ; ++iter) {
86 myNewParams[iter] = tmp[iter];
87 }
88 }
89 }
90 }

102 template < typename DataT, typename IdxT, int VALUE>
103 inline __device__ void invtransform (DataT* tmp, DataT* myNewParams, bool isAr) {
104 //do the ar transformation
105 if (isAr) {
106 for ( int j = VALUE - 1; j > 0; -- j ) {
107 DataT a = myNewParams[ j ];
108
109 for ( int k = 0; k < j ; ++ k ) {
110 tmp[ k ] = (myNewParams[ k ] + a * myNewParams[ j - k - 1]) / (1 - (a * a));
111 }
112
113 for ( int iter = 0; iter < j ; ++iter) {
114 myNewParams[iter] = tmp[iter];
115 }
116 }
117 } else { //do the ma transformation
118 for ( int j = VALUE - 1; j > 0; -- j ) {
119 DataT a = myNewParams[ j ];
120
121 for ( int k = 0; k < j ; ++ k ) {
122 tmp[ k ] = (myNewParams[ k ] - a * myNewParams[ j - k - 1]) / (1 - (a * a));
123 }
124
125 for ( int iter = 0; iter < j ; ++iter) {
126 myNewParams[iter] = tmp[iter];
127 }
128 }
129 }
130
131 for ( int i = 0; i < VALUE; ++ i ) {
132 myNewParams[ i ] = 2 * myATanh (myNewParams[ i ]);
133 }
134 }

149 template < typename DataT, int VALUE, typename IdxT, int BLOCK_DIM_X,
150 int BLOCK_DIM_Y>
151 __global__ void jones_transform_kernel (DataT* newParams, const DataT* params,
152 IdxT batchSize, bool isAr, bool isInv) {
153 //calculating the index of the model that the coefficients belong to
154 IdxT modelIndex = threadIdx.x + ((IdxT)blockIdx.x * blockDim.x);
155
156 DataT tmp[VALUE];
157 DataT myNewParams[VALUE];
158
159 if (modelIndex < batchSize) {
160 //load
161 #pragma unroll
162 for ( int i = 0; i < VALUE; ++ i ) {
163 tmp[ i ] = params[modelIndex * VALUE + i ];
164 myNewParams[ i ] = tmp[ i ];
165 }
166
167 //the transformation/inverse transformation operation
168 if (isInv)
169 invtransform<DataT, IdxT, VALUE>(tmp, myNewParams, isAr);
170 else
171 transform<DataT, IdxT, VALUE>(tmp, myNewParams, isAr);
172
173 //store
174 #pragma unroll
175 for ( int i = 0; i < VALUE; ++ i ) {
176 newParams[modelIndex * VALUE + i ] = myNewParams[ i ];
177 }
178 }
179 }

193 template < typename DataT, typename IdxT = int >
194 void jones_transform ( const DataT* params, IdxT batchSize, IdxT parameter,
195 DataT* newParams, bool isAr, bool isInv,
196 std::shared_ptr<MLCommon::deviceAllocator> allocator,
197 cudaStream_t stream) {
198 ASSERT (batchSize >= 1 && parameter >= 1, "not defined!" );
199
200 IdxT nElements = batchSize * parameter;
201
202 //copying contents
203 copy (newParams, params, ( size_t )nElements, stream);
204
205 //setting the kernel configuration
206 static const int BLOCK_DIM_Y = 1, BLOCK_DIM_X = 256;
207 dim3 numThreadsPerBlock(BLOCK_DIM_X, BLOCK_DIM_Y);
208 dim3 numBlocks(ceildiv<int>(batchSize, numThreadsPerBlock.x), 1);
209
210 //calling the kernel
211
212 switch (parameter) {
213 case 1:
214 jones_transform_kernel<DataT, 1, IdxT, BLOCK_DIM_X, BLOCK_DIM_Y>
215 <<<numBlocks, numThreadsPerBlock, 0, stream>>>(newParams, params,
216 batchSize, isAr, isInv);
217 break ;
218 case 2:
219 jones_transform_kernel<DataT, 2, IdxT, BLOCK_DIM_X, BLOCK_DIM_Y>
220 <<<numBlocks, numThreadsPerBlock, 0, stream>>>(newParams, params,
221 batchSize, isAr, isInv);
222 break ;
223 case 3:
224 jones_transform_kernel<DataT, 3, IdxT, BLOCK_DIM_X, BLOCK_DIM_Y>
225 <<<numBlocks, numThreadsPerBlock, 0, stream>>>(newParams, params,
226 batchSize, isAr, isInv);
227 break ;
228 case 4:
229 jones_transform_kernel<DataT, 4, IdxT, BLOCK_DIM_X, BLOCK_DIM_Y>
230 <<<numBlocks, numThreadsPerBlock, 0, stream>>>(newParams, params,
231 batchSize, isAr, isInv);
232 break ;
233 default :
234 ASSERT ( false , "Unsupported parameter '%d'!" , parameter);
235 }

237 CUDA_CHECK (cudaPeekAtLastError());
238 }

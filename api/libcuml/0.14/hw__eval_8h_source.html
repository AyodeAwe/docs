21 template < typename Dtype>
22 __device__ Dtype holtwinters_eval_device (
23 int tid, const Dtype *ts, int n , int batch_size, int frequency, int shift,
24 Dtype plevel, Dtype ptrend, Dtype *pseason, int pseason_width,
25 const Dtype *start_season, const Dtype *beta, const Dtype *gamma,
26 Dtype alpha_, Dtype beta_, Dtype gamma_, Dtype *level, Dtype *trend,
27 Dtype *season, Dtype *xhat, bool additive_seasonal) {
28 alpha_ = bound_device (alpha_);
29 beta_ = bound_device (beta_);
30 gamma_ = bound_device (gamma_);
31
32 Dtype error_ = .0;
33 Dtype clevel = .0, ctrend = .0, cseason = .0;
34 for ( int i = 0; i < n - shift; i ++) {
35 int s = i % frequency;
36 Dtype pts = ts[ IDX (tid, i + shift, batch_size)];
37 Dtype leveltrend = plevel + ptrend;
38
39 // xhat
40 Dtype stmp;
41 if (gamma)
42 stmp = i < frequency ? start_season[ IDX (tid, i , batch_size)]
43 : pseason[s * pseason_width];
44 else
45 stmp = (!additive_seasonal);
46 Dtype xhat_ = plevel + ptrend;
47 if (additive_seasonal)
48 xhat_ += stmp;
49 else
50 xhat_ *= stmp;
51
52 // Error
53 Dtype diff = pts - xhat_;
54 error_ += diff * diff ;
55
56 // Level
57 if (additive_seasonal) {
58 clevel = alpha_ * (pts - stmp) + (1 - alpha_) * (leveltrend);
59 } else {
60 Dtype stmp_eps = abs(stmp) > STMP_EPS ? stmp : STMP_EPS ;
61 clevel = alpha_ * (pts / stmp_eps) + (1 - alpha_) * (leveltrend);
62 }
63
64 // Trend
65 if (beta) {
66 ctrend = beta_ * (clevel - plevel) + (1 - beta_) * ptrend;
67 ptrend = ctrend;
68 }
69
70 // Seasonal
71 if (gamma) {
72 if (additive_seasonal)
73 cseason = gamma_ * (pts - clevel) + (1 - gamma_) * stmp;
74 else
75 cseason = gamma_ * (pts / clevel) + (1 - gamma_) * stmp;
76 pseason[s * pseason_width] = cseason;
77 }
78
79 plevel = clevel;
80
81 if (level) level[ IDX (tid, i , batch_size)] = clevel;
82 if (trend) trend[ IDX (tid, i , batch_size)] = ctrend;
83 if (season) season[ IDX (tid, i , batch_size)] = cseason;
84 if (xhat) xhat[ IDX (tid, i , batch_size)] = xhat_;
85 }
86 return error_;
87 }

89 template < typename Dtype>
90 __global__ void holtwinters_eval_gpu_shared_kernel (
91 const Dtype *ts, int n , int batch_size, int frequency,
92 const Dtype *start_level, const Dtype *start_trend, const Dtype *start_season,
93 const Dtype *alpha, const Dtype *beta, const Dtype *gamma, Dtype *level,
94 Dtype *trend, Dtype *season, Dtype *xhat, Dtype *error,
95 bool additive_seasonal) {
96 int tid = GET_TID ;
97 extern __shared__ __align__( sizeof (Dtype)) unsigned char pseason_[];
98 Dtype *pseason = reinterpret_cast< Dtype * > (pseason_);
99
100 if (tid < batch_size) {
101 int shift = 1;
102 Dtype plevel = start_level[tid], ptrend = .0;
103 Dtype alpha_ = alpha[tid];
104 Dtype beta_ = beta ? beta[tid] : .0;
105 Dtype gamma_ = gamma ? gamma[tid] : .0;
106
107 if (gamma) {
108 shift = frequency;
109 ptrend = beta ? start_trend[tid] : .0;
110 } else if (beta) {
111 shift = 2;
112 ptrend = start_trend[tid];
113 }
114
115 Dtype error_ = holtwinters_eval_device<Dtype>(
116 tid, ts, n , batch_size, frequency, shift, plevel, ptrend,
117 pseason + threadIdx.x, blockDim.x, start_season, beta, gamma, alpha_,
118 beta_, gamma_, level, trend, season, xhat, additive_seasonal);
119 if (error) error[tid] = error_;
120 }
121 }

157 template < typename Dtype>
158 void holtwinters_eval_gpu ( const ML::cumlHandle_impl &handle, const Dtype *ts,
159 int n , int batch_size, int frequency,
160 const Dtype *start_level, const Dtype *start_trend,
161 const Dtype *start_season, const Dtype *alpha,
162 const Dtype *beta, const Dtype *gamma, Dtype *level,
163 Dtype *trend, Dtype *season, Dtype *xhat,
164 Dtype *error, ML::SeasonalType seasonal) {
165 cudaStream_t stream = handle. getStream ();
166 std::shared_ptr<MLCommon::deviceAllocator> dev_allocator =
167 handle. getDeviceAllocator ();
168
169 int total_blocks = GET_NUM_BLOCKS (batch_size);
170 int threads_per_block = GET_THREADS_PER_BLOCK (batch_size);
171
172 // How much sm needed for shared kernel
173 size_t sm_needed = sizeof (Dtype) * threads_per_block * frequency;
174 bool is_additive = seasonal == ML::SeasonalType::ADDITIVE ;
175
176 if (sm_needed > MLCommon::getSharedMemPerBlock ()) {
177 MLCommon::device_buffer<Dtype> pseason(dev_allocator, stream,
178 batch_size * frequency);
179 holtwinters_eval_gpu_global_kernel<Dtype>
180 <<<total_blocks, threads_per_block, 0, stream>>>(
181 ts, n , batch_size, frequency, start_level, start_trend, start_season,
182 pseason. data (), alpha, beta, gamma, level, trend, season, xhat, error,
183 is_additive);
184 } else {
185 holtwinters_eval_gpu_shared_kernel<Dtype>
186 <<<total_blocks, threads_per_block, sm_needed, stream>>>(
187 ts, n , batch_size, frequency, start_level, start_trend, start_season,
188 alpha, beta, gamma, level, trend, season, xhat, error, is_additive);
189 }

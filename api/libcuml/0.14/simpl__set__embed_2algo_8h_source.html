52 template < typename T>
53 void make_epochs_per_sample (T *weights, int weights_n, int n_epochs, T *result,
54 cudaStream_t stream) {
55 thrust::device_ptr<T> d_weights = thrust::device_pointer_cast(weights);
56 T weights_max = *(thrust::max_element(thrust::cuda::par.on(stream), d_weights,
57 d_weights + weights_n));
58
59 // result = -1.0 * np.ones(
60 // weights.shape[0], dtype=np.float64
61 // )
62 // n_samples = n_epochs * (weights / weights.max())
63 // result[n_samples > 0] = (
64 // float(n_epochs) / n_samples[n_samples > 0]
65 // )
66
67 MLCommon::LinAlg::unaryOp<T>(
68 result, weights, weights_n,
69 [=] __device__(T input) {
70 T v = n_epochs * (input / weights_max);
71 if (v > 0)
72 return T(n_epochs) / v;
73 else
74 return T(-1.0);
75 },
76 stream);
77 }

93 template < typename T>
94 inline void optimization_iteration_finalization ( UMAPParams * params ,
95 T *head_embedding, T &alpha,
96 int n , int n_epochs,
97 uint64_t &seed) {
98 if ( params ->callback) params ->callback->on_epoch_end(head_embedding);
99 alpha = params ->initial_alpha * (1.0 - (T( n ) / T(n_epochs)));
100 seed += 1;
101 }

112 template < int TPB_X, typename T>
113 void optimize_layout (T *head_embedding, int head_n, T *tail_embedding,
114 int tail_n, const int *head, const int *tail, int nnz,
115 T *epochs_per_sample, int n_vertices, float gamma,
116 UMAPParams * params , int n_epochs, bool multicore_implem,
117 std::shared_ptr<deviceAllocator> d_alloc,
118 cudaStream_t stream) {
119 // Are we doing a fit or a transform?
120 bool move_other = head_embedding == tail_embedding;
121
122 if ( params ->optim_batch_size <= 0) {
123 params ->optim_batch_size = 100000 / params ->n_components;
124 }
125
126 T alpha = params ->initial_alpha;
127
128 MLCommon::device_buffer<T> epoch_of_next_negative_sample(d_alloc, stream,
129 nnz);
130 T nsr_inv = T(1.0) / params ->negative_sample_rate;
131 MLCommon::LinAlg::unaryOp<T>(
132 epoch_of_next_negative_sample. data (), epochs_per_sample, nnz,
133 [=] __device__(T input) { return input * nsr_inv; }, stream);
134
135 MLCommon::device_buffer<T> epoch_of_next_sample(d_alloc, stream, nnz);
136 MLCommon::copy (epoch_of_next_sample.data(), epochs_per_sample, nnz, stream);
137
138 dim3 grid( MLCommon::ceildiv (nnz, TPB_X ), 1, 1);
139 dim3 blk( TPB_X , 1, 1);
140 uint64_t seed = params ->random_state;
141
142 int requiredSize = TPB_X * params ->n_components;
143 if (multicore_implem) {
144 requiredSize *= sizeof (T);
145 } else {
146 requiredSize *= sizeof (double);
147 }
148
149 // checks if enough shared memory is available
150 bool use_shared_mem = requiredSize < MLCommon::getSharedMemPerBlock ();
151 MLCommon::FastIntDiv tail_n_fast(tail_n);
152
153 if (multicore_implem) {
154 for ( int n = 0; n < n_epochs; n ++) {
155 call_optimize_batch_kernel<T, TPB_X>(
156 head_embedding, head_n, tail_embedding, tail_n_fast, head, tail, nnz,
157 epochs_per_sample, n_vertices, epoch_of_next_negative_sample. data (),
158 epoch_of_next_sample.data(), alpha, n , gamma, seed, nullptr , move_other,
159 use_shared_mem, params , n , grid, blk, requiredSize, stream);
160 CUDA_CHECK (cudaGetLastError());
161 optimization_iteration_finalization ( params , head_embedding, alpha, n ,
162 n_epochs, seed);
163 }
164 } else {
165 MLCommon::device_buffer<double> embedding_updates_buf(
166 d_alloc, stream, n_vertices * params ->n_components);
167 double *embedding_updates = embedding_updates_buf. data ();
168 dim3 grid2( MLCommon::ceildiv (n_vertices * params ->n_components, TPB_X ));
169
170 for ( int n = 0; n < n_epochs; n ++) {
171 CUDA_CHECK (cudaMemsetAsync(
172 embedding_updates, 0,
173 n_vertices * params ->n_components * sizeof ( double ), stream));
174
175 int toDo = nnz;
176 int offset = 0;
177 while (toDo > 0) {
178 int curBatchSize = min(toDo, params ->optim_batch_size);
179 call_optimize_batch_kernel<T, TPB_X>(
180 head_embedding, head_n, tail_embedding, tail_n_fast, head, tail,
181 offset + curBatchSize, epochs_per_sample, n_vertices,
182 epoch_of_next_negative_sample. data (), epoch_of_next_sample.data(),
183 alpha, n , gamma, seed, embedding_updates, move_other, use_shared_mem,
184 params , n , grid, blk, requiredSize, stream, offset);
185 CUDA_CHECK (cudaGetLastError());
186
187 toDo -= curBatchSize;
188 offset += curBatchSize;
189 }
190
191 apply_optimization_kernel<T, TPB_X><<<grid2, blk, 0, stream>>>(
192 head_embedding, embedding_updates, n_vertices * params ->n_components);
193 CUDA_CHECK (cudaGetLastError());
194 optimization_iteration_finalization ( params , head_embedding, alpha, n ,
195 n_epochs, seed);
196 }
197 }

205 template < int TPB_X, typename T>
206 void launcher ( int m , int n , MLCommon::Sparse::COO<T> *in, UMAPParams * params ,
207 T *embedding, std::shared_ptr<deviceAllocator> d_alloc,
208 cudaStream_t stream) {
209 int nnz = in-> nnz ;
210
214 thrust::device_ptr<const T> d_ptr = thrust::device_pointer_cast(in-> vals ());
215 T max =
216 *(thrust::max_element(thrust::cuda::par.on(stream), d_ptr, d_ptr + nnz));
217
218 int n_epochs = params ->n_epochs;
219 if (n_epochs <= 0) {
220 if ( m <= 10000)
221 n_epochs = 500;
222 else
223 n_epochs = 200;
224 }
225

231 in-> vals (), in-> vals (), nnz,
232 [=] __device__(T input) {
233 if (input < ( max / float (n_epochs)))
234 return 0.0 f ;
235 else
236 return input;
237 },
238 stream);
239
240 MLCommon::Sparse::COO<T> out(d_alloc, stream);
241 MLCommon::Sparse::coo_remove_zeros<TPB_X, T>(in, &out, d_alloc, stream);
242
243 MLCommon::device_buffer<T> epochs_per_sample(d_alloc, stream, out.nnz);
244 CUDA_CHECK (
245 cudaMemsetAsync(epochs_per_sample. data (), 0, out.nnz * sizeof (T), stream));
246
247 make_epochs_per_sample (out.vals(), out.nnz, n_epochs,
248 epochs_per_sample. data (), stream);

251 std::stringstream ss;
252 ss << MLCommon::arr2Str (epochs_per_sample. data (), out.nnz,

254 CUML_LOG_DEBUG (ss.str().c_str());
255 }

257 optimize_layout<TPB_X, T>(embedding, m , embedding, m , out.rows(), out.cols(),
258 out.nnz, epochs_per_sample. data (), m ,
259 params ->repulsion_strength, params , n_epochs,
260 params ->multicore_implem, d_alloc, stream);
261
262 CUDA_CHECK (cudaPeekAtLastError());
263 }

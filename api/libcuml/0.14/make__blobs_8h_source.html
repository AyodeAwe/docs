65 template < typename DataT, typename IdxT>
66 void make_blobs (DataT* out, IdxT* labels, IdxT n_rows, IdxT n_cols,
67 IdxT n_clusters, std::shared_ptr<deviceAllocator> allocator,
68 cudaStream_t stream, const DataT* centers = nullptr ,
69 const DataT* cluster_std = nullptr ,
70 const DataT cluster_std_scalar = (DataT)1.0,
71 bool shuffle = true , DataT center_box_min = (DataT)-10.0,
72 DataT center_box_max = (DataT)10.0, uint64_t seed = 0ULL,
73 GeneratorType type = GenPhilox ) {
74 Rng r (seed, type);
75 // use the right centers buffer for data generation
76 device_buffer<DataT> rand_centers(allocator, stream);
77 const DataT* _centers;
78 if (centers == nullptr ) {
79 rand_centers. resize (n_clusters * n_cols, stream);
80 r .uniform(rand_centers. data (), n_clusters * n_cols, center_box_min,
81 center_box_max, stream);
82 _centers = rand_centers. data ();
83 } else {
84 _centers = centers;
85 }
86 // use the right output buffer
87 device_buffer<DataT> tmp_out(allocator, stream);
88 device_buffer<IdxT> perms(allocator, stream);
89 device_buffer<IdxT> tmp_labels(allocator, stream);
90 DataT* _out;
91 IdxT* _labels;
92 if ( shuffle ) {
93 tmp_out. resize (n_rows * n_cols, stream);
94 perms. resize (n_rows, stream);
95 tmp_labels. resize (n_rows, stream);
96 _out = tmp_out. data ();
97 _labels = tmp_labels. data ();
98 } else {
99 _out = out;
100 _labels = labels;
101 }
102 // get the std info transferred to host
103 std::vector<DataT> h_cluster_std(n_clusters, cluster_std_scalar);
104 if (cluster_std != nullptr ) {
105 updateHost (&(h_cluster_std[0]), cluster_std, n_clusters, stream);
106 CUDA_CHECK (cudaStreamSynchronize(stream));
107 }
108 // generate data points for each cluster (assume equal distribution)
109 IdxT rows_per_cluster = ceildiv (n_rows, n_clusters);
110 for (IdxT i = 0, row_id = 0; i < n_clusters;
111 ++ i , row_id += rows_per_cluster) {
112 IdxT current_rows = std::min(rows_per_cluster, n_rows - row_id);
113 if (current_rows > 0) {
114 r .normalTable<DataT, IdxT>(_out + row_id * n_cols, current_rows, n_cols,
115 _centers + i * n_cols, nullptr ,
116 h_cluster_std[ i ], stream);
117 r .fill(_labels + row_id, current_rows, (IdxT) i , stream);
118 }
119 }
120 // shuffle, if asked for
122 if ( shuffle ) {
123 permute<DataT, IdxT, IdxT>(perms. data (), out, _out, n_cols, n_rows, true ,
124 stream);
125 constexpr long Nthreads = 256;
126 IdxT nblks = ceildiv<IdxT>(n_rows, Nthreads);
127 gatherKernel<<<nblks, Nthreads, 0, stream>>>(labels, _labels, perms. data (),
128 n_rows);
129 }
130 }

142 template < typename TypeV, typename TypeK, bool Greater, int Log2Str id e>
143 DI void bitonicSortStage ( KVPair<TypeV, TypeK> &current) {
144 constexpr int Stride2 = 1 << (Log2Stride + 1);
145 int lid = laneId ();
146 const bool lidMask = lid & Stride2;
147 #pragma unroll
148 for ( int stage = Log2Stride; stage >= 0; --stage) {
149 int stride = 1 << stage;
150 bool group = lidMask;
151 bool phase = lid & stride;
152 bool reverse = phase ^ group;
153 auto other = current. shfl_xor (stride);
154 current. cas <Greater>(other, reverse );
155 }
156 }

203 template < typename TypeV, typename TypeK, int N, bool Greater>
204 struct KVArray {
205 typedef KVPair<TypeV, TypeK> Pair ;
207 Pair arr [ N ];
209 constexpr static int ArrMask = N - 1;
211 constexpr static int WarpMask = WarpSize - 1;
212
214 DI void reset (TypeV iV, TypeK iK) {
215 #pragma unroll
216 for ( int i = 0; i < N ; ++ i ) {
217 arr [ i ]. val = iV;
218 arr [ i ]. key = iK;
219 }
220 }
221
222 DI void topkUpdate ( Pair &other) {
223 #pragma unroll
224 for ( int i = 0; i < N ; ++ i ) {
225 // perform the sort in the reverse order as to minimize the
226 // amount of shfl's needed during the merge phase
227 bitonicSort<TypeV, TypeK, !Greater>(other);
228 arr [ i ]. cas <Greater>(other, false );
229 bitonicSort<TypeV, TypeK, Greater>( arr [ i ]);
230 }
231 }
232
236
237 DI void sort () {
238 // start by sorting along the warp, first
239 warpWideSort();
240 // iteratively merge each of these "warp-wide" sorted arrays
241 #pragma unroll
242 for ( int stride = 1; stride < N ; stride *= 2) {
243 const int s2 = 2 * stride;
244 #pragma unroll
245 for ( int start = 0; start < N ; start += s2) mergeHalves(stride, start );
246 #pragma unroll
247 for ( int start = 0; start < N ; start += stride)
248 postMergeSort(stride, start );
249 warpWideSort();
250 }
251 }
252
253 private :
254 DI void mergeHalves( int stride, int start ) {
255 const int mask = 2 * stride - 1;
256 #pragma unroll
257 for ( int i = 0; i < stride; ++ i ) {
258 int src = i + start ;
259 int dst = ( i + start ) ^ mask;
260 auto srcOtherPair = arr [src]. shfl_xor ( WarpMask );
261 auto dstOtherPair = arr [dst]. shfl_xor ( WarpMask );
262 arr [src]. cas <Greater>(dstOtherPair, true );
263 arr [dst]. cas <Greater>(srcOtherPair, false );
264 }
265 }
266
267 DI void postMergeSort( int stride, int start ) {
268 #pragma unroll
269 for ( int s = stride / 2; s >= 1; s /= 2) {
270 #pragma unroll
271 for ( int j = 0; j < s; ++ j ) {
272 int ij = start + j ;
273 arr [ij]. cas <Greater>( arr [ij + s], true );
274 }
275 }
276 }
277
278 DI void warpWideSort() {
279 #pragma unroll
280 for ( int i = 0; i < N; ++i) warpSort<TypeV, TypeK, Greater>( arr [ i ]);
281 }
282 };

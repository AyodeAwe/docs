29 /* Finds the best guassian bandwith for
30 each row in the dataset */
31 __global__ void sigmas_kernel ( const float * restrict distances,
32 float * restrict P, const float perplexity,
33 const float desired_entropy,
34 float * restrict P_sum, const int epochs,
35 const float tol, const int n , const int k ) {
36 // For every item in row
37 const int i = (blockIdx.x * blockDim.x) + threadIdx.x;
38 if ( i >= n ) return ;
39
40 float beta_min = -INFINITY, beta_max = INFINITY;
41 float beta = 1;
42 float sum_P_row = 0;
43 register const int ik = i * k ;
44
45 for ( int step = 0; step < epochs; step++) {
46 float sum_Pi = FLT_EPSILON;
47
48 // Exponentiate to get guassian
49 for ( int j = 0; j < k ; j ++) {
50 P[ik + j ] = __expf(-distances[ik + j ] * beta);
51 sum_Pi += P[ik + j ];
52 }
53
54 // Normalize
55 float sum_disti_Pi = 0;
56 sum_P_row = 0;
57 const float div = __fdividef(1.0 f , sum_Pi);
58 for ( int j = 0; j < k ; j ++) {
59 P[ik + j ] *= div;
60 sum_disti_Pi += distances[ik + j ] * P[ik + j ];
61 sum_P_row += P[ik + j ];
62 }
63
64 const float entropy = __logf(sum_Pi) + beta * sum_disti_Pi;
65 const float entropy_diff = entropy - desired_entropy;
66 if (fabs(entropy_diff) <= tol) break ;
67
68 // Bisection search
69 if (entropy_diff > 0) {
70 beta_min = beta;
71 if (isinf(beta_max))
72 beta *= 2.0f;
73 else
74 beta = (beta + beta_max) * 0.5 f ;
75 } else {
76 beta_max = beta;
77 if (isinf(beta_min))
78 beta *= 0.5 f ;
79 else
80 beta = (beta + beta_min) * 0.5 f ;
81 }
82 }
83 atomicAdd(P_sum, sum_P_row);
84 }

87 /* Finds the best guassian bandwith for
88 each row in the dataset */
89 __global__ void sigmas_kernel_2d ( const float * restrict distances,
90 float * restrict P, const float perplexity,
91 const float desired_entropy,
92 float * restrict P_sum, const int epochs,
93 const float tol, const int n ) {
94 // For every item in row
95 const int i = (blockIdx.x * blockDim.x) + threadIdx.x;
96 if ( i >= n ) return ;
97
98 float beta_min = -INFINITY, beta_max = INFINITY;
99 float beta = 1;
100 float sum_P_row = 0;
101 register const int ik = i * 2;
102
103 for ( int step = 0; step < epochs; step++) {
104 // Exponentiate to get guassian
105 P[ik] = __expf(-distances[ik] * beta);
106 P[ik + 1] = __expf(-distances[ik + 1] * beta);
107 const float sum_Pi = FLT_EPSILON + P[ik] + P[ik + 1];
108
109 // Normalize
110 const float div = __fdividef(1.0 f , sum_Pi);
111 P[ik] *= div;
112 P[ik + 1] *= div;
113 const float sum_disti_Pi =
114 distances[ik] * P[ik] + distances[ik + 1] * P[ik + 1];
115 sum_P_row = P[ik] + P[ik + 1];
116
117 const float entropy = __logf(sum_Pi) + beta * sum_disti_Pi;
118 const float entropy_diff = entropy - desired_entropy;
119 if (fabs(entropy_diff) <= tol) break ;
120
121 // Bisection search
122 if (entropy_diff > 0) {
123 beta_min = beta;
124 if (isinf(beta_max))
125 beta *= 2.0f;
126 else
127 beta = (beta + beta_max) * 0.5 f ;
128 } else {
129 beta_max = beta;
130 if (isinf(beta_min))
131 beta *= 0.5 f ;
132 else
133 beta = (beta + beta_min) * 0.5 f ;
134 }
135 }
136 atomicAdd(P_sum, sum_P_row);
137 }

140 float perplexity_search ( const float * restrict distances, float * restrict P,
141 const float perplexity, const int epochs,
142 const float tol, const int n , const int dim,
143 const cumlHandle &handle) {
144 const float desired_entropy = logf(perplexity);
145 auto d_alloc = handle. getDeviceAllocator ();
146 cudaStream_t stream = handle. getStream ();
147
148 float *P_sum = ( float *)d_alloc->allocate( sizeof ( float ), stream);
149 CUDA_CHECK (cudaMemsetAsync(P_sum, 0, sizeof ( float ), stream));
150
151 if (dim == 2)
152 sigmas_kernel_2d <<< MLCommon::ceildiv ( n , 1024), 1024, 0, stream>>>(
153 distances, P, perplexity, desired_entropy, P_sum, epochs, tol, n );
154 else
155 sigmas_kernel <<< MLCommon::ceildiv ( n , 1024), 1024, 0, stream>>>(
156 distances, P, perplexity, desired_entropy, P_sum, epochs, tol, n , dim);
157 CUDA_CHECK (cudaPeekAtLastError());
158
159 cudaStreamSynchronize(stream);
160 float sum ;
161 MLCommon::updateHost (& sum , P_sum, 1, stream);
162 d_alloc->deallocate(P_sum, sizeof ( float ), stream);
163
164 return sum ;
165 }

168 /* Compute attractive forces in O(uN) time.
169 Uses only nearest neighbors */
170 __global__ void attractive_kernel (
171 const float * restrict VAL, const int * restrict COL, const int * restrict ROW,
172 const float * restrict Y, const float * restrict norm, float * restrict attract,
173 const int NNZ, const int n , const int dim,
174 const float df_power, // -(df + 1)/2)
175 const float recp_df) // 1 / df
176 {
177 const int index = (blockIdx.x * blockDim.x) + threadIdx.x;
178 if (index >= NNZ) return ;
179 const int i = ROW[index], j = COL[index];
180
181 // Euclidean distances
182 // TODO: can provide any distance ie cosine
183 // #862
184 float d = 0;
185 for ( int k = 0; k < dim; k ++) d += Y[ k * n + i ] * Y[ k * n + j ];
186 const float euclidean_d = -2.0f * d + norm[ i ] + norm[ j ];
187
188 // TODO: Calculate Kullback-Leibler divergence
189 // #863
190 const float PQ =
191 VAL[index] * __powf((1.0 f + euclidean_d * recp_df), df_power); // P*Q
192
193 // Apply forces
194 for ( int k = 0; k < dim; k ++)
195 atomicAdd(&attract[ k * n + i ], PQ * (Y[ k * n + i ] - Y[ k * n + j ]));
196 }

198 /****************************************/
199 /* Special case when dim == 2. Can speed
200 up many calculations up */
201 __global__ void attractive_kernel_2d (
202 const float * restrict VAL, const int * restrict COL, const int * restrict ROW,
203 const float * restrict Y1 , const float * restrict Y2 ,
204 const float * restrict norm, float * restrict attract1 ,
205 float * restrict attract2 , const int NNZ) {
206 const int index = (blockIdx.x * blockDim.x) + threadIdx.x;
207 if (index >= NNZ) return ;
208 const int i = ROW[index], j = COL[index];
209
210 // Euclidean distances
211 // TODO: can provide any distance ie cosine
212 // #862
213 const float euclidean_d =
214 norm[ i ] + norm[ j ] - 2.0f * ( Y1 [ i ] * Y1 [ j ] + Y2 [ i ] * Y2 [ j ]);
215
216 // TODO: Calculate Kullback-Leibler divergence
217 // #863
218 const float PQ = __fdividef(VAL[index], (1.0 f + euclidean_d)); // P*Q
219
220 // Apply forces
221 atomicAdd(& attract1 [ i ], PQ * ( Y1 [ i ] - Y1 [ j ]));
222 atomicAdd(& attract2 [ i ], PQ * ( Y2 [ i ] - Y2 [ j ]));
223 }

250 /****************************************/
251 /* Computes repulsive forces in pseudo-O(N^2)
252 time where many of the math ops are
253 made considerably faster. */
254 __global__ void repulsive_kernel ( const float * restrict Y, float * restrict repel,
255 const float * restrict norm,
256 float * restrict Z_sum1, float * restrict Z_sum2,
257 const int n , const int dim,
258 const float df_power, // -(df + 1)/2)
259 const float recp_df) // 1 / df
260 {
261 const int j =
262 (blockIdx.x * blockDim.x) + threadIdx.x; // for every item in row
263 const int i = (blockIdx.y * blockDim.y) + threadIdx.y; // for every row
264 if ( j >= i || i >= n || j >= n ) return ;
265
266 // Euclidean distances
267 // TODO: can provide any distance ie cosine
268 float d = 0;
269 for ( int k = 0; k < dim; k ++) d += Y[ k * n + i ] * Y[ k * n + j ];
270 const float euclidean_d = -2.0f * d + norm[ i ] + norm[ j ];
271
272 // Q and Q^2
273 const float Q = __powf((1.0 f + euclidean_d * recp_df), df_power);
274 const float Q2 = Q * Q;
275
276 // Apply forces
277 for ( int k = 0; k < dim; k ++) {
278 const float force = Q2 * (Y[ k * n + j ] - Y[ k * n + i ]);
279 atomicAdd(&repel[ k * n + i ], force);
280 atomicAdd(&repel[ k * n + j ], force);
281 }
282
283 // Sum up Z sum
284 if ( i % 2 == 0)
285 atomicAdd(&Z_sum1[ i ], Q);
286 else
287 atomicAdd(&Z_sum2[ i ], Q);
288 }

290 /****************************************/
291 /* Special case when dim == 2. Much faster
292 since calculations are streamlined. */
293 __global__ void repulsive_kernel_2d (
294 const float * restrict Y1 , const float * restrict Y2 , float * restrict repel1 ,
295 float * restrict repel2 , const float * restrict norm, float * restrict Z_sum1,
296 float * restrict Z_sum2, const int n ) {
297 const int j =
298 (blockIdx.x * blockDim.x) + threadIdx.x; // for every item in row
299 const int i = (blockIdx.y * blockDim.y) + threadIdx.y; // for every row
300 if ( j >= i || i >= n || j >= n ) return ;
301
302 // Euclidean distances
303 // TODO: can provide any distance ie cosine
304 // #862
305 const float euclidean_d =
306 norm[ i ] + norm[ j ] - 2.0f * ( Y1 [ i ] * Y1 [ j ] + Y2 [ i ] * Y2 [ j ]);
307 const float Q = __fdividef(1.0 f , (1.0 f + euclidean_d));
308 const float Q2 = Q * Q;
309
310 const float force1 = Q2 * ( Y1 [ j ] - Y1 [ i ]);
311 const float force2 = Q2 * ( Y2 [ j ] - Y2 [ i ]);
312
313 // Add forces
314 atomicAdd(& repel1 [ i ], force1);
315 atomicAdd(& repel1 [ j ], -force1);
316
317 atomicAdd(& repel2 [ i ], force2);
318 atomicAdd(& repel2 [ j ], -force2);
319
320 // Sum up Z sum
321 if ( i % 2 == 0)
322 atomicAdd(&Z_sum1[ i ], Q);
323 else
324 atomicAdd(&Z_sum2[ i ], Q);
325 }

327 /****************************************/
328 template < int TPB_X = 32, int TPB_Y = 32>
329 float repulsive_forces ( const float * restrict Y, float * restrict repel,
330 const float * restrict norm, float * restrict Z_sum,
331 const int n , const int dim,
332 const float df_power, // -(df + 1)/2)
333 const float recp_df, cudaStream_t stream) {
334 CUDA_CHECK (cudaMemsetAsync(Z_sum, 0, sizeof ( float ) * 2 * n , stream));
335 CUDA_CHECK (cudaMemsetAsync(repel, 0, sizeof ( float ) * n * dim, stream));
336
337 const dim3 threadsPerBlock( TPB_X , TPB_Y);
338 const dim3 numBlocks( MLCommon::ceildiv ( n , TPB_X ),
339 MLCommon::ceildiv ( n , TPB_Y));
340
341 // For general embedding dimensions
342 if (dim != 2) {
343 repulsive_kernel<<<numBlocks, threadsPerBlock, 0, stream>>>(
344 Y, repel, norm, Z_sum, Z_sum + n , n , dim, df_power, recp_df);
345 }
346 // For special dim == 2 case
347 else {
348 repulsive_kernel_2d<<<numBlocks, threadsPerBlock, 0, stream>>>(
349 Y, Y + n , repel, repel + n , norm, Z_sum, Z_sum + n , n );
350 }
351 CUDA_CHECK (cudaPeekAtLastError());
352
353 // Find sum(Z_sum)
354 thrust::device_ptr<float> begin = thrust::device_pointer_cast(Z_sum);
355 float Z = thrust::reduce (thrust::cuda::par.on(stream), begin, begin + 2 * n );
356 return 1.0f /
357 (2.0f *
358 ( Z + (float) n )); // Notice + n since diagonal of repulsion sums to n
359 }

361 /****************************************/
362 /* Applys or integrates all forces. Uses
363 more gains and contrains the output
364 for output stability */
365 __global__ void apply_kernel (
366 float * restrict Y, float * restrict velocity, const float * restrict attract,
367 const float * restrict repel, float * restrict means, float * restrict gains,
368 const float Z , // sum(Q)
369 const float learning_rate,
370 const float C, // constant from T-Dist Degrees of Freedom
371 const float momentum ,
372 const int SIZE, // SIZE = n*dim
373 const int n , const float min_gain, float * restrict gradient,
374 const bool check_convergence) {
375 const int index = (blockIdx.x * blockDim.x) + threadIdx.x;
376 if (index >= SIZE) return ;
377
378 const float dy = C * (attract[index] + Z * repel[index]);
379 if (check_convergence) gradient[index] = dy * dy;
380
381 // Find new gain
382 // TODO: Incorporate AadaBound (2019) or Adam
383 // #864
384 if (signbit(dy) != signbit(velocity[index]))
385 gains[index] += 0.2 f ; // Original TSNE is 0.2
386 else
387 gains[index] *= 0.8f; // Original TSNE is 0.8
388 if (gains[index] < min_gain) gains[index] = min_gain;
389
390 velocity[index] =
391 momentum * velocity[index] - learning_rate * dy * gains[index];
392 Y[index] += velocity[index];
393
394 // Add to mean
395 //atomicAdd(&means[index / n], Y[index]);
396 }

398 /****************************************/
399 template < int TPB_X = 32, int TPB_Y = 32>
400 float apply_forces ( float * restrict Y, float * restrict velocity,
401 const float * restrict attract, const float * restrict repel,
402 float * restrict means, float * restrict gains,
403 const float Z , // sum(Q)
404 const float learning_rate,
405 const float C, // constant from T-dist
406 const float momentum , const int dim, const int n ,
407 const float min_gain, float * restrict gradient,
408 const bool check_convergence, cudaStream_t stream) {
409 //cudaMemset(means, 0, sizeof(float) * dim);
410 if (check_convergence)
411 CUDA_CHECK (cudaMemsetAsync(gradient, 0, sizeof ( float ) * n * dim, stream));
412
413 apply_kernel <<< MLCommon::ceildiv ( n * dim, 1024), 1024, 0, stream>>>(
414 Y, velocity, attract, repel, means, gains, Z , learning_rate, C, momentum ,
415 n * dim, n , min_gain, gradient, check_convergence);
416 CUDA_CHECK (cudaPeekAtLastError());
417
418 // Find sum of gradient norms
419 float gradient_norm = INFINITY;
420 if (check_convergence) {
421 thrust::device_ptr<float> begin = thrust::device_pointer_cast(gradient);
422 gradient_norm = sqrtf(
423 thrust::reduce (thrust::cuda::par.on(stream), begin, begin + n * dim));
424 }
425
426 // TODO: Subtract means
427 return gradient_norm;
428 }

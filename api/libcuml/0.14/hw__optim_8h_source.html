22 template < typename Dtype>
23 __device__ Dtype golden_step (Dtype a, Dtype b, Dtype c) {
24 Dtype mid = (a + c) * 0.5;
25 if (b > mid)
26 return (a - b) * GOLD ;
27 else
28 return (c - b) * GOLD ;
29 }

31 template < typename Dtype>
32 __device__ Dtype fix_step (Dtype a, Dtype b, Dtype c, Dtype step, Dtype e) {
33 Dtype min_step = abs_device (e * b) + PG_EPS ;
34 if ( abs_device (step) < min_step) return step > 0 ? min_step : -min_step;
35 if ( abs_device (b + step - a) <= e || abs_device (b + step - c) <= e)
36 return 0.0; // steps are too close to each others
37 return step;
38 }

40 template < typename Dtype>
41 __device__ Dtype calculate_step (Dtype a, Dtype b, Dtype c, Dtype loss_a,
42 Dtype loss_b, Dtype loss_c, Dtype pstep,
43 Dtype e) {
44 // parabola step
45 Dtype p = (b - a) * (loss_b - loss_c);
46 Dtype q = (b - c) * (loss_b - loss_a);
47 Dtype x = q * (b - c) - p * (b - a);
48 Dtype y = (p - q) * 2.;
49 Dtype step = abs_device ( y ) < PG_EPS ? golden_step (a, b, c) : x / y ;
50 step = fix_step (a, b, c, step, e); // ensure point is new
51
52 if ( abs_device (step) > abs_device (pstep / 2) || step == 0.0)
53 step = golden_step (a, b, c);
54 return step;
55 }

57 template < typename Dtype>
58 __device__ void parabolic_interpolation_golden_optim (
59 int tid, const Dtype *ts, int n , int batch_size, int frequency, int shift,
60 Dtype plevel, Dtype ptrend, Dtype *pseason, int pseason_width,
61 const Dtype *start_season, const Dtype *beta, const Dtype *gamma,
62 bool optim_alpha, Dtype *alpha_, bool optim_beta, Dtype *beta_,
63 bool optim_gamma, Dtype *gamma_, Dtype eps, bool ADDITIVE_KERNEL) {
64 Dtype a = (Dtype).0;
65 Dtype b = (Dtype) GOLD ;
66 Dtype c = (Dtype)1.;
67
68 Dtype loss_a = holtwinters_eval_device<Dtype>(
69 tid, ts, n , batch_size, frequency, shift, plevel, ptrend, pseason,
70 pseason_width, start_season, beta, gamma, optim_alpha ? a : *alpha_,
71 optim_beta ? a : *beta_, optim_gamma ? a : *gamma_, nullptr , nullptr ,
72 nullptr , nullptr , ADDITIVE_KERNEL);
73 Dtype loss_b = holtwinters_eval_device<Dtype>(
74 tid, ts, n , batch_size, frequency, shift, plevel, ptrend, pseason,
75 pseason_width, start_season, beta, gamma, optim_alpha ? b : *alpha_,
76 optim_beta ? b : *beta_, optim_gamma ? b : *gamma_, nullptr , nullptr ,
77 nullptr , nullptr , ADDITIVE_KERNEL);
78 Dtype loss_c = holtwinters_eval_device<Dtype>(
79 tid, ts, n , batch_size, frequency, shift, plevel, ptrend, pseason,
80 pseason_width, start_season, beta, gamma, optim_alpha ? c : *alpha_,
81 optim_beta ? c : *beta_, optim_gamma ? c : *gamma_, nullptr , nullptr ,
82 nullptr , nullptr , ADDITIVE_KERNEL);
83
84 Dtype pstep = (c - a) / 2;
85 Dtype cstep = pstep;
86
87 while ( abs_device (c - a) > abs_device (b * eps) + PG_EPS ) {
88 Dtype step = calculate_step (a, b, c, loss_a, loss_b, loss_c, cstep, eps);
89 Dtype optim_val = b + step;
90 Dtype loss_val = holtwinters_eval_device<Dtype>(
91 tid, ts, n , batch_size, frequency, shift, plevel, ptrend, pseason,
92 pseason_width, start_season, beta, gamma,
93 optim_alpha ? optim_val : *alpha_, optim_beta ? optim_val : *beta_,
94 optim_gamma ? optim_val : *gamma_, nullptr , nullptr , nullptr , nullptr ,
95 ADDITIVE_KERNEL);
96 if (loss_val < loss_b) {
97 if (optim_val > b) {
98 SUBSTITUTE (a, b, loss_a, loss_b);
99 } else {
100 SUBSTITUTE (c, b, loss_c, loss_b);
101 }
102 SUBSTITUTE (b, optim_val, loss_b, loss_val);
103 } else {
104 if (optim_val > b) {
105 SUBSTITUTE (c, optim_val, loss_c, loss_val);
106 } else {
107 SUBSTITUTE (a, optim_val, loss_a, loss_val);
108 }
109 }
110 SUBSTITUTE (cstep, pstep, pstep, step);
111 }
112 if (optim_alpha) *alpha_ = b;
113 if (optim_beta) *beta_ = b;
114 if (optim_gamma) *gamma_ = b;
115 }

117 template < typename Dtype>
118 __device__ void holtwinters_finite_gradient_device (
119 int tid, const Dtype *ts, int n , int batch_size, int frequency, int shift,
120 Dtype plevel, Dtype ptrend, Dtype *pseason, int pseason_width,
121 const Dtype *start_season, const Dtype *beta, const Dtype *gamma,
122 Dtype alpha_, Dtype beta_, Dtype gamma_, Dtype *g_alpha, Dtype *g_beta,
123 Dtype *g_gamma, Dtype eps, bool ADDITIVE_KERNEL) {
124 Dtype left_error, right_error;
125 if (g_alpha) { // alpha gradient
126 left_error = holtwinters_eval_device<Dtype>(
127 tid, ts, n , batch_size, frequency, shift, plevel, ptrend, pseason,
128 pseason_width, start_season, beta, gamma, alpha_ - eps, beta_, gamma_,
129 nullptr , nullptr , nullptr , nullptr , ADDITIVE_KERNEL);
130 right_error = holtwinters_eval_device<Dtype>(
131 tid, ts, n , batch_size, frequency, shift, plevel, ptrend, pseason,
132 pseason_width, start_season, beta, gamma, alpha_ + eps, beta_, gamma_,
133 nullptr , nullptr , nullptr , nullptr , ADDITIVE_KERNEL);
134 *g_alpha = (right_error - left_error) / (eps * 2.);
135 }
136 if (g_beta) { // beta gradient
137 left_error = holtwinters_eval_device<Dtype>(
138 tid, ts, n , batch_size, frequency, shift, plevel, ptrend, pseason,
139 pseason_width, start_season, beta, gamma, alpha_, beta_ - eps, gamma_,
140 nullptr , nullptr , nullptr , nullptr , ADDITIVE_KERNEL);
141 right_error = holtwinters_eval_device<Dtype>(
142 tid, ts, n , batch_size, frequency, shift, plevel, ptrend, pseason,
143 pseason_width, start_season, beta, gamma, alpha_, beta_ + eps, gamma_,
144 nullptr , nullptr , nullptr , nullptr , ADDITIVE_KERNEL);
145 *g_beta = (right_error - left_error) / (eps * 2.);
146 }
147 if (g_gamma) { // gamma gradient
148 left_error = holtwinters_eval_device<Dtype>(
149 tid, ts, n , batch_size, frequency, shift, plevel, ptrend, pseason,
150 pseason_width, start_season, beta, gamma, alpha_, beta_, gamma_ - eps,
151 nullptr , nullptr , nullptr , nullptr , ADDITIVE_KERNEL);
152 right_error = holtwinters_eval_device<Dtype>(
153 tid, ts, n , batch_size, frequency, shift, plevel, ptrend, pseason,
154 pseason_width, start_season, beta, gamma, alpha_, beta_, gamma_ + eps,
155 nullptr , nullptr , nullptr , nullptr , ADDITIVE_KERNEL);
156 *g_gamma = (right_error - left_error) / (eps * 2.);
157 }
158 }

160 // There's a bug here, where the wrong values are updated if line search
161 // iter limit is reached. Last values are of nx are put in x, whereas it
162 // should be the nx values which minimised loss. For summary, check
163 // https://github.com/rapidsai/cuml/issues/888
164 template < typename Dtype>
165 __device__ ML::OptimCriterion holtwinters_bfgs_optim_device (
166 int tid, const Dtype *ts, int n , int batch_size, int frequency, int shift,
167 Dtype plevel, Dtype ptrend, Dtype *pseason, int pseason_width,
168 const Dtype *start_season, const Dtype *beta, const Dtype *gamma,
169 bool optim_alpha, Dtype *x1, bool optim_beta, Dtype *x2, bool optim_gamma,
170 Dtype *x3, const ML::OptimParams<Dtype> optim_params, bool ADDITIVE_KERNEL) {
171 Dtype H11 = 1., H12 = .0, H13 = .0, H22 = 1., H23 = .0,
172 H33 = 1.; // Hessian approximiation (Hessian is symmetric)
173 Dtype g1 = .0, g2 = .0, g3 = .0; // gradients
174
175 // initial gradient
176 holtwinters_finite_gradient_device<Dtype>(
177 tid, ts, n , batch_size, frequency, shift, plevel, ptrend, pseason,
178 pseason_width, start_season, beta, gamma, *x1, *x2, *x3,
179 optim_alpha ? &g1 : nullptr , optim_beta ? &g2 : nullptr ,
180 optim_gamma ? &g3 : nullptr , optim_params. eps , ADDITIVE_KERNEL);
181
182 for ( int iter = 0; iter < optim_params. bfgs_iter_limit ; ++iter) {
183 // Step direction
184 Dtype p1 = -H11 * g1 - H12 * g2 - H13 * g3;
185 Dtype p2 = -H12 * g1 - H22 * g2 - H23 * g3;
186 Dtype p3 = -H13 * g1 - H23 * g2 - H33 * g3;
187
188 const Dtype phi = p1 * g1 + p2 * g2 + p3 * g3;
189 if (phi > 0) {
190 H11 = 1.;
191 H12 = 0.;
192 H13 = 0.;
193 H22 = 1.;
194 H23 = 0.;
195 H33 = 1.;
196 p1 = -g1;
197 p2 = -g2;
198 p3 = -g3;
199 }
200
201 // {next_params} = {params}+step_size*p;
202 // start of line search
203
204 // starting step size, we assume the largest distance between x and nx is going to be sqrt(3)/2. where sqrt(3)
205 // is the largest allowed step in a 1x1x1 cube.
206 Dtype step_size;
207 if (optim_params. linesearch_step_size <= 0)
208 step_size = (Dtype)0.866 / sqrt (p1 * p1 + p2 * p2 + p3 * p3);
209 else
210 step_size = optim_params. linesearch_step_size ;
211 Dtype nx1 = *x1 + step_size * p1;
212 Dtype nx2 = *x2 + step_size * p2;
213 Dtype nx3 = *x3 + step_size * p3;
214
215 // line search params
216 const Dtype cauchy =
217 optim_params. linesearch_c * (g1 * p1 + g2 * p2 + g3 * p3);
218 const Dtype loss_ref = holtwinters_eval_device<Dtype>(
219 tid, ts, n , batch_size, frequency, shift, plevel, ptrend, pseason,
220 pseason_width, start_season, beta, gamma, *x1, *x2, *x3, nullptr , nullptr ,
221 nullptr , nullptr , ADDITIVE_KERNEL);
222 Dtype loss = holtwinters_eval_device<Dtype>(
223 tid, ts, n , batch_size, frequency, shift, plevel, ptrend, pseason,
224 pseason_width, start_season, beta, gamma, nx1, nx2, nx3, nullptr , nullptr ,
225 nullptr , nullptr , ADDITIVE_KERNEL);
226
227 for ( int i = 0; i < optim_params. linesearch_iter_limit &&
228 (loss > loss_ref + step_size * cauchy);
229 ++ i ) {
230 step_size *= optim_params. linesearch_tau ;
231 nx1 = *x1 + step_size * p1;
232 nx2 = *x2 + step_size * p2;
233 nx3 = *x3 + step_size * p3;
234 loss = holtwinters_eval_device<Dtype>(
235 tid, ts, n , batch_size, frequency, shift, plevel, ptrend, pseason,
236 pseason_width, start_season, beta, gamma, nx1, nx2, nx3, nullptr ,
237 nullptr , nullptr , nullptr , ADDITIVE_KERNEL);
238 }
239 // end of line search
240
241 // see if new {params} meet stop condition
242 const Dtype dx1 = abs_device (*x1 - nx1);
243 const Dtype dx2 = abs_device (*x2 - nx2);
244 const Dtype dx3 = abs_device (*x3 - nx3);
245 Dtype max = max3 (dx1, dx2, dx3);
246 // update {params}
247 *x1 = nx1;
248 *x2 = nx2;
249 *x3 = nx3;
250 if (optim_params. min_param_diff > max )
251 return ML::OptimCriterion::OPTIM_MIN_PARAM_DIFF ;
252 if (optim_params. min_error_diff > abs_device (loss - loss_ref))
253 return ML::OptimCriterion::OPTIM_MIN_ERROR_DIFF ;
254
255 Dtype ng1 = .0, ng2 = .0, ng3 = .0; // next gradient
256 holtwinters_finite_gradient_device<Dtype>(
257 tid, ts, n , batch_size, frequency, shift, plevel, ptrend, pseason,
258 pseason_width, start_season, beta, gamma, nx1, nx2, nx3,
259 optim_alpha ? &ng1 : nullptr , optim_beta ? &ng2 : nullptr ,
260 optim_gamma ? &ng3 : nullptr , optim_params. eps , ADDITIVE_KERNEL);
261 // see if new gradients meet stop condition
262 max = max3 ( abs_device (ng1), abs_device (ng2), abs_device (ng3));
263 if (optim_params. min_grad_norm > max )
264 return ML::OptimCriterion::OPTIM_MIN_GRAD_NORM ;
265
266 // s = step_size*p;
267 const Dtype s1 = step_size * p1;
268 const Dtype s2 = step_size * p2;
269 const Dtype s3 = step_size * p3;
270
271 // y = next_grad-grad
272 const Dtype y1 = ng1 - g1;
273 const Dtype y2 = ng2 - g2;
274 const Dtype y3 = ng3 - g3;
275
276 // rho_ = y(*)s; rho = 1/rho_
277 const Dtype rho_ = y1 * s1 + y2 * s2 + y3 * s3;
278 const Dtype rho = 1.0 / rho_;
279
280 const Dtype Hy1 = H11 * y1 + H12 * y2 + H13 * y3;
281 const Dtype Hy2 = H12 * y1 + H22 * y2 + H23 * y3;
282 const Dtype Hy3 = H13 * y1 + H23 * y2 + H33 * y3;
283 const Dtype k = rho * rho * (y1 * Hy1 + y2 * Hy2 + y3 * Hy3 + rho_);
284
285 H11 += k * s1 * s1 - 2. * rho * s1 * Hy1;
286 H12 += k * s1 * s2 - rho * (s2 * Hy1 + s1 * Hy2);
287 H13 += k * s1 * s3 - rho * (s3 * Hy1 + s1 * Hy3);
288 H22 += k * s2 * s2 - 2 * rho * s2 * Hy2;
289 H23 += k * s2 * s3 - rho * (s3 * Hy2 + s2 * Hy3);
290 H33 += k * s3 * s3 - 2. * rho * s3 * Hy3;
291
292 g1 = ng1;
293 g2 = ng2;
294 g3 = ng3;
295 }
296
297 return ML::OptimCriterion::OPTIM_BFGS_ITER_LIMIT ;
298 }

300 template < typename Dtype>
301 __global__ void holtwinters_optim_gpu_shared_kernel (
302 const Dtype *ts, int n , int batch_size, int frequency,
303 const Dtype *start_level, const Dtype *start_trend, const Dtype *start_season,
304 Dtype *alpha, bool optim_alpha, Dtype *beta, bool optim_beta, Dtype *gamma,
305 bool optim_gamma, Dtype *level, Dtype *trend, Dtype *season, Dtype *xhat,
306 Dtype *error, ML::OptimCriterion *optim_result,
307 const ML::OptimParams<Dtype> optim_params, bool ADDITIVE_KERNEL,
308 bool single_param) {
309 int tid = GET_TID ;
310 extern __shared__ __align__( sizeof (Dtype)) unsigned char pseason_[];
311 Dtype *pseason = reinterpret_cast< Dtype * > (pseason_);
312
313 if (tid < batch_size) {
314 int shift = 1;
315 ML::OptimCriterion optim;
316 Dtype plevel = start_level[tid], ptrend = .0;
317 Dtype alpha_ = alpha[tid];
318 Dtype beta_ = beta ? beta[tid] : .0;
319 Dtype gamma_ = gamma ? gamma[tid] : .0;
320
321 if (gamma) {
322 shift = frequency;
323 ptrend = beta ? start_trend[tid] : .0;
324 } else if (beta) {
325 shift = 2;
326 ptrend = start_trend[tid];
327 }
328
329 // Optimization
330 if (single_param)
331 parabolic_interpolation_golden_optim<Dtype>(
332 tid, ts, n , batch_size, frequency, shift, plevel, ptrend,
333 pseason + threadIdx.x, blockDim.x, start_season, beta, gamma,
334 optim_alpha, &alpha_, optim_beta, &beta_, optim_gamma, &gamma_,
335 optim_params. eps , ADDITIVE_KERNEL);
336 else
337 optim = holtwinters_bfgs_optim_device<Dtype>(
338 tid, ts, n , batch_size, frequency, shift, plevel, ptrend,
339 pseason + threadIdx.x, blockDim.x, start_season, beta, gamma,
340 optim_alpha, &alpha_, optim_beta, &beta_, optim_gamma, &gamma_,
341 optim_params, ADDITIVE_KERNEL);
342
343 if (optim_alpha) alpha[tid] = bound_device (alpha_);
344 if (optim_beta) beta[tid] = bound_device (beta_);
345 if (optim_gamma) gamma[tid] = bound_device (gamma_);
346 if (!single_param && optim_result) optim_result[tid] = optim;
347
348 if (error || level || trend || season || xhat) {
349 // Final fit
350 Dtype error_ = holtwinters_eval_device<Dtype>(
351 tid, ts, n , batch_size, frequency, shift, plevel, ptrend,
352 pseason + threadIdx.x, blockDim.x, start_season, beta, gamma, alpha_,
353 beta_, gamma_, level, trend, season, xhat, ADDITIVE_KERNEL);
354 if (error) error[tid] = error_;
355 }
356 }

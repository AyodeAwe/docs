42 void HoltWintersBufferSize ( int n , int batch_size, int frequency, bool use_beta,
43 bool use_gamma, int *start_leveltrend_len,
44 int *start_season_len, int *components_len,
45 int *error_len, int *leveltrend_coef_shift,
46 int *season_coef_shift) {
47 int w_len = use_gamma ? frequency : (use_beta ? 2 : 1);
48
49 if (start_leveltrend_len) *start_leveltrend_len = batch_size;
50 if (use_gamma && start_season_len) *start_season_len = frequency * batch_size;
51
52 if (components_len) *components_len = ( n - w_len) * batch_size;
53
54 if (leveltrend_coef_shift)
55 *leveltrend_coef_shift = ( n - w_len - 1) * batch_size;
56 if (use_gamma && season_coef_shift)
57 *season_coef_shift = ( n - w_len - frequency) * batch_size;
58
59 if (error_len) *error_len = batch_size;
60 }

62 template < typename Dtype>
63 void HoltWintersDecompose ( const ML::cumlHandle &handle, const Dtype *ts, int n ,
64 int batch_size, int frequency, Dtype *start_level,
65 Dtype *start_trend, Dtype *start_season,
66 int start_periods, ML::SeasonalType seasonal) {
67 const ML::cumlHandle_impl &handle_impl = handle. getImpl ();
68 ML::detail::streamSyncer _(handle_impl);
69 cudaStream_t stream = handle_impl. getStream ();
70 cublasHandle_t cublas_h = handle_impl. getCublasHandle ();
71
72 if (start_level != nullptr && start_trend == nullptr &&
73 start_season == nullptr ) { // level decomposition
74 MLCommon::copy (start_level, ts, batch_size, stream);
75 } else if (start_level != nullptr && start_trend != nullptr &&
76 start_season == nullptr ) { // trend decomposition
77 MLCommon::copy (start_level, ts + batch_size, batch_size, stream);
78 MLCommon::copy (start_trend, ts + batch_size, batch_size, stream);
79 const Dtype alpha = -1.;
80 CUBLAS_CHECK ( MLCommon::LinAlg::cublasaxpy (cublas_h, batch_size, &alpha, ts,
81 1, start_trend, 1, stream));
82 // cublas::axpy(batch_size, (Dtype)-1., ts, start_trend);
83 } else if (start_level != nullptr && start_trend != nullptr &&
84 start_season != nullptr ) {
85 stl_decomposition_gpu (handle_impl, ts, n , batch_size, frequency,
86 start_periods, start_level, start_trend, start_season,
87 seasonal);
88 }
89 }

118 template < typename Dtype>
119 void HoltWintersOptim ( const ML::cumlHandle &handle, const Dtype *ts, int n ,
120 int batch_size, int frequency, const Dtype *start_level,
121 const Dtype *start_trend, const Dtype *start_season,
122 Dtype *alpha, bool optim_alpha, Dtype *beta,
123 bool optim_beta, Dtype *gamma, bool optim_gamma,
124 Dtype epsilon, Dtype *level, Dtype *trend, Dtype *season,
125 Dtype *xhat, Dtype *error, OptimCriterion *optim_result,
126 OptimParams<Dtype> *optim_params,
127 ML::SeasonalType seasonal) {
128 const ML::cumlHandle_impl &handle_impl = handle. getImpl ();
129 ML::detail::streamSyncer _(handle_impl);
130 cudaStream_t stream = handle_impl. getStream ();
131
132 // default values
133 OptimParams<Dtype> optim_params_;
134 optim_params_. eps = epsilon;
135 optim_params_. min_param_diff = (Dtype)1e-8;
136 optim_params_. min_error_diff = (Dtype)1e-8;
137 optim_params_. min_grad_norm = (Dtype)1e-4;
138 optim_params_. bfgs_iter_limit = 1000;
139 optim_params_. linesearch_iter_limit = 100;
140 optim_params_. linesearch_tau = (Dtype)0.5;
141 optim_params_. linesearch_c = (Dtype)0.8;
142 optim_params_. linesearch_step_size = (Dtype)-1;
143
144 if (optim_params) {
145 if (optim_params-> eps > .0) optim_params_. eps = optim_params-> eps ;
146 if (optim_params-> min_param_diff >= .0)
147 optim_params_. min_param_diff = optim_params-> min_param_diff ;
148 if (optim_params-> min_error_diff >= .0)
149 optim_params_. min_error_diff = optim_params-> min_error_diff ;
150 if (optim_params-> min_grad_norm >= .0)
151 optim_params_. min_grad_norm = optim_params-> min_grad_norm ;
152 if (optim_params-> bfgs_iter_limit > 0)
153 optim_params_. bfgs_iter_limit = optim_params-> bfgs_iter_limit ;
154 if (optim_params-> linesearch_iter_limit > 0)
155 optim_params_. linesearch_iter_limit = optim_params-> linesearch_iter_limit ;
156 if (optim_params-> linesearch_tau > .0)
157 optim_params_. linesearch_tau = optim_params-> linesearch_tau ;
158 if (optim_params-> linesearch_c > .0)
159 optim_params_. linesearch_c = optim_params-> linesearch_c ;
160 if (optim_params-> linesearch_step_size > 0)
161 optim_params_. linesearch_step_size = optim_params-> linesearch_step_size ;
162 }
163
164 ASSERT (alpha && start_level, "HW error in in line %d" , __LINE__);
165 ASSERT (!((!start_trend) != (!beta) || (!start_season) != (!gamma)),
166 "HW error in in line %d" , __LINE__);
167 ASSERT (!(start_season && frequency < 2), "HW error in in line %d" , __LINE__);
168 ASSERT (!(!optim_alpha && !optim_beta && !optim_gamma),
169 "HW error in in line %d" , __LINE__);
170 ASSERT (!((optim_beta && !beta) || (optim_gamma && !gamma)),
171 "HW error in in line %d" , __LINE__);
172 if (!(!alpha && !beta && !gamma & !level && !trend && !season && !xhat &&
173 !error)) {
174 holtwinters_optim_gpu (
175 handle_impl, ts, n , batch_size, frequency, start_level, start_trend,
176 start_season, alpha, optim_alpha, beta, optim_beta, gamma, optim_gamma,
177 level, trend, season, xhat, error, optim_result, seasonal, optim_params_);
178 }
179 }

199 template < typename Dtype>
200 void HoltWintersFitHelper ( const ML::cumlHandle &handle, int n , int batch_size,
201 int frequency, int start_periods,
202 ML::SeasonalType seasonal, Dtype epsilon, Dtype *data,
203 Dtype *level_d, Dtype *trend_d, Dtype *season_d,
204 Dtype *error_d) {
205 const ML::cumlHandle_impl &handle_impl = handle. getImpl ();
206 ML::detail::streamSyncer _(handle_impl);
207 cudaStream_t stream = handle_impl. getStream ();
208 std::shared_ptr<MLCommon::deviceAllocator> dev_allocator =
209 handle_impl. getDeviceAllocator ();
210
211 bool optim_alpha = true , optim_beta = true , optim_gamma = true ;
212 // initial values for alpha, beta and gamma
213 std::vector<Dtype> alpha_h(batch_size, 0.4);
214 std::vector<Dtype> beta_h(batch_size, 0.3);
215 std::vector<Dtype> gamma_h(batch_size, 0.3);
216
217 int leveltrend_seed_len, season_seed_len, components_len;
218 int leveltrend_coef_offset, season_coef_offset;
219 int error_len;
220
221 HoltWintersBufferSize (
222 n , batch_size, frequency, optim_beta, optim_gamma,
223 &leveltrend_seed_len, // = batch_size
224 &season_seed_len, // = frequency*batch_size
225 &components_len, // = (n-w_len)*batch_size
226 &error_len, // = batch_size
227 &leveltrend_coef_offset, // = (n-wlen-1)*batch_size (last row)
228 &season_coef_offset); // = (n-wlen-frequency)*batch_size(last freq rows)
229
230 Dtype *trend_seed_d = nullptr , *start_season_d = nullptr ;
231 Dtype *beta_d = nullptr , *gamma_d = nullptr ;
232
233 MLCommon::device_buffer<Dtype> dataset_d(dev_allocator, stream,
234 batch_size * n );
235 MLCommon::device_buffer<Dtype> alpha_d(dev_allocator, stream, batch_size);
236 MLCommon::updateDevice (alpha_d. data (), alpha_h.data(), batch_size, stream);
237 MLCommon::device_buffer<Dtype> level_seed_d(dev_allocator, stream,
238 leveltrend_seed_len);
239
240 if (optim_beta) {
241 beta_d =
242 (Dtype *)dev_allocator->allocate( sizeof (Dtype) * batch_size, stream);
243 MLCommon::updateDevice (beta_d, beta_h.data(), batch_size, stream);
244 trend_seed_d = (Dtype *)dev_allocator->allocate(
245 sizeof (Dtype) * leveltrend_seed_len, stream);
246 }
247
248 if (optim_gamma) {
249 gamma_d =
250 (Dtype *)dev_allocator->allocate( sizeof (Dtype) * batch_size, stream);
251 MLCommon::updateDevice (gamma_d, gamma_h.data(), batch_size, stream);
252 start_season_d =
253 (Dtype *)dev_allocator->allocate( sizeof (Dtype) * season_seed_len, stream);
254 }
255
256 // Step 1: transpose the dataset (ML expects col major dataset)
257 HWTranspose (handle, data, batch_size, n , dataset_d. data ());
258
259 // Step 2: Decompose dataset to get seed for level, trend and seasonal values
260 HoltWintersDecompose (handle, dataset_d. data (), n , batch_size, frequency,
261 level_seed_d. data (), trend_seed_d, start_season_d,
262 start_periods, seasonal);
263
264 // Step 3: Find optimal alpha, beta and gamma values (seasonal HW)
265 HoltWintersOptim (handle, dataset_d. data (), n , batch_size, frequency,
266 level_seed_d. data (), trend_seed_d, start_season_d,
267 alpha_d. data (), optim_alpha, beta_d, optim_beta, gamma_d,
268 optim_gamma, epsilon, level_d, trend_d, season_d,
269 (Dtype *) nullptr , error_d, ( OptimCriterion *) nullptr ,
270 ( OptimParams<Dtype> *) nullptr , seasonal);
271
272 // Free the allocated memory on GPU
273 dev_allocator->deallocate(trend_seed_d, sizeof (Dtype) * leveltrend_seed_len,
274 stream);
275 dev_allocator->deallocate(start_season_d, sizeof (Dtype) * components_len,
276 stream);
277 dev_allocator->deallocate(beta_d, sizeof (Dtype) * batch_size, stream);
278 dev_allocator->deallocate(gamma_d, sizeof (Dtype) * batch_size, stream);
279 }

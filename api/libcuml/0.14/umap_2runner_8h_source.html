52 template < int TPB_X, typename T>
53 __global__ void init_transform ( int *indices, T *weights, int n ,
54 const T *embeddings, int embeddings_n,
55 int n_components, T *result, int n_neighbors) {
56 // row-based matrix 1 thread per row
57 int row = (blockIdx.x * TPB_X ) + threadIdx.x;
58 int i =
59 row * n_neighbors; // each thread processes one row of the dist matrix
60
61 if (row < n ) {
62 for ( int j = 0; j < n_neighbors; j ++) {
63 for ( int d = 0; d < n_components; d++) {
64 result[row * n_components + d] +=
65 weights[ i + j ] * embeddings[indices[ i + j ] * n_components + d];
66 }
67 }
68 }
69 }

81 template < typename T, int TPB_X>
82 void _fit ( const cumlHandle &handle,
83 T *X, // input matrix
84 int n , // rows
85 int d, // cols
86 int64_t *knn_indices, T *knn_dists, UMAPParams * params ,
87 T *embeddings) {

89 auto d_alloc = handle. getDeviceAllocator ();
90
91 int k = params ->n_neighbors;
92
93 ML::Logger::get (). setLevel ( params ->verbosity);
94
95 CUML_LOG_DEBUG ( "n_neighbors=%d" , params ->n_neighbors);
96
97 MLCommon::device_buffer<int64_t> *knn_indices_b = nullptr ;
98 MLCommon::device_buffer<T> *knn_dists_b = nullptr ;
99
100 if (!knn_indices || !knn_dists) {
101 ASSERT (!knn_indices && !knn_dists,
102 "Either both or none of the KNN parameters should be provided" );
103
107 knn_indices_b =
108 new MLCommon::device_buffer<int64_t> (d_alloc, stream, n * k );
109 knn_dists_b = new MLCommon::device_buffer<T> (d_alloc, stream, n * k );
110
111 knn_indices = knn_indices_b-> data ();
112 knn_dists = knn_dists_b-> data ();
113
114 kNNGraph::run (X, n , X, n , d, knn_indices, knn_dists, k , params , d_alloc,
115 stream);
116 CUDA_CHECK (cudaPeekAtLastError());
117 }
118
119 COO<T> rgraph_coo(d_alloc, stream);
120
121 FuzzySimplSet::run<TPB_X, T>( n , knn_indices, knn_dists, k , &rgraph_coo,
122 params , d_alloc, stream);
123
127 COO<T> cgraph_coo(d_alloc, stream);
128 MLCommon::Sparse::coo_remove_zeros<TPB_X, T>(&rgraph_coo, &cgraph_coo,
129 d_alloc, stream);
130
134 InitEmbed::run (handle, X, n , d, knn_indices, knn_dists, &cgraph_coo, params ,
135 embeddings, stream, params ->init);
136
137 if (knn_indices_b) delete knn_indices_b;
138 if (knn_dists_b) delete knn_dists_b;
139
140 if ( params ->callback) {
141 params ->callback->setup<T>( n , params ->n_components);
142 params ->callback->on_preprocess_end(embeddings);
143 }
144
148 SimplSetEmbed::run<TPB_X, T>(X, n , d, &cgraph_coo, params , embeddings,
149 d_alloc, stream);
150
151 if ( params ->callback) params ->callback->on_train_end(embeddings);
152 }

154 template < typename T, int TPB_X>
155 void _fit ( const cumlHandle &handle,
156 T *X, // input matrix
157 T * y , // labels
158 int n , int d, int64_t *knn_indices, T *knn_dists, UMAPParams * params ,
159 T *embeddings) {
160 std::shared_ptr<deviceAllocator> d_alloc = handle. getDeviceAllocator ();
161 cudaStream_t stream = handle. getStream ();
162
163 int k = params ->n_neighbors;
164
165 ML::Logger::get (). setLevel ( params ->verbosity);
166
167 if ( params ->target_n_neighbors == -1)
168 params ->target_n_neighbors = params ->n_neighbors;
169
170 MLCommon::device_buffer<int64_t> *knn_indices_b = nullptr ;
171 MLCommon::device_buffer<T> *knn_dists_b = nullptr ;
172
173 if (!knn_indices || !knn_dists) {
174 ASSERT (!knn_indices && !knn_dists,
175 "Either both or none of the KNN parameters should be provided" );
176
180 knn_indices_b =
181 new MLCommon::device_buffer<int64_t> (d_alloc, stream, n * k );
182 knn_dists_b = new MLCommon::device_buffer<T> (d_alloc, stream, n * k );
183
184 knn_indices = knn_indices_b-> data ();
185 knn_dists = knn_dists_b-> data ();
186
187 kNNGraph::run (X, n , X, n , d, knn_indices, knn_dists, k , params , d_alloc,
188 stream);
189 CUDA_CHECK (cudaPeekAtLastError());
190 }
191
195 COO<T> rgraph_coo(d_alloc, stream);
196 COO<T> tmp_coo(d_alloc, stream);
197
201 //int nnz = n*k*2;
202 FuzzySimplSet::run<TPB_X, T>( n , knn_indices, knn_dists, params ->n_neighbors,
203 &tmp_coo, params , d_alloc, stream);
204 CUDA_CHECK (cudaPeekAtLastError());
205
206 MLCommon::Sparse::coo_remove_zeros<TPB_X, T>(&tmp_coo, &rgraph_coo, d_alloc,
207 stream);
208
209 COO<T> final_coo(d_alloc, stream);
210
215 if ( params ->target_metric == ML::UMAPParams::MetricType::CATEGORICAL) {
216 CUML_LOG_DEBUG ( "Performing categorical intersection" );
217 Supervised::perform_categorical_intersection<TPB_X, T>(
218 y , &rgraph_coo, &final_coo, params , d_alloc, stream);
219
223 } else {
224 CUML_LOG_DEBUG ( "Performing general intersection" );
225 Supervised::perform_general_intersection<TPB_X, T>(
226 handle, y , &rgraph_coo, &final_coo, params , stream);
227 }
228
232 MLCommon::Sparse::coo_sort<T>(&final_coo, d_alloc, stream);
233
234 COO<T> ocoo(d_alloc, stream);
235 MLCommon::Sparse::coo_remove_zeros<TPB_X, T>(&final_coo, &ocoo, d_alloc,
236 stream);
240 InitEmbed::run (handle, X, n , d, knn_indices, knn_dists, &ocoo, params ,
241 embeddings, stream, params ->init);
242
243 if (knn_indices_b) delete knn_indices_b;
244 if (knn_dists_b) delete knn_dists_b;
245
246 if ( params ->callback) {
247 params ->callback->setup<T>( n , params ->n_components);
248 params ->callback->on_preprocess_end(embeddings);
249 }
250
254 SimplSetEmbed::run<TPB_X, T>(X, n , d, &ocoo, params , embeddings, d_alloc,
255 stream);
256
257 if ( params ->callback) params ->callback->on_train_end(embeddings);
258
259 CUDA_CHECK (cudaPeekAtLastError());
260 }

265 template < typename T, int TPB_X>
266 void _transform ( const cumlHandle &handle, T *X, int n , int d,
267 int64_t *knn_indices, float *knn_dists, T *orig_X, int orig_n,
268 T *embedding, int embedding_n, UMAPParams * params ,
269 T *transformed) {
270 std::shared_ptr<deviceAllocator> d_alloc = handle. getDeviceAllocator ();
271 cudaStream_t stream = handle. getStream ();
272
273 ML::Logger::get (). setLevel ( params ->verbosity);
274
275 CUML_LOG_DEBUG ( "Running transform" );
276
277 CUML_LOG_DEBUG ( "Building KNN Graph" );
278
279 MLCommon::device_buffer<int64_t> *knn_indices_b = nullptr ;
280 MLCommon::device_buffer<T> *knn_dists_b = nullptr ;
281
282 if (!knn_indices || !knn_dists) {
283 ASSERT (!knn_indices && !knn_dists,
284 "Either both or none of the KNN parameters should be provided" );
285
290 int k = params ->n_neighbors;
291
292 knn_indices_b =
293 new MLCommon::device_buffer<int64_t> (d_alloc, stream, n * k );
294 knn_dists_b = new MLCommon::device_buffer<T> (d_alloc, stream, n * k );
295
296 knn_indices = knn_indices_b-> data ();
297 knn_dists = knn_dists_b-> data ();
298
299 kNNGraph::run (orig_X, orig_n, X, n , d, knn_indices, knn_dists,
300 params ->n_neighbors, params , d_alloc, stream);
301 CUDA_CHECK (cudaPeekAtLastError());
302 }
303
304 float adjusted_local_connectivity =
305 max (0.0, params ->local_connectivity - 1.0);
306
307 CUML_LOG_DEBUG ( "Smoothing KNN distances" );
308
312 MLCommon::device_buffer<T> sigmas(d_alloc, stream, n );
313 MLCommon::device_buffer<T> rhos(d_alloc, stream, n );
314 CUDA_CHECK (cudaMemsetAsync(sigmas. data (), 0, n * sizeof (T), stream));
315 CUDA_CHECK (cudaMemsetAsync(rhos. data (), 0, n * sizeof (T), stream));
316
317 dim3 grid_n( MLCommon::ceildiv ( n , TPB_X ), 1, 1);
318 dim3 blk( TPB_X , 1, 1);
319
320 FuzzySimplSetImpl::smooth_knn_dist<TPB_X, T>(
321 n , knn_indices, knn_dists, rhos. data (), sigmas. data (), params ,
322 params ->n_neighbors, adjusted_local_connectivity, d_alloc, stream);
323
328 int nnz = n * params ->n_neighbors;
329
330 dim3 grid_nnz( MLCommon::ceildiv (nnz, TPB_X ), 1, 1);
331
332 CUML_LOG_DEBUG ( "Executing fuzzy simplicial set" );
333
338 COO<T> graph_coo(d_alloc, stream, nnz, n , n );
339
340 FuzzySimplSetImpl::compute_membership_strength_kernel<TPB_X>
341 <<<grid_nnz, blk, 0, stream>>>(knn_indices, knn_dists, sigmas. data (),
342 rhos. data (), graph_coo. vals (),
343 graph_coo. rows (), graph_coo. cols (),
344 graph_coo. n_rows , params ->n_neighbors);
345 CUDA_CHECK (cudaPeekAtLastError());
346
347 if (knn_indices_b) delete knn_indices_b;
348 if (knn_dists_b) delete knn_dists_b;
349
350 MLCommon::device_buffer<int> row_ind(d_alloc, stream, n );
351 MLCommon::device_buffer<int> ia(d_alloc, stream, n );
352
353 MLCommon::Sparse::sorted_coo_to_csr (&graph_coo, row_ind. data (), d_alloc,
354 stream);
355 MLCommon::Sparse::coo_row_count<TPB_X>(&graph_coo, ia. data (), stream);
356
357 MLCommon::device_buffer<T> vals_normed(d_alloc, stream, graph_coo. nnz );
358 CUDA_CHECK (
359 cudaMemsetAsync(vals_normed. data (), 0, graph_coo. nnz * sizeof (T), stream));
360
361 CUML_LOG_DEBUG ( "Performing L1 normalization" );
362
363 MLCommon::Sparse::csr_row_normalize_l1<TPB_X, T>(
364 row_ind. data (), graph_coo. vals (), graph_coo. nnz , graph_coo. n_rows ,
365 vals_normed. data (), stream);
366
367 init_transform<TPB_X, T><<<grid_n, blk, 0, stream>>>(
368 graph_coo. cols (), vals_normed. data (), graph_coo. n_rows , embedding,
369 embedding_n, params ->n_components, transformed, params ->n_neighbors);
370 CUDA_CHECK (cudaPeekAtLastError());
371
372 CUDA_CHECK (cudaMemsetAsync(ia. data (), 0.0, ia. size () * sizeof ( int ), stream));
373
374 CUDA_CHECK (cudaPeekAtLastError());
375
380 thrust::device_ptr<T> d_ptr = thrust::device_pointer_cast(graph_coo. vals ());
381 T max =
382 *(thrust::max_element(thrust::cuda::par.on(stream), d_ptr, d_ptr + nnz));
383
384 int n_epochs = params ->n_epochs;
385 if (n_epochs <= 0) {
386 if ( n <= 10000)
387 n_epochs = 100;
388 else
389 n_epochs = 30;
390 } else {
391 n_epochs /= 3;
392 }
393
394 CUML_LOG_DEBUG ( "n_epochs=%d" , n_epochs);
395
396 MLCommon::LinAlg::unaryOp<T>(
397 graph_coo. vals (), graph_coo. vals (), graph_coo. nnz ,
398 [=] __device__(T input) {
399 if (input < ( max / float (n_epochs)))
400 return 0.0 f ;
401 else
402 return input;
403 },
404 stream);
405
406 CUDA_CHECK (cudaPeekAtLastError());
407
411 MLCommon::Sparse::COO<T> comp_coo(d_alloc, stream);
412 MLCommon::Sparse::coo_remove_zeros<TPB_X, T>(&graph_coo, &comp_coo, d_alloc,
413 stream);
414
415 CUML_LOG_DEBUG ( "Computing # of epochs for training each sample" );
416
417 MLCommon::device_buffer<T> epochs_per_sample(d_alloc, stream, nnz);
418
419 SimplSetEmbedImpl::make_epochs_per_sample (
420 comp_coo. vals (), comp_coo. nnz , n_epochs, epochs_per_sample. data (), stream);
421
422 CUML_LOG_DEBUG ( "Performing optimization" );
423
424 if ( params ->callback) {
425 params ->callback->setup<T>( n , params ->n_components);
426 params ->callback->on_preprocess_end(transformed);
427 }
428
429 params ->initial_alpha /=
430 4.0; // TODO: This value should be passed into "optimize layout" directly to avoid side-effects.
431
432 SimplSetEmbedImpl::optimize_layout<TPB_X, T>(
433 transformed, n , embedding, embedding_n, comp_coo. rows (), comp_coo. cols (),
434 comp_coo. nnz , epochs_per_sample. data (), n , params ->repulsion_strength,
435 params , n_epochs, params ->multicore_implem, d_alloc, stream);
436
437 if ( params ->callback) params ->callback->on_train_end(transformed);
438 }

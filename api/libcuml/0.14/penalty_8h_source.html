64 template < typename math_t>
65 void elasticnet (math_t *out, const math_t *coef, const int len,
66 const math_t alpha, const math_t l1_ratio,
67 cudaStream_t stream) {
68 math_t *out_lasso = NULL;
69 allocate (out_lasso, 1);
70
71 ridge (out, coef, len, alpha * (math_t(1) - l1_ratio), stream);
72 lasso (out_lasso, coef, len, alpha * l1_ratio, stream);
73
74 LinAlg::add (out, out, out_lasso, 1, stream);
75
76 if (out_lasso != NULL) {
77 CUDA_CHECK (cudaFree(out_lasso));
78 }
79 }

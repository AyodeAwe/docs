23 template < typename Type, typename Lambda>
24 __global__ void eltwise2DKernel ( int rows, // m
25 int cols, // n
26 const Type *dotA, const Type *dotB,
27 const Type *pC, Type *pD, Type alpha, Type beta,
28 Lambda op) {
29 auto tid = blockIdx.x * blockDim.x + threadIdx.x;
30 if (tid < cols * rows) {
31 const auto x = tid % cols;
32 const auto y = tid / cols;
33 const auto ab = pD[tid];
34 const auto a = dotA[ y ];
35 const auto b = dotB[ x ];
36 Type accm = alpha * op(a, b, ab);
37
38 if (beta) {
39 accm += beta * pC[tid];
40 }
41 pD[tid] = accm;
42 }
43 }

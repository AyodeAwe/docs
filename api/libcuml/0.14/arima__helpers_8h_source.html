43 template < bool isAr, typename DataT>
44 HDI DataT _param_to_poly ( const DataT* param, int lags, int idx) {
45 if (idx > lags) {
46 return 0.0;
47 } else if (idx) {
48 return isAr ? -param[idx - 1] : param[idx - 1];
49 } else
50 return 1.0;
51 }

68 template < bool isAr, typename DataT>
69 HDI DataT reduced_polynomial ( int bid, const DataT* param, int lags,
70 const DataT* sparam, int slags, int s, int idx) {
71 int idx1 = s ? idx / s : 0;
72 int idx0 = idx - s * idx1;
73 DataT coef0 = _param_to_poly<isAr>(param + bid * lags, lags, idx0);
74 DataT coef1 = _param_to_poly<isAr>(sparam + bid * slags, slags, idx1);
75 return isAr ? -coef0 * coef1 : coef0 * coef1;
76 }

93 template < typename DataT>
94 void prepare_data (DataT* d_out, const DataT* d_in, int batch_size, int n_obs,
95 int d, int D, int s, cudaStream_t stream) {
96 // Only one difference (simple or seasonal)
97 if (d + D == 1) {
98 int period = d ? 1 : s;
99 int tpb = (n_obs - period) > 512 ? 256 : 128; // quick heuristics
100 MLCommon::LinAlg::Batched::
101 batched_diff_kernel<<<batch_size, tpb, 0, stream>>>(d_in, d_out, n_obs,
102 period);
103 CUDA_CHECK (cudaPeekAtLastError());
104 }
105 // Two differences (simple or seasonal or both)
106 else if (d + D == 2) {
107 int period1 = d ? 1 : s;
108 int period2 = d == 2 ? 1 : s;
109 int tpb = (n_obs - period1 - period2) > 512 ? 256 : 128;
110 MLCommon::LinAlg::Batched::
111 batched_second_diff_kernel<<<batch_size, tpb, 0, stream>>>(
112 d_in, d_out, n_obs, period1, period2);
113 CUDA_CHECK (cudaPeekAtLastError());
114 }
115 // If no difference and the pointers are different, copy in to out
116 else if (d + D == 0 && d_in != d_out) {
117 MLCommon::copy (d_out, d_in, n_obs * batch_size, stream);
118 }
119 // Other cases: no difference and the pointers are the same, nothing to do
120 }

139 template < bool double _diff, typename DataT>
140 __global__ void _undiff_kernel (DataT* d_fc, const DataT* d_in, int num_steps,
141 int batch_size, int in_ld, int n_in, int s0,
142 int s1 = 0) {
143 int bid = blockIdx.x * blockDim.x + threadIdx.x;
144 if (bid < batch_size) {
145 DataT* b_fc = d_fc + bid * num_steps;
146 const DataT* b_in = d_in + bid * in_ld;
147 for ( int i = 0; i < num_steps; i ++) {
148 if (!double_diff) { // One simple or seasonal difference
149 b_fc[ i ] += _select_read (b_in, n_in, b_fc, i - s0);
150 } else { // Two differences (simple, seasonal or both)
151 DataT fc_acc = - _select_read (b_in, n_in, b_fc, i - s0 - s1);
152 fc_acc += _select_read (b_in, n_in, b_fc, i - s0);
153 fc_acc += _select_read (b_in, n_in, b_fc, i - s1);
154 b_fc[ i ] += fc_acc;
155 }
156 }
157 }

177 template < typename DataT>
178 void finalize_forecast (DataT* d_fc, const DataT* d_in, int num_steps,
179 int batch_size, int in_ld, int n_in, int d, int D, int s,
180 cudaStream_t stream) {
181 // Undifference
182 constexpr int TPB = 64; // One thread per series -> avoid big blocks
183 if (d + D == 1) {
184 _undiff_kernel<false>
185 <<<MLCommon::ceildiv<int>(batch_size, TPB), TPB, 0, stream>>>(
186 d_fc, d_in, num_steps, batch_size, in_ld, n_in, d ? 1 : s);
187 CUDA_CHECK (cudaPeekAtLastError());
188 } else if (d + D == 2) {
189 _undiff_kernel<true>
190 <<<MLCommon::ceildiv<int>(batch_size, TPB), TPB, 0, stream>>>(
191 d_fc, d_in, num_steps, batch_size, in_ld, n_in, d ? 1 : s,
192 d == 2 ? 1 : s);
193 CUDA_CHECK (cudaPeekAtLastError());
194 }

210 template < typename DataT>
211 void batched_jones_transform ( const ML::ARIMAOrder & order, int batch_size,
212 bool isInv, const ML::ARIMAParams<DataT> & params,
213 const ML::ARIMAParams<DataT> & Tparams,
214 std::shared_ptr<deviceAllocator> allocator,
215 cudaStream_t stream) {
216 if (order. p )
217 jones_transform (params. ar , batch_size, order. p , Tparams. ar , true , isInv,
218 allocator, stream);
219 if (order. q )
220 jones_transform (params. ma , batch_size, order. q , Tparams. ma , false , isInv,
221 allocator, stream);
222 if (order. P )
223 jones_transform (params. sar , batch_size, order. P , Tparams. sar , true , isInv,
224 allocator, stream);
225 if (order. Q )
226 jones_transform (params. sma , batch_size, order. Q , Tparams. sma , false , isInv,
227 allocator, stream);
228
229 // Constrain sigma2 to be strictly positive
230 constexpr DataT min_sigma2 = 1e-6;
231 LinAlg::unaryOp<DataT>(
232 Tparams. sigma2 , params. sigma2 , batch_size,
233 [=] __device__(DataT input) { return max (input, min_sigma2); }, stream);
234 }

202 template < typename T>
203 void gemm (cublasHandle_t handle, T *z, T * x , T * y , int _M, int _N, int _K,
204 bool isZColMajor, bool isXColMajor, bool isYColMajor,
205 cudaStream_t stream, T alpha = T(1.0), T beta = T(0.0)) {
206 cublasOperation_t trans_a, trans_b;
207 T *a, *b, *c;
208 int lda , ldb, ldc;
209 int M, N , K;
210 // This function performs c = a * b. Based on the required output layout,
211 // either a = x, b = y or a = y, b = x. In either case c = z.
212 if (isZColMajor == true ) {
213 // Result c is required in column major layout. Thus we perform,
214 // z = x * y
215 // Using BLAS call c = a * b. Therefore a = x, b = y and c = z
216
217 a = x ;
218 // If x is in row major layout, cublas needs to transpose x first,
219 // therefore trans_x needs to be CUBLAS_OP_T. If x is in column major
220 // layout, trans_b needs to be CUBLAS_OP_N.
221 trans_a = isXColMajor == true ? CUBLAS_OP_N : CUBLAS_OP_T;
222 // Set leading dimension appropriately
223 lda = isXColMajor == true ? _M : _K;
224
225 b = y ;
226 // If y is in row major layout, cublas needs to transpose y first,
227 // therefore trans_x needs to be CUBLAS_OP_T. If x is in column major
228 // layout, trans_b needs to be CUBLAS_OP_N.
229 trans_b = isYColMajor == true ? CUBLAS_OP_N : CUBLAS_OP_T;
230 ldb = isYColMajor == true ? _K : _N;
231
232 c = z;
233 ldc = _M;
234 M = _M;
235 N = _N;
236 K = _K;
237 } else {
238 // Result c is required in row major layout Thus we pick
239 // a = y, b = x and c = a * b = y * x
240 // cublas produces output matrix only in column major layout. To get output
241 // matrix on row major layout, we need to produce transpose of output
242 // in column major layout. Therefore we perform,
243 // tr(z) = tr(y) * tr(x)
244 // we model this using cublas call for c = a * b
245 // therefore a = tr(y), b = tr(x) and c = tr(z)
246
247 a = y ;
248 // If y is in row major layout, it can be/ interpreted as tr(y) on column
249 // major layout. Therefore we can pass trans_a as CUBLAS_OP_N. If y is in
250 // column major layout, cublas needs to transpose y first, therefore
251 // trans_a needs to be CUBLAS_OP_T
252 trans_a = isYColMajor == true ? CUBLAS_OP_T : CUBLAS_OP_N;
253 // Set leading dimension appropriately
254 lda = isYColMajor == true ? _K : _N;
255
256 b = x ;
257 // If x is in row major layout, it can be interpreted as tr(x) on column
258 // major layout. Therefore we can pass trans_b as CUBLAS_OP_N. If x is in
259 // column major layout, cublas needs to trasponse x first, therefore
260 // trans_b needs to be CUBLAS_OP_T
261 trans_b = isXColMajor == true ? CUBLAS_OP_T : CUBLAS_OP_N;
262 // Set leading dimension appropriately
263 ldb = isXColMajor == true ? _M : _K;
264
265 c = z;
266 ldc = _N;
267
268 M = _N;
269 N = _M;
270 K = _K;
271 }
272 // Actual cuBLAS call
273 CUBLAS_CHECK ( LinAlg::cublasgemm (handle, trans_a, trans_b, M, N , K, &alpha, a,
274 lda , b, ldb, &beta, c, ldc, stream));
275 }

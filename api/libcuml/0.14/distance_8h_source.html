151 template < DistanceType distanceType, typename InType, typename AccType,
152 typename OutType, typename Index_ = int >
153 size_t getWorkspaceSize ( const InType * x , const InType * y , Index_ m , Index_ n ,
154 Index_ k ) {
155 size_t worksize = 0;
156 constexpr bool is_allocated = distanceType <= EucExpandedCosine ;
157 if (is_allocated) {
158 worksize += m * sizeof (AccType);
159 if ( x != y ) worksize += n * sizeof (AccType);
160 }
161 return worksize;
162 }

281 template < typename Type, typename Index_ = int >
282 void pairwiseDistance ( const Type * x , const Type * y , Type *dist, Index_ m ,
283 Index_ n , Index_ k , device_buffer<char> &workspace,
284 DistanceType metric, cudaStream_t stream,
285 bool isRowMajor = true ) {
286 switch (metric) {
287 case DistanceType::EucExpandedL2 :
288 pairwiseDistanceImpl<Type, Index_, DistanceType::EucExpandedL2>(
289 x , y , dist, m , n , k , workspace, stream, isRowMajor);
290 break ;
291 case DistanceType::EucExpandedL2Sqrt :
292 pairwiseDistanceImpl<Type, Index_, DistanceType::EucExpandedL2Sqrt>(
293 x , y , dist, m , n , k , workspace, stream, isRowMajor);
294 break ;
295 case DistanceType::EucExpandedCosine :
296 pairwiseDistanceImpl<Type, Index_, DistanceType::EucExpandedCosine>(
297 x , y , dist, m , n , k , workspace, stream, isRowMajor);
298 break ;
299 case DistanceType::EucUnexpandedL1 :
300 pairwiseDistanceImpl<Type, Index_, DistanceType::EucUnexpandedL1>(
301 x , y , dist, m , n , k , workspace, stream, isRowMajor);
302 break ;
303 case DistanceType::EucUnexpandedL2 :
304 pairwiseDistanceImpl<Type, Index_, DistanceType::EucUnexpandedL2>(
305 x , y , dist, m , n , k , workspace, stream, isRowMajor);
306 break ;
307 case DistanceType::EucUnexpandedL2Sqrt :
308 pairwiseDistanceImpl<Type, Index_, DistanceType::EucUnexpandedL2Sqrt>(
309 x , y , dist, m , n , k , workspace, stream, isRowMajor);
310 break ;
311 default :
312 THROW ( "Unknown distance metric '%d'!" , metric);
313 };

65 template < typename DataT, typename OutT, typename IdxT, bool Sqrt,
66 typename Policy, typename ReduceOpT,
67 typename BaseClass = LinAlg::Contractions_NT<DataT, IdxT, Policy>>
68 struct FusedL2NN : public BaseClass {
69 private :
70 typedef Policy P;
71
72 const DataT* xn;
73 const DataT* yn;
74 OutT* min;
75 int * mutex;
76
77 DataT *sxNorm, *syNorm;
78 cub::KeyValuePair<IdxT, DataT>* sRed;
79
80 DataT maxVal;
81
82 DataT acc[P::AccRowsPerTh][P::AccColsPerTh];
83
84 ReduceOpT redOp;
85
86 static const DataT Two = (DataT)2.0;
87
88 public :
89 DI FusedL2NN (OutT* _min, const DataT* _x, const DataT* _y, const DataT* _xn,
90 const DataT* _yn, IdxT _m, IdxT _n, IdxT _k, char * _smem,
91 DataT _mv, int * _mut, ReduceOpT op)
92 : BaseClass(_x, _y, _m, _n, _k, _smem),
93 xn (_xn),
94 yn (_yn),
95 min(_min),
96 mutex (_mut),
97 sxNorm((DataT*)_smem),
98 syNorm(&(sxNorm[P::Mblk])),
99 sRed((cub::KeyValuePair<IdxT, DataT>*)_smem),
100 maxVal (_mv),
101 redOp (op) {}
102
103 DI void run () {
104 prolog();
105 loop();
106 __syncthreads (); // so that we can safely reuse smem
107 epilog();
108 }
109
110 private :
111 DI void prolog() {
112 this->ldgsts(0);
113 this->pageWr ^= 1;
114 #pragma unroll
115 for ( int i = 0; i < P::AccRowsPerTh; ++ i ) {
116 #pragma unroll
117 for ( int j = 0; j < P::AccColsPerTh; ++ j ) {
118 acc[ i ][ j ] = BaseClass::Zero;
119 }
120 }
121 __syncthreads ();
122 }
123
124 DI void loop() {
125 for ( int kidx = P::Kblk; kidx < this-> k ; kidx += P::Kblk) {
126 this->ldgsts(kidx);
127 accumulate(); // on the previous k-block
128 __syncthreads ();
129 this->pageWr ^= 1;
130 this->pageRd ^= 1;
131 }
132 accumulate(); // last iteration
133 }
134
135 DI void epilog() {
136 for ( int i = threadIdx.x; i < P::Mblk; i += P::Nthreads) {
137 auto idx = blockIdx.x * P::Mblk + i ;
138 sxNorm[ i ] = idx < this-> m ? xn[idx] : maxVal ;
139 }
140 for ( int i = threadIdx.x; i < P::Nblk; i += P::Nthreads) {
141 auto idx = blockIdx.y * P::Nblk + i ;
142 syNorm[ i ] = idx < this-> n ? yn[idx] : maxVal ;
143 }
144 __syncthreads ();
145 DataT regxn[P::AccRowsPerTh], regyn[P::AccColsPerTh];
146 #pragma unroll
147 for ( int i = 0; i < P::AccRowsPerTh; ++ i ) {
148 regxn[ i ] = sxNorm[ i * P::AccThRows + this->accrowid];
149 }
150 #pragma unroll
151 for ( int i = 0; i < P::AccColsPerTh; ++ i ) {
152 regyn[ i ] = syNorm[ i * P::AccThCols + this->acccolid];
153 }
154 #pragma unroll
155 for ( int i = 0; i < P::AccRowsPerTh; ++ i ) {
156 #pragma unroll
157 for ( int j = 0; j < P::AccColsPerTh; ++ j ) {
158 acc[ i ][ j ] = regxn[ i ] + regyn[ j ] - Two * acc[ i ][ j ];
159 }
160 }
161 if (Sqrt) {
162 #pragma unroll
163 for ( int i = 0; i < P::AccRowsPerTh; ++ i ) {
164 #pragma unroll
165 for ( int j = 0; j < P::AccColsPerTh; ++ j ) {
166 acc[ i ][ j ] = mySqrt (acc[ i ][ j ]);
167 }
168 }
169 }
170 // reduce
171 cub::KeyValuePair<IdxT, DataT> val[P::AccRowsPerTh];
172 KVPMinReduce<IdxT, DataT> pairRedOp;
173 auto lid = laneId ();
174 #pragma unroll
175 for ( int i = 0; i < P::AccRowsPerTh; ++ i ) {
176 val[ i ] = {-1, maxVal };
177 #pragma unroll
178 for ( int j = 0; j < P::AccColsPerTh; ++ j ) {
179 auto tmpkey = this->acccolid + j * P::AccThCols + blockIdx.y * P::Nblk;
180 cub::KeyValuePair<IdxT, DataT> tmp = {tmpkey, acc[ i ][ j ]};
181 if (tmpkey < this-> n ) val[ i ] = pairRedOp(tmp, val[ i ]);
182 }
183 __syncthreads ();
184 #pragma unroll
185 for ( int j = P::AccThCols / 2; j > 0; j >>= 1) {
186 auto tmpkey = shfl (val[ i ].key, lid + j );
187 auto tmpvalue = shfl (val[ i ].value, lid + j );
188 cub::KeyValuePair<IdxT, DataT> tmp = {tmpkey, tmpvalue};
189 val[ i ] = pairRedOp(tmp, val[ i ]);
190 }
191 }
192 if (lid % P::AccThCols == 0) {
193 #pragma unroll
194 for ( int i = 0; i < P::AccRowsPerTh; ++ i ) {
195 sRed[ i * P::AccThCols + this->accrowid] = val[ i ];
196 }
197 }
198 __syncthreads ();
199 updateResults();
200 }
201
202 /*
203 * todo: From Volta onwards see if "coalesced" atomicCAS approach as
204 * written below helps improve perf
205 * ```
206 * auto tid = threadIdx.x;
207 * auto rid = IdxT(blockIdx.x) * P::Mblk + tid;
208 * if (rid < m) {
209 * auto val = sRed[i];
210 * while (atomicCAS(mutex + rid, 0, 1) == 1)
211 * ;
212 * __threadfence();
213 * redOp(min + rid, val);
214 * __threadfence();
215 * atomicCAS(mutex + rid, 1, 0);
216 * }
217 * ```
218 */
219 DI void updateResults() {
220 // for now have first lane from each warp update a unique output row. This
221 // will resolve hang issues with pre-Volta architectures
222 auto nWarps = blockDim.x / WarpSize;
223 auto lid = laneId ();
224 auto ridx = IdxT(blockIdx.x) * P::Mblk;
225 if (lid == 0) {
226 for ( int i = threadIdx.x / WarpSize; i < P::Mblk; i += nWarps) {
227 auto rid = ridx + i ;
228 if (rid < this-> m ) {
229 auto val = sRed[ i ];
230 while (atomicCAS(mutex + rid, 0, 1) == 1)
231 ;
232 __threadfence();
233 redOp (min + rid, val);
234 __threadfence();
235 atomicCAS(mutex + rid, 1, 0);
236 }
237 }
238 }
239 }
240
241 DI void accumulate() {
242 #pragma unroll
243 for ( int ki = 0; ki < P::Kblk; ki += P::Veclen) {
244 this->ldsXY(ki);
245 #pragma unroll
246 for ( int i = 0; i < P::AccRowsPerTh; ++ i ) {
247 #pragma unroll
248 for ( int j = 0; j < P::AccColsPerTh; ++ j ) {
249 #pragma unroll
250 for ( int v = 0; v < P::Veclen; ++v) {
251 acc[ i ][ j ] += this->regx[ i ][v] * this->regy[ j ][v];
252 }
253 }
254 }
255 }
256 }
257 }; // struct FusedL2NN

278 template < typename DataT, typename OutT, typename IdxT, int VecLen,
279 typename ReduceOpT>
280 void fusedL2NNImpl (OutT* min, const DataT* x , const DataT* y , const DataT* xn ,
281 const DataT* yn , IdxT m , IdxT n , IdxT k , int * workspace,
282 ReduceOpT redOp , bool sqrt , bool initOutBuffer,
283 cudaStream_t stream) {
284 typedef typename LinAlg::Policy4x4<DataT, VecLen>::Policy Policy;
285 dim3 grid(ceildiv<int>( m , Policy::Mblk), ceildiv<int>( n , Policy::Nblk));
286 dim3 blk(Policy::Nthreads);
287 auto nblks = ceildiv<int>( m , Policy::Nthreads);
288 auto maxVal = std::numeric_limits<DataT>::max ();
289 CUDA_CHECK (cudaMemsetAsync(workspace, 0, sizeof ( int ) * m , stream));
290 if (initOutBuffer) {
291 initKernel<DataT, OutT, IdxT, ReduceOpT>
292 <<<nblks, Policy::Nthreads, 0, stream>>>(min, m , maxVal , redOp );
293 CUDA_CHECK (cudaGetLastError());
294 }
295 if ( sqrt ) {
296 fusedL2NNkernel<DataT, OutT, IdxT, true, Policy, ReduceOpT>
297 <<<grid, blk, Policy::SmemSize, stream>>>(min, x , y , xn , yn , m , n , k ,
298 maxVal , workspace, redOp );
299 } else {
300 fusedL2NNkernel<DataT, OutT, IdxT, false, Policy, ReduceOpT>
301 <<<grid, blk, Policy::SmemSize, stream>>>(min, x , y , xn , yn , m , n , k ,
302 maxVal , workspace, redOp );
303 }
304 CUDA_CHECK (cudaGetLastError());
305 }

342 template < typename DataT, typename OutT, typename IdxT, typename ReduceOpT>
343 void fusedL2NN (OutT* min, const DataT* x , const DataT* y , const DataT* xn ,
344 const DataT* yn , IdxT m , IdxT n , IdxT k , void * workspace,
345 ReduceOpT redOp , bool sqrt , bool initOutBuffer,
346 cudaStream_t stream) {

348 if (16 % sizeof (DataT) == 0 && bytes % 16 == 0) {

351 stream);
352 } else if (8 % sizeof (DataT) == 0 && bytes % 8 == 0) {
353 fusedL2NNImpl <DataT, OutT, IdxT, 8 / sizeof (DataT), ReduceOpT>(
354 min, x , y , xn , yn , m , n , k , ( int *)workspace, redOp , sqrt , initOutBuffer,
355 stream);
356 } else {
357 fusedL2NNImpl<DataT, OutT, IdxT, 1, ReduceOpT>(min, x , y , xn , yn , m , n , k ,

48 void Barnes_Hut ( float *VAL, const int *COL, const int *ROW, const int NNZ,
49 const cumlHandle &handle, float *Y, const int n ,
50 const float theta = 0.5 f , const float epssq = 0.0025,
51 const float early_exaggeration = 12.0 f ,
52 const int exaggeration_iter = 250, const float min_gain = 0.01 f ,
53 const float pre_learning_rate = 200.0 f ,
54 const float post_learning_rate = 500.0 f ,
55 const int max_iter = 1000, const float min_grad_norm = 1e-7,
56 const float pre_momentum = 0.5, const float post_momentum = 0.8,
57 const long long random_state = -1) {
58 auto d_alloc = handle. getDeviceAllocator ();
59 cudaStream_t stream = handle. getStream ();
60
61 // Get device properites
62 //---------------------------------------------------
63 const int blocks = MLCommon::getMultiProcessorCount ();
64
65 int nnodes = n * 2;
66 if (nnodes < 1024 * blocks) nnodes = 1024 * blocks;
67 while ((nnodes & (32 - 1)) != 0) nnodes++;
68 nnodes--;
69 CUML_LOG_DEBUG ( "N_nodes = %d blocks = %d" , nnodes, blocks);
70
71 // Allocate more space
72 //---------------------------------------------------
73 //int *errl = (int *)d_alloc->allocate(sizeof(int), stream);
74 unsigned * limiter = ( unsigned *)d_alloc->allocate( sizeof ( unsigned ), stream);
75 int * maxdepthd = ( int *)d_alloc->allocate( sizeof ( int ), stream);
76 int * bottomd = ( int *)d_alloc->allocate( sizeof ( int ), stream);
77 float * radiusd = ( float *)d_alloc->allocate( sizeof ( float ), stream);
78
79 TSNE::InitializationKernel<<<1, 1, 0, stream>>>( /*errl,*/ limiter , maxdepthd ,
80 radiusd );
81 CUDA_CHECK (cudaPeekAtLastError());
82
83 const int FOUR_NNODES = 4 * nnodes;
84 const int FOUR_N = 4 * n ;
85 const float theta_squared = theta * theta;
86 const int NNODES = nnodes;
87
88 // Actual mallocs
89 int *startl = ( int *)d_alloc->allocate( sizeof ( int ) * (nnodes + 1), stream);
90 int *childl =
91 ( int *)d_alloc->allocate( sizeof ( int ) * (nnodes + 1) * 4, stream);
92 float *massl =
93 ( float *)d_alloc->allocate( sizeof ( float ) * (nnodes + 1), stream);
94 thrust::device_ptr<float> begin_massl = thrust::device_pointer_cast(massl);
95 thrust::fill(thrust::cuda::par.on(stream), begin_massl,
96 begin_massl + (nnodes + 1), 1.0 f );
97
98 float *maxxl =
99 ( float *)d_alloc->allocate( sizeof ( float ) * blocks * FACTOR1 , stream);
100 float *maxyl =
101 ( float *)d_alloc->allocate( sizeof ( float ) * blocks * FACTOR1 , stream);
102 float *minxl =
103 ( float *)d_alloc->allocate( sizeof ( float ) * blocks * FACTOR1 , stream);
104 float *minyl =
105 ( float *)d_alloc->allocate( sizeof ( float ) * blocks * FACTOR1 , stream);
106
107 // SummarizationKernel
108 int *countl = ( int *)d_alloc->allocate( sizeof ( int ) * (nnodes + 1), stream);
109
110 // SortKernel
111 int *sortl = ( int *)d_alloc->allocate( sizeof ( int ) * (nnodes + 1), stream);
112
113 // RepulsionKernel
114 float *rep_forces =
115 ( float *)d_alloc->allocate( sizeof ( float ) * (nnodes + 1) * 2, stream);
116 float *attr_forces = ( float *)d_alloc->allocate(
117 sizeof ( float ) * n * 2, stream); // n*2 double for reduction sum
118
119 float *norm_add1 = ( float *)d_alloc->allocate( sizeof ( float ) * n , stream);
120 float *norm = ( float *)d_alloc->allocate( sizeof ( float ) * n , stream);
121 float * Z_norm = ( float *)d_alloc->allocate( sizeof ( float ), stream);
122
123 float * radiusd_squared = ( float *)d_alloc->allocate( sizeof ( float ), stream);
124
125 // Apply
126 float *gains_bh = ( float *)d_alloc->allocate( sizeof ( float ) * n * 2, stream);
127 thrust::device_ptr<float> begin_gains_bh =
128 thrust::device_pointer_cast(gains_bh);
129 thrust::fill(thrust::cuda::par.on(stream), begin_gains_bh,
130 begin_gains_bh + ( n * 2), 1.0 f );
131
132 float *old_forces = ( float *)d_alloc->allocate( sizeof ( float ) * n * 2, stream);
133 CUDA_CHECK (cudaMemsetAsync(old_forces, 0, sizeof ( float ) * n * 2, stream));
134
135 float *YY =
136 ( float *)d_alloc->allocate( sizeof ( float ) * (nnodes + 1) * 2, stream);
137 random_vector (YY, -0.0001 f , 0.0001 f , (nnodes + 1) * 2, stream, random_state);
138 ASSERT (YY != NULL && rep_forces != NULL, "[ERROR] Possibly no more memory" );
139
140 // Set cache levels for faster algorithm execution
141 //---------------------------------------------------
142 cudaFuncSetCacheConfig(TSNE::BoundingBoxKernel, cudaFuncCachePreferShared);
143 cudaFuncSetCacheConfig(TSNE::TreeBuildingKernel, cudaFuncCachePreferL1);
144 cudaFuncSetCacheConfig(TSNE::ClearKernel1, cudaFuncCachePreferL1);
145 cudaFuncSetCacheConfig(TSNE::ClearKernel2, cudaFuncCachePreferL1);
146 cudaFuncSetCacheConfig(TSNE::SummarizationKernel, cudaFuncCachePreferShared);
147 cudaFuncSetCacheConfig(TSNE::SortKernel, cudaFuncCachePreferL1);
148 cudaFuncSetCacheConfig(TSNE::RepulsionKernel, cudaFuncCachePreferL1);
149 cudaFuncSetCacheConfig( TSNE::attractive_kernel_bh , cudaFuncCachePreferL1);
150 cudaFuncSetCacheConfig(TSNE::IntegrationKernel, cudaFuncCachePreferL1);
151
152 // Do gradient updates
153 //---------------------------------------------------
154 CUML_LOG_DEBUG ( "Start gradient updates!" );
155
156 float momentum = pre_momentum;
157 float learning_rate = pre_learning_rate;
158
159 for ( int iter = 0; iter < max_iter; iter++) {
160 CUDA_CHECK (
161 cudaMemsetAsync(rep_forces, 0, sizeof ( float ) * (nnodes + 1) * 2, stream));
162 CUDA_CHECK (cudaMemsetAsync(attr_forces, 0, sizeof ( float ) * n * 2, stream));
163 TSNE::Reset_Normalization<<<1, 1, 0, stream>>>( Z_norm , radiusd_squared ,
164 bottomd , NNODES , radiusd );
165 CUDA_CHECK (cudaPeekAtLastError());
166
167 if (iter == exaggeration_iter) {
168 momentum = post_momentum;
169 // Divide perplexities
170 const float div = 1.0f / early_exaggeration;
171 MLCommon::LinAlg::scalarMultiply (VAL, VAL, div, NNZ, stream);
172 }
173
174 START_TIMER ;
175 TSNE::BoundingBoxKernel<<<blocks * FACTOR1, THREADS1, 0, stream>>>(
176 startl, childl, massl, YY, YY + nnodes + 1, maxxl, maxyl, minxl, minyl,
177 FOUR_NNODES , NNODES , n , limiter , radiusd );
178 CUDA_CHECK (cudaPeekAtLastError());
179
180 END_TIMER ( BoundingBoxKernel_time );
181
182 START_TIMER ;
183 TSNE::ClearKernel1<<<blocks, 1024, 0, stream>>>(childl, FOUR_NNODES ,
184 FOUR_N );
185 CUDA_CHECK (cudaPeekAtLastError());
186
187 END_TIMER ( ClearKernel1_time );
188
189 START_TIMER ;
190 TSNE::TreeBuildingKernel<<<blocks * FACTOR2, THREADS2, 0, stream>>>(
191 /*errl,*/ childl, YY, YY + nnodes + 1, NNODES , n , maxdepthd , bottomd ,
192 radiusd );
193 CUDA_CHECK (cudaPeekAtLastError());
194
195 END_TIMER ( TreeBuildingKernel_time );
196
197 START_TIMER ;
198 TSNE::ClearKernel2<<<blocks * 1, 1024, 0, stream>>>(startl, massl, NNODES ,
199 bottomd );
200 CUDA_CHECK (cudaPeekAtLastError());
201
202 END_TIMER ( ClearKernel2_time );
203
204 START_TIMER ;
205 TSNE::SummarizationKernel<<<blocks * FACTOR3, THREADS3, 0, stream>>>(
206 countl, childl, massl, YY, YY + nnodes + 1, NNODES , n , bottomd );
207 CUDA_CHECK (cudaPeekAtLastError());
208
209 END_TIMER ( SummarizationKernel_time );
210
211 START_TIMER ;
212 TSNE::SortKernel<<<blocks * FACTOR4, THREADS4, 0, stream>>>(
213 sortl, countl, startl, childl, NNODES , n , bottomd );
214 CUDA_CHECK (cudaPeekAtLastError());
215
216 END_TIMER ( SortKernel_time );
217
218 START_TIMER ;
219 TSNE::RepulsionKernel<<<blocks * FACTOR5, THREADS5, 0, stream>>>(
220 /*errl,*/ theta, epssq, sortl, childl, massl, YY, YY + nnodes + 1,
221 rep_forces, rep_forces + nnodes + 1, Z_norm , theta_squared , NNODES ,
222 FOUR_NNODES , n , radiusd_squared , maxdepthd );
223 CUDA_CHECK (cudaPeekAtLastError());
224
225 END_TIMER ( RepulsionTime );
226
227 START_TIMER ;
228 TSNE::Find_Normalization<<<1, 1, 0, stream>>>( Z_norm , n );
229 CUDA_CHECK (cudaPeekAtLastError());
230
231 END_TIMER ( Reduction_time );
232
233 START_TIMER ;
234 TSNE::get_norm <<< MLCommon::ceildiv ( n , 1024), 1024, 0, stream>>>(
235 YY, YY + nnodes + 1, norm, norm_add1, n );
236 CUDA_CHECK (cudaPeekAtLastError());
237
238 // TODO: Calculate Kullback-Leibler divergence
239 // For general embedding dimensions
240 TSNE::
241 attractive_kernel_bh<<< MLCommon::ceildiv (NNZ, 1024), 1024, 0, stream>>>(
242 VAL, COL, ROW, YY, YY + nnodes + 1, norm, norm_add1, attr_forces,
243 attr_forces + n , NNZ);
244 CUDA_CHECK (cudaPeekAtLastError());
245 END_TIMER ( attractive_time );
246
247 START_TIMER ;
248 TSNE::IntegrationKernel<<<blocks * FACTOR6, THREADS6, 0, stream>>>(
249 learning_rate, momentum , early_exaggeration, YY, YY + nnodes + 1,
250 attr_forces, attr_forces + n , rep_forces, rep_forces + nnodes + 1,
251 gains_bh, gains_bh + n , old_forces, old_forces + n , Z_norm , n );
252 CUDA_CHECK (cudaPeekAtLastError());
253
254 END_TIMER ( IntegrationKernel_time );
255 }
256 PRINT_TIMES ;
257
258 // Copy final YY into true output Y
259 thrust::device_ptr<float> Y_begin = thrust::device_pointer_cast(Y);
260 thrust::copy (thrust::cuda::par.on(stream), YY, YY + n , Y_begin);
261 CUDA_CHECK (cudaPeekAtLastError());
262
263 thrust::copy (thrust::cuda::par.on(stream), YY + nnodes + 1,
264 YY + nnodes + 1 + n , Y_begin + n );
265 CUDA_CHECK (cudaPeekAtLastError());
266
267 // Deallocate everything
268 //d_alloc->deallocate(errl, sizeof(int), stream);
269 d_alloc->deallocate( limiter , sizeof ( unsigned ), stream);
270 d_alloc->deallocate( maxdepthd , sizeof ( int ), stream);
271 d_alloc->deallocate( bottomd , sizeof ( int ), stream);
272 d_alloc->deallocate( radiusd , sizeof ( float ), stream);
273
274 d_alloc->deallocate(startl, sizeof ( int ) * (nnodes + 1), stream);
275 d_alloc->deallocate(childl, sizeof ( int ) * (nnodes + 1) * 4, stream);
276 d_alloc->deallocate(massl, sizeof ( float ) * (nnodes + 1), stream);
277
278 d_alloc->deallocate(maxxl, sizeof ( float ) * blocks * FACTOR1 , stream);
279 d_alloc->deallocate(maxyl, sizeof ( float ) * blocks * FACTOR1 , stream);
280 d_alloc->deallocate(minxl, sizeof ( float ) * blocks * FACTOR1 , stream);
281 d_alloc->deallocate(minyl, sizeof ( float ) * blocks * FACTOR1 , stream);
282
283 d_alloc->deallocate(countl, sizeof ( int ) * (nnodes + 1), stream);
284 d_alloc->deallocate(sortl, sizeof ( int ) * (nnodes + 1), stream);
285
286 d_alloc->deallocate(rep_forces, sizeof ( float ) * (nnodes + 1) * 2, stream);
287 d_alloc->deallocate(attr_forces, sizeof ( float ) * n * 2, stream);
288 d_alloc->deallocate(norm, sizeof ( float ) * n , stream);
289 d_alloc->deallocate(norm_add1, sizeof ( float ) * n , stream);
290
291 d_alloc->deallocate( Z_norm , sizeof ( float ), stream);
292 d_alloc->deallocate( radiusd_squared , sizeof ( float ), stream);
293
294 d_alloc->deallocate(gains_bh, sizeof ( float ) * n * 2, stream);
295 d_alloc->deallocate(old_forces, sizeof ( float ) * n * 2, stream);
296
297 d_alloc->deallocate(YY, sizeof ( float ) * (nnodes + 1) * 2, stream);
298 }

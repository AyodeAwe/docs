48 template < typename math_t>
49 void getUniqueLabels (math_t * y , size_t n , math_t **y_unique, int *n_unique,
50 cudaStream_t stream,
51 std::shared_ptr<deviceAllocator> allocator) {
52 device_buffer<math_t> y2(allocator, stream, n );
53 device_buffer<math_t> y3(allocator, stream, n );
54 device_buffer<int> d_num_selected(allocator, stream, 1);
55 size_t bytes = 0;
56 size_t bytes2 = 0;
57
58 // Query how much temporary storage we will need for cub operations
59 // and allocate it
60 cub::DeviceRadixSort::SortKeys(NULL, bytes, y , y2. data (), n );
61 cub::DeviceSelect::Unique(NULL, bytes2, y2. data (), y3. data (),
62 d_num_selected. data (), n );
63 bytes = max (bytes, bytes2);
64 device_buffer<char> cub_storage(allocator, stream, bytes);
65
66 // Select Unique classes
67 cub::DeviceRadixSort::SortKeys(cub_storage. data (), bytes, y , y2. data (), n );
68 cub::DeviceSelect::Unique(cub_storage. data (), bytes, y2. data (), y3. data (),
69 d_num_selected. data (), n );
70 updateHost (n_unique, d_num_selected. data (), 1, stream);
71 CUDA_CHECK (cudaStreamSynchronize(stream));
72
73 // Copy unique classes to output
74 *y_unique = (math_t *)allocator->allocate(*n_unique * sizeof (math_t), stream);
75 copy (*y_unique, y3. data (), *n_unique, stream);
76 }

96 template < typename math_t>
97 void getOvrLabels (math_t * y , int n , math_t *y_unique, int n_classes,
98 math_t *y_out, int idx, cudaStream_t stream) {
99 ASSERT (idx < n_classes,
100 "Parameter idx should not be larger than the number "
101 "of classes" );
102 LinAlg::unaryOp (
103 y_out, y , n ,
104 [idx, y_unique] __device__(math_t y ) {
105 return y == y_unique[idx] ? +1 : -1;
106 },
107 stream);
108 CUDA_CHECK (cudaPeekAtLastError());
109 }

114 template < typename Type, int TPB_X, typename Lambda>
115 __global__ void map_label_kernel (Type *map_ids, size_t N_labels, Type *in,
116 Type *out, size_t N , Lambda filter_op) {
117 int tid = threadIdx.x + blockIdx.x * TPB_X ;
118 if (tid < N ) {
119 if (!filter_op(in[tid])) {
120 for ( size_t i = 0; i < N_labels; i ++) {
121 if (in[tid] == map_ids[ i ]) {
122 out[tid] = i + 1;
123 break ;
124 }
125 }
126 }
127 }
128 }

149 template < typename Type, typename Lambda>
150 void make_monotonic (Type *out, Type *in, size_t N , cudaStream_t stream,
151 Lambda filter_op) {
152 static const size_t TPB_X = 256;
153
154 dim3 blocks( ceildiv ( N , TPB_X ));
155 dim3 threads( TPB_X );
156
157 std::shared_ptr<deviceAllocator> allocator( new defaultDeviceAllocator );
158
159 Type *map_ids;
160 int num_clusters;
161 getUniqueLabels (in, N , &map_ids, &num_clusters, stream, allocator);
162
163 map_label_kernel<Type, TPB_X><<<blocks, threads, 0, stream>>>(
164 map_ids, num_clusters, in, out, N , filter_op);
165
166 allocator->deallocate(map_ids, num_clusters * sizeof (Type), stream);
167 }

34 template < typename IteratorT1, typename IteratorT2>
35 void __global__ convert_array_kernel (IteratorT1 dst, IteratorT2 src, int n ) {
36 for ( int idx = blockDim.x * blockIdx.x + threadIdx.x; idx < n ;
37 idx += gridDim.x * blockDim.x) {
38 dst[idx] = src[idx];
39 }
40 }

76 // Reduce by keys
77 // We need to sum each dimension by labels
78 // The labels are not adjacent

81 //
82 // Reduce by keys - for keys <= 4
83 //
84
85 #define SUM_ROWS_SMALL_K_DIMX 256
86 #define SUM_ROWS_BY_KEY_SMALL_K_MAX_K 4
87 template < typename DataIteratorT>
88 __launch_bounds__ ( SUM_ROWS_SMALL_K_DIMX , 4) __global__
89 void sum_rows_by_key_small_nkeys_kernel(const DataIteratorT d_A, int lda ,
90 char * d_keys , int nrows , int ncols ,
91 int nkeys , DataIteratorT d_sums ) {
92 typedef typename std::iterator_traits<DataIteratorT>::value_type DataType;
93 typedef cub::BlockReduce<quad<DataType>, SUM_ROWS_SMALL_K_DIMX > BlockReduce ;
94 __shared__ typename BlockReduce::TempStorage temp_storage ;
95
96 for ( int idim = blockIdx.y; idim < ncols ; idim += gridDim.y) {
97 if (idim != blockIdx.y) __syncthreads (); // we're reusing temp_storage
98
99 // threadIdx.x stores partial sum for current dim and key=threadIdx.x in this reg
100 quad<DataType> thread_sums;
101 thread_sums. x = 0.0;
102 thread_sums. y = 0.0;
103 thread_sums. z = 0.0;
104 thread_sums. w = 0.0;
105
106 // May use vectorized load - not necessary for doubles
107 for ( int block_offset_irow = blockIdx.x * blockDim.x;
108 block_offset_irow <
109 nrows ; // we will syncthreads() inside the loop, no CTA divergence
110 block_offset_irow += blockDim.x * gridDim.x) {
111 int irow = block_offset_irow + threadIdx.x;
112 DataType val = (irow < nrows ) ? d_A[irow * lda + idim] : 0.0;
113 // we are not reusing the keys - after profiling
114 // d_keys is mainly loaded from L2, and this kernel is DRAM BW bounded
115 // (experimentation gave a 10% speed up - not worth the many code lines added)
116 int row_key = (irow < nrows ) ? d_keys [irow] : -1;
117
118 thread_sums. x += (row_key == 0) ? val : 0.0;
119 thread_sums. y += (row_key == 1) ? val : 0.0;
120 thread_sums. z += (row_key == 2) ? val : 0.0;
121 thread_sums. w += (row_key == 3) ? val : 0.0;
122 }
123
124 // End of column
125 // Saving local sums back to global mem
126
127 // Strided access
128
129 // Reducing by key
130 thread_sums =
131 BlockReduce ( temp_storage ).Reduce(thread_sums, quadSum<DataType> ());
132
133 if (threadIdx.x < 32) {
134 // We only need 4
135 thread_sums = cub::ShuffleIndex<32>(thread_sums, 0, 0xffffffff);
136 if (threadIdx.x < nkeys ) {
137 if (threadIdx.x == 0)
138 myAtomicAdd (& d_sums [threadIdx.x * ncols + idim], thread_sums. x );
139 if (threadIdx.x == 1)
140 myAtomicAdd (& d_sums [threadIdx.x * ncols + idim], thread_sums. y );
141 if (threadIdx.x == 2)
142 myAtomicAdd (& d_sums [threadIdx.x * ncols + idim], thread_sums. z );
143 if (threadIdx.x == 3)
144 myAtomicAdd (& d_sums [threadIdx.x * ncols + idim], thread_sums. w );
145 }
146 }
147 }
148 }

150 template < typename DataIteratorT>
151 void sum_rows_by_key_small_nkeys ( const DataIteratorT d_A, int lda , char * d_keys ,
152 int nrows , int ncols , int nkeys ,
153 DataIteratorT d_sums , cudaStream_t st) {
154 dim3 grid, block;
155 block.x = SUM_ROWS_SMALL_K_DIMX ;
156 block.y = 1; // Necessary
157
158 grid.x = ceildiv ( nrows , ( int )block.x);
159 grid.x = std::min(grid.x, 32u);
160 grid.y = ncols ;
161 grid.y = std::min(grid.y, MAX_BLOCKS );
162 sum_rows_by_key_small_nkeys_kernel<<<grid, block, 0, st>>>(
163 d_A, lda , d_keys , nrows , ncols , nkeys , d_sums );

166 //
167 // Reduce by keys - large number of keys
168 // Computing a "weigthed histogram" with local histograms in smem
169 // Keeping it simple - not optimized
170 //
171
172 #define SUM_ROWS_BY_KEY_LARGE_K_MAX_K 1024
173 template < typename DataIteratorT, typename KeysIteratorT>
174 __global__ void sum_rows_by_key_large_nkeys_kernel_colmajor (
175 const DataIteratorT d_A, int lda , KeysIteratorT d_keys , int nrows , int ncols ,
176 int key_offset, int nkeys , DataIteratorT d_sums ) {
177 typedef typename std::iterator_traits<KeysIteratorT>::value_type KeyType;
178 typedef typename std::iterator_traits<DataIteratorT>::value_type DataType;
179 __shared__ DataType local_sums[ SUM_ROWS_BY_KEY_LARGE_K_MAX_K ];
180
181 for ( int local_key = threadIdx.x; local_key < nkeys ; local_key += blockDim.x)
182 local_sums[local_key] = 0.0;
183
184 for ( int idim = blockIdx.y; idim < ncols ; idim += gridDim.y) {
185 __syncthreads (); // local_sums
186
187 // At this point local_sums if full of zeros
188
189 for ( int irow = blockIdx.x * blockDim.x + threadIdx.x; irow < nrows ;
190 irow += blockDim.x * gridDim.x) {
191 // Branch div in this loop - not an issue with current code
192 DataType val = d_A[idim * lda + irow];
193 int local_key = d_keys [irow] - key_offset;
194
195 // We could load next val here
196 myAtomicAdd (&local_sums[local_key], val);
197 }
198
199 __syncthreads (); // local_sums
200
201 for ( int local_key = threadIdx.x; local_key < nkeys ;
202 local_key += blockDim.x) {
203 DataType local_sum = local_sums[local_key];
204
205 if (local_sum != 0.0) {
206 KeyType global_key = key_offset + local_key;
207 myAtomicAdd (& d_sums [global_key * ncols + idim], local_sum);
208 local_sums[local_key] = 0.0;
209 }
210 }
211 }

234 template < typename DataIteratorT, typename KeysIteratorT>
235 __global__ void sum_rows_by_key_large_nkeys_kernel_rowmajor (
236 const DataIteratorT d_A, int lda , KeysIteratorT d_keys , int nrows , int ncols ,
237 int key_offset, int nkeys , DataIteratorT d_sums ) {
238 typedef typename std::iterator_traits<KeysIteratorT>::value_type KeyType;
239 typedef typename std::iterator_traits<DataIteratorT>::value_type DataType;
240
241 #ifdef RRBK_SHMEM
242 __shared__ KeyType sh_keys[ RRBK_SHMEM_SZ ];
243 #endif
244 int rows_per_partition = nrows / gridDim.z + 1;
245 int start_row = blockIdx.z * rows_per_partition;
246 int end_row = start_row + rows_per_partition;
247 end_row = end_row > nrows ? nrows : end_row;
248
249 KeyType local_key = blockIdx.y;
250 if (local_key >= nkeys ) return ;
251 int this_col = threadIdx.x + blockIdx.x * blockDim.x;
252 if (this_col >= ncols ) return ;
253
254 DataType sum = 0.0;
255 KeyType global_key = key_offset + local_key;
256 #ifdef RRBK_SHMEM
257 int sh_key_inx = 0;
258 #endif
259 for ( int r = start_row; r < end_row; r ++) {
260 #ifdef RRBK_SHMEM
261 if (0 == sh_key_inx % RRBK_SHMEM_SZ ) {
262 for ( int x = threadIdx.x; x < RRBK_SHMEM_SZ ; x += blockDim.x)
263 sh_keys[ x ] = d_keys [ r + x ];
264 __syncthreads ();
265 }
266 if (sh_keys[sh_key_inx] != global_key)
267 continue ; //No divergence since global_key is the
268 // same for the whole block
269 sh_key_inx++;
270 #else
271 if ( d_keys [ r ] != global_key)
272 continue ; //No divergence since global_key is the
273 // same for the whole block
274 #endif
275 //if ((end_row-start_row) / (r-start_row) != global_key) continue;
276 sum += __ldcg(&d_A[ r * lda + this_col]);
277 }
278
279 if ( sum != 0.0) myAtomicAdd (& d_sums [global_key * ncols + this_col], sum );
280 }

321 template < typename DataIteratorT, typename KeysIteratorT>
322 void reduce_rows_by_key ( const DataIteratorT d_A, int lda ,
323 const KeysIteratorT d_keys , char *d_keys_char,
324 int nrows , int ncols , int nkeys , DataIteratorT d_sums ,
325 cudaStream_t stream) {
326 typedef typename std::iterator_traits<KeysIteratorT>::value_type KeyType;
327 typedef typename std::iterator_traits<DataIteratorT>::value_type DataType;
328
329 // Following kernel needs memset
330 cudaMemsetAsync( d_sums , 0, ncols * nkeys * sizeof (DataType), stream);
331
332 if ( nkeys <= SUM_ROWS_BY_KEY_SMALL_K_MAX_K ) {
333 // sum_rows_by_key_small_k is BW bounded. d_keys is loaded ncols time - avoiding wasting BW
334 // with doubles we have ~20% speed up - with floats we can hope something around 2x
335 // Converting d_keys to char
336 convert_array (d_keys_char, d_keys , nrows , stream);
337 sum_rows_by_key_small_nkeys (d_A, lda , d_keys_char, nrows , ncols , nkeys ,
338 d_sums , stream);
339 } else {
340 for (KeyType key_offset = 0; key_offset < nkeys ;
341 key_offset += SUM_ROWS_BY_KEY_LARGE_K_MAX_K ) {
342 KeyType this_call_nkeys = std::min( SUM_ROWS_BY_KEY_LARGE_K_MAX_K , nkeys );
343 sum_rows_by_key_large_nkeys_rowmajor (d_A, lda , d_keys , nrows , ncols ,
344 key_offset, this_call_nkeys, d_sums ,
345 stream);
346 }
347 }

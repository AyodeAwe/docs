54 template < typename T>
55 void svdQR (T *in, int n_rows, int n_cols, T *sing_vals, T *left_sing_vecs,
56 T *right_sing_vecs, bool trans_right, bool gen_left_vec,
57 bool gen_right_vec, cusolverDnHandle_t cusolverH,
58 cublasHandle_t cublasH, std::shared_ptr<deviceAllocator> allocator,
59 cudaStream_t stream) {
60 #if CUDART_VERSION >= 10010
61 // 46340: sqrt of max int value
62 ASSERT (n_rows <= 46340,
63 "svd solver is not supported for the data that has more than 46340 "
64 "samples (rows) "
65 "if you are using CUDA version 10.1. Please use other solvers such as "
66 "eig if it is available." );
67 #endif
68
69 const int m = n_rows;
70 const int n = n_cols;
71
72 device_buffer<int> devInfo(allocator, stream, 1);
73 T *d_rwork = nullptr ;
74
75 int lwork = 0;
76 CUSOLVER_CHECK (
77 cusolverDngesvd_bufferSize<T>(cusolverH, n_rows, n_cols, &lwork));
78 device_buffer<T> d_work(allocator, stream, lwork);
79
80 char jobu = 'S' ;
81 char jobvt = 'A' ;
82
83 if (!gen_left_vec) {
84 char new_u = 'N' ;
85 strcpy(&jobu, &new_u);
86 }
87
88 if (!gen_right_vec) {
89 char new_vt = 'N' ;
90 strcpy(&jobvt, &new_vt);
91 }
92
93 CUSOLVER_CHECK ( cusolverDngesvd (
94 cusolverH, jobu, jobvt, m , n , in, m , sing_vals, left_sing_vecs, m ,
95 right_sing_vecs, n , d_work. data (), lwork, d_rwork, devInfo. data (), stream));
96
97 // Transpose the right singular vector back
98 if (trans_right) transpose (right_sing_vecs, n_cols, stream);
99
100 CUDA_CHECK (cudaGetLastError());
101
102 int dev_info;
103 updateHost (&dev_info, devInfo. data (), 1, stream);
104 CUDA_CHECK (cudaStreamSynchronize(stream));
105 ASSERT (dev_info == 0,
106 "svd.h: svd couldn't converge to a solution. "
107 "This usually occurs when some of the features do not vary enough." );
108 }

110 template < typename T>
111 void svdEig (T *in, int n_rows, int n_cols, T *S, T *U, T *V, bool gen_left_vec,
112 cublasHandle_t cublasH, cusolverDnHandle_t cusolverH,
113 cudaStream_t stream, std::shared_ptr<deviceAllocator> allocator) {
114 int len = n_cols * n_cols;
115 device_buffer<T> in_cross_mult(allocator, stream, len);
116
117 T alpha = T(1);
118 T beta = T(0);
119 gemm (in, n_rows, n_cols, in, in_cross_mult. data (), n_cols, n_cols,
120 CUBLAS_OP_T, CUBLAS_OP_N, alpha, beta, cublasH, stream);
121
122 eigDC (in_cross_mult. data (), n_cols, n_cols, V, S, cusolverH, stream,
123 allocator);
124
125 Matrix::colReverse (V, n_cols, n_cols, stream);
126 Matrix::rowReverse (S, n_cols, 1, stream);
127
128 Matrix::seqRoot (S, S, alpha, n_cols, stream, true );
129
130 if (gen_left_vec) {
131 gemm (in, n_rows, n_cols, V, U, n_rows, n_cols, CUBLAS_OP_N, CUBLAS_OP_N,
132 alpha, beta, cublasH, stream);
133 Matrix::matrixVectorBinaryDivSkipZero (U, S, n_rows, n_cols, false , true ,
134 stream);
135 }

234 template < typename math_t>
235 bool evaluateSVDByL2Norm (math_t *A_d, math_t *U, math_t *S_vec, math_t *V,
236 int n_rows, int n_cols, int k , math_t tol,
237 cublasHandle_t cublasH, cudaStream_t stream,
238 std::shared_ptr<deviceAllocator> allocator) {
239 int m = n_rows, n = n_cols;
240
241 // form product matrix
242 device_buffer<math_t> P_d(allocator, stream, m * n );
243 device_buffer<math_t> S_mat(allocator, stream, k * k );
244 CUDA_CHECK (cudaMemsetAsync(P_d. data (), 0, sizeof (math_t) * m * n , stream));
245 CUDA_CHECK (cudaMemsetAsync(S_mat. data (), 0, sizeof (math_t) * k * k , stream));
246
247 Matrix::initializeDiagonalMatrix (S_vec, S_mat. data (), k , k , stream);
248 svdReconstruction (U, S_mat. data (), V, P_d. data (), m , n , k , cublasH, stream,
249 allocator);
250
251 // get norms of each
252 math_t normA = Matrix::getL2Norm (A_d, m * n , cublasH, stream);
253 math_t normU = Matrix::getL2Norm (U, m * k , cublasH, stream);
254 math_t normS = Matrix::getL2Norm (S_mat. data (), k * k , cublasH, stream);
255 math_t normV = Matrix::getL2Norm (V, n * k , cublasH, stream);
256 math_t normP = Matrix::getL2Norm (P_d. data (), m * n , cublasH, stream);
257
258 // calculate percent error
259 const math_t alpha = 1.0, beta = -1.0;
260 device_buffer<math_t> A_minus_P(allocator, stream, m * n );
261 CUDA_CHECK (
262 cudaMemsetAsync(A_minus_P. data (), 0, sizeof (math_t) * m * n , stream));
263
264 CUBLAS_CHECK ( cublasgeam (cublasH, CUBLAS_OP_N, CUBLAS_OP_N, m , n , &alpha, A_d,
265 m , &beta, P_d. data (), m , A_minus_P. data (), m ,
266 stream));
267
268 math_t norm_A_minus_P =
269 Matrix::getL2Norm (A_minus_P. data (), m * n , cublasH, stream);
270 math_t percent_error = 100.0 * norm_A_minus_P / normA;
271 return (percent_error / 100.0 < tol);
272 }

27 template < typename math_t>
28 class KernelFactory {
29 public :
30 static GramMatrixBase<math_t> * create ( KernelParams params,
31 cublasHandle_t cublas_handle) {
32 GramMatrixBase<math_t> * res;
33 // KernelParams is not templated, we convert the parameters to math_t here:
34 math_t coef0 = params. coef0 ;
35 math_t gamma = params. gamma ;
36 switch (params. kernel ) {
37 case LINEAR :
38 res = new GramMatrixBase<math_t> (cublas_handle);
39 break ;
40 case POLYNOMIAL :
41 res = new PolynomialKernel<math_t, int> (params. degree , gamma, coef0,
42 cublas_handle);
43 break ;
44 case TANH :
45 res = new TanhKernel<math_t> (gamma, coef0, cublas_handle);
46 break ;
47 case RBF :
48 res = new RBFKernel<math_t> (gamma);
49 break ;
50 default :
51 throw MLCommon::Exception ( "Kernel not implemented" );
52 }
53 return res;
54 }
55 };

39 math_t gain, math_t offset) {
40 for ( int tid = threadIdx.x + blockIdx.x * blockDim.x; tid < len;

56 template < typename math_t, typename exp_t>
57 __global__ void polynomial_kernel (math_t *inout, int ld, int rows, int cols,
58 exp_t exponent, math_t gain, math_t offset) {
59 for ( int tidy = threadIdx.y + blockIdx.y * blockDim.y; tidy < cols;
60 tidy += blockDim.y * gridDim.y)
61 for ( int tidx = threadIdx.x + blockIdx.x * blockDim.x; tidx < rows;
62 tidx += blockDim.x * gridDim.x) {
63 inout[tidx + tidy * ld] =
64 pow(gain * inout[tidx + tidy * ld] + offset, exponent);
65 }

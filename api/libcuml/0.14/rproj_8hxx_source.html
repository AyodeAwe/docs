43 template < typename math_t>
44 void gaussian_random_matrix ( const cumlHandle & h, rand_mat<math_t> *random_matrix,
45 paramsRPROJ & params )
46 {
47 cudaStream_t stream = h. getStream ();
48 auto d_alloc = h. getDeviceAllocator ();
49 int len = params .n_components * params .n_features;
50 random_matrix-> dense_data = (math_t*)d_alloc->allocate(len * sizeof (math_t), stream);
51 auto rng = Random::Rng ( params .random_state);
52 math_t scale = 1.0 / sqrt ( double ( params .n_components));
53 rng.normal(random_matrix-> dense_data , len, math_t(0), scale, stream);
54 }

62 template < typename math_t>
63 void sparse_random_matrix ( const cumlHandle & h, rand_mat<math_t> *random_matrix,
64 paramsRPROJ & params )
65 {
66 cudaStream_t stream = h. getStream ();
67 auto d_alloc = h. getDeviceAllocator ();
68
69 if ( params .density == 1.0f)
70 {
71 int len = params .n_components * params .n_features;
72 random_matrix-> dense_data = (math_t*)d_alloc->allocate(len * sizeof (math_t), stream);
73 auto rng = Random::Rng ( params .random_state);
74 math_t scale = 1.0 / sqrt (math_t( params .n_components));
75 rng.scaled_bernoulli(random_matrix-> dense_data , len, math_t(0.5), scale, stream);
76 }
77 else
78 {
79 auto alloc = h. getHostAllocator ();
80
81 double max_total_density = params .density * 1.2;
82 size_t indices_alloc = ( params .n_features * params .n_components * max_total_density) * sizeof ( int );
83 size_t indptr_alloc = ( params .n_components + 1) * sizeof ( int );
84 int * indices = ( int *)alloc->allocate(indices_alloc, stream);
85 int * indptr = ( int *)alloc->allocate(indptr_alloc, stream);
86
87 size_t offset = 0;
88 size_t indices_idx = 0;
89 size_t indptr_idx = 0;
90
91 for ( size_t i = 0; i < params .n_components; i ++)
92 {
93 int n_nonzero = binomial (h, params .n_features, params .density, params .random_state);
94 sample_without_replacement ( params .n_features, n_nonzero, indices, indices_idx);
95 indptr[indptr_idx] = offset;
96 indptr_idx++;
97 offset += n_nonzero;
98 }
99
100 indptr[indptr_idx] = offset;
101
102 size_t len = offset;
103 random_matrix-> indices = ( int *)d_alloc->allocate(len * sizeof ( int ), stream);
104 updateDevice (random_matrix-> indices , indices, len, stream);
105 alloc->deallocate(indices, indices_alloc, stream);
106
107 len = indptr_idx+1;
108 random_matrix-> indptr = ( int *)d_alloc->allocate(len * sizeof ( int ), stream);
109 updateDevice (random_matrix-> indptr , indptr, len, stream);
110 alloc->deallocate(indptr, indptr_alloc, stream);
111
112 len = offset;
113 random_matrix-> sparse_data = (math_t*)d_alloc->allocate(len * sizeof (math_t), stream);
114 auto rng = Random::Rng ( params .random_state);
115 math_t scale = sqrt (1.0 / params .density) / sqrt ( params .n_components);
116 rng.scaled_bernoulli(random_matrix-> sparse_data , len, math_t(0.5), scale, stream);
117
118 random_matrix-> sparse_data_size = len;
119 }
120 }

128 template < typename math_t>
129 void RPROJfit ( const cumlHandle & handle, rand_mat<math_t> *random_matrix, paramsRPROJ * params )
130 {
131 random_matrix-> reset ();
132
133 build_parameters (* params );
134 check_parameters (* params );
135
136 if ( params ->gaussian_method)
137 {
138 gaussian_random_matrix<math_t>(handle, random_matrix, * params );
139 }
140 else
141 {
142 sparse_random_matrix<math_t>(handle, random_matrix, * params );

154 template < typename math_t>
155 void RPROJtransform ( const cumlHandle & handle, math_t *input, rand_mat<math_t> *random_matrix,
156 math_t *output, paramsRPROJ * params )
157 {
158 cudaStream_t stream = handle. getStream ();
159
160 check_parameters (* params );
161
162 if (random_matrix-> dense_data )
163 {
164 cublasHandle_t cublas_handle = handle. getImpl (). getCublasHandle ();
165
166 const math_t alfa = 1;
167 const math_t beta = 0;
168
169 int & m = params ->n_samples;
170 int & n = params ->n_components;
171 int & k = params ->n_features;
172
173 int & lda = m ;
174 int & ldb = k ;
175 int & ldc = m ;
176
177 CUBLAS_CHECK ( cublasgemm (cublas_handle, CUBLAS_OP_N, CUBLAS_OP_N, m , n , k ,
178 &alfa, input, lda , random_matrix-> dense_data , ldb, &beta, output, ldc, stream));
179
180 }
181 else if (random_matrix-> sparse_data )
182 {
183 cusparseHandle_t cusparse_handle = handle. getImpl (). getcusparseHandle ();
184 CUSPARSE_CHECK (cusparseSetStream(cusparse_handle, stream));
185
186 const math_t alfa = 1;
187 const math_t beta = 0;
188
189 int & m = params ->n_samples;
190 int & n = params ->n_components;
191 int & k = params ->n_features;
192 size_t & nnz = random_matrix-> sparse_data_size ;
193
194 int & lda = m ;
195 int & ldc = m ;
196
197 CUSPARSE_CHECK ( cusparsegemmi (cusparse_handle, m , n , k , nnz, &alfa, input, lda ,
198 random_matrix-> sparse_data , random_matrix-> indptr ,
199 random_matrix-> indices , &beta, output, ldc));
200 }
201 else
202 {
203 ASSERT ( false ,
204 "Could not find a random matrix. Please perform a fit operation before applying transformation" );
205 }
206 }

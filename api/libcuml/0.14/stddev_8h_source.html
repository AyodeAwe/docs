27 template < typename Type, typename IdxType, int TPB, int ColsPerBlk = 32>
28 __global__ void stddevKernelRowMajor (Type *std, const Type *data, IdxType D,
29 IdxType N ) {
30 const int RowsPerBlkPerIter = TPB / ColsPerBlk;
31 IdxType thisColId = threadIdx.x % ColsPerBlk;
32 IdxType thisRowId = threadIdx.x / ColsPerBlk;
33 IdxType colId = thisColId + ((IdxType)blockIdx.y * ColsPerBlk);
34 IdxType rowId = thisRowId + ((IdxType)blockIdx.x * RowsPerBlkPerIter);
35 Type thread_data = Type(0);
36 const IdxType stride = RowsPerBlkPerIter * gridDim.x;
37 for (IdxType i = rowId; i < N ; i += stride) {
38 Type val = (colId < D) ? data[ i * D + colId] : Type(0);
39 thread_data += val * val;
40 }
41 __shared__ Type sstd[ColsPerBlk];
42 if (threadIdx.x < ColsPerBlk) sstd[threadIdx.x] = Type(0);
43 __syncthreads ();
44 myAtomicAdd (sstd + thisColId, thread_data);
45 __syncthreads ();
46 if (threadIdx.x < ColsPerBlk) myAtomicAdd (std + colId, sstd[thisColId]);
47 }

49 template < typename Type, typename IdxType, int TPB>
50 __global__ void stddevKernelColMajor (Type *std, const Type *data,
51 const Type *mu, IdxType D, IdxType N ) {
52 typedef cub::BlockReduce<Type, TPB> BlockReduce ;
53 __shared__ typename BlockReduce::TempStorage temp_storage ;
54 Type thread_data = Type(0);
55 IdxType colStart = N * blockIdx.x;
56 Type m = mu[blockIdx.x];
57 for (IdxType i = threadIdx.x; i < N ; i += TPB) {
58 IdxType idx = colStart + i ;
59 Type diff = data[idx] - m ;
60 thread_data += diff * diff ;
61 }
62 Type acc = BlockReduce ( temp_storage ).Sum(thread_data);
63 if (threadIdx.x == 0) {
64 std[blockIdx.x] = mySqrt (acc / N );
65 }
66 }

105 template < typename Type, typename IdxType = int >
106 void stddev (Type *std, const Type *data, const Type *mu, IdxType D, IdxType N ,
107 bool sample, bool rowMajor, cudaStream_t stream) {
108 static const int TPB = 256;
109 if (rowMajor) {
110 static const int RowsPerThread = 4;
111 static const int ColsPerBlk = 32;
112 static const int RowsPerBlk = (TPB / ColsPerBlk) * RowsPerThread;
113 dim3 grid( ceildiv ( N , (IdxType)RowsPerBlk), ceildiv (D, (IdxType)ColsPerBlk));
114 CUDA_CHECK (cudaMemset(std, 0, sizeof (Type) * D));
115 stddevKernelRowMajor<Type, IdxType, TPB, ColsPerBlk>
116 <<<grid, TPB, 0, stream>>>(std, data, D, N );
117 Type ratio = Type(1) / (sample ? Type( N - 1) : Type( N ));
118 LinAlg::binaryOp (
119 std, std, mu, D,
120 [ ratio ] __device__(Type a, Type b) { return mySqrt (a * ratio - b * b); },
121 stream);
122 } else {
123 stddevKernelColMajor<Type, IdxType, TPB>
124 <<<D, TPB, 0, stream>>>(std, data, mu, D, N );
125 }
126 CUDA_CHECK (cudaPeekAtLastError());
127 }

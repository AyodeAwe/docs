269 void coo_sort ( int m , int n , int nnz, int *rows, int *cols, T *vals,
270 std::shared_ptr<deviceAllocator> d_alloc, cudaStream_t stream) {
271 cusparseHandle_t handle = NULL;
272
273 size_t pBufferSizeInBytes = 0;
274
275 CUSPARSE_CHECK (cusparseCreate(&handle));
276 CUSPARSE_CHECK (cusparseSetStream(handle, stream));
277 CUSPARSE_CHECK (cusparseXcoosort_bufferSizeExt(handle, m , n , nnz, rows, cols,
278 &pBufferSizeInBytes));
279
280 device_buffer<int> d_P(d_alloc, stream, nnz);
281 device_buffer<char> pBuffer(d_alloc, stream, pBufferSizeInBytes);
282
283 CUSPARSE_CHECK (cusparseCreateIdentityPermutation(handle, nnz, d_P. data ()));
284
285 CUSPARSE_CHECK (cusparseXcoosortByRow(handle, m , n , nnz, rows, cols,
286 d_P. data (), pBuffer. data ()));
287
288 device_buffer<T> vals_sorted(d_alloc, stream, nnz);
289
290 CUSPARSE_CHECK (
291 cusparsegthr<T>(handle, nnz, vals, vals_sorted. data (), d_P. data (), stream));
292
293 CUDA_CHECK (cudaStreamSynchronize(stream));
294
295 copy (vals, vals_sorted. data (), nnz, stream);
296
297 CUSPARSE_CHECK (cusparseDestroy(handle));
298 }

721 template < int TPB_X, typename T, typename Lambda>
722 __global__ void coo_symmetrize_kernel ( int *row_ind, int *rows, int *cols,
723 T *vals, int *orows, int *ocols, T *ovals,
724 int n , int cnnz, Lambda reduction_op) {
725 int row = (blockIdx.x * TPB_X ) + threadIdx.x;
726
727 if (row < n ) {
728 int start_idx = row_ind[row]; // each thread processes one row
729 int stop_idx = MLCommon::Sparse::get_stop_idx (row, n , cnnz, row_ind);
730
731 int row_nnz = 0;
732 int out_start_idx = start_idx * 2;
733
734 for ( int idx = 0; idx < stop_idx - start_idx; idx++) {
735 int cur_row = rows[idx + start_idx];
736 int cur_col = cols[idx + start_idx];
737 T cur_val = vals[idx + start_idx];
738
739 int lookup_row = cur_col;
740 int t_start = row_ind[lookup_row]; // Start at
741 int t_stop = MLCommon::Sparse::get_stop_idx (lookup_row, n , cnnz, row_ind);
742
743 T transpose = 0.0;
744
745 bool found_match = false ;
746 for ( int t_idx = t_start; t_idx < t_stop; t_idx++) {
747 // If we find a match, let's get out of the loop. We won't
748 // need to modify the transposed value, since that will be
749 // done in a different thread.
750 if (cols[t_idx] == cur_row && rows[t_idx] == cur_col) {
751 // If it exists already, set transposed value to existing value
752 transpose = vals[t_idx];
753 found_match = true ;
754 break ;
755 }
756 }
757
758 // Custom reduction op on value and its transpose, which enables
759 // specialized weighting.
760 // If only simple X+X.T is desired, this op can just sum
761 // the two values.
762 T res = reduction_op(cur_row, cur_col, cur_val, transpose );
763
764 // if we didn't find an exact match, we need to add
765 // the computed res into our current matrix to guarantee
766 // symmetry.
767 // Note that if we did find a match, we don't need to
768 // compute `res` on it here because it will be computed
769 // in a different thread.
770 if (!found_match && vals[idx] != 0.0) {
771 orows[out_start_idx + row_nnz] = cur_col;
772 ocols[out_start_idx + row_nnz] = cur_row;
773 ovals[out_start_idx + row_nnz] = res;
774 ++row_nnz;
775 }
776
777 if (res != 0.0) {
778 orows[out_start_idx + row_nnz] = cur_row;
779 ocols[out_start_idx + row_nnz] = cur_col;
780 ovals[out_start_idx + row_nnz] = res;
781 ++row_nnz;
782 }
783 }
784 }
785 }

798 template < int TPB_X, typename T, typename Lambda>
799 void coo_symmetrize ( COO<T> *in, COO<T> *out,
800 Lambda reduction_op, // two-argument reducer

880 template < typename math_t>
881 __global__ static void symmetric_sum( int * restrict edges,
882 const math_t * restrict data,
883 const long * restrict indices,
884 math_t * restrict VAL, int * restrict COL,
885 int * restrict ROW, const int n ,
886 const int k ) {
887 const int j =
888 (blockIdx.x * blockDim.x) + threadIdx.x; // for every item in row
889 const int row = (blockIdx.y * blockDim.y) + threadIdx.y; // for every row
890 if (row >= n || j >= k ) return ;
891
892 const int col = indices[row * k + j ];
893 const int original = atomicAdd(&edges[row], 1);
894 const int transpose = atomicAdd(&edges[col], 1);
895
896 VAL[ transpose ] = VAL[original] = data[row * k + j ];
897 // Notice swapped ROW, COL since transpose
898 ROW[original] = row;
899 COL[original] = col;
900
901 ROW[ transpose ] = col;
902 COL[ transpose ] = row;
903 }

923 template < typename math_t, int TPB_X = 32, int TPB_Y = 32>
924 void from_knn_symmetrize_matrix ( const long * restrict knn_indices,
925 const math_t * restrict knn_dists, const int n ,
926 const int k , COO<math_t> *out,
927 cudaStream_t stream,
928 std::shared_ptr<deviceAllocator> d_alloc) {
929 // (1) Find how much space needed in each row
930 // We look through all datapoints and increment the count for each row.
931 const dim3 threadsPerBlock( TPB_X , TPB_Y);
932 const dim3 numBlocks( MLCommon::ceildiv ( k , TPB_X ),
933 MLCommon::ceildiv ( n , TPB_Y));
934
935 // Notice n+1 since we can reuse these arrays for transpose_edges, original_edges in step (4)
936 device_buffer<int> row_sizes(d_alloc, stream, n );
937 CUDA_CHECK (cudaMemsetAsync(row_sizes. data (), 0, sizeof ( int ) * n , stream));
938
939 device_buffer<int> row_sizes2(d_alloc, stream, n );
940 CUDA_CHECK (cudaMemsetAsync(row_sizes2. data (), 0, sizeof ( int ) * n , stream));
941
942 symmetric_find_size<<<numBlocks, threadsPerBlock, 0, stream>>>(
943 knn_dists, knn_indices, n , k , row_sizes. data (), row_sizes2. data ());
944 CUDA_CHECK (cudaPeekAtLastError());
945
946 reduce_find_size<<< MLCommon::ceildiv ( n , 1024), 1024, 0, stream>>>(
947 n , k , row_sizes. data (), row_sizes2. data ());
948 CUDA_CHECK (cudaPeekAtLastError());
949
950 // (2) Compute final space needed (n*k + sum(row_sizes)) == 2*n*k
951 // Notice we don't do any merging and leave the result as 2*NNZ
952 const int NNZ = 2 * n * k ;
953
954 // (3) Allocate new space
955 out-> allocate (NNZ, n , n , stream);
956
957 // (4) Prepare edges for each new row
958 // This mirrors CSR matrix's row Pointer, were maximum bounds for each row
959 // are calculated as the cumulative rolling sum of the previous rows.
960 // Notice reusing old row_sizes2 memory
961 int *edges = row_sizes2. data ();
962 thrust::device_ptr<int> __edges = thrust::device_pointer_cast(edges);
963 thrust::device_ptr<int> __row_sizes =
964 thrust::device_pointer_cast(row_sizes. data ());
965
966 // Rolling cumulative sum
967 thrust::exclusive_scan(thrust::cuda::par.on(stream), __row_sizes,
968 __row_sizes + n , __edges);
969
970 // (5) Perform final data + data.T operation in tandem with memcpying
971 symmetric_sum<<<numBlocks, threadsPerBlock, 0, stream>>>(
972 edges, knn_dists, knn_indices, out-> vals (), out-> cols (), out-> rows (), n , k );
973 CUDA_CHECK (cudaPeekAtLastError());
974 }

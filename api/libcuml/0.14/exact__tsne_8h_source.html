48 void Exact_TSNE ( float *VAL, const int *COL, const int *ROW, const int NNZ,
49 const cumlHandle &handle, float *Y, const int n , const int dim,
50 const float early_exaggeration = 12.0 f ,
51 const int exaggeration_iter = 250, const float min_gain = 0.01 f ,
52 const float pre_learning_rate = 200.0 f ,
53 const float post_learning_rate = 500.0 f ,
54 const int max_iter = 1000, const float min_grad_norm = 1e-7,
55 const float pre_momentum = 0.5, const float post_momentum = 0.8,
56 const long long random_state = -1,
57 const bool intialize_embeddings = true ) {
58 auto d_alloc = handle. getDeviceAllocator ();
59 cudaStream_t stream = handle. getStream ();
60
61 if (intialize_embeddings)
62 random_vector (Y, -0.0001 f , 0.0001 f , n * dim, stream, random_state);
63
64 // Allocate space
65 //---------------------------------------------------
66 CUML_LOG_DEBUG ( "Now allocating memory for TSNE." );
67 float *norm = ( float *)d_alloc->allocate( sizeof ( float ) * n , stream);
68 float *Z_sum = ( float *)d_alloc->allocate( sizeof ( float ) * 2 * n , stream);
69 float *means = ( float *)d_alloc->allocate( sizeof ( float ) * dim, stream);
70
71 float *attract = ( float *)d_alloc->allocate( sizeof ( float ) * n * dim, stream);
72 float *repel = ( float *)d_alloc->allocate( sizeof ( float ) * n * dim, stream);
73
74 float *velocity = ( float *)d_alloc->allocate( sizeof ( float ) * n * dim, stream);
75 CUDA_CHECK (cudaMemsetAsync(velocity, 0, sizeof ( float ) * n * dim, stream));
76
77 float *gains = ( float *)d_alloc->allocate( sizeof ( float ) * n * dim, stream);
78 thrust::device_ptr<float> begin = thrust::device_pointer_cast(gains);
79 thrust::fill(thrust::cuda::par.on(stream), begin, begin + n * dim, 1.0 f );
80
81 float *gradient = ( float *)d_alloc->allocate( sizeof ( float ) * n * dim, stream);
82 //---------------------------------------------------
83
84 // Calculate degrees of freedom
85 //---------------------------------------------------
86 const float degrees_of_freedom = fmaxf(dim - 1, 1);
87 const float df_power = -(degrees_of_freedom + 1.0f) / 2.0 f ;
88 const float recp_df = 1.0f / degrees_of_freedom;
89 const float C = 2.0f * (degrees_of_freedom + 1.0f) / degrees_of_freedom;
90
91 CUML_LOG_DEBUG ( "Start gradient updates!" );
92 float momentum = pre_momentum;
93 float learning_rate = pre_learning_rate;
94 bool check_convergence = false ;
95
96 for ( int iter = 0; iter < max_iter; iter++) {
97 check_convergence = ((iter % 10) == 0) and (iter > exaggeration_iter);
98
99 if (iter == exaggeration_iter) {
100 momentum = post_momentum;
101 // Divide perplexities
102 const float div = 1.0f / early_exaggeration;
103 MLCommon::LinAlg::scalarMultiply (VAL, VAL, div, NNZ, stream);
104 learning_rate = post_learning_rate;
105 }
106
107 // Get row norm of Y
108 MLCommon::LinAlg::rowNorm (norm, Y, dim, n , MLCommon::LinAlg::L2Norm , false ,
109 stream);
110
111 // Compute attractive forces
112 TSNE::attractive_forces (VAL, COL, ROW, Y, norm, attract, NNZ, n , dim,
113 df_power, recp_df, stream);
114 // Compute repulsive forces
115 const float Z = TSNE::repulsive_forces (Y, repel, norm, Z_sum, n , dim,
116 df_power, recp_df, stream);
117
118 // Apply / integrate forces
119 const float gradient_norm = TSNE::apply_forces (
120 Y, velocity, attract, repel, means, gains, Z , learning_rate, C, momentum ,
121 dim, n , min_gain, gradient, check_convergence, stream);
122
123 if (check_convergence) {
124 CUML_LOG_DEBUG ( "Z at iter = %d = %f and gradient norm = %f" , iter, Z ,
125 gradient_norm);
126 if (gradient_norm < min_grad_norm) {
127 CUML_LOG_DEBUG (
128 "Gradient norm = %f <= min_grad_norm = %f. Early stopped at iter = "
129 "%d" ,
130 gradient_norm, min_grad_norm, iter);
131 break ;
132 }
133 } else {
134 CUML_LOG_DEBUG ( "Z at iter = %d = %f" , iter, Z );
135 }
136 }
137
138 d_alloc->deallocate(norm, sizeof ( float ) * n , stream);
139 d_alloc->deallocate(Z_sum, sizeof ( float ) * 2 * n , stream);
140 d_alloc->deallocate(means, sizeof ( float ) * dim, stream);
141
142 d_alloc->deallocate(attract, sizeof ( float ) * n * dim, stream);
143 d_alloc->deallocate(repel, sizeof ( float ) * n * dim, stream);
144
145 d_alloc->deallocate(velocity, sizeof ( float ) * n * dim, stream);
146 d_alloc->deallocate(gains, sizeof ( float ) * n * dim, stream);
147 d_alloc->deallocate(gradient, sizeof ( float ) * n * dim, stream);
148 }

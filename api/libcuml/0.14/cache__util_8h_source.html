124 int DI arg_first_ge ( const int *array, int n , int val) {
125 int start = 0;
126 int end = n - 1;
127 if (array[0] == val) return 0;
128 if (array[ end ] < val) return n ;
129 while ( start + 1 < end ) {
130 int q = ( start + end + 1) / 2;
131 //invariants:
132 // start < end
133 // start < q <=end
134 // array[start] < val && array[end] <=val
135 // at every iteration d = end-start is decreasing
136 // when d==0, then array[end] will be the first element >= val.
137 if (array[q] >= val) {
138 end = q;
139 } else {
140 start = q;
141 }
142 }
143 return end ;
144 }
160 int DI find_nth_occurrence ( const int *array, int n , int val, int k ) {
161 int q = arg_first_ge (array, n , val);
162 if (q + k < n && array[q + k ] == val) {
163 q += k ;
164 } else {
165 q = -1;
166 }
167 return q;
168 }

198 template < int nthreads, int associativity>
199 DI void rank_set_entries ( const int *cache_time, int n_cache_sets, int *rank) {
200 const int items_per_thread = ceildiv (associativity, nthreads);
201 typedef cub::BlockRadixSort<int, nthreads, items_per_thread, int>
202 BlockRadixSort;
203 __shared__ typename BlockRadixSort::TempStorage temp_storage ;
204
205 int key[items_per_thread];
206 int val[items_per_thread];
207
208 int block_offset = blockIdx.x * associativity;
209
210 for ( int j = 0; j < items_per_thread; j ++) {
211 int k = threadIdx.x + j * nthreads;
212 int t = ( k < associativity) ? cache_time[block_offset + k ] : 32768;
213 key[ j ] = t;
214 val[ j ] = k ;
215 }
216
217 BlockRadixSort( temp_storage ).Sort(key, val);
218
219 for ( int j = 0; j < items_per_thread; j ++) {
220 if (val[ j ] < associativity) {
221 rank[val[ j ]] = threadIdx.x * items_per_thread + j ;
222 }
223 }
224 __syncthreads ();
225 }

254 template < int nthreads, int associativity>
255 __global__ void assign_cache_idx ( const int *keys, int n , const int *cache_set,
256 int *cached_keys, int n_cache_sets,
257 int *cache_time, int time, int *cache_idx) {
258 int block_offset = blockIdx.x * associativity;
259
260 const int items_per_thread = ceildiv (associativity, nthreads);
261
262 // the size of rank limits how large associativity can be used in practice
263 __shared__ int rank[items_per_thread * nthreads];
264 rank_set_entries<nthreads, associativity>(cache_time, n_cache_sets, rank);
265
266 // Each thread will fill items_per_thread items in the cache.
267 // It uses a place, only if it was not udated at the current time step
268 // (cache_time != time).
269 // We rank the places acconding to the time stamp, least recently used
270 // elements come to the front.
271 // We fill the least recently used elements with the working set.
272 // there might be elements which cannot be assigned to cache loc.
273 // these elements are assigned -1.
274
275 for ( int j = 0; j < items_per_thread; j ++) {
276 int i = threadIdx.x + j * nthreads;
277 int t_idx = block_offset + i ;
278 bool mask = ( i < associativity);
279 // whether this slot is available for writing
280 mask = mask && (cache_time[t_idx] != time);
281
282 // rank[i] tells which element to store by this thread
283 // we look up where is the corresponding key stored in the input array
284 if (mask) {
285 int k = find_nth_occurrence (cache_set, n , blockIdx.x, rank[ i ]);
286 if ( k > -1) {
287 int key_val = keys[ k ];
288 cached_keys[t_idx] = key_val;
289 cache_idx[ k ] = t_idx;
290 cache_time[t_idx] = time;
291 }
292 }
293 }
294 }

323 __global__ void get_cache_idx( int *keys, int n , int *cached_keys,
324 int n_cache_sets, int associativity,
325 int *cache_time, int *cache_idx, bool *is_cached,
326 int time) {
327 int tid = threadIdx.x + blockIdx.x * blockDim.x;
328 if (tid < n ) {
329 int widx = keys[tid];
330 int sidx = hash (widx, n_cache_sets, associativity);
331 int cidx = sidx * associativity;
332 int i = 0;
333 bool found = false ;
334 // search for empty spot and the least recently used spot
335 while ( i < associativity && !found) {
336 found = (cache_time[cidx + i ] > 0 && cached_keys[cidx + i ] == widx);
337 i ++;
338 }
339 is_cached[tid] = found;
340 if (found) {
341 cidx = cidx + i - 1;
342 cache_time[cidx] = time; //update time stamp
343 cache_idx[tid] = cidx; //exact cache idx
344 } else {
345 cache_idx[tid] = sidx; // assign cache set
346 }
347 }
348 }

66 DI bool isLast() { return threadIdx.x == blockDim.x - 1; }

68 DI void sumDone(Type sum ) {
69 volatile Flags* myFlag = flags + blockIdx.x;
70 __syncthreads ();
71 if (isLast()) myFlag->sum = sum ;
72 __threadfence();
73 // prefix sum update to be done for the first block
74 if (isLast() && blockIdx.x == 0) myFlag->incl_prefix = sum ;
75 // make sure that sum never crosses flag update!
76 __threadfence();
77 if (isLast()) myFlag->status = blockIdx.x == 0 ? 2 : 1;
78 __threadfence();
79 }

81 DI Type predecessorSum() {
82 __shared__ char s_buff[ sizeof (Type)];
83 auto * s_excl_sum = (Type*)s_buff;
84 if (isLast()) {
85 int bidx = blockIdx.x - 1;
86 Type excl_sum = 0;
87 while (bidx >= 0) {
88 volatile Flags* others = flags + bidx;
89 int status;
90 do {
91 status = others->status;
92 __threadfence();
93 } while (status == 0);
94 // one of the predecessors has computed their inclusive sum
95 if (status == 2) {
96 excl_sum += others->incl_prefix;
97 __threadfence();
98 break ;
99 }
100 // one of the predessors has only computed it's reduction sum
101 if (status == 1) excl_sum += others->sum;
102 --bidx;
103 __threadfence();
104 }
105 s_excl_sum[0] = excl_sum;
106 }
107 __syncthreads ();
108 return s_excl_sum[0];
109 }
110
111 DI void communicateDone(Type prefix, Type sum ) {
112 if (blockIdx.x > 0) {
113 volatile Flags* myFlag = flags + blockIdx.x;
114 __syncthreads ();
115 // make sure that the sum never crosses flag update!
116 if (isLast()) myFlag->incl_prefix = prefix + sum ;
117 __threadfence();
118 if (isLast()) myFlag->status = 2;
119 __threadfence();
120 }
121 }

25 template < typename InType, int VecLen, typename Lambda, typename OutType,
26 typename IdxType>
27 __global__ void unaryOpKernel (OutType *out, const InType *in, IdxType len,
28 Lambda op) {
29 typedef TxN_t<InType, VecLen> InVecType;
30 typedef TxN_t<OutType, VecLen> OutVecType;
31 InVecType a;
32 OutVecType b;
33 IdxType idx = threadIdx.x + ((IdxType)blockIdx.x * blockDim.x);
34 idx *= InVecType::Ratio;
35 if (idx >= len) return ;
36 a. load (in, idx);
37 #pragma unroll
38 for ( int i = 0; i < InVecType::Ratio; ++ i ) {
39 b.val.data[ i ] = op(a.val.data[ i ]);
40 }
41 b.store(out, idx);
42 }

45 typename IdxType, int TPB>
46 void unaryOpImpl (OutType *out, const InType *in, IdxType len, Lambda op,

49 unaryOpKernel<InType, VecLen, Lambda, OutType, IdxType>
50 <<<nblks, TPB, 0, stream>>>(out, in, len, op);

69 template < typename InType, typename Lambda, typename IdxType = int,
70 typename OutType = InType, int TPB = 256>
71 void unaryOp (OutType *out, const InType *in, IdxType len, Lambda op,
72 cudaStream_t stream) {
73 if (len <= 0) return ; //silently skip in case of 0 length input
74 constexpr auto maxSize =
75 sizeof (InType) >= sizeof (OutType) ? sizeof (InType) : sizeof (OutType);
76 size_t bytes = len * maxSize;
77 uint64_t inAddr = uint64_t(in);
78 uint64_t outAddr = uint64_t(out);
79 if (16 / maxSize && bytes % 16 == 0 && inAddr % 16 == 0 &&
80 outAddr % 16 == 0) {
81 unaryOpImpl <InType, 16 / maxSize, Lambda, OutType, IdxType, TPB>(
82 out, in, len, op, stream);
83 } else if (8 / maxSize && bytes % 8 == 0 && inAddr % 8 == 0 &&
84 outAddr % 8 == 0) {
85 unaryOpImpl <InType, 8 / maxSize, Lambda, OutType, IdxType, TPB>(
86 out, in, len, op, stream);
87 } else if (4 / maxSize && bytes % 4 == 0 && inAddr % 4 == 0 &&
88 outAddr % 4 == 0) {
89 unaryOpImpl <InType, 4 / maxSize, Lambda, OutType, IdxType, TPB>(
90 out, in, len, op, stream);
91 } else if (2 / maxSize && bytes % 2 == 0 && inAddr % 2 == 0 &&
92 outAddr % 2 == 0) {
93 unaryOpImpl <InType, 2 / maxSize, Lambda, OutType, IdxType, TPB>(
94 out, in, len, op, stream);
95 } else if (1 / maxSize) {
96 unaryOpImpl <InType, 1 / maxSize, Lambda, OutType, IdxType, TPB>(
97 out, in, len, op, stream);
98 } else {
99 unaryOpImpl<InType, 1, Lambda, OutType, IdxType, TPB>(out, in, len, op,
100 stream);
101 }

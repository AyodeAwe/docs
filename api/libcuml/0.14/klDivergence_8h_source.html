40 template < typename Type>
41 struct KLDOp {
42 HDI Type operator() (Type modelPDF, Type candidatePDF) {
43 if (modelPDF == 0.0)
44 return 0;
45
46 else
47 return modelPDF * (log(modelPDF) - log(candidatePDF));
48 }
49 };

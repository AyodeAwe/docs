32 template < typename OutputIterator>
33 CUTLASS_HOST_DEVICE void extract_index_from_iterator(
34 OutputIterator &iterator, typename OutputIterator::Pointer base_ptr,
35 int index[OutputIterator::Fragment::kElements]) {
36 int st = 0;
37 typename OutputIterator::Pointer current_ptr = iterator.params.pointer;
38 typename OutputIterator::Index current_pred_offset =
39 iterator.params.predicate_offset;
40 for ( int d = 0; d < OutputIterator::Iterations::kD; ++d) {
41 for ( int h = 0; h < OutputIterator::Iterations::kH; ++h) {
42 for ( int w = 0; w < OutputIterator::Iterations::kW; ++w) {
43 int const imm = cutlass::ComputeOffsetFromStrides<
44 typename OutputIterator::Base::ImmediateOffsetStrides>::get(0, 0, w,
45 0);
46 index[st++] = iterator.valid(d, h, w, 0)
47 ? (&iterator.params.pointer[imm] - base_ptr)
48 : -1;
49 if (w < OutputIterator::Iterations::kW - 1) {
50 iterator.inc_w();
51 }
52 }
53 if (h < OutputIterator::Iterations::kH - 1) {
54 iterator.inc_h();
55 }
56 }
57 if (d < OutputIterator::Iterations::kD - 1) {
58 iterator.inc_d();
59 }
60 }
61 iterator.inc_advance();
62 iterator.params.pointer = current_ptr;
63 iterator.params.predicate_offset = current_pred_offset;
64 }

180 template < typename FinalLambda>
181 CUTLASS_DEVICE void epilogue (cutlass::Coord<3> const &block,
182 Accumulators &accumulators, FinalLambda fin_op) {
183 // The problem size.
184 cutlass::Coord<3> const bounds = cutlass::make_Coord(0, this-> n , this-> m );
185
186 // The functor.
187 Functor functor(this-> params .functor);
188
189 // The AA fragment
190 typename GlobalLoadIteratorAA::Fragment fragment_aa;
191 // The BB fragment
192 typename GlobalLoadIteratorBB::Fragment fragment_bb;
193
194 // The BB column vector size.
195 cutlass::Coord<3> const bb_bounds = cutlass::make_Coord(0, 1, this-> m );
196 // The BB block size.
197 cutlass::Coord<3> const bb_block = cutlass::make_Coord(0, 0, block[2]);
198
199 // The iterator to load the elements of the BB row vector.
200 GlobalLoadIteratorBB global_load_iterator_bb(
201 this-> params .functor.iterator_bb, bb_bounds, bb_block, 0, 0);
202 iterator_load(global_load_iterator_bb, fragment_bb);
203
204 // Preserve the base pointer of the output D matrix
205 typename GlobalStoreIteratorD::Pointer global_base_ptr =
206 this-> params .iterator_d.pointer;
207
208 CUTLASS_PRAGMA_UNROLL
209 for ( int h = 0; h < Iterations::kH; ++h) {
210 // Compute pointer and predicate offsets for C and D global iterators.
211 int const pointer_offset =
212 ((this-> params .iterator_d.inc_h *
213 (GlobalStoreIteratorD::Iterations::kH - 1) +
214 this-> params .iterator_d.inc_advance) *
215 Iterations::kW +
216 this-> params .stride_h) *
217 h;
218 int const predicate_offset =
219 ((this-> params .iterator_d.predicate_inc_h *
220 (GlobalStoreIteratorD::Iterations::kH - 1) +
221 this-> params .iterator_d.predicate_inc_advance) *
222 Iterations::kW +
223 Traits::Delta::kH) *
224 h;
225
226 // The transformer for D.
227 GlobalTransformerD transformer_d;
228 // The iterator to store into the D matrix.
229 GlobalStoreIteratorD global_store_iterator(this-> params .iterator_d,
230 bounds, block, pointer_offset,
231 predicate_offset);
232
233 // The transformer to transform before storing to shared memory.
234 SharedStoreTransformerD shared_store_transformer;
235 typename SharedStoreTransformerD::OutputFragment
236 shared_store_transformed_d;
237
238 // The iterator to store to shared memory.
239 SharedStoreIteratorD shared_store_iterator(
240 this-> params .shared_store_iterator_d,
241 this->shared_storage.shared_stream.store);
242
243 // The iterator to load from shared memory. TODO: Use a stream.
244 SharedLoadIteratorD shared_load_iterator(
245 this-> params .shared_load_iterator_d,
246 this->shared_storage.shared_stream.load);
247
248 // The AA column vector size.
249 cutlass::Coord<3> const aa_bounds = cutlass::make_Coord(0, this-> n , 1);
250 // The AA block size.
251 cutlass::Coord<3> const aa_block = cutlass::make_Coord(0, block[1], 0);
252
253 // Compute pointer and predicate offsets for AA global iterators.
254 int const aa_pointer_offset =
255 ((this-> params .functor.iterator_aa.inc_h *
256 (GlobalLoadIteratorAA::Iterations::kH - 1) +
257 this-> params .functor.iterator_aa.inc_advance) *
258 Iterations::kW +
259 Traits::Delta::kH) *
260 h;
261 int const aa_predicate_offset =
262 ((this-> params .functor.iterator_aa.predicate_inc_h *
263 (GlobalLoadIteratorAA::Iterations::kH - 1) +
264 this-> params .functor.iterator_aa.predicate_inc_advance) *
265 Iterations::kW +
266 Traits::Delta::kH) *
267 h;
268 // The iterator to load the elements of the AA column vector.
269 GlobalLoadIteratorAA global_load_iterator_aa(
270 this-> params .functor.iterator_aa, aa_bounds, aa_block,
271 aa_pointer_offset, aa_predicate_offset);
272
273 CUTLASS_PRAGMA_UNROLL
274 for ( int w = 0; w < Iterations::kW; ++w) {
275 iterator_load(global_load_iterator_aa, fragment_aa);
276
277 // Make sure we can write to shared memory.
278 shared_load_fence ();
279
280 // Copy the accumulators to shared memory.
281 int const offset =
282 (h * Iterations::kW + w) * SharedStoreIteratorD::Fragment::kElements;
283
284 shared_store_transformer.transform(accumulators, offset,
285 shared_store_transformed_d);
286 shared_iterator_store(shared_store_iterator,
287 shared_store_transformed_d);
288
289 // Make sure the data is in shared memory.
290 shared_store_fence ();
291
292 // Copy the accumulators back to registers from shared memory.
293 typename SharedLoadIteratorD::Fragment fetched_d;
294 shared_iterator_load(shared_load_iterator, fetched_d);
295
296 // Do the math.
297 typename GlobalTransformerD::InputFragment fragment_d;
298
299 // extract the global pointer index for each fragment element
300 int index[GlobalStoreIteratorD::Fragment::kElements];
301 extract_index_from_iterator(global_store_iterator, global_base_ptr,
302 index);
303
304 functor.evaluate(fetched_d, fragment_d, index, fragment_aa, fragment_bb,
305 fin_op);
306
307 // Transform D fragment.
308 typename GlobalTransformerD::OutputFragment transformed_d;
309 transformer_d.transform(fragment_d, transformed_d);
310
311 // Copy the results to global memory.
312 iterator_store(global_store_iterator, transformed_d);
313 }
314 }
315 }
316 }; // end struct ExpandedDistanceGemmEpilogue

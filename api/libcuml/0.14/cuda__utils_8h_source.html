76 template < int ItemsPerThread, typename L>
77 DI void forEach ( int num, L lambda) {
78 int idx = (blockDim.x * blockIdx.x) + threadIdx.x;
79 const int numThreads = blockDim.x * gridDim.x;
80 #pragma unroll
81 for ( int itr = 0; itr < ItemsPerThread; ++itr, idx += numThreads) {
82 if (idx < num) lambda(idx, itr);
83 }

115 // Ref:
116 // http://on-demand.gputechconf.com/gtc/2013/presentations/S3101-Atomic-Memory-Operations.pdf
117 template <>
118 DI void myAtomicAdd ( double *address, double val) {
119 unsigned long long int *address_as_ull = ( unsigned long long int *)address;
120 unsigned long long int old = *address_as_ull, assumed;
121 do {
122 assumed = old;
123 old = atomicCAS(address_as_ull, assumed,
124 __double_as_longlong(val + __longlong_as_double(assumed)));
125 } while (assumed != old);
126 }
127 #else
128 #define myAtomicAdd(a, b) atomicAdd(a, b)
129 #endif // __CUDA_ARCH__

576 template < typename T>
577 DI T warpReduce (T val) {
578 #pragma unroll
579 for ( int i = WarpSize / 2; i > 0; i >>= 1) {
580 T tmp = shfl (val, laneId () + i );
581 val += tmp;
582 }
583 return val;

597 template < typename T>
598 DI T blockReduce (T val, char *smem) {
599 auto *sTemp = reinterpret_cast< T * > (smem);
600 int nWarps = (blockDim.x + WarpSize - 1) / WarpSize;
601 int lid = laneId ();
602 int wid = threadIdx.x / WarpSize;
603 val = warpReduce (val);
604 if (lid == 0) sTemp[wid] = val;
605 __syncthreads ();
606 val = lid < nWarps ? sTemp[lid] : T(0);
607 return warpReduce (val);
608 }

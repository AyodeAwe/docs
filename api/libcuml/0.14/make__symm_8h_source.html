28 // Ref: https://devblogs.nvidia.com/efficient-matrix-transpose-cuda-cc/
30 template < typename DataT, typename IdxT, typename EpilogueOp>
31 __global__ void symmKernel (DataT* out, const DataT* in, IdxT batchSize, IdxT n ,
32 EpilogueOp op) {
33 __shared__ DataT smem[TileDim][TileDim + 1]; // +1 to avoid bank conflicts
34 IdxT batchOffset = blockIdx.z * n * n ;
35 IdxT myRowStart = blockIdx.y * TileDim + threadIdx.y;
36 IdxT myColStart = blockIdx.x * TileDim + threadIdx.x;
37 IdxT myIdx = batchOffset + myRowStart * n + myColStart;
38 // load the transpose part
39 IdxT otherRowStart = blockIdx.x * TileDim + threadIdx.y;
40 IdxT otherColStart = blockIdx.y * TileDim + threadIdx.x;
41 IdxT otherIdx = batchOffset + otherRowStart * n + otherColStart;
42 if (otherColStart < n ) {
43 #pragma unroll
44 for ( int i = 0; i < TileDim; i += BlockRows) {
45 if (otherRowStart + i < n ) {
46 smem[threadIdx.y + i ][threadIdx.x] = in[otherIdx + i * n ];
47 }
48 }
49 }
50 __syncthreads ();
51 if (myColStart < n ) {
52 #pragma unroll
53 for ( int i = 0; i < TileDim; i += BlockRows) {
54 auto offset = myIdx + i * n ;
55 if (myRowStart + i < n ) {
56 auto sum = smem[threadIdx.x][threadIdx.y + i ] + in[offset];
57 out[offset] = op( sum * DataT(0.5), offset);
58 }
59 }
60 }
61 }

76 template < typename DataT, typename IdxT, typename EpilogueOp = Nop<DataT, IdxT>>
77 void make_symm (DataT* out, const DataT* in, IdxT batchSize, IdxT n ,
78 cudaStream_t stream, EpilogueOp op = Nop<DataT, IdxT> ()) {
79 dim3 blk(TileDim, BlockRows);
80 auto nblks = ceildiv<int>( n , TileDim);
81 dim3 grid(nblks, nblks, batchSize);
82 symmKernel<DataT, IdxT, EpilogueOp>
83 <<<grid, blk, 0, stream>>>(out, in, batchSize, n , op);
84 CUDA_CHECK (cudaGetLastError());
85 }

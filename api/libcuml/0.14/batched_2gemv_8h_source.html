40 template < typename DataT, typename IdxT, int VecLen>
41 DI DataT dotProduct ( const DataT (& x )[VecLen], const DataT (& y )[VecLen],
42 char * smem, bool broadcast = false ) {
43 auto val = DataT(0.0);
44 #pragma unroll
45 for ( int i = 0; i < VecLen; ++ i ) {
46 val += x [ i ] * y [ i ];
47 }
48 auto dot = blockReduce (val, smem);
49 if (broadcast) {
50 auto * sDot = reinterpret_cast< DataT* > (smem);
51 __syncthreads ();
52 if (threadIdx.x == 0) {
53 sDot[0] = dot;
54 }
55 __syncthreads ();
56 dot = sDot[0];
57 }
58 return dot;
59 }

61 template < typename DataT, typename IdxT, int VecLenAx, int VecLenY,
62 typename EpilogueOp>
63 __global__ void gemvKernel (DataT* y , const DataT* A, const DataT* x ,
64 const DataT* z, DataT alpha, DataT beta, IdxT m ,
65 IdxT n , EpilogueOp op) {
66 typedef TxN_t<DataT, VecLenAx> VecTypeAx;
67 typedef TxN_t<DataT, VecLenY> VecTypeY;
68 static constexpr DataT Zero = DataT(0.0);
69 extern __shared__ char smem[];
70 auto * sdot = smem;
71 VecTypeAx _x, _a;
72 VecTypeY _y, _z;
73 IdxT idx = threadIdx.x * VecTypeAx::Ratio;
74 IdxT batch = blockIdx.y;
75 IdxT rowId = blockIdx.x * VecTypeY::Ratio;
76 auto rowOffset = batch * m * n + rowId * n ;
77 _x.fill(Zero);
78 _z.fill(Zero);
79 if (idx < n ) {
80 _x.load( x , batch * n + idx);
81 }
82 #pragma unroll
83 for (IdxT j = 0; j < VecTypeY::Ratio; ++ j ) {
84 _a.fill(Zero);
85 if (idx < n ) {
86 _a.load(A, rowOffset + j * n + idx);
87 }
88 _y.val.data[ j ] = dotProduct<DataT, IdxT, VecTypeAx::Ratio>(
89 _a.val.data, _x.val.data, sdot, false );
90 __syncthreads ();
91 }
92 if (threadIdx.x == 0) {
93 auto yidx = batch * m + rowId;
94 if (beta != Zero) {
95 _z.load( y , yidx);
96 }
97 #pragma unroll
98 for (IdxT j = 0; j < VecTypeY::Ratio; ++ j ) {
99 _y.val.data[ j ] =
100 op(alpha * _y.val.data[ j ] + beta * _z.val.data[ j ], yidx + j );
101 }
102 _y.store( y , yidx);
103 }
104 }

106 template < typename DataT, typename IdxT, int VecLenAx, int VecLenY,
107 typename EpilogueOp>
108 void gemvImplY (DataT* y , const DataT* A, const DataT* x , const DataT* z,
109 DataT alpha, DataT beta, IdxT m , IdxT n , IdxT batchSize,
110 EpilogueOp op, cudaStream_t stream) {
111 auto nAligned = VecLenAx ? n / VecLenAx : n ;
112 int tpb = alignTo<int>(nAligned, WarpSize);
113 int nWarps = tpb / WarpSize;
114 size_t smemSize = sizeof (DataT) * nWarps;
115 auto mAligned = VecLenY ? ceildiv ( m , VecLenY) : m ;
116 dim3 nblks(mAligned, batchSize);
117 gemvKernel<DataT, IdxT, VecLenAx, VecLenY, EpilogueOp>
118 <<<nblks, tpb, smemSize, stream>>>( y , A, x , z, alpha, beta, m , n , op);
119 CUDA_CHECK (cudaGetLastError());
120 }

28 template < typename Type, typename IntType, typename IdxType, int TPB,
29 bool rowMajor>
30 __global__ void permuteKernel (IntType* perms, Type* out, const Type* in,
31 IdxType a, IdxType b, IdxType N , IdxType D) {
32 namespace cg = cooperative_groups;
33 const int WARP_SIZE = 32;
34
35 int tid = threadIdx.x + blockIdx.x * blockDim.x;
36
37 // having shuffled input indices and coalesced output indices appears
38 // to be preferrable to the reverse, especially for column major
39 IntType inIdx = ((a * int64_t(tid)) + b) % N ;
40 IntType outIdx = tid;
41
42 if (perms != nullptr && tid < N ) {
43 perms[outIdx] = inIdx;
44 }
45
46 if (out == nullptr || in == nullptr ) {
47 return ;
48 }
49
50 if (rowMajor) {
51 cg::thread_block_tile<WARP_SIZE> warp =
52 cg::tiled_partition<WARP_SIZE>(cg::this_thread_block());
53
54 __shared__ IntType inIdxShm[TPB];
55 __shared__ IntType outIdxShm[TPB];
56 inIdxShm[threadIdx.x] = inIdx;
57 outIdxShm[threadIdx.x] = outIdx;
58 warp.sync();
59
60 int warpID = threadIdx.x / WARP_SIZE;
61 int laneID = threadIdx.x % WARP_SIZE;
62 for ( int i = warpID * WARP_SIZE; i < warpID * WARP_SIZE + WARP_SIZE; ++ i ) {
63 if (outIdxShm[ i ] < N ) {
64 #pragma unroll
65 for ( int j = laneID; j < D; j += WARP_SIZE) {
66 out[outIdxShm[ i ] * D + j ] = in[inIdxShm[ i ] * D + j ];
67 }
68 }
69 }
70 } else {
71 #pragma unroll
72 for ( int j = 0; j < D; ++ j ) {
73 if (tid < N ) {
74 out[outIdx + j * N ] = in[inIdx + j * N ];
75 }
76 }
77 }
78 }

81 template < typename Type, typename IntType, typename IdxType, int TPB,
82 bool rowMajor, int VLen>
83 struct permute_impl_t {
84 static void permuteImpl (IntType* perms, Type* out, const Type* in, IdxType N ,
85 IdxType D, int nblks, IdxType a, IdxType b,
86 cudaStream_t stream) {
87 //determine vector type and set new pointers
88 typedef typename MLCommon::IOType<Type, VLen>::Type VType;
89 VType* vout = reinterpret_cast< VType* > (out);
90 const VType* vin = reinterpret_cast< const VType* > (in);
91
92 // check if we can execute at this vector length
93 if (D % VLen == 0 && is_aligned (vout, sizeof (VType)) &&
94 is_aligned (vin, sizeof (VType))) {
95 permuteKernel<VType, IntType, IdxType, TPB, rowMajor>
96 <<<nblks, TPB, 0, stream>>>(perms, vout, vin, a, b, N , D / VLen);
97 CUDA_CHECK (cudaPeekAtLastError());
98 } else { // otherwise try the next lower vector length
99 permute_impl_t <Type, IntType, IdxType, TPB, rowMajor,
100 VLen / 2> ::permuteImpl (perms, out, in, N , D, nblks, a, b,
101 stream);
102 }
103 }
104 };

146 auto nblks = ceildiv ( N , (IntType)TPB);

148 // always keep 'a' to be coprime to N

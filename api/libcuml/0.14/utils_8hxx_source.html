28 inline void sample_without_replacement ( size_t n_population, size_t n_samples,
29 int * indices, size_t & indices_idx)
30 {
31 std::random_device dev;
32 std::mt19937 gen(dev());
33
34 std::uniform_int_distribution<int> uni_dist(0, n_population-1);
35
36 std::unordered_set<int> s;
37
38 for ( size_t i = 0; i < n_samples; i ++)
39 {
40 int rand_idx = uni_dist(gen);
41
42 while (s.find(rand_idx) != s.end())
43 {
44 rand_idx = uni_dist(gen);
45 }
46 s.insert(rand_idx);
47 indices[indices_idx] = rand_idx;
48 indices_idx++;
49 }
50 }

62 inline size_t binomial ( const ML::cumlHandle & h, size_t n , double p, int random_state)
63 {
64
65 auto alloc = h. getDeviceAllocator ();
66
67 struct timeval tp;
68 gettimeofday(&tp, NULL);
69 long long seed = tp.tv_sec * 1000 + tp.tv_usec;
70
71 auto rng = MLCommon::Random::Rng (random_state+seed);
72
73 bool *rand_array = ( bool *)alloc->allocate( n * sizeof ( bool ), h. getStream ());
74 int *successes = ( int *)alloc->allocate( sizeof ( int ), h. getStream ());
75
76 rng.bernoulli(rand_array, n , p, h. getStream ());
77
78 cudaMemsetAsync(successes, 0, sizeof ( int ), h. getStream ());
79
80 dim3 grid_n( MLCommon::ceildiv ( n , ( size_t ) TPB_X ), 1, 1);
81 dim3 blk( TPB_X , 1, 1);
82
83 sum_bools <<<grid_n, blk, 0, h. getStream ()>>>(rand_array, n , successes);
84 CUDA_CHECK (cudaPeekAtLastError());
85
86 int ret = 0;
87 MLCommon::updateHost (&ret, successes,1, h. getStream ());
88 cudaStreamSynchronize(h. getStream ());
89 CUDA_CHECK (cudaPeekAtLastError());
90
91 alloc->deallocate(rand_array, n * sizeof ( bool ), h. getStream ());
92 alloc->deallocate(successes, sizeof ( int ), h. getStream ());
93
94 return n -ret;
95 }

113 size_t johnson_lindenstrauss_min_dim ( size_t n_samples, double eps)
114 {
115 ASSERT (eps > 0.0 && eps < 1.0,
116 "Parameter eps: must be in range (0, 1)" );
117 ASSERT (n_samples > 0,
118 "Parameter n_samples: must be strictly positive" );
119
120 double denominator = (pow(eps, 2.0) / 2.0) - (pow(eps, 3) / 3.0);
121 size_t res = 4.0 * log(n_samples) / denominator;
122 return res;
123 }

307 // Computes the argmax(d_in) column-wise in a DxN matrix
308 template < typename T, int TPB>
309 __global__ void argmaxKernel ( const T *d_in, int D, int N , T * argmax ) {
310 typedef cub::BlockReduce<cub::KeyValuePair<int, T>, TPB> BlockReduce ;
311 __shared__ typename BlockReduce::TempStorage temp_storage ;
312
313 // compute maxIndex=argMax index for column
314 using KVP = cub::KeyValuePair<int, T>;
315 int rowStart = blockIdx.x * D;
316 KVP thread_data(-1, -myInf<T>());
317
318 for ( int i = threadIdx.x; i < D; i += TPB) {
319 int idx = rowStart + i ;
320 thread_data = cub::ArgMax()(thread_data, KVP( i , d_in[idx]));
321 }
322
323 auto maxKV = BlockReduce ( temp_storage ).Reduce(thread_data, cub::ArgMax());
324
325 if (threadIdx.x == 0) {
326 argmax [blockIdx.x] = maxKV.key;
327 }
328 }

338 template < typename math_t>
339 void argmax ( const math_t *in, int n_rows, int n_cols, math_t *out,
340 cudaStream_t stream) {
341 int D = n_rows;
342 int N = n_cols;
343 if (D <= 32) {
344 argmaxKernel<math_t, 32><<< N , 32, 0, stream>>>(in, D, N , out);
345 } else if (D <= 64) {
346 argmaxKernel<math_t, 64><<< N , 64, 0, stream>>>(in, D, N , out);
347 } else if (D <= 128) {
348 argmaxKernel<math_t, 128><<< N , 128, 0, stream>>>(in, D, N , out);
349 } else {
350 argmaxKernel<math_t, 256><<< N , 256, 0, stream>>>(in, D, N , out);
351 }
352 CUDA_CHECK (cudaPeekAtLastError());
353 }

355 // Utility kernel needed for signFlip.
356 // Computes the argmax(abs(d_in)) column-wise in a DxN matrix followed by
357 // flipping the sign if the |max| value for each column is negative.
358 template < typename T, int TPB>
359 __global__ void signFlipKernel (T *d_in, int D, int N ) {
360 typedef cub::BlockReduce<cub::KeyValuePair<int, T>, TPB> BlockReduce ;
361 __shared__ typename BlockReduce::TempStorage temp_storage ;
362
363 // compute maxIndex=argMax (with abs()) index for column
364 using KVP = cub::KeyValuePair<int, T>;
365 int rowStart = blockIdx.x * D;
366 KVP thread_data(0, 0);
367 for ( int i = threadIdx.x; i < D; i += TPB) {
368 int idx = rowStart + i ;
369 thread_data = cub::ArgMax()(thread_data, KVP(idx, abs(d_in[idx])));
370 }
371 auto maxKV = BlockReduce ( temp_storage ).Reduce(thread_data, cub::ArgMax());
372
373 // flip column sign if d_in[maxIndex] < 0
374 __shared__ bool need_sign_flip;
375 if (threadIdx.x == 0) {
376 need_sign_flip = d_in[maxKV.key] < T(0);
377 }
378 __syncthreads ();
379
380 if (need_sign_flip) {
381 for ( int i = threadIdx.x; i < D; i += TPB) {
382 int idx = rowStart + i ;
383 d_in[idx] = -d_in[idx];
384 }
385 }
386 }

396 template < typename math_t>
397 void signFlip (math_t *inout, int n_rows, int n_cols, cudaStream_t stream) {
398 int D = n_rows;
399 int N = n_cols;
400 auto data = inout;
401 if (D <= 32) {
402 signFlipKernel<math_t, 32><<< N , 32, 0, stream>>>(data, D, N );
403 } else if (D <= 64) {
404 signFlipKernel<math_t, 64><<< N , 64, 0, stream>>>(data, D, N );
405 } else if (D <= 128) {
406 signFlipKernel<math_t, 128><<< N , 128, 0, stream>>>(data, D, N );
407 } else {
408 signFlipKernel<math_t, 256><<< N , 256, 0, stream>>>(data, D, N );
409 }
410 CUDA_CHECK (cudaPeekAtLastError());
411 }

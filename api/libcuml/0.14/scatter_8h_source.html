24 template < typename DataT, int VecLen, typename Lambda, typename IdxT>
25 __global__ void scatterKernel (DataT *out, const DataT *in, const IdxT *idx,
26 IdxT len, Lambda op) {
27 typedef TxN_t<DataT, VecLen> DataVec;
28 typedef TxN_t<IdxT, VecLen> IdxVec;
29 IdxT tid = threadIdx.x + ((IdxT)blockIdx.x * blockDim.x);
30 tid *= VecLen;
31 if (tid >= len) return ;
32 IdxVec idxIn;
33 idxIn. load (idx, tid);
34 DataVec dataIn;
35 #pragma unroll
36 for ( int i = 0; i < VecLen; ++ i ) {
37 auto inPos = idxIn.val.data[ i ];
38 dataIn.val.data[ i ] = op(in[inPos], tid + i );
39 }
40 dataIn.store(out, tid);
41 }

43 template < typename DataT, int VecLen, typename Lambda, typename IdxT, int TPB>
44 void scatterImpl (DataT *out, const DataT *in, const IdxT *idx, IdxT len,
45 Lambda op, cudaStream_t stream) {
46 const IdxT nblks = ceildiv (VecLen ? len / VecLen : len, (IdxT)TPB);
47 scatterKernel<DataT, VecLen, Lambda, IdxT>
48 <<<nblks, TPB, 0, stream>>>(out, in, idx, len, op);
49 CUDA_CHECK (cudaGetLastError());
50 }

68 template < typename DataT, typename IdxT, typename Lambda = Nop<DataT, IdxT>,
69 int TPB = 256>
70 void scatter (DataT *out, const DataT *in, const IdxT *idx, IdxT len,
71 cudaStream_t stream, Lambda op = Nop<DataT, IdxT> ()) {
72 if (len <= 0) return ;
73 constexpr size_t DataSize = sizeof (DataT);
74 constexpr size_t IdxSize = sizeof (IdxT);
75 constexpr size_t MaxPerElem = DataSize > IdxSize ? DataSize : IdxSize;
76 size_t bytes = len * MaxPerElem;
77 if (16 / MaxPerElem && bytes % 16 == 0) {
78 scatterImpl <DataT, 16 / MaxPerElem, Lambda, IdxT, TPB>(out, in, idx, len,
79 op, stream);
80 } else if (8 / MaxPerElem && bytes % 8 == 0) {
81 scatterImpl <DataT, 8 / MaxPerElem, Lambda, IdxT, TPB>(out, in, idx, len, op,
82 stream);
83 } else if (4 / MaxPerElem && bytes % 4 == 0) {
84 scatterImpl <DataT, 4 / MaxPerElem, Lambda, IdxT, TPB>(out, in, idx, len, op,
85 stream);
86 } else if (2 / MaxPerElem && bytes % 2 == 0) {
87 scatterImpl <DataT, 2 / MaxPerElem, Lambda, IdxT, TPB>(out, in, idx, len, op,
88 stream);
89 } else if (1 / MaxPerElem) {
90 scatterImpl <DataT, 1 / MaxPerElem, Lambda, IdxT, TPB>(out, in, idx, len, op,
91 stream);
92 } else {
93 scatterImpl<DataT, 1, Lambda, IdxT, TPB>(out, in, idx, len, op, stream);
94 }
95 }

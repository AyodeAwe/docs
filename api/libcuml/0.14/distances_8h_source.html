63 void normalize_distances ( const int n , float *distances, const int n_neighbors,
64 cudaStream_t stream) {
65 // Now D / max(abs(D)) to allow exp(D) to not explode
66 thrust::device_ptr<float> begin = thrust::device_pointer_cast(distances);
67 float maxNorm = *thrust::max_element(thrust::cuda::par.on(stream), begin,
68 begin + n * n_neighbors);
69 if (maxNorm == 0.0 f ) maxNorm = 1.0f;
70
71 // Divide distances inplace by max
72 const float div = 1.0f / maxNorm; // Mult faster than div
73 MLCommon::LinAlg::scalarMultiply (distances, distances, div, n * n_neighbors,
74 stream);
75 }

51 template < int TPB_X, typename T>
52 __global__ void fast_intersection_kernel ( int *rows, int *cols, T *vals, int nnz,
53 T *target, float unknown_dist = 1.0,
54 float far_dist = 5.0) {
55 int row = (blockIdx.x * TPB_X ) + threadIdx.x;
56 if (row < nnz) {
57 int i = rows[row];
58 int j = cols[row];
59 if (target[ i ] == T(-1.0) || target[ j ] == T(-1.0))
60 vals[row] *= exp(-unknown_dist);
61 else if (target[ i ] != target[ j ])
62 vals[row] *= exp(-far_dist);
63 }
64 }

66 template < typename T, int TPB_X>
67 void reset_local_connectivity ( COO<T> *in_coo, COO<T> *out_coo,
68 std::shared_ptr<deviceAllocator> d_alloc,
69 cudaStream_t stream // size = nnz*2
70 ) {
71 MLCommon::device_buffer<int> row_ind(d_alloc, stream, in_coo-> n_rows );
72
73 MLCommon::Sparse::sorted_coo_to_csr (in_coo, row_ind. data (), d_alloc, stream);
74
75 // Perform l_inf normalization
76 MLCommon::Sparse::csr_row_normalize_max<TPB_X, T>(
77 row_ind. data (), in_coo-> vals (), in_coo-> nnz , in_coo-> n_rows , in_coo-> vals (),
78 stream);
79 CUDA_CHECK (cudaPeekAtLastError());
80
81 MLCommon::Sparse::coo_symmetrize<TPB_X, T>(
82 in_coo, out_coo,
83 [] __device__( int row, int col, T result, T transpose ) {
84 T prod_matrix = result * transpose ;
85 return result + transpose - prod_matrix;
86 },
87 d_alloc, stream);
88
89 CUDA_CHECK (cudaPeekAtLastError());
90 }

99 template < typename T, int TPB_X>
100 void categorical_simplicial_set_intersection ( COO<T> *graph_coo, T *target,
101 cudaStream_t stream,
102 float far_dist = 5.0,
103 float unknown_dist = 1.0) {
104 dim3 grid( MLCommon::ceildiv (graph_coo-> nnz , TPB_X ), 1, 1);
105 dim3 blk( TPB_X , 1, 1);
106 fast_intersection_kernel<TPB_X, T><<<grid, blk, 0, stream>>>(
107 graph_coo-> rows (), graph_coo-> cols (), graph_coo-> vals (), graph_coo-> nnz ,
108 target, unknown_dist, far_dist);
109 }

111 template < typename T, int TPB_X>
112 __global__ void sset_intersection_kernel ( int *row_ind1, int *cols1, T *vals1,
113 int nnz1, int *row_ind2, int *cols2,
114 T *vals2, int nnz2, int *result_ind,
115 int *result_cols, T *result_vals,
116 int nnz, T left_min, T right_min,
117 int m , float mix_weight = 0.5) {
118 int row = (blockIdx.x * TPB_X ) + threadIdx.x;
119
120 if (row < m ) {
121 int start_idx_res = result_ind[row];
122 int stop_idx_res = MLCommon::Sparse::get_stop_idx (row, m , nnz, result_ind);
123
124 int start_idx1 = row_ind1[row];
125 int stop_idx1 = MLCommon::Sparse::get_stop_idx (row, m , nnz1, row_ind1);
126
127 int start_idx2 = row_ind2[row];
128 int stop_idx2 = MLCommon::Sparse::get_stop_idx (row, m , nnz2, row_ind2);
129
130 for ( int j = start_idx_res; j < stop_idx_res; j ++) {
131 int col = result_cols[ j ];
132
133 T left_val = left_min;
134 for ( int k = start_idx1; k < stop_idx1; k ++) {
135 if (cols1[ k ] == col) {
136 left_val = vals1[ k ];
137 }
138 }
139
140 T right_val = right_min;
141 for ( int k = start_idx2; k < stop_idx2; k ++) {
142 if (cols2[ k ] == col) {
143 right_val = vals2[ k ];
144 }
145 }
146
147 if (left_val > left_min || right_val > right_min) {
148 if (mix_weight < 0.5) {
149 result_vals[ j ] =
150 left_val * powf(right_val, mix_weight / (1.0 - mix_weight));
151 } else {
152 result_vals[ j ] =
153 powf(left_val, (1.0 - mix_weight) / mix_weight) * right_val;
154 }
155 }
156 }
157 }

164 template < typename T, int TPB_X>
165 void general_simplicial_set_intersection (
166 int *row1_ind, COO<T> *in1, int *row2_ind, COO<T> *in2, COO<T> *result,
167 float weight, std::shared_ptr<deviceAllocator> d_alloc, cudaStream_t stream) {
168 MLCommon::device_buffer<int> result_ind(d_alloc, stream, in1-> n_rows );
169 CUDA_CHECK (
170 cudaMemsetAsync(result_ind. data (), 0, in1-> n_rows * sizeof ( int ), stream));
171
172 int result_nnz = MLCommon::Sparse::csr_add_calc_inds<float, 32>(
173 row1_ind, in1-> cols (), in1-> vals (), in1-> nnz , row2_ind, in2-> cols (),
174 in2-> vals (), in2-> nnz , in1-> n_rows , result_ind. data (), d_alloc, stream);
175
176 result-> allocate (result_nnz, in1-> n_rows , stream);
177
181 MLCommon::Sparse::csr_add_finalize<float, 32>(
182 row1_ind, in1-> cols (), in1-> vals (), in1-> nnz , row2_ind, in2-> cols (),
183 in2-> vals (), in2-> nnz , in1-> n_rows , result_ind. data (), result-> cols (),
184 result-> vals (), stream);
185
186 //@todo: Write a wrapper function for this
187 MLCommon::Sparse::csr_to_coo<TPB_X>(result_ind. data (), result-> n_rows ,
188 result-> rows (), result-> nnz , stream);
189
190 thrust::device_ptr<const T> d_ptr1 = thrust::device_pointer_cast(in1-> vals ());
191 T min1 = *(thrust::min_element(thrust::cuda::par.on(stream), d_ptr1,
192 d_ptr1 + in1-> nnz ));
193
194 thrust::device_ptr<const T> d_ptr2 = thrust::device_pointer_cast(in2-> vals ());
195 T min2 = *(thrust::min_element(thrust::cuda::par.on(stream), d_ptr2,
196 d_ptr2 + in2-> nnz ));
197
198 T left_min = max (min1 / 2.0, 1e-8);
199 T right_min = max (min2 / 2.0, 1e-8);
200
201 dim3 grid( MLCommon::ceildiv (in1-> nnz , TPB_X ), 1, 1);
202 dim3 blk( TPB_X , 1, 1);
203
204 sset_intersection_kernel<T, TPB_X><<<grid, blk, 0, stream>>>(
205 row1_ind, in1-> cols (), in1-> vals (), in1-> nnz , row2_ind, in2-> cols (),
206 in2-> vals (), in2-> nnz , result_ind. data (), result-> cols (), result-> vals (),
207 result-> nnz , left_min, right_min, in1-> n_rows , weight);
208 CUDA_CHECK (cudaGetLastError());
209
210 dim3 grid_n( MLCommon::ceildiv (result-> nnz , TPB_X ), 1, 1);
211 }

213 template < int TPB_X, typename T>
214 void perform_categorical_intersection (T * y , COO<T> *rgraph_coo,
215 COO<T> *final_coo, UMAPParams * params ,
216 std::shared_ptr<deviceAllocator> d_alloc,
217 cudaStream_t stream) {
218 float far_dist = 1.0e12; // target weight
219 if ( params ->target_weights < 1.0)
220 far_dist = 2.5 * (1.0 / (1.0 - params ->target_weights));
221
222 categorical_simplicial_set_intersection<T, TPB_X>(rgraph_coo, y , stream,
223 far_dist);
224
225 COO<T> comp_coo(d_alloc, stream);
226 coo_remove_zeros<TPB_X, T>(rgraph_coo, &comp_coo, d_alloc, stream);
227
228 reset_local_connectivity<T, TPB_X>(&comp_coo, final_coo, d_alloc, stream);
229
230 CUDA_CHECK (cudaPeekAtLastError());
231 }

257 CUML_LOG_DEBUG ( "%s" , ss1.str().c_str());
258 ss2 << MLCommon::arr2Str (y_knn_dists. data (),

275 CUML_LOG_DEBUG ( "Target Fuzzy Simplicial Set" );

277 ss << ygraph_coo;
278 CUML_LOG_DEBUG (ss.str().c_str());

287 CUDA_CHECK (cudaMemsetAsync(xrow_ind. data (), 0,

297 MLCommon::Sparse::sorted_coo_to_csr (rgraph_coo, xrow_ind. data (), d_alloc,
298 stream);
299
300 COO<T> result_coo(d_alloc, stream);
301 general_simplicial_set_intersection<T, TPB_X>(
302 xrow_ind. data (), rgraph_coo, yrow_ind. data (), &cygraph_coo, &result_coo,
303 params ->target_weights, d_alloc, stream);
304
308 COO<T> out(d_alloc, stream);
309 coo_remove_zeros<TPB_X, T>(&result_coo, &out, d_alloc, stream);
310
311 reset_local_connectivity<T, TPB_X>(&out, final_coo, d_alloc, stream);
312
313 CUDA_CHECK (cudaPeekAtLastError());
314 }

345 template < typename FinalLambda>
346 static void launch ( Params const & params, FinalLambda fin_op,
347 cudaStream_t stream) {
348 // Setup the grid.
349 dim3 grid;
350 grid.x = ceildiv<int>(params.m, Traits::OutputTile::kW);
351 grid.y = ceildiv<int>(params.n, Traits::OutputTile::kH);
352 // The number of threads.
353 dim3 block;
354 block.x = BaseClass::kThreads;
355 // Launch the kernel.
356 void const * args[] = {&params, &fin_op};
357 cudaLaunchKernel(
358 reinterpret_cast< void * > (&custom_gemm_kernel<This_, FinalLambda>), grid,
359 block, const_cast< void ** > (args), 0, stream);
360 CUDA_CHECK (cudaPeekAtLastError());
361 }

364 BaseClass::launch(params, stream);
365 }

373 template < typename FinalLambda>
374 CUTLASS_DEVICE void multiply_add (FinalLambda fin_op) {
375 // Swizzle the IDs of the block (to enable better cache behavior).
376 typename Traits::BlockSwizzle block_swizzle;
377 dim3 block = block_swizzle.swizzle();
378
379 // Scale the id.
380 block.x *= Traits::OutputTile::kW;
381 block.y *= Traits::OutputTile::kH;
382
383 // We may want to use shared memory to clear the registers.
384 typedef typename Traits::ClearAccumulators ClearAccumulators;
385
386 // The streams to read A/B from global memory to shared memory.
387 typename Traits::GlobalLoadStream global_stream(
388 BaseClass::params, BaseClass::shared_storage, block);
389
390 // Create the accumulator clear.
391 ClearAccumulators clear(BaseClass::shared_storage.main_loop.clear);
392
393 // By how much we unroll the main loop.
394 Index const kUnroll = static_cast< Index > (Traits::OutputTile::kD);
395
396 // If we do not have enough steps in the main loop, trigger the residue code
397 global_stream.move_to_residue< true >(BaseClass::params.k);
398
399 // Fetch the fragments for A and B from global memory.
400 global_stream.copy();
401
402 // Copy the elements to shared memory (after transformation if needed).
403 global_stream.commit();
404
405 // Make sure the data is in shared memory.
406 Traits::shared_store_fence( false );
407
408 // Rollback to the beginning of the GEMM-K dimension. It may have no impact.
409 global_stream.rollback();
410
411 // The unrolling steps for the main loop.
412 int const kUnrollingSteps = Traits::MultiplyAdd::AccumulatorsPerWarp::kD /
413 Traits::MultiplyAdd::InstructionShape::kD;
414
415 // Make sure we have at least 2 unrolling steps or our pipeling is not
416 // going to work.
417 static_assert(kUnrollingSteps >= 2,
418 "The pipelining assumes at least two steps" );
419
420 // The stream of data from shared memory to fragments.
421 typename Traits::SharedLoadStream shared_load_stream(
422 BaseClass::params, BaseClass::shared_storage);
423
424 // Trigger the copy from shared memory for the 1st stream.
425 shared_load_stream.copy(0);
426
427 // Allocate the accumulators.
428 typename Traits::MultiplyAdd::Accumulators accumulators;
429 // Clear the accumulators.
430 clear.clear(accumulators);
431
432 // The loop index.
433 Index outer_k = BaseClass::params.k - kUnroll;
434
435 // Enter the main loop and iterate.
436 for (; outer_k > 0; outer_k -= kUnroll) {
437 BaseClass::consume_tile<false>(global_stream, shared_load_stream,
438 accumulators, outer_k);
439 }
440
441 // Residual loop.
442 for (; outer_k > -kUnroll; outer_k -= kUnroll) {
443 BaseClass::consume_tile<true>(global_stream, shared_load_stream,
444 accumulators, outer_k);
445 }
446
447 // Epilogue.
448 typedef typename Traits::Epilogue Epilogue;
449 Epilogue epilogue(BaseClass::params.epilogue,
450 BaseClass::shared_storage.epilogue, BaseClass::params. m ,
451 BaseClass::params. n );
452 epilogue.epilogue(cutlass::make_Coord(0, block.y, block.x), accumulators,
453 fin_op);
454 }
455 }; // end struct CustomGemm

39 HDI int encode ( float val) {
40 int i = *( int *)&val;
41 return i >= 0 ? i : (1 << 31) | ~ i ;

44 HDI long long encode ( double val) {

90 template < typename T, typename E>
91 __global__ void minmaxKernel ( const T* data, const unsigned int * rowids,
92 const unsigned int * colids, int nrows , int ncols ,
93 int row_stride, T* g_min, T* g_max, T* sampledcols,
94 T init_min_val, int batch_ncols, int num_batches) {
95 int tid = threadIdx.x + blockIdx.x * blockDim.x;
96 extern __shared__ char shmem[];
97 T* s_min = (T*)shmem;
98 T* s_max = (T*)(shmem + sizeof (T) * batch_ncols);
99
100 int last_batch_ncols = ncols % batch_ncols;
101 if (last_batch_ncols == 0) {
102 last_batch_ncols = batch_ncols;
103 }
104 int orig_batch_ncols = batch_ncols;
105
106 for ( int batch_id = 0; batch_id < num_batches; batch_id++) {
107 if (batch_id == num_batches - 1) {
108 batch_ncols = last_batch_ncols;
109 }
110
111 for ( int i = threadIdx.x; i < batch_ncols; i += blockDim.x) {
112 *(E*)&s_min[ i ] = encode (init_min_val);
113 *(E*)&s_max[ i ] = encode (-init_min_val);
114 }
115 __syncthreads ();
116
117 for ( int i = tid; i < nrows * batch_ncols; i += blockDim.x * gridDim.x) {
118 int col = (batch_id * orig_batch_ncols) + ( i / nrows );
119 int row = i % nrows ;
120 if (colids != nullptr ) {
121 col = colids[col];
122 }
123 if (rowids != nullptr ) {
124 row = rowids[row];
125 }
126 int index = row + col * row_stride;
127 T coldata = data[index];
128 if (!isnan(coldata)) {
129 //Min max values are saved in shared memory and global memory as per the shuffled colids.
130 atomicMinBits<T, E>(&s_min[( int )( i / nrows )], coldata);
131 atomicMaxBits<T, E>(&s_max[( int )( i / nrows )], coldata);
132 }
133 if (sampledcols != nullptr ) {
134 sampledcols[batch_id * orig_batch_ncols + i ] = coldata;
135 }
136 }
137 __syncthreads ();
138
139 // finally, perform global mem atomics
140 for ( int j = threadIdx.x; j < batch_ncols; j += blockDim.x) {
141 atomicMinBits<T, E>(&g_min[batch_id * orig_batch_ncols + j ],
142 decode (*(E*)&s_min[ j ]));
143 atomicMaxBits<T, E>(&g_max[batch_id * orig_batch_ncols + j ],
144 decode (*(E*)&s_max[ j ]));
145 }
146 __syncthreads ();
147 }
148 }

175 template < typename T, int TPB = 512>
176 void minmax ( const T* data, const unsigned * rowids, const unsigned * colids,
177 int nrows , int ncols , int row_stride, T* globalmin, T* globalmax,
178 T* sampledcols, cudaStream_t stream) {
179 using E = typename encode_traits<T>::E ;
180 int nblks = ceildiv ( ncols , TPB);
181 T init_val = std::numeric_limits<T>::max ();
182 minmaxInitKernel<T, E>
183 <<<nblks, TPB, 0, stream>>>( ncols , globalmin, globalmax, init_val);
184 CUDA_CHECK (cudaPeekAtLastError());
185 nblks = ceildiv ( nrows * ncols , TPB);
186 nblks = min(nblks, 65536);
187 size_t smemSize = sizeof (T) * 2 * ncols ;
188
189 // Compute the batch_ncols, in [1, ncols] range, that meet the available
190 // shared memory constraints.
191 auto smemPerBlk = getSharedMemPerBlock ();
192 int batch_ncols = min( ncols , ( int )(smemPerBlk / ( sizeof (T) * 2)));
193 int num_batches = ceildiv ( ncols , batch_ncols);
194 smemSize = sizeof (T) * 2 * batch_ncols;
195
196 minmaxKernel<T, E><<<nblks, TPB, smemSize, stream>>>(
197 data, rowids, colids, nrows , ncols , row_stride, globalmin, globalmax,
198 sampledcols, init_val, batch_ncols, num_batches);
199 CUDA_CHECK (cudaPeekAtLastError());
200 decodeKernel<T, E><<<nblks, TPB, 0, stream>>>(globalmin, globalmax, ncols );
201 CUDA_CHECK (cudaPeekAtLastError());
202 }

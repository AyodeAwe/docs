32 T* out, IdxType nrows , IdxType ncols ,
33 IdxType nkeys ) {
34 IdxType idx = blockIdx.x * blockDim.x + threadIdx.x;
35 if (idx >= ( nrows * ncols )) return ;
37 IdxType colId = idx % ncols ;
38 IdxType rowId = idx / ncols ;

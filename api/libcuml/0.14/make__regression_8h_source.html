39 /* Internal auxiliary function to help build the singular profile */
40 template < typename DataT, typename IdxT>
41 static __global__ void _singular_profile_kernel(DataT* out, IdxT n ,
42 DataT tail_strength,
43 IdxT rank) {
44 IdxT tid = threadIdx.x + blockIdx.x * blockDim.x;
45 if (tid < n ) {
46 DataT sval = static_cast< DataT > (tid) / rank;
47 DataT low_rank = ((DataT)1.0 - tail_strength) * myExp (-sval * sval);
48 DataT tail = tail_strength * myExp ((DataT)-0.1 * sval);
49 out[tid] = low_rank + tail;
50 }
51 }

53 /* Internal auxiliary function to generate a low-rank matrix */
54 template < typename DataT, typename IdxT>
55 static void _make_low_rank_matrix(DataT* out, IdxT n_rows, IdxT n_cols,
56 IdxT effective_rank, DataT tail_strength,
57 Rng& r , cublasHandle_t cublas_handle,
58 cusolverDnHandle_t cusolver_handle,
59 std::shared_ptr<deviceAllocator> allocator,
60 cudaStream_t stream) {
61 IdxT n = std::min(n_rows, n_cols);
62
63 // Generate random (ortho normal) vectors with QR decomposition
64 device_buffer<DataT> rd_mat_0(allocator, stream);
65 device_buffer<DataT> rd_mat_1(allocator, stream);
66 rd_mat_0.resize(n_rows * n , stream);
67 rd_mat_1.resize(n_cols * n , stream);
68 r .normal(rd_mat_0.data(), n_rows * n , (DataT)0.0, (DataT)1.0, stream);
69 r .normal(rd_mat_1.data(), n_cols * n , (DataT)0.0, (DataT)1.0, stream);
70 device_buffer<DataT> q0(allocator, stream);
71 device_buffer<DataT> q1(allocator, stream);
72 q0.resize(n_rows * n , stream);
73 q1.resize(n_cols * n , stream);
74 LinAlg::qrGetQ (rd_mat_0.data(), q0.data(), n_rows, n , cusolver_handle, stream,
75 allocator);
76 LinAlg::qrGetQ (rd_mat_1.data(), q1.data(), n_cols, n , cusolver_handle, stream,
77 allocator);
78
79 // Build the singular profile by assembling signal and noise components
80 device_buffer<DataT> singular_vec(allocator, stream);
81 device_buffer<DataT> singular_mat(allocator, stream);
82 singular_vec.resize( n , stream);
83 _singular_profile_kernel<<<ceildiv<IdxT>( n , 256), 256, 0, stream>>>(
84 singular_vec.data(), n , tail_strength, effective_rank);
85 CUDA_CHECK (cudaPeekAtLastError());
86 singular_mat.resize( n * n , stream);
87 CUDA_CHECK (
88 cudaMemsetAsync(singular_mat.data(), 0, n * n * sizeof (DataT), stream));
89 Matrix::initializeDiagonalMatrix (singular_vec.data(), singular_mat.data(), n ,
90 n , stream);
91
92 // Generate the column-major matrix
93 device_buffer<DataT> temp_q0s(allocator, stream);
94 device_buffer<DataT> temp_out(allocator, stream);
95 temp_q0s.resize(n_rows * n , stream);
96 temp_out.resize(n_rows * n_cols, stream);
97 DataT alpha = 1.0, beta = 0.0;
98 LinAlg::cublasgemm (cublas_handle, CUBLAS_OP_N, CUBLAS_OP_N, n_rows, n , n ,
99 &alpha, q0.data(), n_rows, singular_mat.data(), n , &beta,
100 temp_q0s.data(), n_rows, stream);
101 LinAlg::cublasgemm (cublas_handle, CUBLAS_OP_N, CUBLAS_OP_T, n_rows, n_cols, n ,
102 &alpha, temp_q0s.data(), n_rows, q1.data(), n_cols, &beta,
103 temp_out.data(), n_rows, stream);
104
105 // Transpose from column-major to row-major
106 LinAlg::transpose (temp_out.data(), out, n_rows, n_cols, cublas_handle,
107 stream);
108 }

110 /* Internal auxiliary function to permute rows in the given matrix according
111 * to a given permutation vector */
112 template < typename DataT, typename IdxT>
113 static __global__ void _gather2d_kernel(DataT* out, const DataT* in,
114 const IdxT* perms, IdxT n_rows,
115 IdxT n_cols) {
116 IdxT tid = blockIdx.x * blockDim.x + threadIdx.x;
117
118 if (tid < n_rows) {
119 const DataT* row_in = in + n_cols * perms[tid];
120 DataT* row_out = out + n_cols * tid;
121
122 for (IdxT i = 0; i < n_cols; i ++) {
123 row_out[ i ] = row_in[ i ];
124 }
125 }
126 }

165 template < typename DataT, typename IdxT>
166 void make_regression (DataT* out, DataT* values, IdxT n_rows, IdxT n_cols,
167 IdxT n_informative, cublasHandle_t cublas_handle,
168 cusolverDnHandle_t cusolver_handle,
169 std::shared_ptr<deviceAllocator> allocator,
170 cudaStream_t stream, DataT* coef = nullptr ,
171 IdxT n_targets = (IdxT)1, DataT bias = (DataT)0.0,
172 IdxT effective_rank = (IdxT)-1,
173 DataT tail_strength = (DataT)0.5, DataT noise = (DataT)0.0,
174 bool shuffle = true , uint64_t seed = 0ULL,
175 GeneratorType type = GenPhilox ) {
176 n_informative = std::min(n_informative, n_cols);
177 cublasSetPointerMode(cublas_handle, CUBLAS_POINTER_MODE_HOST);
178 Rng r (seed, type);
179
180 if (effective_rank < 0) {
181 // Randomly generate a well conditioned input set
182 r .normal(out, n_rows * n_cols, (DataT)0.0, (DataT)1.0, stream);
183 } else {
184 // Randomly generate a low rank, fat tail input set
185 _make_low_rank_matrix(out, n_rows, n_cols, effective_rank, tail_strength, r ,
186 cublas_handle, cusolver_handle, allocator, stream);
187 }
188
189 // Use the right output buffer for the values
190 device_buffer<DataT> tmp_values(allocator, stream);
191 DataT* _values;
192 if ( shuffle ) {
193 tmp_values. resize (n_rows * n_targets, stream);
194 _values = tmp_values. data ();
195 } else {
196 _values = values;
197 }
198 // Create a column-major matrix of output values only if it has more
199 // than 1 column
200 device_buffer<DataT> values_col(allocator, stream);
201 DataT* _values_col;
202 if (n_targets > 1) {
203 values_col. resize (n_rows * n_targets, stream);
204 _values_col = values_col. data ();
205 } else {
206 _values_col = _values;
207 }
208
209 // Use the right buffer for the coefficients
210 device_buffer<DataT> tmp_coef(allocator, stream);
211 DataT* _coef;
212 if (coef != nullptr && ! shuffle ) {
213 _coef = coef;
214 } else {
215 tmp_coef. resize (n_cols * n_targets, stream);
216 _coef = tmp_coef. data ();
217 }
218
219 // Generate a ground truth model with only n_informative features
220 r .uniform(_coef, n_informative * n_targets, (DataT)1.0, (DataT)100.0, stream);
221 if (coef && n_informative != n_cols) {
222 CUDA_CHECK (cudaMemsetAsync(
223 _coef + n_informative * n_targets, 0,
224 (n_cols - n_informative) * n_targets * sizeof (DataT), stream));
225 }
226
227 // Compute the output values
228 DataT alpha = (DataT)1.0, beta = (DataT)0.0;
229 CUBLAS_CHECK ( LinAlg::cublasgemm (
230 cublas_handle, CUBLAS_OP_T, CUBLAS_OP_T, n_rows, n_targets, n_informative,
231 &alpha, out, n_cols, _coef, n_targets, &beta, _values_col, n_rows, stream));
232
233 // Transpose the values from column-major to row-major if needed
234 if (n_targets > 1) {
235 LinAlg::transpose (_values_col, _values, n_rows, n_targets, cublas_handle,
236 stream);
237 }
238
239 if (bias != 0.0) {
240 // Add bias
241 LinAlg::addScalar (_values, _values, bias, n_rows * n_targets, stream);
242 }
243
244 device_buffer<DataT> white_noise(allocator, stream);
245 if (noise != 0.0) {
246 // Add white noise
247 white_noise. resize (n_rows * n_targets, stream);
248 r .normal(white_noise. data (), n_rows * n_targets, (DataT)0.0, noise, stream);
249 LinAlg::add (_values, _values, white_noise. data (), n_rows * n_targets,
250 stream);
251 }
252
253 if ( shuffle ) {
254 device_buffer<DataT> tmp_out(allocator, stream);
255 device_buffer<IdxT> perms_samples(allocator, stream);
256 device_buffer<IdxT> perms_features(allocator, stream);
257 tmp_out. resize (n_rows * n_cols, stream);
258 perms_samples. resize (n_rows, stream);
259 perms_features. resize (n_cols, stream);
260
261 constexpr IdxT Nthreads = 256;
262
263 // Shuffle the samples from out to tmp_out
264 permute<DataT, IdxT, IdxT>(perms_samples. data (), tmp_out. data (), out,
265 n_cols, n_rows, true , stream);
266 IdxT nblks_rows = ceildiv<IdxT>(n_rows, Nthreads);
267 _gather2d_kernel<<<nblks_rows, Nthreads, 0, stream>>>(
268 values, _values, perms_samples. data (), n_rows, n_targets);
269 CUDA_CHECK (cudaPeekAtLastError());
270
271 // Shuffle the features from tmp_out to out
272 permute<DataT, IdxT, IdxT>(perms_features. data (), out, tmp_out. data (),
273 n_rows, n_cols, false , stream);
274
275 // Shuffle the coefficients accordingly
276 if (coef != nullptr ) {
277 IdxT nblks_cols = ceildiv<IdxT>(n_cols, Nthreads);
278 _gather2d_kernel<<<nblks_cols, Nthreads, 0, stream>>>(
279 coef, _coef, perms_features. data (), n_cols, n_targets);
280 CUDA_CHECK (cudaPeekAtLastError());
281 }
282 }

17 /*
18 * This file contains an implementation of some batched sparse matrix
19 * operations in Compressed Sparse Row representation.
20 *
21 * Important: the implementation is designed to give good performance on
22 * large batches of relatively small matrices (typically one or two
23 * elements per row). In other use cases it might be slower than using
24 * the dense counterparts!
25 */

67 template < typename T>
68 static __global__ void dense_to_csr_kernel( const T* dense, const int * col_index,
69 const int * row_index, T* values,
70 int batch_size, int m , int n ,
71 int nnz) {
72 int bid = blockIdx.x * blockDim.x + threadIdx.x;
73
74 if (bid < batch_size) {
75 int stride = m * n ;
76 for ( int i = 0; i < m ; i ++) {
77 for ( int idx = row_index[ i ]; idx < row_index[ i + 1]; idx++) {
78 int j = col_index[idx];
79 values[bid * nnz + idx] = dense[bid * stride + j * m + i ];
80 }
81 }
82 }
83 }

213 static CSR<T> from_dense ( const LinAlg::Batched::Matrix<T> & dense,
214 const std::vector<bool>& mask,
215 cusolverSpHandle_t cusolverSpHandle ) {
216 std::pair<int, int> shape = dense. shape ();
217
218 // Create the index arrays from the mask
219 std::vector<int> h_col_index;
220 std::vector<int> h_row_index = std::vector<int>( shape .first + 1);
221 int nnz = 0;
222 for ( int i = 0; i < shape .first; i ++) {
223 h_row_index[ i ] = nnz ;
224 for ( int j = 0; j < shape .second; j ++) {
225 if (mask[ j * shape .first + i ]) {
226 h_col_index.push_back( j );
227 nnz ++;
228 }
229 }
230 }
231 h_row_index[ shape .first] = nnz ;
232
233 CSR<T> out = CSR<T> ( shape .first, shape .second, nnz , dense. batches (),
234 dense. cublasHandle (), cusolverSpHandle ,
235 dense. allocator (), dense. stream ());
236
237 // Copy the host index arrays to the device
238 MLCommon::copy (out. get_col_index (), h_col_index.data(), nnz , out. stream ());
239 MLCommon::copy (out. get_row_index (), h_row_index.data(), shape .first + 1,
240 out. stream ());
241
242 // Copy the data from the dense matrix to its sparse representation
243 constexpr int TPB = 256;
244 dense_to_csr_kernel<<<MLCommon::ceildiv<int>(out. batches (), TPB), TPB, 0,
245 out. stream ()>>>(
246 dense. raw_data (), out. get_col_index (), out. get_row_index (),
247 out. get_values (), out. batches (), shape .first, shape .second, nnz );
248 CUDA_CHECK (cudaPeekAtLastError());
249
250 return out;
251 }

351 template < typename T>
352 __global__ void batched_spmv_kernel (T alpha, const int * A_col_index,
353 const int * A_row_index, const T* A_values,
354 const T* x , T beta, T* y , int m , int n ,
355 int batch_size) {
356 int bid = blockIdx.x * blockDim.x + threadIdx.x;
357
358 if (bid < batch_size) {
359 int nnz = A_row_index[ m ];
360 for ( int i = 0; i < m ; i ++) {
361 T acc = 0.0;
362 for ( int idx = A_row_index[ i ]; idx < A_row_index[ i + 1]; idx++) {
363 int j = A_col_index[idx];
364 acc += A_values[bid * nnz + idx] * x [bid * n + j ];
365 }
366 y [bid * m + i ] =
367 alpha * acc + (beta == 0.0 ? 0.0 : beta * y [bid * m + i ]);
368 }
369 }
370 }

386 template < typename T>
387 void b_spmv (T alpha, const CSR<T> & A, const LinAlg::Batched::Matrix<T> & x ,
388 T beta, LinAlg::Batched::Matrix<T> & y ) {
389 int m = A. shape ().first;
390 int n = A. shape ().second;
391 // A few checks
392 ASSERT (std::min( x .shape().first, x .shape().second) == 1 &&
393 std::max ( x .shape().first, x .shape().second) == n ,
394 "SpMV: Dimension mismatch: x" );
395 ASSERT (std::min( y .shape().first, y .shape().second) == 1 &&
396 std::max ( y .shape().first, y .shape().second) == m ,
397 "SpMV: Dimension mismatch: y" );
398 ASSERT (A. batches () == x .batches(),
399 "SpMV: A and x must have the same batch size" );
400 ASSERT (A. batches () == y .batches(),
401 "SpMV: A and y must have the same batch size" );
402
403 // Execute the kernel
404 constexpr int TPB = 256;
405 batched_spmv_kernel<<<MLCommon::ceildiv<int>(A. batches (), TPB), TPB, 0,
406 A. stream ()>>>(
407 alpha, A. get_col_index (), A. get_row_index (), A. get_values (), x .raw_data(),
408 beta, y .raw_data(), m , n , A. batches ());
409 CUDA_CHECK (cudaPeekAtLastError());
410 }

431 template < typename T>
432 __global__ void batched_spmm_kernel (T alpha, const int * A_col_index,
433 const int * A_row_index, const T* A_values,
434 const T* B, T beta, T* C, int m , int k ,
435 int n , int batch_size,
436 int threads_per_bid) {
437 int thread_idx = blockIdx.x * blockDim.x + threadIdx.x;
438 int bid = thread_idx / threads_per_bid;
439
440 if (bid < batch_size) {
441 int nnz = A_row_index[ m ];
442 const T* b_A_values = A_values + bid * nnz;
443 const T* b_B = B + bid * k * n ;
444 for ( int j = thread_idx % threads_per_bid; j < n ; j += threads_per_bid) {
445 for ( int i = 0; i < m ; i ++) {
446 T acc = 0.0;
447 for ( int idx = A_row_index[ i ]; idx < A_row_index[ i + 1]; idx++) {
448 int ik = A_col_index[idx];
449 acc += b_A_values[idx] * b_B[ j * k + ik];
450 }
451 int ci = bid * m * n + j * m + i ;
452 C[ci] = alpha * acc + (beta == 0.0 ? 0.0 : beta * C[ci]);
453 }
454 }
455 }

477 template < typename T>
478 __global__ void batched_spmm_kernel_shared_mem (T alpha, const int * A_col_index,
479 const int * A_row_index,
480 const T* A_values, const T* B,
481 T beta, T* C, int m , int k ,
482 int n , int nnz) {
483 int bid = blockIdx.x;
484 int j = threadIdx.x;
485
486 // Using dynamic shared memory
487 extern __shared__ int8_t shared_mem[];
488 // Mapping arrays to shared mem ; note: T before int for alignment!
489 T* s_A_values = (T*)shared_mem;
490 T* s_B = (T*)(shared_mem + nnz * sizeof (T));
491 int * s_A_col_index = ( int *)(shared_mem + (nnz + k * n ) * sizeof (T));
492 int * s_A_row_index =
493 ( int *)(shared_mem + (nnz + k * n ) * sizeof (T) + nnz * sizeof ( int ));
494
495 // Load A in shared memory
496 const T* b_A_values = A_values + bid * nnz;
497 for ( int i_nnz = j ; i_nnz < nnz; i_nnz += blockDim.x) {
498 s_A_col_index[i_nnz] = A_col_index[i_nnz];
499 s_A_values[i_nnz] = b_A_values[i_nnz];
500 }
501 for ( int i_m = j ; i_m < m ; i_m += blockDim.x) {
502 s_A_row_index[i_m] = A_row_index[i_m];
503 }
504 if ( j == 0) s_A_row_index[ m ] = nnz;
505
506 // Load B in shared memory
507 const T* b_B = B + bid * k * n ;
508 for ( int i_kn = j ; i_kn < k * n ; i_kn += blockDim.x) {
509 s_B[i_kn] = b_B[i_kn];
510 }
511
512 __syncthreads ();
513
514 for ( int i = 0; i < m ; i ++) {
515 T acc = 0.0;
516 for ( int idx = s_A_row_index[ i ]; idx < s_A_row_index[ i + 1]; idx++) {
517 int ik = s_A_col_index[idx];
518 acc += s_A_values[idx] * s_B[ j * k + ik];
519 }
520 int ci = bid * m * n + j * m + i ;
521 C[ci] = alpha * acc + (beta == 0.0 ? 0.0 : beta * C[ci]);
522 }
523 }

539 template < typename T>
540 void b_spmm (T alpha, const CSR<T> & A, const LinAlg::Batched::Matrix<T> & B,
541 T beta, LinAlg::Batched::Matrix<T> & C, bool use_shared_mem = true ) {
542 int m = A. shape ().first;
543 int n = B. shape ().second;
544 int k = A. shape ().second;
545 int nb = A. batches ();
546 int nnz = A. nnz ();
547 // Check the parameters
548 ASSERT (B. batches () == nb, "SpMM: A and B must have the same batch size" );
549 ASSERT (C. batches () == nb, "SpMM: A and C must have the same batch size" );
550 ASSERT (B. shape ().first == k , "SpMM: Dimension mismatch: A and B" );
551 ASSERT (C. shape ().first == m && C. shape ().second == n ,
552 "SpMM: Dimension mismatch: C" );
553
554 // Execute the kernel
555 if (use_shared_mem) { // Shared memory kernel (large matrices)
556 size_t shared_mem_size =
557 (nnz + m + 1) * sizeof ( int ) + (nnz + k * n ) * sizeof (T);
558 batched_spmm_kernel_shared_mem <<<nb, n , shared_mem_size, A. stream ()>>>(
559 alpha, A. get_col_index (), A. get_row_index (), A. get_values (), B. raw_data (),
560 beta, C. raw_data (), m , k , n , nnz);
561 CUDA_CHECK (cudaPeekAtLastError());
562 } else { // No shared memory (small matrices)
563 constexpr int TPB = 256;
564 int threads_per_bid =
565 nb <= 1024 ? 8 : (nb <= 2048 ? 4 : (nb <= 4096 ? 2 : 1));
566 batched_spmm_kernel<<<MLCommon::ceildiv<int>(nb * threads_per_bid, TPB),
567 TPB, 0, A. stream ()>>>(
568 alpha, A. get_col_index (), A. get_row_index (), A. get_values (), B. raw_data (),
569 beta, C. raw_data (), m , k , n , nb, threads_per_bid);
570 CUDA_CHECK (cudaPeekAtLastError());
571 }

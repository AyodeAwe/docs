159 struct forest_params_t {
160 // total number of nodes; ignored for dense forests
161 int num_nodes ;
162 // maximum depth; ignored for sparse forests
163 int depth ;
164 // ntrees is the number of trees
165 int num_trees ;
166 // num_cols is the number of columns in the data
167 int num_cols ;
168 // leaf_payload_type determines what the leaves store (predict)
169 leaf_value_t leaf_payload_type ;
170 // algo is the inference algorithm;
171 // sparse forests do not distinguish between NAIVE and TREE_REORG
172 algo_t algo ;
173 // output is the desired output type
174 output_t output ;
175 // threshold is used to for classification if leaf_payload_type == FLOAT_SCALAR && (output & OUTPUT_CLASS) != 0 && !predict_proba,
176 // and is ignored otherwise
177 float threshold ;
178 // global_bias is added to the sum of tree predictions
179 // (after averaging, if it is used, but before any further transformations)
180 float global_bias ;
181 // only used for INT_CLASS_LABEL inference. since we're storing the
182 // labels in leaves instead of the whole vector, this keeps track
183 // of the number of classes
184 int num_classes ;
185 };

188 struct treelite_params_t {

195 // and is ignored otherwise. threshold is ignored if leaves store

197 // is returned regardless of the absolute vote count

199 // storage_type indicates whether the forest should be imported as dense or sparse

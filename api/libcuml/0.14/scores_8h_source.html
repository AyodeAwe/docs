50 template < typename math_t, typename knn_index_t>
51 __global__ void compute_rank (math_t *ind_X, knn_index_t *ind_X_embedded, int n ,
52 int n_neighbors, int work, double *rank) {
53 int i = blockIdx.x * blockDim.x + threadIdx.x;
54 if ( i >= work) return ;
55
56 int n_idx = i / n_neighbors;
57 int nn_idx = ( i % n_neighbors) + 1;
58
59 knn_index_t idx = ind_X_embedded[n_idx * (n_neighbors + 1) + nn_idx];
60 math_t *sample_i = &ind_X[n_idx * n ];
61
62 // TODO: This could probably be binary searched, based on
63 // the distances, as well. (re: https://github.com/rapidsai/cuml/issues/1698)
64 for ( int r = 1; r < n ; r ++) {
65 if (sample_i[ r ] == idx) {
66 int tmp = r - n_neighbors;
67 if (tmp > 0) atomicAdd(rank, tmp);
68 break ;
69 }
70 }
71 }

118 template < typename math_t, Distance::DistanceType distance_type>
119 double trustworthiness_score (math_t *X, math_t *X_embedded, int n , int m , int d,
120 int n_neighbors,
121 std::shared_ptr<deviceAllocator> d_alloc,
122 cudaStream_t stream, int batchSize = 512) {
123 const int TMP_SIZE = batchSize * n ;
124
125 typedef cutlass::Shape<8, 128, 128> OutputTile_t;
126
127 math_t *d_pdist_tmp =
128 (math_t *)d_alloc->allocate(TMP_SIZE * sizeof (math_t), stream);
129 int *d_ind_X_tmp = ( int *)d_alloc->allocate(TMP_SIZE * sizeof ( int ), stream);
130
131 int64_t *ind_X_embedded =
132 get_knn_indices (X_embedded, n , d, n_neighbors + 1, d_alloc, stream);
133
134 double t_tmp = 0.0;
135 double t = 0.0;
136 double *d_t = ( double *)d_alloc->allocate( sizeof ( double ), stream);
137
138 int toDo = n ;
139 while (toDo > 0) {
140 int curBatchSize = min(toDo, batchSize);
141
142 // Takes at most batchSize vectors at a time
143
144 size_t workspaceSize = 0;
145
146 MLCommon::Distance::distance <distance_type, math_t, math_t, math_t,
147 OutputTile_t>(
148 &X[( n - toDo) * m ], X, d_pdist_tmp, curBatchSize, n , m , ( void *) nullptr ,
149 workspaceSize, stream);
150 CUDA_CHECK (cudaPeekAtLastError());
151
152 size_t colSortWorkspaceSize = 0;
153 bool bAllocWorkspace = false ;
154 char *sortColsWorkspace;
155
156 MLCommon::Selection::sortColumnsPerRow(
157 d_pdist_tmp, d_ind_X_tmp, curBatchSize, n , bAllocWorkspace, nullptr ,
158 colSortWorkspaceSize, stream);
159
160 if (bAllocWorkspace) {
161 sortColsWorkspace =
162 ( char *)d_alloc->allocate(colSortWorkspaceSize, stream);
163
164 MLCommon::Selection::sortColumnsPerRow(
165 d_pdist_tmp, d_ind_X_tmp, curBatchSize, n , bAllocWorkspace,
166 sortColsWorkspace, colSortWorkspaceSize, stream);
167 }
168 CUDA_CHECK (cudaPeekAtLastError());
169
170 t_tmp = 0.0;
171 updateDevice (d_t, &t_tmp, 1, stream);
172
173 int work = curBatchSize * n_neighbors;
174 int n_blocks = ceildiv (work, N_THREADS );
175 compute_rank<<<n_blocks, N_THREADS, 0, stream>>>(
176 d_ind_X_tmp, &ind_X_embedded[( n - toDo) * (n_neighbors + 1)], n ,
177 n_neighbors, curBatchSize * n_neighbors, d_t);
178 CUDA_CHECK (cudaPeekAtLastError());
179
180 updateHost (&t_tmp, d_t, 1, stream);
181 CUDA_CHECK (cudaStreamSynchronize(stream));
182
183 if (bAllocWorkspace) {
184 d_alloc->deallocate(sortColsWorkspace, colSortWorkspaceSize, stream);
185 }
186
187 t += t_tmp;
188
189 toDo -= curBatchSize;
190 }
191
192 t =
193 1.0 -
194 ((2.0 / (( n * n_neighbors) * ((2.0 * n ) - (3.0 * n_neighbors) - 1.0))) * t);
195
196 d_alloc->deallocate(ind_X_embedded, n * (n_neighbors + 1) * sizeof (int64_t),
197 stream);
198 d_alloc->deallocate(d_pdist_tmp, TMP_SIZE * sizeof (math_t), stream);
199 d_alloc->deallocate(d_ind_X_tmp, TMP_SIZE * sizeof ( int ), stream);
200 d_alloc->deallocate(d_t, sizeof ( double ), stream);
201
202 return t;
203 }

220 template < typename math_t>
221 math_t r2_score (math_t * y , math_t *y_hat, int n , cudaStream_t stream) {
222 math_t *y_bar;
223 MLCommon::allocate (y_bar, 1);
224
225 MLCommon::Stats::mean (y_bar, y , 1, n , false , false , stream);
226 CUDA_CHECK (cudaPeekAtLastError());
227
228 math_t *sse_arr;
229 MLCommon::allocate (sse_arr, n );
230
231 MLCommon::LinAlg::eltwiseSub (sse_arr, y , y_hat, n , stream);
232 MLCommon::LinAlg::powerScalar (sse_arr, sse_arr, math_t(2.0), n , stream);
233 CUDA_CHECK (cudaPeekAtLastError());
234
235 math_t *ssto_arr;
236 MLCommon::allocate (ssto_arr, n );
237
238 MLCommon::LinAlg::subtractDevScalar (ssto_arr, y , y_bar, n , stream);
239 MLCommon::LinAlg::powerScalar (ssto_arr, ssto_arr, math_t(2.0), n , stream);
240 CUDA_CHECK (cudaPeekAtLastError());
241
242 thrust::device_ptr<math_t> d_sse = thrust::device_pointer_cast(sse_arr);
243 thrust::device_ptr<math_t> d_ssto = thrust::device_pointer_cast(ssto_arr);
244
245 math_t sse = thrust::reduce (thrust::cuda::par.on(stream), d_sse, d_sse + n );
246 math_t ssto =
247 thrust::reduce (thrust::cuda::par.on(stream), d_ssto, d_ssto + n );
248
249 CUDA_CHECK (cudaFree(y_bar));
250 CUDA_CHECK (cudaFree(sse_arr));
251 CUDA_CHECK (cudaFree(ssto_arr));
252
253 return 1.0 - sse / ssto;
254 }

285 template < typename T>
286 __global__ void reg_metrics_kernel ( const T *predictions,
287 const T *ref_predictions, int n ,
288 double *abs_diffs, double *tmp_sums) {
289 int tid = threadIdx.x + blockIdx.x * blockDim.x;
290 __shared__ double shmem[2]; // {abs_difference_sum, squared difference sum}
291
292 for ( int i = threadIdx.x; i < 2; i += blockDim.x) {
293 shmem[ i ] = 0;
294 }
295 __syncthreads ();
296
297 for ( int i = tid; i < n ; i += blockDim.x * gridDim.x) {
298 double diff = predictions[ i ] - ref_predictions[ i ];
299 double abs_diff = abs( diff );
300 atomicAdd(&shmem[0], abs_diff);
301 atomicAdd(&shmem[1], diff * diff );
302
303 // update absolute difference in global memory for subsequent abs. median computation
304 abs_diffs[ i ] = abs_diff;
305 }
306 __syncthreads ();
307
308 // Update tmp_sum w/ total abs_difference_sum and squared difference sum.
309 for ( int i = threadIdx.x; i < 2; i += blockDim.x) {
310 atomicAdd(&tmp_sums[ i ], shmem[ i ]);
311 }
312 }

326 template < typename T>
327 void regression_metrics ( const T *predictions, const T *ref_predictions, int n ,
328 std::shared_ptr<deviceAllocator> d_alloc,
329 cudaStream_t stream, double &mean_abs_error,
330 double &mean_squared_error, double &median_abs_error) {
331 std::vector<double> mean_errors(2);
332 std::vector<double> h_sorted_abs_diffs( n );
333 int thread_cnt = 256;
334 int block_cnt = ceildiv ( n , thread_cnt);
335
336 int array_size = n * sizeof (double);
337 double *abs_diffs_array = ( double *)d_alloc->allocate(array_size, stream);
338 double *sorted_abs_diffs = ( double *)d_alloc->allocate(array_size, stream);
339 double *tmp_sums = ( double *)d_alloc->allocate(2 * sizeof ( double ), stream);
340 CUDA_CHECK (cudaMemsetAsync(tmp_sums, 0, 2 * sizeof ( double ), stream));
341
342 reg_metrics_kernel<T><<<block_cnt, thread_cnt, 0, stream>>>(
343 predictions, ref_predictions, n , abs_diffs_array, tmp_sums);
344 CUDA_CHECK (cudaGetLastError());
345 MLCommon::updateHost (&mean_errors[0], tmp_sums, 2, stream);
346 CUDA_CHECK (cudaStreamSynchronize(stream));
347
348 mean_abs_error = mean_errors[0] / n ;
349 mean_squared_error = mean_errors[1] / n ;
350
351 // Compute median error. Sort diffs_array and pick median value
352 char * temp_storage = nullptr ;
353 size_t temp_storage_bytes;
354 CUDA_CHECK (cub::DeviceRadixSort::SortKeys(
355 ( void *) temp_storage , temp_storage_bytes, abs_diffs_array, sorted_abs_diffs,
356 n , 0, 8 * sizeof ( double ), stream));
357 temp_storage = ( char *)d_alloc->allocate(temp_storage_bytes, stream);
358 CUDA_CHECK (cub::DeviceRadixSort::SortKeys(
359 ( void *) temp_storage , temp_storage_bytes, abs_diffs_array, sorted_abs_diffs,
360 n , 0, 8 * sizeof ( double ), stream));
361
362 MLCommon::updateHost (h_sorted_abs_diffs.data(), sorted_abs_diffs, n , stream);
363 CUDA_CHECK (cudaStreamSynchronize(stream));
364
365 int middle = n / 2;
366 if ( n % 2 == 1) {
367 median_abs_error = h_sorted_abs_diffs[middle];
368 } else {
369 median_abs_error =
370 (h_sorted_abs_diffs[middle] + h_sorted_abs_diffs[middle - 1]) / 2;
371 }
372
373 d_alloc->deallocate(abs_diffs_array, array_size, stream);
374 d_alloc->deallocate(sorted_abs_diffs, array_size, stream);
375 d_alloc->deallocate( temp_storage , temp_storage_bytes, stream);
376 d_alloc->deallocate(tmp_sums, 2 * sizeof ( double ), stream);
377 }

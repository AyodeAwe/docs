55 template < typename ScalarT, typename IdxT>
56 void information_criterion (ScalarT* d_ic, const ScalarT* d_loglikelihood,
57 IC_Type ic_type, IdxT n_params, IdxT batch_size,
58 IdxT n_samples, cudaStream_t stream) {
59 ScalarT ic_base;
60 ScalarT N = static_cast< ScalarT > (n_params);
61 ScalarT T = static_cast< ScalarT > (n_samples);
62 switch (ic_type) {
63 case AIC :
64 ic_base = (ScalarT)2.0 * N ;
65 break ;
66 case AICc :
67 ic_base =
68 (ScalarT)2.0 * ( N + ( N * ( N + (ScalarT)1.0)) / (T - N - (ScalarT)1.0));
69 break ;
70 case BIC :
71 ic_base = std::log(T) * N ;
72 break ;
73 }

78 return ic_base - (ScalarT)2.0 * loglike;

80 stream);
81 }

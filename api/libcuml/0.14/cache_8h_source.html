121 Cache (std::shared_ptr<deviceAllocator> allocator, cudaStream_t stream,
122 int n_vec, float cache_size = 200)
123 : allocator(allocator),
124 n_vec(n_vec),
125 cache_size(cache_size),
126 cache(allocator, stream),
127 cached_keys(allocator, stream),
128 cache_time(allocator, stream),
129 is_cached(allocator, stream),
130 ws_tmp(allocator, stream),
131 idx_tmp(allocator, stream),
132 d_num_selected_out(allocator, stream, 1),
133 d_temp_storage(allocator, stream) {
134 ASSERT (n_vec > 0, "Parameter n_vec: shall be larger than zero" );
135 ASSERT (associativity > 0, "Associativity shall be larger than zero" );
136 ASSERT (cache_size >= 0, "Cache size should not be negative" );
137
138 // Calculate how many vectors would fit the cache
139 int n_cache_vecs = (cache_size * 1024 * 1024) / ( sizeof (math_t) * n_vec);
140
141 // The available memory shall be enough for at least one cache set
142 if (n_cache_vecs >= associativity) {
143 n_cache_sets = n_cache_vecs / associativity;
144 n_cache_vecs = n_cache_sets * associativity;
145 cache.resize(n_cache_vecs * n_vec, stream);
146 cached_keys.resize(n_cache_vecs, stream);
147 cache_time.resize(n_cache_vecs, stream);
148 CUDA_CHECK (cudaMemsetAsync(cached_keys.data(), 0,
149 cached_keys.size() * sizeof ( int ), stream));
150 CUDA_CHECK (cudaMemsetAsync(cache_time.data(), 0,
151 cache_time.size() * sizeof ( int ), stream));
152 } else {
153 if (cache_size > 0) {
154 CUML_LOG_WARN (
155 "Warning: not enough memory to cache a single set of "
156 "rows, not using cache" );
157 }
158 n_cache_sets = 0;
159 cache_size = 0;
160 }
161 CUML_LOG_DEBUG (
162 "Creating cache with size=%f MiB, to store %d vectors, in "
163 "%d sets with associativity=%d" ,
164 cache_size, n_cache_vecs, n_cache_sets, associativity);
165 }

272 void GetCacheIdxPartitioned ( int *keys, int n , int *cache_idx, int *n_cached,
273 cudaStream_t stream) {
274 ResizeTmpBuffers( n , stream);
275
276 GetCacheIdx (keys, n , ws_tmp.data(), is_cached.data(), stream);
277
278 // Group cache indices as [already cached, non_cached]
279 cub::DevicePartition::Flagged(d_temp_storage.data(), d_temp_storage_size,
280 ws_tmp.data(), is_cached.data(), cache_idx,
281 d_num_selected_out.data(), n , stream);
282
283 updateHost (n_cached, d_num_selected_out.data(), 1, stream);
284
285 // Similarily re-group the input indices
286 copy (ws_tmp.data(), keys, n , stream);
287 cub::DevicePartition::Flagged(d_temp_storage.data(), d_temp_storage_size,
288 ws_tmp.data(), is_cached.data(), keys,
289 d_num_selected_out.data(), n , stream);
290
291 CUDA_CHECK (cudaStreamSynchronize(stream));
292 }

306 void AssignCacheIdx ( int *keys, int n , int *cidx, cudaStream_t stream) {
307 if ( n <= 0) return ;
308 cub::DeviceRadixSort::SortPairs(d_temp_storage.data(), d_temp_storage_size,
309 cidx, ws_tmp.data(), keys, idx_tmp.data(),
310 n , 0, sizeof ( int ) * 8, stream);
311
312 copy (keys, idx_tmp.data(), n , stream);
313
314 // set it to -1
315 CUDA_CHECK (cudaMemsetAsync(cidx, 255, n * sizeof ( int ), stream));
316 const int nthreads = associativity <= 32 ? associativity : 32;
317
318 assign_cache_idx<nthreads, associativity>
319 <<<n_cache_sets, nthreads, 0, stream>>>(keys, n , ws_tmp.data(),
320 cached_keys.data(), n_cache_sets,
321 cache_time.data(), n_iter, cidx);
322
323 CUDA_CHECK (cudaPeekAtLastError());
324 if (debug_mode) CUDA_CHECK (cudaDeviceSynchronize());
325 }

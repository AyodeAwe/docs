23 template < typename Dtype>
24 __global__ void conv1d_kernel ( const Dtype *input, int batch_size,
25 const Dtype *filter, int filter_size,
26 Dtype *output, int output_size) {
27 const int tid = GET_TID ;
28 if (tid < batch_size) {
29 for ( int o = 0; o < output_size; ++o) {
30 Dtype out = 0.;
31 for ( int i = 0; i < filter_size; ++ i )
32 out += filter[ i ] * input[tid + ( i + o) * batch_size];
33 output[tid + o * batch_size] = out;
34 }
35 }
36 }

50 template < typename Dtype>
51 __global__ void season_mean_kernel ( const Dtype *season, int len, int batch_size,
52 Dtype *start_season, int frequency,
53 int half_filter_size, bool ADDITIVE_KERNEL) {
54 int tid = GET_TID ;
55 if (tid < batch_size) {
56 Dtype mean = 0.0;
57 for ( int i = 0; i < frequency; ++ i ) {
58 Dtype period_mean = 0.;
59 int c = 0;
60 for ( int k = i ; k < len; k = k + frequency) {
61 period_mean += season[ k * batch_size + tid];
62 c++;
63 }
64 int count = 1 + ((len - i - 1) / frequency);
65 period_mean /= count;
66 int ss_idx = ( i + half_filter_size) % frequency;
67 start_season[ss_idx * batch_size + tid] = period_mean;
68 mean += period_mean;
69 }
70 mean /= frequency;
71 for ( int i = 0; i < frequency; ++ i ) {
72 if (ADDITIVE_KERNEL)
73 start_season[ i * batch_size + tid] -= mean ;
74 else // MULTIPLOCATIVE
75 start_season[ i * batch_size + tid] /= mean ;
76 }
77 }
78 }

92 template < typename Dtype>
93 __global__ void RinvKernel ( const Dtype *A, Dtype *Rinv, int trend_len) {
94 // Inverse of R (2x2 upper triangular matrix)
95 int tid = GET_TID ;
96 if (tid == 0) {
97 Dtype a = A[0], b = A[trend_len], d = A[trend_len + 1];
98 Dtype factor = 1. / (a * d);
99 Rinv[0] = factor * d;
100 Rinv[1] = 0.;
101 Rinv[2] = -factor * b;
102 Rinv[3] = factor * a;
103 }
104 }

106 template < typename Dtype>
107 __global__ void batched_ls_solver_kernel ( const Dtype *B, const Dtype *rq,
108 int batch_size, int len, Dtype *level,
109 Dtype *trend) {
110 int tid = GET_TID ;
111 if (tid < batch_size) {
112 Dtype level_ = 0., trend_ = 0.;
113 for ( int i = 0; i < len; ++ i ) {
114 Dtype b = B[tid + i * batch_size];
115 level_ += rq[2 * i ] * b;
116 trend_ += rq[2 * i + 1] * b;
117 }
118 level[tid] = level_;
119 trend[tid] = trend_;
120 }
121 }

123 template < typename Dtype>
124 void batched_ls ( const ML::cumlHandle_impl &handle, const Dtype *data,
125 int trend_len, int batch_size, Dtype *level, Dtype *trend) {
126 cudaStream_t stream = handle. getStream ();
127 cublasHandle_t cublas_h = handle. getCublasHandle ();
128 cusolverDnHandle_t cusolver_h = handle. getcusolverDnHandle ();
129 std::shared_ptr<MLCommon::deviceAllocator> dev_allocator =
130 handle. getDeviceAllocator ();
131
132 const Dtype one = (Dtype)1.;
133 const Dtype zero = (Dtype)0.;
134 int geqrf_buffer;
135 int orgqr_buffer;
136 int lwork_size;
137
138 // Allocate memory
139 std::vector<Dtype> A_h(2 * trend_len);
140
141 MLCommon::device_buffer<Dtype> A_d(dev_allocator, stream, 2 * trend_len);
142 MLCommon::device_buffer<Dtype> tau_d(dev_allocator, stream, 2);
143 MLCommon::device_buffer<Dtype> Rinv_d(dev_allocator, stream, 4);
144 MLCommon::device_buffer<Dtype> R1Qt_d(dev_allocator, stream, 2 * trend_len);
145 MLCommon::device_buffer<int> dev_info_d(dev_allocator, stream, 1);
146
147 // Prepare A
148 for ( int i = 0; i < trend_len; ++ i ) {
149 A_h[ i ] = (Dtype)1.;
150 A_h[trend_len + i ] = (Dtype)( i + 1);
151 }
152 MLCommon::updateDevice (A_d. data (), A_h.data(), 2 * trend_len, stream);
153
154 CUSOLVER_CHECK (MLCommon::LinAlg::cusolverDngeqrf_bufferSize<Dtype>(
155 cusolver_h, trend_len, 2, A_d. data (), 2, &geqrf_buffer));
156
157 CUSOLVER_CHECK (MLCommon::LinAlg::cusolverDnorgqr_bufferSize<Dtype>(
158 cusolver_h, trend_len, 2, 2, A_d. data (), 2, tau_d. data (), &orgqr_buffer));
159
160 lwork_size = geqrf_buffer > orgqr_buffer ? geqrf_buffer : orgqr_buffer;
161 MLCommon::device_buffer<Dtype> lwork_d(dev_allocator, stream, lwork_size);
162
163 // QR decomposition of A
164 CUSOLVER_CHECK (MLCommon::LinAlg::cusolverDngeqrf<Dtype>(
165 cusolver_h, trend_len, 2, A_d. data (), trend_len, tau_d. data (),
166 lwork_d. data (), lwork_size, dev_info_d. data (), stream));
167
168 // Single thread kenrel to inverse R
169 RinvKernel<Dtype><<<1, 1, 0, stream>>>(A_d. data (), Rinv_d. data (), trend_len);
170
171 // R1QT = inv(R)*transpose(Q)
172 CUSOLVER_CHECK (MLCommon::LinAlg::cusolverDnorgqr<Dtype>(
173 cusolver_h, trend_len, 2, 2, A_d. data (), trend_len, tau_d. data (),
174 lwork_d. data (), lwork_size, dev_info_d. data (), stream));
175
176 CUBLAS_CHECK (MLCommon::LinAlg::cublasgemm<Dtype>(
177 cublas_h, CUBLAS_OP_N, CUBLAS_OP_T, 2, trend_len, 2, &one, Rinv_d. data (), 2,
178 A_d. data (), trend_len, &zero, R1Qt_d. data (), 2, stream));
179
180 batched_ls_solver_kernel<Dtype>
181 <<< GET_NUM_BLOCKS (batch_size), GET_THREADS_PER_BLOCK (batch_size), 0,
182 stream>>>(data, R1Qt_d. data (), batch_size, trend_len, level, trend);
183 }

185 template < typename Dtype>
186 void stl_decomposition_gpu ( const ML::cumlHandle_impl &handle, const Dtype *ts,
187 int n , int batch_size, int frequency,
188 int start_periods, Dtype *start_level,
189 Dtype *start_trend, Dtype *start_season,
190 ML::SeasonalType seasonal) {
191 cudaStream_t stream = handle. getStream ();
192 cublasHandle_t cublas_h = handle. getCublasHandle ();
193 std::shared_ptr<MLCommon::deviceAllocator> dev_allocator =
194 handle. getDeviceAllocator ();
195
196 const int end = start_periods * frequency;
197 const int filter_size = (frequency / 2) * 2 + 1;
198 const int trend_len = end - filter_size + 1;
199
200 // Set filter
201 std::vector<Dtype> filter_h(filter_size, 1. / frequency);
202 if (frequency % 2 == 0) {
203 filter_h.front() /= 2;
204 filter_h.back() /= 2;
205 }
206
207 MLCommon::device_buffer<Dtype> filter_d(dev_allocator, stream, filter_size);
208 MLCommon::updateDevice (filter_d. data (), filter_h.data(), filter_size, stream);
209
210 // Set Trend
211 MLCommon::device_buffer<Dtype> trend_d(dev_allocator, stream,
212 batch_size * trend_len);
213 conv1d<Dtype>(handle, ts, batch_size, filter_d. data (), filter_size,
214 trend_d. data (), trend_len);
215
216 MLCommon::device_buffer<Dtype> season_d(dev_allocator, stream,
217 batch_size * trend_len);
218
219 const int ts_offset = (filter_size / 2) * batch_size;
220 if (seasonal == ML::SeasonalType::ADDITIVE ) {
221 const Dtype one = 1.;
222 const Dtype minus_one = -1.;
223 CUBLAS_CHECK (MLCommon::LinAlg::cublasgeam<Dtype>(
224 cublas_h, CUBLAS_OP_N, CUBLAS_OP_N, trend_len, batch_size, &one,
225 ts + ts_offset, trend_len, &minus_one, trend_d. data (), trend_len,
226 season_d. data (), trend_len, stream));
227 } else {
228 MLCommon::device_buffer<Dtype> aligned_ts(dev_allocator, stream,
229 batch_size * trend_len);
230 MLCommon::copy (aligned_ts. data (), ts + ts_offset, batch_size * trend_len,
231 stream);
232 MLCommon::LinAlg::eltwiseDivide<Dtype>(season_d. data (), aligned_ts. data (),
233 trend_d. data (),
234 trend_len * batch_size, stream);
235 }
236
237 season_mean (handle, season_d. data (), trend_len, batch_size, start_season,
238 frequency, filter_size / 2, seasonal);
239
240 batched_ls (handle, trend_d. data (), trend_len, batch_size, start_level,
241 start_trend);
242 }

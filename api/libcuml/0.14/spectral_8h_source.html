36 template < typename T>
37 void coo2csr (cusparseHandle_t handle, const int *srcRows, const int *srcCols,
38 const T *srcVals, int nnz, int m , int *dst_offsets, int *dstCols,
39 T *dstVals, std::shared_ptr<deviceAllocator> d_alloc,
40 cudaStream_t stream) {
41 device_buffer<int> dstRows(d_alloc, stream, nnz);
42 CUDA_CHECK (cudaMemcpyAsync(dstRows. data (), srcRows, sizeof ( int ) * nnz,
43 cudaMemcpyDeviceToDevice, stream));
44 CUDA_CHECK (cudaMemcpyAsync(dstCols, srcCols, sizeof ( int ) * nnz,
45 cudaMemcpyDeviceToDevice, stream));
46 auto buffSize = Sparse::cusparsecoosort_bufferSizeExt (
47 handle, m , m , nnz, srcRows, srcCols, stream);
48 device_buffer<char> pBuffer(d_alloc, stream, buffSize);
49 device_buffer<int> P(d_alloc, stream, nnz);
50 CUSPARSE_CHECK (cusparseCreateIdentityPermutation(handle, nnz, P. data ()));
51 Sparse::cusparsecoosortByRow (handle, m , m , nnz, dstRows. data (), dstCols,
52 P. data (), pBuffer. data (), stream);
53 Sparse::cusparsegthr (handle, nnz, srcVals, dstVals, P. data (), stream);
54 Sparse::cusparsecoo2csr (handle, dstRows. data (), nnz, m , dst_offsets, stream);
55 CUDA_CHECK (cudaDeviceSynchronize());
56 }

58 template < typename T>
59 void fit_clusters (cusparseHandle_t handle, int *rows, int *cols, T *vals,
60 int nnz, int n , int n_clusters, float eigen_tol, int *out,
61 std::shared_ptr<deviceAllocator> d_alloc,
62 cudaStream_t stream) {
63 nvgraphHandle_t graphHandle;
64 cudaDataType_t edge_dimT = CUDA_R_32F;
65 NVGRAPH_CHECK (nvgraphCreate(&graphHandle));
66
67 // conversion
68 device_buffer<int> src_offsets(d_alloc, stream, n + 1);
69 device_buffer<int> dst_cols(d_alloc, stream, nnz);
70 device_buffer<T> dst_vals(d_alloc, stream, nnz);
71 coo2csr (handle, rows, cols, vals, nnz, n , src_offsets. data (), dst_cols. data (),
72 dst_vals. data (), d_alloc, stream);
73
74 nvgraphCSRTopology32I_st CSR_input;
75 CSR_input.destination_indices = dst_cols. data ();
76 CSR_input.nedges = nnz;
77 CSR_input.nvertices = n ;
78 CSR_input.source_offsets = src_offsets. data ();
79
80 int weight_index = 0;
81
82 device_buffer<T> eigVals(d_alloc, stream, n_clusters);
83 device_buffer<T> embedding(d_alloc, stream, n * n_clusters);
84
85 CUDA_CHECK (cudaStreamSynchronize(stream));
86
87 // Spectral clustering parameters
88 struct SpectralClusteringParameter clustering_params;
89 clustering_params.n_clusters = n_clusters;
90 clustering_params.n_eig_vects = n_clusters;
91 clustering_params.algorithm = NVGRAPH_BALANCED_CUT_LANCZOS;
92 clustering_params.evs_tolerance = eigen_tol;
93 clustering_params.evs_max_iter = 0;
94 clustering_params.kmean_tolerance = 0.0f;
95 clustering_params.kmean_max_iter = 0;
96
97 nvgraphGraphDescr_t graph;
98 NVGRAPH_CHECK (nvgraphCreateGraphDescr(graphHandle, &graph));
99 NVGRAPH_CHECK (nvgraphSetGraphStructure(graphHandle, graph, ( void *)&CSR_input,
100 NVGRAPH_CSR_32));
101 NVGRAPH_CHECK (nvgraphAllocateEdgeData(graphHandle, graph, 1, &edge_dimT));
102 NVGRAPH_CHECK (
103 nvgraphSetEdgeData(graphHandle, graph, ( void *)dst_vals. data (), 0));
104
105 NVGRAPH_CHECK (nvgraphSpectralClustering(graphHandle, graph, weight_index,
106 &clustering_params, out,
107 eigVals. data (), embedding. data ()));
108
109 NVGRAPH_CHECK (nvgraphDestroyGraphDescr(graphHandle, graph));
110 NVGRAPH_CHECK (nvgraphDestroy(graphHandle));
111 }

255 template < int TPB_X, typename T>
256 __global__ void csr_row_normalize_l1_kernel (
257 const int *ia, // csr row ex_scan (sorted by row)
258 const T *vals, int nnz, // array of values and number of non-zeros
259 int m , // num rows in csr
260 T *result) { // output array
261
262 // row-based matrix 1 thread per row
263 int row = (blockIdx.x * TPB_X ) + threadIdx.x;
264
265 // sum all vals_arr for row and divide each val by sum
266 if (row < m ) {
267 int start_idx = ia[row];
268 int stop_idx = 0;
269 if (row < m - 1) {
270 stop_idx = ia[row + 1];
271 } else
272 stop_idx = nnz;
273
274 T sum = T(0.0);
275 for ( int j = start_idx; j < stop_idx; j ++) {
276 sum = sum + fabs(vals[ j ]);
277 }
278
279 for ( int j = start_idx; j < stop_idx; j ++) {
280 if ( sum != 0.0) {
281 T val = vals[ j ];
282 result[ j ] = val / sum ;
283 } else {
284 result[ j ] = 0.0;
285 }
286 }
287 }
288 }

316 template < int TPB_X = 32, typename T>
317 __global__ void csr_row_normalize_max_kernel (
318 const int *ia, // csr row ind array (sorted by row)
319 const T *vals, int nnz, // array of values and number of non-zeros
320 int m , // num total rows in csr
321 T *result) { // output array
322
323 // row-based matrix 1 thread per row
324 int row = (blockIdx.x * TPB_X ) + threadIdx.x;
325
326 // find max across columns and divide
327 if (row < m ) {
328 int start_idx = ia[row];
329 int stop_idx = 0;
330 if (row < m - 1) {
331 stop_idx = ia[row + 1];
332 } else
333 stop_idx = nnz;
334
335 T max = MIN_FLOAT;
336 for ( int j = start_idx; j < stop_idx; j ++) {
337 if (vals[ j ] > max ) max = vals[ j ];
338 }
339
340 // divide nonzeros in current row by max
341 for ( int j = start_idx; j < stop_idx; j ++) {
342 if ( max != 0.0 && max > MIN_FLOAT) {
343 T val = vals[ j ];
344 result[ j ] = val / max ;
345 } else {
346 result[ j ] = 0.0;
347 }
348 }
349 }
350 }

418 template < typename T, int TPB_X = 32>
419 __global__ void csr_add_calc_row_counts_kernel (
420 const int *a_ind, const int *a_indptr, const T *a_val, int nnz1,
421 const int *b_ind, const int *b_indptr, const T *b_val, int nnz2, int m ,
422 int *out_rowcounts) {
423 // loop through columns in each set of rows and
424 // calculate number of unique cols across both rows
425 int row = (blockIdx.x * TPB_X ) + threadIdx.x;
426
427 if (row < m ) {
428 int a_start_idx = a_ind[row];
429 int a_stop_idx = get_stop_idx (row, m , nnz1, a_ind);
430
431 int b_start_idx = b_ind[row];
432 int b_stop_idx = get_stop_idx (row, m , nnz2, b_ind);
433
438 int max_size = (a_stop_idx - a_start_idx) + (b_stop_idx - b_start_idx);
439
440 int *arr = new int [max_size];
441 int cur_arr_idx = 0;
442 for ( int j = a_start_idx; j < a_stop_idx; j ++) {
443 arr[cur_arr_idx] = a_indptr[ j ];
444 cur_arr_idx++;
445 }
446
447 int arr_size = cur_arr_idx;
448 int final_size = arr_size;
449
450 for ( int j = b_start_idx; j < b_stop_idx; j ++) {
451 int cur_col = b_indptr[ j ];
452 bool found = false ;
453 for ( int k = 0; k < arr_size; k ++) {
454 if (arr[ k ] == cur_col) {
455 found = true ;
456 break ;
457 }
458 }
459
460 if (!found) {
461 final_size++;
462 }
463 }
464
465 out_rowcounts[row] = final_size;
466 atomicAdd(out_rowcounts + m , final_size);
467
468 delete arr;
469 }
470 }

472 template < typename T, int TPB_X = 32>
473 __global__ void csr_add_kernel ( const int *a_ind, const int *a_indptr,
474 const T *a_val, int nnz1, const int *b_ind,
475 const int *b_indptr, const T *b_val, int nnz2,
476 int m , int *out_ind, int *out_indptr,
477 T *out_val) {
478 // 1 thread per row
479 int row = (blockIdx.x * TPB_X ) + threadIdx.x;
480
481 if (row < m ) {
482 int a_start_idx = a_ind[row];
483 int a_stop_idx = get_stop_idx (row, m , nnz1, a_ind);
484
485 int b_start_idx = b_ind[row];
486 int b_stop_idx = get_stop_idx (row, m , nnz2, b_ind);
487
488 int o_idx = out_ind[row];
489
490 int cur_o_idx = o_idx;
491 for ( int j = a_start_idx; j < a_stop_idx; j ++) {
492 out_indptr[cur_o_idx] = a_indptr[ j ];
493 out_val[cur_o_idx] = a_val[ j ];
494 cur_o_idx++;
495 }
496
497 int arr_size = cur_o_idx - o_idx;
498 for ( int j = b_start_idx; j < b_stop_idx; j ++) {
499 int cur_col = b_indptr[ j ];
500 bool found = false ;
501 for ( int k = o_idx; k < o_idx + arr_size; k ++) {
502 // If we found a match, sum the two values
503 if (out_indptr[ k ] == cur_col) {
504 out_val[ k ] += b_val[ j ];
505 found = true ;
506 break ;
507 }
508 }
509
510 // if we didn't find a match, add the value for b
511 if (!found) {
512 out_indptr[o_idx + arr_size] = cur_col;
513 out_val[o_idx + arr_size] = b_val[ j ];
514 arr_size++;
515 }
516 }
517 }
518 }

713 template < typename Index_, int TPB_X = 32>
714 __global__ void weak_cc_label_device (Index_ *labels, const Index_ *row_ind,
715 const Index_ *row_ind_ptr, Index_ nnz,
716 bool *fa, bool *xa, bool * m ,
717 Index_ startVertexId, Index_ batchSize,
718 Index_ N ) {
719 Index_ tid = threadIdx.x + blockIdx.x * TPB_X ;
720
721 if (tid < batchSize && tid + startVertexId < N ) {
722 if (fa[tid + startVertexId]) {
723 fa[tid + startVertexId] = false ;
724 Index_ row_ind_val = row_ind[tid];
725
726 Index_ start = row_ind_val;
727 Index_ ci, cj;
728 bool ci_mod = false ;
729 ci = labels[tid + startVertexId];
730
731 Index_ degree = get_stop_idx (tid, batchSize, nnz, row_ind) - row_ind_val;
732 for (Index_ j = 0; j < degree;
733 j ++) { // TODO: Can't this be calculated from the ex_scan?
734 Index_ j_ind = row_ind_ptr[ start + j ];
735 cj = labels[j_ind];
736 if (ci < cj) {
737 if ( sizeof (Index_) == 4)
738 atomicMin(( int *)(labels + j_ind), ci);
739 else if ( sizeof (Index_) == 8)
740 atomicMin(( long long int *)(labels + j_ind), ci);
741 xa[j_ind] = true ;
742 m [0] = true ;
743 } else if (ci > cj) {
744 ci = cj;
745 ci_mod = true ;
746 }
747 }
748 if (ci_mod) {
749 if ( sizeof (Index_) == 4)
750 atomicMin(( int *)(labels + startVertexId + tid), ci);
751 else if ( sizeof (Index_) == 8)
752 atomicMin(( long long int *)(labels + startVertexId + tid), ci);
753 xa[startVertexId + tid] = true ;
754 m [0] = true ;
755 }
756 }
757 }
758 }

784 template < typename Index_, int TPB_X = 32, typename Lambda>
785 void weak_cc_label_batched (Index_ *labels, const Index_ *row_ind,
786 const Index_ *row_ind_ptr, Index_ nnz, Index_ N ,
787 WeakCCState *state, Index_ startVertexId,
788 Index_ batchSize, cudaStream_t stream,
789 Lambda filter_op) {
790 ASSERT ( sizeof (Index_) == 4 || sizeof (Index_) == 8,
791 "Index_ should be 4 or 8 bytes" );
792
793 bool host_m;
794 bool *host_fa = ( bool *)malloc( sizeof ( bool ) * N );
795 bool *host_xa = ( bool *)malloc( sizeof ( bool ) * N );
796
797 dim3 blocks( ceildiv (batchSize, Index_( TPB_X )));
798 dim3 threads( TPB_X );
799 Index_ MAX_LABEL = std::numeric_limits<Index_>::max ();
800
801 weak_cc_init_label_kernel<Index_, TPB_X><<<blocks, threads, 0, stream>>>(
802 labels, startVertexId, batchSize, MAX_LABEL, filter_op);
803 CUDA_CHECK (cudaPeekAtLastError());
804
805 int n_iters = 0;
806 do {

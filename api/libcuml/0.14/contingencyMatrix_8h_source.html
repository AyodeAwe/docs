35 template < typename T, typename OutT = int >
36 __global__ void devConstructContingencyMatrix ( const T *groundTruth,
37 const T *predicted, int nSamples,
38 OutT *outMat, int outIdxOffset,
39 int outMatWidth) {
40 auto elementId = threadIdx.x + blockDim.x * blockIdx.x;
41 if (elementId < nSamples) {
42 T gt = groundTruth[elementId];
43 T pd = predicted[elementId];
44 auto outputIdx = (gt - outIdxOffset) * outMatWidth + pd - outIdxOffset;
45 myAtomicAdd (outMat + outputIdx, OutT(1));
46 }
47 }

49 template < typename T, typename OutT = int >
50 void computeCMatWAtomics ( const T *groundTruth, const T *predictedLabel,
51 int nSamples, OutT *outMat, int outIdxOffset,
52 int outDimN, cudaStream_t stream) {
53 CUDA_CHECK (cudaFuncSetCacheConfig(devConstructContingencyMatrix<T, OutT>,
54 cudaFuncCachePreferL1));
55 static const int block = 128;
56 auto grid = ceildiv (nSamples, block);
57 devConstructContingencyMatrix<T, OutT><<<grid, block, 0, stream>>>(
58 groundTruth, predictedLabel, nSamples, outMat, outIdxOffset, outDimN);
59 CUDA_CHECK (cudaGetLastError());
60 }

62 template < typename T, typename OutT = int >
63 __global__ void devConstructContingencyMatrixSmem ( const T *groundTruth,
64 const T *predicted,
65 int nSamples, OutT *outMat,
66 int outIdxOffset,
67 int outMatWidth) {
68 extern __shared__ char smem[];
69 auto *sMemMatrix = reinterpret_cast< OutT * > (smem);
70 for ( auto smemIdx = threadIdx.x; smemIdx < outMatWidth * outMatWidth;
71 smemIdx += blockDim.x) {
72 sMemMatrix[smemIdx] = 0;
73 }
74 __syncthreads ();
75 auto elementId = threadIdx.x + blockDim.x * blockIdx.x;
76 if (elementId < nSamples) {
77 T gt = groundTruth[elementId];
78 T pd = predicted[elementId];
79 auto outputIdx = (gt - outIdxOffset) * outMatWidth + pd - outIdxOffset;
80 myAtomicAdd (sMemMatrix + outputIdx, OutT(1));
81 }
82 __syncthreads ();
83 for ( auto smemIdx = threadIdx.x; smemIdx < outMatWidth * outMatWidth;
84 smemIdx += blockDim.x) {
85 myAtomicAdd (outMat + smemIdx, sMemMatrix[smemIdx]);
86 }
87 }

102 template < typename T, typename OutT = int >
103 void contingencyMatrixWSort ( const T *groundTruth, const T *predictedLabel,
104 int nSamples, OutT *outMat, T minLabel, T maxLabel,
105 void *workspace, size_t workspaceSize,
106 cudaStream_t stream) {
107 T *outKeys = reinterpret_cast< T * > (workspace);
108 auto alignedBufferSz = alignTo<size_t>(nSamples * sizeof (T), 256);
109 T *outValue = reinterpret_cast< T * > ((size_t)workspace + alignedBufferSz);
110 void *pWorkspaceCub =
111 reinterpret_cast< void * > ((size_t)workspace + 2 * alignedBufferSz);
112 auto bitsToSort = log2<int>(maxLabel);
113 if (! isPo2 (maxLabel)) ++bitsToSort;
114 // we dont really need perfect sorting, should get by with some sort of
115 // binning-reordering operation
117 CUDA_CHECK (cub::DeviceRadixSort::SortPairs(
118 pWorkspaceCub, workspaceSize, groundTruth, outKeys, predictedLabel,
119 outValue, nSamples, 0, bitsToSort, stream));
120 auto outDimM_N = int(maxLabel - minLabel + 1);
121 computeCMatWAtomics<T, OutT>(outKeys, outValue, nSamples, outMat, minLabel,
122 outDimM_N, stream);
123 }

125 template < typename OutT = int >
126 ContingencyMatrixImplType getImplVersion (OutT outDimN) {
127 int currDevice = 0;
128 int l2CacheSize = 0;
129 // no way to query this from CUDA APIs, value for CC 7.0, 3.0
130 int maxBlocksResidentPerSM = 16;
131 CUDA_CHECK (cudaGetDevice(&currDevice));
132 CUDA_CHECK (
133 cudaDeviceGetAttribute(&l2CacheSize, cudaDevAttrL2CacheSize, currDevice));
134 auto maxSmemPerBlock = getSharedMemPerBlock ();
135 ContingencyMatrixImplType implVersion = IMPL_NONE ;
136 // keeping 8 block per SM to get good utilization
137 // can go higher but reduced L1 size degrades perf
138 int upperLimitSmemAtomics = std::floor(
139 std::sqrt (maxSmemPerBlock / ( sizeof (OutT) * (maxBlocksResidentPerSM / 2))));
140 int upperLimitL2Atomics = std::floor( std::sqrt (l2CacheSize / sizeof (OutT)));
141 if (outDimN <= upperLimitSmemAtomics)
142 implVersion = SMEM_ATOMICS ;
143 else if (outDimN <= upperLimitL2Atomics)
144 implVersion = GLOBAL_ATOMICS ;
145 else
146 implVersion = SORT_AND_GATOMICS ;
147 return implVersion;
148 }

159 template < typename T>
160 void getInputClassCardinality ( const T *groundTruth, const int nSamples,
161 cudaStream_t stream, T &minLabel, T &maxLabel) {
162 thrust::device_ptr<const T> dTrueLabel =
163 thrust::device_pointer_cast(groundTruth);
164 auto min_max = thrust::minmax_element(thrust::cuda::par.on(stream),
165 dTrueLabel, dTrueLabel + nSamples);
166 minLabel = *min_max.first;
167 maxLabel = *min_max.second;
168 }

180 template < typename T, typename OutT = int >
181 size_t getContingencyMatrixWorkspaceSize (
182 int nSamples, const T *groundTruth, cudaStream_t stream,
183 T minLabel = std::numeric_limits<T>::max (),
184 T maxLabel = std::numeric_limits<T>::max ()) {
185 size_t workspaceSize = 0;
186 // below is a redundant computation - can be avoided
187 if (minLabel == std::numeric_limits<T>::max () ||
188 maxLabel == std::numeric_limits<T>::max ()) {
189 getInputClassCardinality<T>(groundTruth, nSamples, stream, minLabel,
190 maxLabel);
191 }
192 auto outDimN = OutT(maxLabel - minLabel + 1);
193 ContingencyMatrixImplType implVersion = getImplVersion<OutT>(outDimN);
194 if (implVersion == SORT_AND_GATOMICS ) {
195 void *pWorkspaceCub = nullptr ;
196 size_t tmpStorageBytes = 0;
197 // bunch of no-op pointers to get workspace size
198 T *pTmpKey, *pTmpValue, *pTmpKeyOut, *pTmpValueOut;
199 CUDA_CHECK (cub::DeviceRadixSort::SortPairs(pWorkspaceCub, tmpStorageBytes,
200 pTmpKey, pTmpValue, pTmpKeyOut,
201 pTmpValueOut, nSamples));
202 auto tmpStagingMemorySize = alignTo<size_t>(nSamples * sizeof (T), 256);
203 tmpStagingMemorySize *= 2;
204 workspaceSize = tmpStagingMemorySize + tmpStorageBytes;
205 }
206 return workspaceSize;
207 }

226 template < typename T, typename OutT = int >
227 void contingencyMatrix ( const T *groundTruth, const T *predictedLabel,
228 int nSamples, OutT *outMat, cudaStream_t stream,
229 void *workspace = nullptr , size_t workspaceSize = 0,
230 T minLabel = std::numeric_limits<T>::max (),
231 T maxLabel = std::numeric_limits<T>::max ()) {
232 // assumptions:
233 // output is not at par with scikit learn - output will be square matrix
234 // always with numRows = numColumns = numOfClassesInTrueLabel
235 // it is also assumed that true labels are monotically increasing
236 // if for some reason groundTruth completely skips some labels
237 // eg: {0,1,2,5} instead of {0,1,2,3}.
238 // Output matrix will still have empty rows for label value {3,4}
239 // Users can use "make_monotonic" to convert their discontinuous input label
240 // range to a monotonically increasing one //
241 // this also serves as way to measure co-occurence/joint counts for NLP tasks which
242 // can be used to then compute pointwise mutual information and mutual information
243 if (minLabel == std::numeric_limits<T>::max () ||
244 maxLabel == std::numeric_limits<T>::max ()) {
245 getInputClassCardinality<T>(groundTruth, nSamples, stream, minLabel,
246 maxLabel);
247 }
248 auto outDimM_N = OutT(maxLabel - minLabel + 1);
249 CUDA_CHECK (
250 cudaMemsetAsync(outMat, 0, sizeof (OutT) * outDimM_N * outDimM_N, stream));
251 ContingencyMatrixImplType implVersion = getImplVersion<OutT>(outDimM_N);
252 switch (implVersion) {
253 case SMEM_ATOMICS :
254 // smem atomics and then single global mem atomics only works
255 // when all label count can fit in smem for a block
256 // helps when GLOBAL_ATOMICS performance blocked by atomic update
257 // serialization -when very less labels ~10 labels
258 computeCMatWSmemAtomics<T, OutT>(groundTruth, predictedLabel, nSamples,
259 outMat, minLabel, outDimM_N, stream);
260 break ;
261 case GLOBAL_ATOMICS :
262 // launch kernel - global atomic ops per (groundTruth,predictedValue) pair
263 computeCMatWAtomics<T, OutT>(groundTruth, predictedLabel, nSamples,
264 outMat, minLabel, outDimM_N, stream);
265 break ;
266 // more L2 thrashing if atomic OPs land in completely different mem
267 // segment - when more labels
268 case SORT_AND_GATOMICS :
269 contingencyMatrixWSort<T, OutT>(groundTruth, predictedLabel, nSamples,
270 outMat, minLabel, maxLabel, workspace,
271 workspaceSize, stream);
272 break ;
273 }
274 }

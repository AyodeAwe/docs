27 // Kernel to perform reductions along the strided dimension
28 // of the matrix, i.e. reduce along columns for row major or reduce along rows
29 // for column major layout
30 template < typename Type, typename MainLambda>
31 __global__ void stridedSummationKernel (Type *dots, const Type *data, int D,
32 int N , Type init, MainLambda main_op) {
33 // Thread reduction
34 Type thread_data = Type(init);
35 int colStart = blockIdx.x * blockDim.x + threadIdx.x;
36 if (colStart < D) {
37 int rowStart = blockIdx.y * blockDim.y + threadIdx.y;
38 int stride = blockDim.y * gridDim.y;
39 for ( int j = rowStart; j < N ; j += stride) {
40 int idx = colStart + j * D;
41 thread_data += main_op(data[idx], j );
42 }
43 }
44
45 // Block reduction
46 extern __shared__ char tmp[]; // One element per thread in block
47 Type *temp = (Type *)tmp; // Cast to desired type
48 int myidx = threadIdx.x + blockDim.x * threadIdx.y;
49 temp[myidx] = thread_data;
50 __syncthreads ();
51 for ( int j = blockDim.y / 2; j > 0; j /= 2) {
52 if (threadIdx.y < j ) temp[myidx] += temp[myidx + j * blockDim.x];
53 __syncthreads ();
54 }
55
56 // Grid reduction
57 if ((colStart < D) && (threadIdx.y == 0))
58 myAtomicAdd (dots + colStart, temp[myidx]);
59 }

61 // Kernel to perform reductions along the strided dimension
62 // of the matrix, i.e. reduce along columns for row major or reduce along rows
63 // for column major layout
64 template < typename InType, typename OutType, typename IdxType,
65 typename MainLambda, typename ReduceLambda>
66 __global__ void stridedReductionKernel (OutType *dots, const InType *data, int D,
67 int N , OutType init, MainLambda main_op,
68 ReduceLambda reduce_op) {
69 // Thread reduction
70 OutType thread_data = init;
71 IdxType colStart = blockIdx.x * blockDim.x + threadIdx.x;
72 if (colStart < D) {
73 IdxType rowStart = blockIdx.y * blockDim.y + threadIdx.y;
74 IdxType stride = blockDim.y * gridDim.y;
75 for (IdxType j = rowStart; j < N ; j += stride) {
76 IdxType idx = colStart + j * D;
77 thread_data = reduce_op(thread_data, main_op(data[idx], j ));
78 }
79 }
80
81 // Block reduction
82 extern __shared__ char tmp[]; // One element per thread in block
83 auto *temp = (OutType *)tmp; // Cast to desired type
84 IdxType myidx = threadIdx.x + ((IdxType)blockDim.x * (IdxType)threadIdx.y);
85 temp[myidx] = thread_data;
86 __syncthreads ();
87 for ( int j = blockDim.y / 2; j > 0; j /= 2) {
88 if (threadIdx.y < j )
89 temp[myidx] = reduce_op(temp[myidx], temp[myidx + j * blockDim.x]);
90 __syncthreads ();
91 }
92
93 // Grid reduction
94 if ((colStart < D) && (threadIdx.y == 0))
95 myAtomicReduce (dots + colStart, temp[myidx], reduce_op);
96 }

125 template < typename InType, typename OutType = InType, typename IdxType = int,
126 typename MainLambda = Nop<InType, IdxType> ,
127 typename ReduceLambda = Sum<OutType> ,
128 typename FinalLambda = Nop<OutType> >
129 void stridedReduction (OutType *dots, const InType *data, IdxType D, IdxType N ,
130 OutType init, cudaStream_t stream, bool inplace = false ,
131 MainLambda main_op = Nop<InType, IdxType> (),
132 ReduceLambda reduce_op = Sum<OutType> (),
133 FinalLambda final_op = Nop<OutType> ()) {
136 if (!inplace)
137 unaryOp (
138 dots, dots, D, [init] __device__(OutType a) { return init; }, stream);
139
140 // Arbitrary numbers for now, probably need to tune
141 const dim3 thrds(32, 16);
142 IdxType elemsPerThread = ceildiv ( N , (IdxType)thrds.y);
143 elemsPerThread = (elemsPerThread > 8) ? 8 : elemsPerThread;
144 const dim3 nblks( ceildiv (D, (IdxType)thrds.x),
145 ceildiv ( N , (IdxType)thrds.y * elemsPerThread));
146 const size_t shmemSize = sizeof (OutType) * thrds.x * thrds.y;
147
150 if (std::is_same<ReduceLambda, Sum<OutType> >::value &&
151 std::is_same<InType, OutType>::value)
152 stridedSummationKernel<InType>
153 <<<nblks, thrds, shmemSize, stream>>>(dots, data, D, N , init, main_op);
154 else
155 stridedReductionKernel<InType, OutType, IdxType>
156 <<<nblks, thrds, shmemSize, stream>>>(dots, data, D, N , init, main_op,
157 reduce_op);
158
161 // Perform final op on output data
162 if (!std::is_same<FinalLambda, Nop<OutType> >::value)
163 unaryOp (dots, dots, D, final_op, stream);
164 }

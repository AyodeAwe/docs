46 template < typename math_t>
47 void linearRegLossGrads (math_t *input, int n_rows, int n_cols,
48 const math_t *labels, const math_t *coef, math_t *grads,
49 penalty pen, math_t alpha, math_t l1_ratio,
50 cublasHandle_t cublas_handle,
51 std::shared_ptr<deviceAllocator> allocator,
52 cudaStream_t stream) {
53 device_buffer<math_t> labels_pred(allocator, stream, n_rows);
54
55 linearRegH (input, n_rows, n_cols, coef, labels_pred. data (), math_t(0),
56 cublas_handle, stream);
57 LinAlg::subtract (labels_pred. data (), labels_pred. data (), labels, n_rows,
58 stream);
59 Matrix::matrixVectorBinaryMult (input, labels_pred. data (), n_rows, n_cols,
60 false , false , stream);
61
62 Stats::mean (grads, input, n_cols, n_rows, false , false , stream);
63 LinAlg::scalarMultiply (grads, grads, math_t(2), n_cols, stream);
64
65 device_buffer<math_t> pen_grads(allocator, stream, 0);
66
67 if (pen != penalty::NONE ) pen_grads. resize (n_cols, stream);
68
69 if (pen == penalty::L1 ) {
70 lassoGrad (pen_grads. data (), coef, n_cols, alpha, stream);
71 } else if (pen == penalty::L2 ) {
72 ridgeGrad (pen_grads. data (), coef, n_cols, alpha, stream);
73 } else if (pen == penalty::ELASTICNET ) {
74 elasticnetGrad (pen_grads. data (), coef, n_cols, alpha, l1_ratio, stream);
75 }
76
77 if (pen != penalty::NONE ) {
78 LinAlg::add (grads, grads, pen_grads. data (), n_cols, stream);
79 }
80 }

82 template < typename math_t>
83 void linearRegLoss (math_t *input, int n_rows, int n_cols, const math_t *labels,
84 const math_t *coef, math_t *loss, penalty pen, math_t alpha,
85 math_t l1_ratio, cublasHandle_t cublas_handle,
86 std::shared_ptr<deviceAllocator> allocator,
87 cudaStream_t stream) {
88 device_buffer<math_t> labels_pred(allocator, stream, n_rows);
89
90 linearRegH (input, n_rows, n_cols, coef, labels_pred. data (), math_t(0),
91 cublas_handle, stream);
92
93 LinAlg::subtract (labels_pred. data (), labels, labels_pred. data (), n_rows,
94 stream);
95 Matrix::power (labels_pred. data (), n_rows, stream);
96 Stats::mean (loss, labels_pred. data (), 1, n_rows, false , false , stream);
97
98 device_buffer<math_t> pen_val(allocator, stream, 0);
99
100 if (pen != penalty::NONE ) pen_val. resize (1, stream);
101
102 if (pen == penalty::L1 ) {
103 lasso (pen_val. data (), coef, n_cols, alpha, stream);
104 } else if (pen == penalty::L2 ) {
105 ridge (pen_val. data (), coef, n_cols, alpha, stream);
106 } else if (pen == penalty::ELASTICNET ) {
107 elasticnet (pen_val. data (), coef, n_cols, alpha, l1_ratio, stream);
108 }
109
110 if (pen != penalty::NONE ) {
111 LinAlg::add (loss, loss, pen_val. data (), 1, stream);
112 }
113 }

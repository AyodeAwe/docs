154 DI void next (uint32_t& ret) {
155 uint32_t MWC;
156 z = 36969 * (z & 65535) + (z >> 16);
157 w = 18000 * (w & 65535) + (w >> 16);
158 MWC = ((z << 16) + w);
159 jsr ^= (jsr << 17);
160 jsr ^= (jsr >> 13);
161 jsr ^= (jsr << 5);
162 jcong = 69069 * jcong + 1234567;
163 MWC = ((MWC ^ jcong) + jsr);
164 ret = MWC;

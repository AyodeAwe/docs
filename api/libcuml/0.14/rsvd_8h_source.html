57 template < typename math_t>
58 void rsvdFixedRank (math_t *M, int n_rows, int n_cols, math_t *&S_vec,
59 math_t *&U, math_t *&V, int k , int p, bool use_bbt,
60 bool gen_left_vec, bool gen_right_vec, bool use_jacobi,
61 math_t tol, int max_sweeps, cusolverDnHandle_t cusolverH,
62 cublasHandle_t cublasH, cudaStream_t stream,
63 std::shared_ptr<deviceAllocator> allocator) {
64 // All the notations are following Algorithm 4 & 5 in S. Voronin's paper:
65 // https://arxiv.org/abs/1502.05366
66
67 int m = n_rows, n = n_cols;
68 int l =
69 k + p; // Total number of singular values to be computed before truncation
70 int q = 2; // Number of power sampling counts
71 int s = 1; // Frequency controller for QR decomposition during power sampling
72 // scheme. s = 1: 2 QR per iteration; s = 2: 1 QR per iteration; s
73 // > 2: less frequent QR
74
75 const math_t alpha = 1.0, beta = 0.0;
76
77 // Build temporary U, S, V matrices
78 device_buffer<math_t> S_vec_tmp(allocator, stream, l);
79 CUDA_CHECK (cudaMemsetAsync(S_vec_tmp. data (), 0, sizeof (math_t) * l, stream));
80
81 // build random matrix
82 device_buffer<math_t> RN(allocator, stream, n * l);
83 Random::Rng rng(484);
84 rng. normal (RN. data (), n * l, math_t(0.0), alpha, stream);
85
86 // multiply to get matrix of random samples Y
87 device_buffer<math_t> Y(allocator, stream, m * l);
88 gemm (M, m , n , RN. data (), Y. data (), m , l, CUBLAS_OP_N, CUBLAS_OP_N, alpha,
89 beta, cublasH, stream);
90
91 // now build up (M M^T)^q R
92 device_buffer<math_t> Z (allocator, stream, n * l);
93 device_buffer<math_t> Yorth(allocator, stream, m * l);
94 device_buffer<math_t> Zorth(allocator, stream, n * l);
95 CUDA_CHECK (cudaMemsetAsync( Z .data(), 0, sizeof (math_t) * n * l, stream));
96 CUDA_CHECK (cudaMemsetAsync(Yorth. data (), 0, sizeof (math_t) * m * l, stream));
97 CUDA_CHECK (cudaMemsetAsync(Zorth. data (), 0, sizeof (math_t) * n * l, stream));
98
99 // power sampling scheme
100 for ( int j = 1; j < q; j ++) {
101 if ((2 * j - 2) % s == 0) {
102 qrGetQ (Y. data (), Yorth. data (), m , l, cusolverH, stream, allocator);
103 gemm (M, m , n , Yorth. data (), Z .data(), n , l, CUBLAS_OP_T, CUBLAS_OP_N,
104 alpha, beta, cublasH, stream);
105 } else {
106 gemm (M, m , n , Y. data (), Z .data(), n , l, CUBLAS_OP_T, CUBLAS_OP_N, alpha,
107 beta, cublasH, stream);
108 }
109
110 if ((2 * j - 1) % s == 0) {
111 qrGetQ ( Z .data(), Zorth. data (), n , l, cusolverH, stream, allocator);
112 gemm (M, m , n , Zorth. data (), Y. data (), m , l, CUBLAS_OP_N, CUBLAS_OP_N,
113 alpha, beta, cublasH, stream);
114 } else {
115 gemm (M, m , n , Z .data(), Y. data (), m , l, CUBLAS_OP_N, CUBLAS_OP_N, alpha,
116 beta, cublasH, stream);
117 }
118 }
119
120 // orthogonalize on exit from loop to get Q
121 device_buffer<math_t> Q(allocator, stream, m * l);
122 CUDA_CHECK (cudaMemsetAsync(Q. data (), 0, sizeof (math_t) * m * l, stream));
123 qrGetQ (Y. data (), Q. data (), m , l, cusolverH, stream, allocator);
124
125 // either QR of B^T method, or eigendecompose BB^T method
126 if (!use_bbt) {
127 // form Bt = Mt*Q : nxm * mxl = nxl
128 device_buffer<math_t> Bt(allocator, stream, n * l);
129 CUDA_CHECK (cudaMemsetAsync(Bt. data (), 0, sizeof (math_t) * n * l, stream));
130 gemm (M, m , n , Q. data (), Bt. data (), n , l, CUBLAS_OP_T, CUBLAS_OP_N, alpha,
131 beta, cublasH, stream);
132
133 // compute QR factorization of Bt
134 // M is mxn ; Q is mxn ; R is min(m,n) x min(m,n) */
135 device_buffer<math_t> Qhat(allocator, stream, n * l);
136 CUDA_CHECK (cudaMemsetAsync(Qhat. data (), 0, sizeof (math_t) * n * l, stream));
137 device_buffer<math_t> Rhat(allocator, stream, l * l);
138 CUDA_CHECK (cudaMemsetAsync(Rhat. data (), 0, sizeof (math_t) * l * l, stream));
139 qrGetQR (Bt. data (), Qhat. data (), Rhat. data (), n , l, cusolverH, stream,
140 allocator);
141
142 // compute SVD of Rhat (lxl)
143 device_buffer<math_t> Uhat(allocator, stream, l * l);
144 CUDA_CHECK (cudaMemsetAsync(Uhat. data (), 0, sizeof (math_t) * l * l, stream));
145 device_buffer<math_t> Vhat(allocator, stream, l * l);
146 CUDA_CHECK (cudaMemsetAsync(Vhat. data (), 0, sizeof (math_t) * l * l, stream));
147 if (use_jacobi)
148 svdJacobi (Rhat. data (), l, l, S_vec_tmp. data (), Uhat. data (), Vhat. data (),
149 true , true , tol, max_sweeps, cusolverH, stream, allocator);
150 else
151 svdQR (Rhat. data (), l, l, S_vec_tmp. data (), Uhat. data (), Vhat. data (), true ,
152 true , true , cusolverH, cublasH, allocator, stream);
153 Matrix::sliceMatrix (S_vec_tmp. data (), 1, l, S_vec, 0, 0, 1, k ,
154 stream); // First k elements of S_vec
155
156 // Merge step 14 & 15 by calculating U = Q*Vhat[:,1:k] mxl * lxk = mxk
157 if (gen_left_vec) {
158 gemm (Q. data (), m , l, Vhat. data (), U, m ,
159 k /*used to be l and needs slicing*/ , CUBLAS_OP_N, CUBLAS_OP_N,
160 alpha, beta, cublasH, stream);
161 }
162
163 // Merge step 14 & 15 by calculating V = Qhat*Uhat[:,1:k] nxl * lxk = nxk
164 if (gen_right_vec) {
165 gemm (Qhat. data (), n , l, Uhat. data (), V, n ,
166 k /*used to be l and needs slicing*/ , CUBLAS_OP_N, CUBLAS_OP_N,
167 alpha, beta, cublasH, stream);
168 }
169 } else {
170 // build the matrix B B^T = Q^T M M^T Q column by column
171 // Bt = M^T Q ; nxm * mxk = nxk
172 device_buffer<math_t> B(allocator, stream, n * l);
173 gemm (Q. data (), m , l, M, B. data (), l, n , CUBLAS_OP_T, CUBLAS_OP_N, alpha,
174 beta, cublasH, stream);
175
176 device_buffer<math_t> BBt(allocator, stream, l * l);
177 gemm (B. data (), l, n , B. data (), BBt. data (), l, l, CUBLAS_OP_N, CUBLAS_OP_T,
178 alpha, beta, cublasH, stream);
179
180 // compute eigendecomposition of BBt
181 device_buffer<math_t> Uhat(allocator, stream, l * l);
182 CUDA_CHECK (cudaMemsetAsync(Uhat. data (), 0, sizeof (math_t) * l * l, stream));
183 device_buffer<math_t> Uhat_dup(allocator, stream, l * l);
184 CUDA_CHECK (
185 cudaMemsetAsync(Uhat_dup. data (), 0, sizeof (math_t) * l * l, stream));
186 Matrix::copyUpperTriangular (BBt. data (), Uhat_dup. data (), l, l, stream);
187 if (use_jacobi)
188 eigJacobi (Uhat_dup. data (), l, l, Uhat. data (), S_vec_tmp. data (), cusolverH,
189 stream, allocator, tol, max_sweeps);
190 else
191 eigDC (Uhat_dup. data (), l, l, Uhat. data (), S_vec_tmp. data (), cusolverH,
192 stream, allocator);
193 Matrix::seqRoot (S_vec_tmp. data (), l, stream);
194 Matrix::sliceMatrix (S_vec_tmp. data (), 1, l, S_vec, 0, p, 1, l,
195 stream); // Last k elements of S_vec
196 Matrix::colReverse (S_vec, 1, k , stream);
197
198 // Merge step 14 & 15 by calculating U = Q*Uhat[:,(p+1):l] mxl * lxk = mxk
199 if (gen_left_vec) {
200 gemm (Q. data (), m , l, Uhat. data () + p * l, U, m , k , CUBLAS_OP_N,
201 CUBLAS_OP_N, alpha, beta, cublasH, stream);
202 Matrix::colReverse (U, m , k , stream);
203 }
204
205 // Merge step 14 & 15 by calculating V = B^T Uhat[:,(p+1):l] *
206 // Sigma^{-1}[(p+1):l, (p+1):l] nxl * lxk * kxk = nxk
207 if (gen_right_vec) {
208 device_buffer<math_t> Sinv(allocator, stream, k * k );
209 CUDA_CHECK (
210 cudaMemsetAsync(Sinv. data (), 0, sizeof (math_t) * k * k , stream));
211 device_buffer<math_t> UhatSinv(allocator, stream, l * k );
212 CUDA_CHECK (
213 cudaMemsetAsync(UhatSinv. data (), 0, sizeof (math_t) * l * k , stream));
214 Matrix::reciprocal (S_vec_tmp. data (), l, stream);
215 Matrix::initializeDiagonalMatrix (S_vec_tmp. data () + p, Sinv. data (), k , k ,
216 stream);
217
218 gemm (Uhat. data () + p * l, l, k , Sinv. data (), UhatSinv. data (), l, k ,
219 CUBLAS_OP_N, CUBLAS_OP_N, alpha, beta, cublasH, stream);
220 gemm (B. data (), l, n , UhatSinv. data (), V, n , k , CUBLAS_OP_T, CUBLAS_OP_N,
221 alpha, beta, cublasH, stream);
222 Matrix::colReverse (V, n , k , stream);
223 }
224 }
225 }

25 // Kernel (based on norm.h) to perform reductions along the coalesced dimension
26 // of the matrix, i.e. reduce along rows for row major or reduce along columns
27 // for column major layout. Kernel does an inplace reduction adding to original
28 // values of dots.
29 template < typename InType, typename OutType, typename IdxType, int TPB,
30 typename MainLambda, typename ReduceLambda, typename FinalLambda>
31 __global__ void coalescedReductionKernel (OutType *dots, const InType *data,
32 int D, int N , OutType init,
33 MainLambda main_op,
34 ReduceLambda reduce_op,
35 FinalLambda final_op,
36 bool inplace = false ) {
37 typedef cub::BlockReduce<OutType, TPB> BlockReduce ;
38 __shared__ typename BlockReduce::TempStorage temp_storage ;
39 OutType thread_data = init;
40 IdxType rowStart = blockIdx.x * D;
41 for (IdxType i = threadIdx.x; i < D; i += TPB) {
42 IdxType idx = rowStart + i ;
43 thread_data = reduce_op(thread_data, main_op(data[idx], i ));
44 }
45 OutType acc = BlockReduce ( temp_storage ).Reduce(thread_data, reduce_op);
46 if (threadIdx.x == 0) {
47 if (inplace) {
48 dots[blockIdx.x] = final_op(reduce_op(dots[blockIdx.x], acc));
49 } else {
50 dots[blockIdx.x] = final_op(acc);
51 }
52 }
53 }

82 template < typename InType, typename OutType = InType, typename IdxType = int,
83 typename MainLambda = Nop<InType, IdxType> ,
84 typename ReduceLambda = Sum<OutType> ,
85 typename FinalLambda = Nop<OutType> >
86 void coalescedReduction (OutType *dots, const InType *data, int D, int N ,
87 OutType init, cudaStream_t stream, bool inplace = false ,
88 MainLambda main_op = Nop<InType, IdxType> (),
89 ReduceLambda reduce_op = Sum<OutType> (),
90 FinalLambda final_op = Nop<OutType> ()) {
91 // One block per reduction
92 // Efficient only for large leading dimensions
93 if (D <= 32) {
94 coalescedReductionKernel<InType, OutType, IdxType, 32>
95 <<< N , 32, 0, stream>>>(dots, data, D, N , init, main_op, reduce_op,
96 final_op, inplace);
97 } else if (D <= 64) {
98 coalescedReductionKernel<InType, OutType, IdxType, 64>
99 <<< N , 64, 0, stream>>>(dots, data, D, N , init, main_op, reduce_op,
100 final_op, inplace);
101 } else if (D <= 128) {
102 coalescedReductionKernel<InType, OutType, IdxType, 128>
103 <<< N , 128, 0, stream>>>(dots, data, D, N , init, main_op, reduce_op,
104 final_op, inplace);
105 } else {
106 coalescedReductionKernel<InType, OutType, IdxType, 256>
107 <<< N , 256, 0, stream>>>(dots, data, D, N , init, main_op, reduce_op,
108 final_op, inplace);
109 }
110 CUDA_CHECK (cudaPeekAtLastError());
111 }

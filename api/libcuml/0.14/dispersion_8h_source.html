31 template < typename DataT, typename IdxT, int TPB, int ColsPerBlk = 32>
32 __global__ void weightedMeanKernel (DataT *mu, const DataT *data,
33 const IdxT *counts, IdxT D, IdxT N ) {
34 constexpr int RowsPerBlkPerIter = TPB / ColsPerBlk;
35 IdxT thisColId = threadIdx.x % ColsPerBlk;
36 IdxT thisRowId = threadIdx.x / ColsPerBlk;
37 IdxT colId = thisColId + ((IdxT)blockIdx.y * ColsPerBlk);
38 IdxT rowId = thisRowId + ((IdxT)blockIdx.x * RowsPerBlkPerIter);
39 DataT thread_data = DataT(0);
40 const IdxT stride = RowsPerBlkPerIter * gridDim.x;
41 __shared__ DataT smu[ColsPerBlk];
42 if (threadIdx.x < ColsPerBlk) smu[threadIdx.x] = DataT(0);
43 for (IdxT i = rowId; i < N ; i += stride) {
44 thread_data +=
45 (colId < D) ? data[ i * D + colId] * (DataT)counts[ i ] : DataT(0);
46 }
47 __syncthreads ();
48 myAtomicAdd (smu + thisColId, thread_data);
49 __syncthreads ();
50 if (threadIdx.x < ColsPerBlk && colId < D)
51 myAtomicAdd (mu + colId, smu[thisColId]);
52 }

54 template < typename DataT, typename IdxT, int TPB>
55 __global__ void dispersionKernel (DataT *result, const DataT *clusters,
56 const IdxT *clusterSizes, const DataT *mu,
57 IdxT dim, IdxT nClusters) {
58 IdxT tid = threadIdx.x + blockIdx.x * blockDim.x;
59 IdxT len = dim * nClusters;
60 IdxT stride = blockDim.x * gridDim.x;
61 DataT sum = DataT(0);
62 for (; tid < len; tid += stride) {
63 IdxT col = tid % dim;
64 IdxT row = tid / dim;
65 DataT diff = clusters[tid] - mu[col];
66 sum += diff * diff * DataT(clusterSizes[row]);
67 }
68 typedef cub::BlockReduce<DataT, TPB> BlockReduce ;
69 __shared__ typename BlockReduce::TempStorage temp_storage ;
70 __syncthreads ();
71 auto acc = BlockReduce ( temp_storage ).Sum( sum );
72 __syncthreads ();
73 if (threadIdx.x == 0) myAtomicAdd (result, acc);
74 }

95 template < typename DataT, typename IdxT = int , int TPB = 256>
96 DataT dispersion ( const DataT *centroids, const IdxT *clusterSizes,
97 DataT *globalCentroid, IdxT nClusters, IdxT nPoints, IdxT dim,
98 std::shared_ptr<deviceAllocator> allocator,
99 cudaStream_t stream) {
100 static const int RowsPerThread = 4;
101 static const int ColsPerBlk = 32;
102 static const int RowsPerBlk = (TPB / ColsPerBlk) * RowsPerThread;
103 dim3 grid( ceildiv (nPoints, (IdxT)RowsPerBlk), ceildiv (dim, (IdxT)ColsPerBlk));
104 device_buffer<DataT> mean (allocator, stream);
105 device_buffer<DataT> result(allocator, stream, 1);
106 DataT *mu = globalCentroid;
107 if (globalCentroid == nullptr ) {
108 mean .resize(dim, stream);
109 mu = mean .data();
110 }
111 CUDA_CHECK (cudaMemsetAsync(mu, 0, sizeof (DataT) * dim, stream));
112 CUDA_CHECK (cudaMemsetAsync(result. data (), 0, sizeof (DataT), stream));
113 weightedMeanKernel<DataT, IdxT, TPB, ColsPerBlk>
114 <<<grid, TPB, 0, stream>>>(mu, centroids, clusterSizes, dim, nClusters);
115 CUDA_CHECK (cudaGetLastError());
116 DataT ratio = DataT(1) / DataT(nPoints);
117 LinAlg::scalarMultiply (mu, mu, ratio , dim, stream);
118 // finally, compute the dispersion
119 constexpr int ItemsPerThread = 4;
120 int nblks = ceildiv<int>(dim * nClusters, TPB * ItemsPerThread);
121 dispersionKernel<DataT, IdxT, TPB><<<nblks, TPB, 0, stream>>>(
122 result. data (), centroids, clusterSizes, mu, dim, nClusters);
123 CUDA_CHECK (cudaGetLastError());
124 DataT h_result;
125 updateHost (&h_result, result. data (), 1, stream);
126 CUDA_CHECK (cudaStreamSynchronize(stream));
127 return sqrt (h_result);
128 }

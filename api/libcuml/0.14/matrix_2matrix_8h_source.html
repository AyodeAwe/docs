50 void copyRows ( const m_t *in, int n_rows, int n_cols, m_t *out,
51 const int *indices, int n_rows_indices, cudaStream_t stream,
52 bool rowMajor = false ) {
53 if (rowMajor) {
54 ASSERT ( false , "matrix.h: row major is not supported yet!" );

57 auto size = n_rows_indices * n_cols;
58 auto counting = thrust::make_counting_iterator<int>(0);
59
60 thrust::for_each(thrust::cuda::par.on(stream), counting, counting + size,
61 [=] __device__( int idx) {
62 int row = idx % n_rows_indices;
63 int col = idx / n_rows_indices;
64
65 out[col * n_rows_indices + row] =
66 in[col * n_rows + indices[row]];
67 });

120 template < typename m_t>
121 void colReverse (m_t *inout, int n_rows, int n_cols, cudaStream_t stream) {
122 auto n = n_cols;
123 auto m = n_rows;
124 auto size = n_rows * n_cols;
125 auto d_q = inout;
126 auto d_q_reversed = inout;
127 auto counting = thrust::make_counting_iterator<int>(0);
128
129 thrust::for_each(thrust::cuda::par.on(stream), counting,
130 counting + (size / 2), [=] __device__( int idx) {
131 int dest_row = idx % m ;
132 int dest_col = idx / m ;
133 int src_row = dest_row;
134 int src_col = ( n - dest_col) - 1;
135 m_t temp = (m_t)d_q_reversed[idx];
136 d_q_reversed[idx] = d_q[src_col * m + src_row];
137 d_q[src_col * m + src_row] = temp;
138 });
139 }

223 int x2, int y2) {
224 int idx = threadIdx.x + blockDim.x * blockIdx.x;

229 dst_d[idx] = src_d[is + js * m ];

265 int k ) {
266 int idx = threadIdx.x + blockDim.x * blockIdx.x;
267 int m = n_rows, n = n_cols;

269 int i = idx % m , j = idx / m ;
270 if ( i < k && j < k && j >= i ) {
271 dst[ i + j * k ] = src[idx];

43 template < typename T>
44 void launcher ( const cumlHandle &handle, const T *X, int n , int d,
45 const int64_t *knn_indices, const T *knn_dists,
46 MLCommon::Sparse::COO<float> *coo, UMAPParams * params ,
47 T *embedding) {
48 cudaStream_t stream = handle. getStream ();
49
50 ASSERT ( n > params ->n_components,
51 "Spectral layout requires n_samples > n_components" );
52
53 MLCommon::device_buffer<T> tmp_storage(handle. getDeviceAllocator (), stream,
54 n * params ->n_components);
55
56 Spectral::fit_embedding (handle, coo-> rows (), coo-> cols (), coo-> vals (),
57 coo-> nnz , n , params ->n_components,
58 tmp_storage. data ());
59
60 MLCommon::LinAlg::transpose (tmp_storage. data (), embedding, n ,
61 params ->n_components,
62 handle. getImpl (). getCublasHandle (), stream);
63
64 MLCommon::LinAlg::unaryOp<T>(
65 tmp_storage. data (), tmp_storage. data (), n * params ->n_components,
66 [=] __device__(T input) { return fabsf(input); }, stream);
67
68 thrust::device_ptr<T> d_ptr = thrust::device_pointer_cast(tmp_storage. data ());
69 T max = *(thrust::max_element(thrust::cuda::par.on(stream), d_ptr,
70 d_ptr + ( n * params ->n_components)));
71
72 uint64_t seed = params ->random_state;
73
74 // Reuse tmp_storage to add random noise
75 MLCommon::Random::Rng r (seed);
76 r .normal(tmp_storage. data (), n * params ->n_components, 0.0f, 0.0001f, stream);
77
78 MLCommon::LinAlg::unaryOp<T>(
79 embedding, embedding, n * params ->n_components,
80 [=] __device__(T input) { return (10.0 f / max ) * input; }, stream);
81
82 MLCommon::LinAlg::add (embedding, embedding, tmp_storage. data (),
83 n * params ->n_components, stream);
84
85 CUDA_CHECK (cudaPeekAtLastError());
86 }

24 // gatherKernel conditionally copies rows from the source matrix 'in' into the destination matrix 'out' according to a map (or a transformed map)
25 template < typename MatrixIteratorT, typename MapIteratorT,
26 typename StencilIteratorT, int TPB, typename PredicateOp,
27 typename MapTransformOp, typename IndexT = int >
28 __global__ void gatherKernel (MatrixIteratorT in, IndexT D, IndexT N ,
29 MapIteratorT map, StencilIteratorT stencil,
30 MatrixIteratorT out, PredicateOp pred_op,
31 MapTransformOp transform_op) {
32 typedef typename std::iterator_traits<MapIteratorT>::value_type MapValueT;
33 typedef
34 typename std::iterator_traits<StencilIteratorT>::value_type StencilValueT;
35
36 IndexT outRowStart = blockIdx.x * D;
37 MapValueT map_val = map[blockIdx.x];
38 StencilValueT stencil_val = stencil[blockIdx.x];
39
40 bool predicate = pred_op(stencil_val);
41 if (predicate) {
42 IndexT inRowStart = transform_op(map_val) * D;
43 for ( int i = threadIdx.x; i < D; i += TPB) {
44 out[outRowStart + i ] = in[inRowStart + i ];
45 }
46 }
47 }

76 // skip in case of 0 length input
77 if (map_length <= 0 || N <= 0 || D <= 0) return ;

79 // signed integer type for indexing or global offsets
80 typedef int IndexT;

91 MapTransformOpReturnT;

97 typedef typename std::result_of<decltype(pred_op)(StencilValueT)>::type
98 PredicateOpReturnT;
99 static_assert(
100 (std::is_convertible<PredicateOpReturnT, bool>::value),
101 "UnaryPredicateOp's result type must be convertible to bool type" );
102
103 if (D <= 32) {
104 gatherKernel <MatrixIteratorT, MapIteratorT, StencilIteratorT, 32,
105 UnaryPredicateOp, MapTransformOp>
106 <<<map_length, 32, 0, stream>>>(in, D, N , map, stencil, out, pred_op,
107 transform_op);
108 } else if (D <= 64) {

111 <<<map_length, 64, 0, stream>>>(in, D, N , map, stencil, out, pred_op,
112 transform_op);
113 } else if (D <= 128) {

115 UnaryPredicateOp, MapTransformOp>

117 transform_op);
118 } else {
119 gatherKernel <MatrixIteratorT, MapIteratorT, StencilIteratorT, 256,
120 UnaryPredicateOp, MapTransformOp>
121 <<<map_length, 256, 0, stream>>>(in, D, N , map, stencil, out, pred_op,
122 transform_op);

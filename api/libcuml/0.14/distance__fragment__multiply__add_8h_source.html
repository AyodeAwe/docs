37 template < bool enable_sqrt_, typename FragmentB_, typename FragmentCd_,
38 typename FragmentCol_, typename FragmentRow_, typename Lambda_>
39 CUTLASS_DEVICE void multiply (FragmentB_ const &b, FragmentCd_ &d,
40 const int index[FragmentCd_::kElements],
41 FragmentCol_ const &col, FragmentRow_ const &row,
42 Lambda_ fin_op) {
43 FusedDistance fd;
44 int const kReduction = FragmentB_::kElements / FragmentCd_::kElements;
45 int const width = FragmentCd_::kElements / FragmentCol_::kElements;
46 for ( int j = 0; j < FragmentCd_::kElements; ++ j ) {
47 d[ j ] = b[ j * kReduction + 0];
48 for ( int k = 1; k < kReduction; ++ k ) {
49 d[ j ] += b[ j * kReduction + k ];
50 }
51 if (index[ j ] != -1) {
52 fd.fused_distance<enable_sqrt_>(d[ j ], col[ j / width], row[ j % width]);
53 d[ j ] = fin_op(d[ j ], index[ j ]);
54 }
55 }
56 }

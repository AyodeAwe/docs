25 template < typename math_t, int veclen_, typename Lambda>
26 __global__ void reverseKernel (math_t *out, const math_t *in, int nrows ,
27 int ncols , bool rowMajor, bool alongRows, int len,
28 Lambda op) {
29 typedef TxN_t<math_t, veclen_> VecType;
30 int idx = (threadIdx.x + (blockIdx.x * blockDim.x)) * VecType::Ratio;
31 if (idx >= len) return ;
32 int srcIdx, dstIdx;
33 if (!rowMajor && !alongRows) {
34 int srcRow = idx % nrows ;
35 int srcCol = idx / nrows ;
36 int dstRow = srcRow;
37 int dstCol = ncols - srcCol - 1;
38 srcIdx = idx;
39 dstIdx = dstCol * nrows + dstRow;
40 } else if (!rowMajor && alongRows) {
41 int mod = ceildiv ( nrows , 2);
42 int srcRow = idx % mod;
43 int srcCol = idx / mod;
44 int dstRow = nrows - srcRow - VecType::Ratio;
45 int dstCol = srcCol;
46 srcIdx = srcCol * nrows + srcRow;
47 dstIdx = dstCol * nrows + dstRow;
48 } else if (rowMajor && !alongRows) {
49 int mod = ceildiv ( ncols , 2);
50 int srcRow = idx / mod;
51 int srcCol = idx % mod;
52 int dstRow = srcRow;
53 int dstCol = ncols - srcCol - VecType::Ratio;
54 srcIdx = srcCol + srcRow * ncols ;
55 dstIdx = dstCol + dstRow * ncols ;
56 } else {
57 int srcRow = idx / ncols ;
58 int srcCol = idx % ncols ;
59 int dstRow = nrows - srcRow - 1;
60 int dstCol = srcCol;
61 srcIdx = idx;
62 dstIdx = dstCol + dstRow * ncols ;
63 }
64 VecType a, b;
65 a.load(in, srcIdx);
66 b.load(in, dstIdx);
67 // while reversing along coalesced dimension, also reverse the elements
68 if ((rowMajor && !alongRows) || (!rowMajor && alongRows)) {
69 #pragma unroll
70 for ( int i = 0; i < VecType::Ratio; ++ i ) {
71 swap (a.val.data[ i ], a.val.data[VecType::Ratio - i - 1]);
72 swap (b.val.data[ i ], b.val.data[VecType::Ratio - i - 1]);
73 }
74 }
75 #pragma unroll
76 for ( int i = 0; i < VecType::Ratio; ++ i ) {
77 a.val.data[ i ] = op(a.val.data[ i ]);
78 b.val.data[ i ] = op(b.val.data[ i ]);
79 }
80 a.store(out, dstIdx);
81 b.store(out, srcIdx);
82 }

84 template < typename math_t, int veclen_, typename Lambda, int TPB>
85 void reverseImpl (math_t *out, const math_t *in, int nrows , int ncols ,
86 bool rowMajor, bool alongRows, Lambda op,
87 cudaStream_t stream) {
88 int len = alongRows ? ceildiv ( nrows , 2) * ncols : nrows * ceildiv ( ncols , 2);
89 const int nblks = ceildiv (veclen_ ? len / veclen_ : len, TPB);
90 reverseKernel<math_t, veclen_, Lambda><<<nblks, TPB, 0, stream>>>(
91 out, in, nrows , ncols , rowMajor, alongRows, len, op);
92 CUDA_CHECK (cudaPeekAtLastError());
93 }

110 template < typename math_t, typename Lambda = Nop<math_t>, int TPB = 256>
111 void reverse (math_t *out, const math_t *in, int nrows , int ncols , bool rowMajor,
112 bool alongRows, cudaStream_t stream, Lambda op = Nop<math_t> ()) {
113 size_t bytes = (rowMajor ? ncols : nrows ) * sizeof (math_t);
114 if (16 / sizeof (math_t) && bytes % 16 == 0) {
115 reverseImpl <math_t, 16 / sizeof (math_t), Lambda, TPB>(
116 out, in, nrows , ncols , rowMajor, alongRows, op, stream);
117 } else if (8 / sizeof (math_t) && bytes % 8 == 0) {
118 reverseImpl <math_t, 8 / sizeof (math_t), Lambda, TPB>(
119 out, in, nrows , ncols , rowMajor, alongRows, op, stream);
120 } else if (4 / sizeof (math_t) && bytes % 4 == 0) {
121 reverseImpl <math_t, 4 / sizeof (math_t), Lambda, TPB>(
122 out, in, nrows , ncols , rowMajor, alongRows, op, stream);
123 } else if (2 / sizeof (math_t) && bytes % 2 == 0) {
124 reverseImpl <math_t, 2 / sizeof (math_t), Lambda, TPB>(
125 out, in, nrows , ncols , rowMajor, alongRows, op, stream);

128 out, in, nrows , ncols , rowMajor, alongRows, op, stream);

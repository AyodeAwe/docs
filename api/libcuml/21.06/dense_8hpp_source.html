59  // Implemented GEMM as a static method here to improve readability
60  inline static void gemm(const raft::handle_t &handle, const T alpha,
61  const SimpleDenseMat<T> &A, const bool transA,
62  const SimpleDenseMat<T> &B, const bool transB,
63  const T beta, SimpleDenseMat<T> &C,
64  cudaStream_t stream) {
65  int kA = A.n;
66  int kB = B.m;
67 
68  if (transA) {
69  ASSERT(A.n == C.m, "GEMM invalid dims: m");
70  kA = A.m;
71  } else {
72  ASSERT(A.m == C.m, "GEMM invalid dims: m");
73  }
74 
75  if (transB) {
76  ASSERT(B.m == C.n, "GEMM invalid dims: n");
77  kB = B.n;
78  } else {
79  ASSERT(B.n == C.n, "GEMM invalid dims: n");
80  }
81  ASSERT(kA == kB, "GEMM invalid dims: k");
82 
83  if (A.ord == COL_MAJOR && B.ord == COL_MAJOR && C.ord == COL_MAJOR) {
84  raft::linalg::cublasgemm(handle.get_cublas_handle(), // handle
85  transA ? CUBLAS_OP_T : CUBLAS_OP_N, // transA
86  transB ? CUBLAS_OP_T : CUBLAS_OP_N, // transB
87  C.m, C.n, kA, // dimensions m,n,k
88  &alpha, A.data,
89  A.m, // lda
90  B.data, B.m, // ldb
91  &beta, C.data,
92  C.m, // ldc,
93  stream);
94  return;
95  }
96  if (A.ord == ROW_MAJOR) {
97  const SimpleDenseMat<T> Acm(A.data, A.n, A.m, COL_MAJOR);
98  gemm(handle, alpha, Acm, !transA, B, transB, beta, C, stream);
99  return;
100  }
101  if (B.ord == ROW_MAJOR) {
102  const SimpleDenseMat<T> Bcm(B.data, B.n, B.m, COL_MAJOR);
103  gemm(handle, alpha, A, transA, Bcm, !transB, beta, C, stream);
104  return;
105  }
106  if (C.ord == ROW_MAJOR) {
107  SimpleDenseMat<T> Ccm(C.data, C.n, C.m, COL_MAJOR);
108  gemm(handle, alpha, B, !transB, A, !transA, beta, Ccm, stream);
109  return;
110  }
111  }

252 inline T dot(const SimpleVec<T> &u, const SimpleVec<T> &v, T *tmp_dev,
253  cudaStream_t stream) {
254  auto f = [] __device__(const T x, const T y) { return x * y; };
255  raft::linalg::mapThenSumReduce(tmp_dev, u.len, f, stream, u.data, v.data);
256  T tmp_host;
257  raft::update_host(&tmp_host, tmp_dev, 1, stream);
258  cudaStreamSynchronize(stream);
259  return tmp_host;
260 }

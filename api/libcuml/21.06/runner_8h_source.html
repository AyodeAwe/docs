45 template <typename value_idx, typename value_t>
46 struct FixConnectivitiesRedOp {
47  value_idx *colors;
48  value_t *core_dists;
49  value_idx m;
50 
51  FixConnectivitiesRedOp(value_idx *colors_, value_t *core_dists_, value_idx m_)
52  : colors(colors_), core_dists(core_dists_), m(m_){};
53 
54  typedef typename cub::KeyValuePair<value_idx, value_t> KVP;
55  DI void operator()(value_idx rit, KVP *out, const KVP &other) {
56  if (rit < m && other.value < std::numeric_limits<value_t>::max() &&
57  colors[rit] != colors[other.key]) {
58  value_t core_dist_rit = core_dists[rit];
59  value_t core_dist_other =
60  max(core_dist_rit, max(core_dists[other.key], other.value));
61  value_t core_dist_out =
62  max(core_dist_rit, max(core_dists[out->key], out->value));
63 
64  bool smaller = core_dist_other < core_dist_out;
65  out->key = smaller ? other.key : out->key;
66  out->value = smaller ? core_dist_other : core_dist_out;
67  }
68  }
69 
70  DI KVP operator()(value_idx rit, const KVP &a, const KVP &b) {
71  if (rit < m && a.key > -1 && colors[rit] != colors[a.key]) {
72  value_t core_dist_rit = core_dists[rit];
73  value_t core_dist_a = max(core_dist_rit, max(core_dists[a.key], a.value));
74  value_t core_dist_b = max(core_dist_rit, max(core_dists[b.key], b.value));
75 
76  return core_dist_a < core_dist_b ? KVP(a.key, core_dist_a)
77  : KVP(b.key, core_dist_b);
78  }
79 
80  return b;
81  }
82 
83  DI void init(value_t *out, value_t maxVal) { *out = maxVal; }
84  DI void init(KVP *out, value_t maxVal) {
85  out->key = -1;
86  out->value = maxVal;
87  }
88 };

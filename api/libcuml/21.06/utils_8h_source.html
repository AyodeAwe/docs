86 template <typename value_idx, typename value_t>
87 Common::CondensedHierarchy<value_idx, value_t> make_cluster_tree(
88  const raft::handle_t &handle,
89  Common::CondensedHierarchy<value_idx, value_t> &condensed_tree) {
90  auto stream = handle.get_stream();
91  auto thrust_policy = rmm::exec_policy(stream);
92  auto parents = condensed_tree.get_parents();
93  auto children = condensed_tree.get_children();
94  auto lambdas = condensed_tree.get_lambdas();
95  auto sizes = condensed_tree.get_sizes();
96 
97  value_idx cluster_tree_edges = thrust::transform_reduce(
98  thrust_policy, sizes, sizes + condensed_tree.get_n_edges(),
99  [=] __device__(value_idx a) { return a > 1; }, 0,
100  thrust::plus<value_idx>());
101 
102  // remove leaves from condensed tree
103  rmm::device_uvector<value_idx> cluster_parents(cluster_tree_edges, stream);
104  rmm::device_uvector<value_idx> cluster_children(cluster_tree_edges, stream);
105  rmm::device_uvector<value_t> cluster_lambdas(cluster_tree_edges, stream);
106  rmm::device_uvector<value_idx> cluster_sizes(cluster_tree_edges, stream);
107 
108  auto in = thrust::make_zip_iterator(
109  thrust::make_tuple(parents, children, lambdas, sizes));
110 
111  auto out = thrust::make_zip_iterator(
112  thrust::make_tuple(cluster_parents.data(), cluster_children.data(),
113  cluster_lambdas.data(), cluster_sizes.data()));
114 
115  thrust::copy_if(thrust_policy, in, in + (condensed_tree.get_n_edges()), sizes,
116  out, [=] __device__(value_idx a) { return a > 1; });
117 
118  auto n_leaves = condensed_tree.get_n_leaves();
119  thrust::transform(
120  thrust_policy, cluster_parents.begin(), cluster_parents.end(),
121  cluster_parents.begin(),
122  [n_leaves] __device__(value_idx a) { return a - n_leaves; });
123  thrust::transform(
124  thrust_policy, cluster_children.begin(), cluster_children.end(),
125  cluster_children.begin(),
126  [n_leaves] __device__(value_idx a) { return a - n_leaves; });
127 
128  return Common::CondensedHierarchy<value_idx, value_t>(
129  handle, condensed_tree.get_n_leaves(), cluster_tree_edges,
130  condensed_tree.get_n_clusters(), std::move(cluster_parents),
131  std::move(cluster_children), std::move(cluster_lambdas),
132  std::move(cluster_sizes));
133 }

29 inline const char *cublasErr2Str(cublasStatus_t err) {
30  switch (err) {
31  _CUBLAS_ERR_TO_STR(CUBLAS_STATUS_SUCCESS);
32  _CUBLAS_ERR_TO_STR(CUBLAS_STATUS_NOT_INITIALIZED);
33  _CUBLAS_ERR_TO_STR(CUBLAS_STATUS_ALLOC_FAILED);
34  _CUBLAS_ERR_TO_STR(CUBLAS_STATUS_INVALID_VALUE);
35  _CUBLAS_ERR_TO_STR(CUBLAS_STATUS_ARCH_MISMATCH);
36  _CUBLAS_ERR_TO_STR(CUBLAS_STATUS_MAPPING_ERROR);
37  _CUBLAS_ERR_TO_STR(CUBLAS_STATUS_EXECUTION_FAILED);
38  _CUBLAS_ERR_TO_STR(CUBLAS_STATUS_INTERNAL_ERROR);
39  _CUBLAS_ERR_TO_STR(CUBLAS_STATUS_NOT_SUPPORTED);
40  _CUBLAS_ERR_TO_STR(CUBLAS_STATUS_LICENSE_ERROR);
41  default:
42  return "CUBLAS_STATUS_UNKNOWN";
43  };
44 }

49  do { \
50  cublasStatus_t err = call; \
51  ASSERT(err == CUBLAS_STATUS_SUCCESS, \
52  "CUBLAS call='%s' got errorcode=%d err=%s", #call, err, \
53  MLCommon::LinAlg::cublasErr2Str(err)); \
54  } while (0)

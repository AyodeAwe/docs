182  void ucp_isend(ucp_request *req, ucp_ep_h ep_ptr, const void *buf, int size,
183  int tag, ucp_tag_t tag_mask, int rank) const {
184  ucp_tag_t ucp_tag = build_message_tag(rank, tag);
185 
186  CUML_LOG_DEBUG("Sending tag: %ld", ucp_tag);
187 
188  ucs_status_ptr_t send_result = (*(send_func))(
189  ep_ptr, buf, size, ucp_dt_make_contig(1), ucp_tag, send_callback);
190  struct ucx_context *ucp_req = (struct ucx_context *)send_result;
191  if (UCS_PTR_IS_ERR(send_result)) {
192  ASSERT(!UCS_PTR_IS_ERR(send_result),
193  "unable to send UCX data message (%d)\n",
194  UCS_PTR_STATUS(send_result));
199  } else if (UCS_PTR_STATUS(send_result) != UCS_OK) {
205  req->needs_release = true;
206  } else {
207  req->needs_release = false;
208  }
209 
210  req->other_rank = rank;
211  req->is_send_request = true;
212  req->req = ucp_req;
213  }

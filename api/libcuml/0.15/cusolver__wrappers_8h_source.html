27 #define _CUSOLVER_ERR_TO_STR(err) \
28  case err: \
29  return #err;
30 inline const char *cusolverErr2Str(cusolverStatus_t err) {
31  switch (err) {
32  _CUSOLVER_ERR_TO_STR(CUSOLVER_STATUS_SUCCESS);
33  _CUSOLVER_ERR_TO_STR(CUSOLVER_STATUS_NOT_INITIALIZED);
34  _CUSOLVER_ERR_TO_STR(CUSOLVER_STATUS_ALLOC_FAILED);
35  _CUSOLVER_ERR_TO_STR(CUSOLVER_STATUS_INVALID_VALUE);
36  _CUSOLVER_ERR_TO_STR(CUSOLVER_STATUS_ARCH_MISMATCH);
37  _CUSOLVER_ERR_TO_STR(CUSOLVER_STATUS_EXECUTION_FAILED);
38  _CUSOLVER_ERR_TO_STR(CUSOLVER_STATUS_INTERNAL_ERROR);
39  _CUSOLVER_ERR_TO_STR(CUSOLVER_STATUS_MATRIX_TYPE_NOT_SUPPORTED);
40  _CUSOLVER_ERR_TO_STR(CUSOLVER_STATUS_ZERO_PIVOT);
41  _CUSOLVER_ERR_TO_STR(CUSOLVER_STATUS_NOT_SUPPORTED);
42  default:
43  return "CUSOLVER_STATUS_UNKNOWN";
44  };
45 }
46 #undef _CUSOLVER_ERR_TO_STR

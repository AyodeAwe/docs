33 __device__ double atomicMin(double* addr, double val)
34 {
35  unsigned long long int* address_as_ll = reinterpret_cast<unsigned long long int*>(addr);
36  unsigned long long int old = __double_as_longlong(*addr);
37  unsigned long long int assumed;
38 
39  do {
40  assumed = old;
41  old = atomicCAS(
42  address_as_ll, assumed, __double_as_longlong(std::min(val, __longlong_as_double(assumed))));
43 
44  // Note: uses integer comparison to avoid hang in case of NaN (since NaN != NaN)
45  } while (assumed != old);
46 
47  return __longlong_as_double(old);
48 }

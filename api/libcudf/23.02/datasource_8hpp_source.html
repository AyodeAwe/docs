402  public:
408  explicit arrow_io_source(std::string_view arrow_uri)
409  {
410  const std::string uri_start_delimiter = "//";
411  const std::string uri_end_delimiter = "?";
412 
413  arrow::Result<std::shared_ptr<arrow::fs::FileSystem>> result =
414  arrow::fs::FileSystemFromUri(static_cast<std::string>(arrow_uri));
415  CUDF_EXPECTS(result.ok(), "Failed to generate Arrow Filesystem instance from URI.");
416  filesystem = result.ValueOrDie();
417 
418  // Parse the path from the URI
419  size_t start = arrow_uri.find(uri_start_delimiter) == std::string::npos
420  ? 0
421  : arrow_uri.find(uri_start_delimiter) + uri_start_delimiter.size();
422  size_t end = arrow_uri.find(uri_end_delimiter) - start;
423  std::string_view path = arrow_uri.substr(start, end);
424 
425  arrow::Result<std::shared_ptr<arrow::io::RandomAccessFile>> in_stream =
426  filesystem->OpenInputFile(static_cast<std::string>(path).c_str());
427  CUDF_EXPECTS(in_stream.ok(), "Failed to open Arrow RandomAccessFile");
428  arrow_file = in_stream.ValueOrDie();
429  }

1419 ColumnDeviceView* child_columns_to_device_array(ColumnViewIterator child_begin,
1420  ColumnViewIterator child_end,
1421  void* h_ptr,
1422  void* d_ptr)
1423 {
1424  ColumnDeviceView* d_children = detail::align_ptr_for_type<ColumnDeviceView>(d_ptr);
1425  auto num_children = std::distance(child_begin, child_end);
1426  if (num_children > 0) {
1427  // The beginning of the memory must be the fixed-sized ColumnDeviceView
1428  // struct objects in order for d_children to be used as an array.
1429  auto h_column = detail::align_ptr_for_type<ColumnDeviceView>(h_ptr);
1430  auto d_column = d_children;
1431 
1432  // Any child data is assigned past the end of this array: h_end and d_end.
1433  auto h_end = reinterpret_cast<int8_t*>(h_column + num_children);
1434  auto d_end = reinterpret_cast<int8_t*>(d_column + num_children);
1435  std::for_each(child_begin, child_end, [&](auto const& col) {
1436  // inplace-new each child into host memory
1437  new (h_column) ColumnDeviceView(col, h_end, d_end);
1438  h_column++; // advance to next child
1439  // update the pointers for holding this child column's child data
1440  auto col_child_data_size = ColumnDeviceView::extent(col) - sizeof(ColumnDeviceView);
1441  h_end += col_child_data_size;
1442  d_end += col_child_data_size;
1443  });
1444  }
1445  return d_children;
1446 }

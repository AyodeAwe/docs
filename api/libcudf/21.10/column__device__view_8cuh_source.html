1432 {
1433  ColumnDeviceView* d_children = detail::align_ptr_for_type<ColumnDeviceView>(d_ptr);
1434  auto num_children = std::distance(child_begin, child_end);
1435  if (num_children > 0) {
1436  // The beginning of the memory must be the fixed-sized ColumnDeviceView
1437  // struct objects in order for d_children to be used as an array.
1438  auto h_column = detail::align_ptr_for_type<ColumnDeviceView>(h_ptr);
1439  auto d_column = d_children;
1440 
1441  // Any child data is assigned past the end of this array: h_end and d_end.
1442  auto h_end = reinterpret_cast<int8_t*>(h_column + num_children);
1443  auto d_end = reinterpret_cast<int8_t*>(d_column + num_children);
1444  std::for_each(child_begin, child_end, [&](auto const& col) {
1445  // inplace-new each child into host memory
1446  new (h_column) ColumnDeviceView(col, h_end, d_end);
1447  h_column++; // advance to next child
1448  // update the pointers for holding this child column's child data
1449  auto col_child_data_size = ColumnDeviceView::extent(col) - sizeof(ColumnDeviceView);
1450  h_end += col_child_data_size;
1451  d_end += col_child_data_size;
1452  });
1453  }
1454  return d_children;

74 template <typename Rep,
75  Radix Base,
76  typename T,
77  typename cuda::std::enable_if_t<(cuda::std::is_same_v<int32_t, T> &&
78  is_supported_representation_type<Rep>())>* = nullptr>
79 CUDA_HOST_DEVICE_CALLABLE Rep ipow(T exponent)
80 {
81  cudf_assert(exponent >= 0 && "integer exponentiation with negative exponent is not possible.");
82  if (exponent == 0) return static_cast<Rep>(1);
83  auto extra = static_cast<Rep>(1);
84  auto square = static_cast<Rep>(Base);
85  while (exponent > 1) {
86  if (exponent & 1 /* odd */) {
87  extra *= square;
88  exponent -= 1;
89  }
90  exponent /= 2;
91  square *= square;
92  }
93  return square * extra;
94 }

143 template <typename Rep, Radix Rad, typename T>
144 CUDA_HOST_DEVICE_CALLABLE constexpr T shift(T const& val, scale_type const& scale)
145 {
146  if (scale == 0)
147  return val;
148  else if (scale > 0)
149  return right_shift<Rep, Rad>(val, scale);

539  CUDA_HOST_DEVICE_CALLABLE fixed_point<Rep, Rad> rescaled(scale_type scale) const
540  {
541  if (scale == _scale) return *this;
542  Rep const value = detail::shift<Rep, Rad>(_value, scale_type{scale - _scale});
543  return fixed_point<Rep, Rad>{scaled_integer<Rep>{value, scale}};
544  }

549  explicit operator std::string() const
550  {
551  if (_scale < 0) {
552  auto const av = std::abs(_value);
553  int64_t const n = std::pow(10, -_scale);
554  int64_t const f = av % n;
555  auto const num_zeros =
556  std::max(0, (-_scale - static_cast<int32_t>(std::to_string(f).size())));
557  auto const zeros = std::string(num_zeros, '0');
558  auto const sign = _value < 0 ? std::string("-") : std::string();
559  return sign + std::to_string(av / n) + std::string(".") + zeros + std::to_string(av % n);
560  } else {
561  auto const zeros = std::string(_scale, '0');
562  return std::to_string(_value) + zeros;
563  }

606 template <typename Rep, typename T>
607 CUDA_HOST_DEVICE_CALLABLE auto division_overflow(T lhs, T rhs)
608 {
609  return lhs == cuda::std::numeric_limits<Rep>::min() && rhs == -1;
610 }

620 template <typename Rep, typename T>
621 CUDA_HOST_DEVICE_CALLABLE auto multiplication_overflow(T lhs, T rhs)
622 {
623  auto const min = cuda::std::numeric_limits<Rep>::min();
624  auto const max = cuda::std::numeric_limits<Rep>::max();
625  if (rhs > 0)
626  return lhs > max / rhs || lhs < min / rhs;
627  else if (rhs < -1)
628  return lhs > min / rhs || lhs < max / rhs;
629  else
630  return rhs == -1 && lhs == min;
631 }

634 template <typename Rep1, Radix Rad1>
635 CUDA_HOST_DEVICE_CALLABLE fixed_point<Rep1, Rad1> operator+(fixed_point<Rep1, Rad1> const& lhs,
636  fixed_point<Rep1, Rad1> const& rhs)
637 {
638  auto const scale = std::min(lhs._scale, rhs._scale);
639  auto const sum = lhs.rescaled(scale)._value + rhs.rescaled(scale)._value;
640 
641 #if defined(__CUDACC_DEBUG__)

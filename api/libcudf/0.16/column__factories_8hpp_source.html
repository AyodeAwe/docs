310 template <typename B>
311 std::unique_ptr<column> make_fixed_width_column(
312  data_type type,
313  size_type size,
314  B&& null_mask,
315  size_type null_count = cudf::UNKNOWN_NULL_COUNT,
316  cudaStream_t stream = 0,
317  rmm::mr::device_memory_resource* mr = rmm::mr::get_current_device_resource())
318 {
319  CUDF_EXPECTS(is_fixed_width(type), "Invalid, non-fixed-width type.");
320  if (is_timestamp(type)) {
321  return make_timestamp_column(type, size, std::forward<B>(null_mask), null_count, stream, mr);

324  } else if (is_fixed_point(type)) {
325  return make_fixed_point_column(type, size, std::forward<B>(null_mask), null_count, stream, mr);
326  }
327  return make_numeric_column(type, size, std::forward<B>(null_mask), null_count, stream, mr);

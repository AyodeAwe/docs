167 template <>
168 inline std::pair<thrust::host_vector<std::string>, std::vector<bitmask_type>> to_host(column_view c)
169 {
170  auto strings_data = cudf::strings::create_offsets(strings_column_view(c));
171  thrust::host_vector<char> h_chars(strings_data.first);
172  thrust::host_vector<size_type> h_offsets(strings_data.second);
173 
174  // build std::string vector from chars and offsets
175  std::vector<std::string> host_data;
176  host_data.reserve(c.size());
177 
178  // When C++17, replace this loop with std::adjacent_difference()
179  for (size_type idx = 0; idx < c.size(); ++idx) {
180  auto offset = h_offsets[idx];
181  auto length = h_offsets[idx + 1] - offset;
182  host_data.push_back(std::string(h_chars.data() + offset, length));
183  }
184 
185  return {host_data, bitmask_to_host(c)};
186 }

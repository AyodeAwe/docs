112 template <typename From, typename To>
113 struct fixed_width_type_converter {
114  // Are the types same - simply copy elements from [begin, end) to out
115  template <typename FromT = From,
116  typename ToT = To,
117  typename std::enable_if<std::is_same<FromT, ToT>::value, void>::type* = nullptr>
118  ToT operator()(FromT element) const
119  {
120  return element;
121  }
122 
123  // Are the types convertible or can target be constructed from source?
124  template <typename FromT = From,
125  typename ToT = To,
126  typename std::enable_if<!std::is_same<FromT, ToT>::value &&
127  (cudf::is_convertible<FromT, ToT>::value ||
128  std::is_constructible<ToT, FromT>::value),
129  void>::type* = nullptr>
130  ToT operator()(FromT element) const
131  {
132  return static_cast<ToT>(element);
133  }
134 
135  // Convert integral values to timestamps
136  template <
137  typename FromT = From,
138  typename ToT = To,
139  typename std::enable_if<std::is_integral<FromT>::value && cudf::is_timestamp_t<ToT>::value,
140  void>::type* = nullptr>
141  ToT operator()(FromT element) const
142  {
143  return ToT{typename ToT::duration{element}};
144  }
145 };

230 template <typename StringsIterator, typename ValidityIterator>
231 auto make_chars_and_offsets(StringsIterator begin, StringsIterator end, ValidityIterator v)
232 {
233  std::vector<char> chars{};
234  std::vector<int32_t> offsets(1, 0);
235  for (auto str = begin; str < end; ++str) {
236  std::string tmp = (*v++) ? std::string(*str) : std::string{};
237  chars.insert(chars.end(), std::cbegin(tmp), std::cend(tmp));
238  offsets.push_back(offsets.back() + tmp.length());
239  }
240  return std::make_pair(std::move(chars), std::move(offsets));

904  void build_from_nested(std::initializer_list<lists_column_wrapper<T, SourceElementT>> elements,
905  std::vector<bool> const& v)
906  {
907  auto valids = cudf::test::make_counting_transform_iterator(
908  0, [&v](auto i) { return v.empty() ? true : v[i]; });
909 
910  // compute the expected hierarchy and depth
911  auto const hierarchy_and_depth = std::accumulate(
912  elements.begin(),
913  elements.end(),
914  std::pair<column_view, int32_t>{{}, -1},
915  [](auto acc, lists_column_wrapper const& lcw) {
916  return lcw.depth > acc.second ? std::make_pair(lcw.get_view(), lcw.depth) : acc;
917  });
918  column_view expected_hierarchy = hierarchy_and_depth.first;
919  int32_t const expected_depth = hierarchy_and_depth.second;
920 
921  // preprocess columns so that every column_view in 'cols' is an equivalent hierarchy
922  std::vector<std::unique_ptr<column>> stubs;
923  std::vector<column_view> cols;
924  std::tie(cols, stubs) = preprocess_columns(elements, expected_hierarchy, expected_depth);
925 
926  // generate offsets
927  size_type count = 0;
928  std::vector<size_type> offsetv;
929  std::transform(cols.begin(),
930  cols.end(),
931  valids,
932  std::back_inserter(offsetv),
933  [&](cudf::column_view const& col, bool valid) {
934  // nulls are represented as a repeated offset
935  size_type ret = count;
936  if (valid) { count += col.size(); }
937  return ret;
938  });
939  // add the final offset
940  offsetv.push_back(count);
941  auto offsets =
942  cudf::test::fixed_width_column_wrapper<size_type>(offsetv.begin(), offsetv.end()).release();
943 
944  // concatenate them together, skipping children that are null.
945  std::vector<column_view> children;
946  for (size_t idx = 0; idx < cols.size(); idx++) {
947  if (!valids[idx]) { continue; }
948  children.push_back(cols[idx]);
949  }
950  auto data = children.empty() ? cudf::empty_like(expected_hierarchy) : concatenate(children);
951 
952  // increment depth
953  depth = expected_depth + 1;
954 
955  // construct the list column
956  wrapped =
957  make_lists_column(cols.size(),
958  std::move(offsets),
959  std::move(data),
960  v.size() <= 0 ? 0 : cudf::UNKNOWN_NULL_COUNT,
961  v.size() <= 0 ? rmm::device_buffer{0}
962  : cudf::test::detail::make_null_mask(v.begin(), v.end()));
963  }

1028  std::unique_ptr<column> normalize_column(column_view const& col,
1029  column_view const& expected_hierarchy)
1030  {

1032  if (col.type().id() != type_id::LIST) {

1048  std::pair<std::vector<column_view>, std::vector<std::unique_ptr<column>>> preprocess_columns(
1049  std::initializer_list<lists_column_wrapper<T, SourceElementT>> const& elements,
1050  column_view& expected_hierarchy,
1051  int expected_depth)
1052  {
1053  std::vector<std::unique_ptr<column>> stubs;
1054  std::vector<column_view> cols;
1055 
1056  // preprocess the incoming lists.
1057  // - unwrap any "root" lists
1058  // - handle incomplete hierarchies
1059  std::transform(elements.begin(),
1060  elements.end(),
1061  std::back_inserter(cols),
1062  [&](lists_column_wrapper const& l) -> column_view {
1063  // depth mismatch. attempt to normalize the short column.
1064  // this function will also catch if this is a legitmately broken
1065  // set of input
1066  if (l.depth < expected_depth) {
1067  if (l.root) {
1068  // this exception distinguishes between the following two cases:
1069  //
1070  // { {{{1, 2, 3}}}, {} }
1071  // In this case, row 0 is a List<List<List<int>>>, whereas row 1 is
1072  // just a List<> which is an apparent mismatch. However, because row 1
1073  // is empty we will allow that to semantically mean
1074  // "a List<List<List<int>>> that's empty at the top level"
1075  //
1076  // { {{{1, 2, 3}}}, {4, 5, 6} }
1077  // In this case, row 1 is a a concrete List<int> with actual values. There
1078  // is no way to rectify the differences so we will treat it as a true
1079  // column mismatch.
1080  CUDF_EXPECTS(l.wrapped->size() == 0, "Mismatch in column types!");
1081  stubs.push_back(empty_like(expected_hierarchy));
1082  } else {
1083  stubs.push_back(normalize_column(l.get_view(), expected_hierarchy));
1084  }
1085  return *(stubs.back());
1086  }
1087  // the empty hierarchy case
1088  return l.get_view();
1089  });
1090 
1091  return {std::move(cols), std::move(stubs)};
1092  }
1093 
1094  column_view get_view() const { return root ? lists_column_view(*wrapped).child() : *wrapped; }
1095 
1096  int depth = 0;
1097  bool root = false;

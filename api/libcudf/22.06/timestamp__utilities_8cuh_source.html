46 template <typename T, bool nullable = false>
47 inline cudf::test::fixed_width_column_wrapper<T, int64_t> generate_timestamps(int32_t count,
48  time_point_ms start,
49  time_point_ms stop)
50 {
51  using Rep = typename T::rep;
52  using Period = typename T::period;
53  using ToDuration = cuda::std::chrono::duration<Rep, Period>;
54 
55  auto lhs = start.time_since_epoch().count();
56  auto rhs = stop.time_since_epoch().count();
57 
58  auto const min = std::min(lhs, rhs);
59  auto const max = std::max(lhs, rhs);
60  auto const range = max - min;
61  auto iter = cudf::detail::make_counting_transform_iterator(0, [=](auto i) {
62  return cuda::std::chrono::floor<ToDuration>(
63  cuda::std::chrono::milliseconds(min + (range / count) * i))
64  .count();
65  });
66 
67  if (nullable) {
68  auto mask =
69  cudf::detail::make_counting_transform_iterator(0, [](auto i) { return i % 2 == 0; });
70  return cudf::test::fixed_width_column_wrapper<T, int64_t>(iter, iter + count, mask);
71  } else {
72  // This needs to be in an else to quash `statement_not_reachable` warnings
73  return cudf::test::fixed_width_column_wrapper<T, int64_t>(iter, iter + count);
74  }
75 }

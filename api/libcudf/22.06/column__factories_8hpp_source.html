316 template <typename B>
317 std::unique_ptr<column> make_fixed_width_column(
318  data_type type,
319  size_type size,
320  B&& null_mask,
321  size_type null_count = cudf::UNKNOWN_NULL_COUNT,
322  rmm::cuda_stream_view stream = rmm::cuda_stream_default,
323  rmm::mr::device_memory_resource* mr = rmm::mr::get_current_device_resource())
324 {
325  CUDF_EXPECTS(is_fixed_width(type), "Invalid, non-fixed-width type.");
326  if (is_timestamp(type)) {
327  return make_timestamp_column(type, size, std::forward<B>(null_mask), null_count, stream, mr);
328  } else if (is_duration(type)) {
329  return make_duration_column(type, size, std::forward<B>(null_mask), null_count, stream, mr);
330  } else if (is_fixed_point(type)) {
331  return make_fixed_point_column(type, size, std::forward<B>(null_mask), null_count, stream, mr);
332  }
333  return make_numeric_column(type, size, std::forward<B>(null_mask), null_count, stream, mr);
334 }

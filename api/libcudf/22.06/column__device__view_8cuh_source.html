1303 template <typename ColumnDeviceView, typename ColumnViewIterator>
1304 ColumnDeviceView* child_columns_to_device_array(ColumnViewIterator child_begin,
1305  ColumnViewIterator child_end,
1306  void* h_ptr,
1307  void* d_ptr)
1308 {
1309  ColumnDeviceView* d_children = detail::align_ptr_for_type<ColumnDeviceView>(d_ptr);
1310  auto num_children = std::distance(child_begin, child_end);
1311  if (num_children > 0) {
1312  // The beginning of the memory must be the fixed-sized ColumnDeviceView
1313  // struct objects in order for d_children to be used as an array.
1314  auto h_column = detail::align_ptr_for_type<ColumnDeviceView>(h_ptr);
1315  auto d_column = d_children;
1316 
1317  // Any child data is assigned past the end of this array: h_end and d_end.
1318  auto h_end = reinterpret_cast<int8_t*>(h_column + num_children);
1319  auto d_end = reinterpret_cast<int8_t*>(d_column + num_children);
1320  std::for_each(child_begin, child_end, [&](auto const& col) {
1321  // inplace-new each child into host memory
1322  new (h_column) ColumnDeviceView(col, h_end, d_end);

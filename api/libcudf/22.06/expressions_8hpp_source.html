272  {
273  auto const table = [&] {
274  if (get_table_source() == table_reference::LEFT) {
275  return left_table;

277  return right_table;
278  } else {
279  CUDF_FAIL("Column reference data type cannot be determined from unknown table.");
280  }
281  }();

283  }

294  table_view const& right,

94 void tdigest_minmax_compare(cudf::tdigest::tdigest_column_view const& tdv,
95  column_view const& input_values)
96 {
97  // verify min/max
98  thrust::host_vector<device_span<T const>> h_spans;
99  h_spans.push_back({input_values.begin<T>(), static_cast<size_t>(input_values.size())});
100  thrust::device_vector<device_span<T const>> spans(h_spans);
101 
102  auto expected_min = cudf::make_fixed_width_column(
103  data_type{type_id::FLOAT64}, spans.size(), mask_state::UNALLOCATED);
104  thrust::transform(rmm::exec_policy(rmm::cuda_stream_default),
105  spans.begin(),
106  spans.end(),
107  expected_min->mutable_view().template begin<double>(),
108  column_min<T>{});
109  column_view result_min(data_type{type_id::FLOAT64}, tdv.size(), tdv.min_begin());
110  CUDF_TEST_EXPECT_COLUMNS_EQUAL(result_min, *expected_min);
111 
112  auto expected_max = cudf::make_fixed_width_column(
113  data_type{type_id::FLOAT64}, spans.size(), mask_state::UNALLOCATED);
114  thrust::transform(rmm::exec_policy(rmm::cuda_stream_default),
115  spans.begin(),
116  spans.end(),
117  expected_max->mutable_view().template begin<double>(),
118  column_max<T>{});
119  column_view result_max(data_type{type_id::FLOAT64}, tdv.size(), tdv.max_begin());
120  CUDF_TEST_EXPECT_COLUMNS_EQUAL(result_max, *expected_max);
121 }

427 // Note: there is no need to test different types here as the internals of a tdigest are always
428 // the same regardless of input.
429 template <typename Func, typename MergeFunc>
430 void tdigest_merge_simple(Func op, MergeFunc merge_op)
431 {
432  auto values = generate_standardized_percentile_distribution(data_type{type_id::FLOAT64});
433  CUDF_EXPECTS(values->size() == 750000, "Unexpected distribution size");
434 
435  auto split_values = cudf::split(*values, {250000, 500000});
436 
437  int const delta = 1000;
438 
439  // generate seperate digests
440  std::vector<std::unique_ptr<column>> parts;
441  auto iter = thrust::make_counting_iterator(0);
442  std::transform(
443  iter, iter + split_values.size(), std::back_inserter(parts), [&split_values, delta, op](int i) {
444  return op(split_values[i], delta);
445  });
446  std::vector<column_view> part_views;
447  std::transform(parts.begin(),
448  parts.end(),
449  std::back_inserter(part_views),
450  [](std::unique_ptr<column> const& col) { return col->view(); });
451 
452  // merge delta = 1000
453  {
454  int const merge_delta = 1000;
455 
456  // merge them
457  auto merge_input = cudf::concatenate(part_views);
458  auto result = merge_op(*merge_input, merge_delta);
459  cudf::tdigest::tdigest_column_view tdv(*result);
460 
461  // verify centroids
462  std::vector<expected_value> expected{{0, 0.00013945158577498588, 2},
463  {10, 0.04804393446447510763, 50},
464  {59, 1.68846964439246893797, 284},
465  {250, 33.36323141295877547918, 1479},
466  {368, 65.36307727957283475462, 2292},
467  {409, 73.95399208218296394080, 1784},
468  {490, 87.67566167909056673579, 1570},
469  {491, 87.83119717763385381204, 1570},
470  {500, 89.24891838334393412424, 1555},
471  {578, 95.87182997389099625707, 583},
472  {625, 98.20470345147104751504, 405},
473  {700, 99.96818381983835877236, 56},
474  {711, 99.99970905482754801596, 1}};
475  tdigest_sample_compare(tdv, expected);
476 
477  // verify min/max
478  tdigest_minmax_compare<double>(tdv, *values);
479  }

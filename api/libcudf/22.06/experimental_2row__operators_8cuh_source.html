158  template <typename Element,
159  CUDF_ENABLE_IF(cudf::is_relationally_comparable<Element, Element>())>
160  __device__ cuda::std::pair<weak_ordering, int> operator()(
161  size_type const lhs_element_index, size_type const rhs_element_index) const noexcept
162  {
163  if (_nulls) {
164  bool const lhs_is_null{_lhs.is_null(lhs_element_index)};
165  bool const rhs_is_null{_rhs.is_null(rhs_element_index)};
166 
167  if (lhs_is_null or rhs_is_null) { // at least one is null
168  return cuda::std::make_pair(null_compare(lhs_is_null, rhs_is_null, _null_precedence),
169  _depth);
170  }
171  }
172 
173  return cuda::std::make_pair(relational_compare(_lhs.element<Element>(lhs_element_index),
174  _rhs.element<Element>(rhs_element_index)),
175  std::numeric_limits<int>::max());
176  }

178  template <typename Element,
179  CUDF_ENABLE_IF(not cudf::is_relationally_comparable<Element, Element>() and
180  not std::is_same_v<Element, cudf::struct_view>),
181  typename... Args>
182  __device__ cuda::std::pair<weak_ordering, int> operator()(Args...)
183  {
184  CUDF_UNREACHABLE("Attempted to compare elements of uncomparable types.");
185  }
186 
187  template <typename Element, CUDF_ENABLE_IF(std::is_same_v<Element, cudf::struct_view>)>
188  __device__ cuda::std::pair<weak_ordering, int> operator()(size_type const lhs_element_index,
189  size_type const rhs_element_index)
190  {
191  column_device_view lcol = _lhs;
192  column_device_view rcol = _rhs;
193  int depth = _depth;
194  while (lcol.type().id() == type_id::STRUCT) {
195  bool const lhs_is_null{lcol.is_null(lhs_element_index)};
196  bool const rhs_is_null{rcol.is_null(rhs_element_index)};
197 
198  if (lhs_is_null or rhs_is_null) { // at least one is null
199  weak_ordering state = null_compare(lhs_is_null, rhs_is_null, _null_precedence);
200  return cuda::std::make_pair(state, depth);
201  }
202 
203  if (lcol.num_child_columns() == 0) {
204  return cuda::std::make_pair(weak_ordering::EQUIVALENT, depth);
205  }
206 
207  // Non-empty structs have been modified to only have 1 child when using this.
208  lcol = lcol.children()[0];
209  rcol = rcol.children()[0];
210  ++depth;
211  }
212 
213  auto const comparator = element_comparator{_nulls, lcol, rcol, _null_precedence, depth};
214  return cudf::type_dispatcher<dispatch_void_if_nested>(
215  lcol.type(), comparator, lhs_element_index, rhs_element_index);
216  }

220  column_device_view const _rhs;
221  Nullate const _nulls;
222  null_order const _null_precedence;
223  int const _depth;

226  public:
235  __device__ bool operator()(size_type const lhs_index, size_type const rhs_index) const noexcept
236  {
237  int last_null_depth = std::numeric_limits<int>::max();
238  for (size_type i = 0; i < _lhs.num_columns(); ++i) {
239  int const depth = _depth.has_value() ? (*_depth)[i] : 0;
240  if (depth > last_null_depth) { continue; }
241 
242  bool const ascending =
243  _column_order.has_value() ? (*_column_order)[i] == order::ASCENDING : true;
244 
245  null_order const null_precedence =
246  _null_precedence.has_value() ? (*_null_precedence)[i] : null_order::BEFORE;
247 
248  auto const comparator =
249  element_comparator{_nulls, _lhs.column(i), _rhs.column(i), null_precedence, depth};
250 
251  weak_ordering state;
252  cuda::std::tie(state, last_null_depth) =
253  cudf::type_dispatcher(_lhs.column(i).type(), comparator, lhs_index, rhs_index);
254 
255  if (state == weak_ordering::EQUIVALENT) { continue; }
256 
257  return state == (ascending ? weak_ordering::LESS : weak_ordering::GREATER);
258  }
259  return false;
260  }

508  template <typename Element, CUDF_ENABLE_IF(cudf::is_equality_comparable<Element, Element>())>
509  __device__ bool operator()(size_type const lhs_element_index,
510  size_type const rhs_element_index) const noexcept
511  {
512  if (nulls) {
513  bool const lhs_is_null{lhs.is_null(lhs_element_index)};
514  bool const rhs_is_null{rhs.is_null(rhs_element_index)};
515  if (lhs_is_null and rhs_is_null) {
516  return nulls_are_equal == null_equality::EQUAL;
517  } else if (lhs_is_null != rhs_is_null) {
518  return false;
519  }
520  }
521 
522  return equality_compare(lhs.element<Element>(lhs_element_index),
523  rhs.element<Element>(rhs_element_index));
524  }

528  not cudf::is_nested<Element>()),

535  template <typename Element, CUDF_ENABLE_IF(cudf::is_nested<Element>())>
536  __device__ bool operator()(size_type const lhs_element_index,
537  size_type const rhs_element_index) const noexcept
538  {
539  column_device_view lcol = lhs.slice(lhs_element_index, 1);
540  column_device_view rcol = rhs.slice(rhs_element_index, 1);
541  while (is_nested(lcol.type())) {
542  if (nulls) {
543  auto lvalid = detail::make_validity_iterator<true>(lcol);
544  auto rvalid = detail::make_validity_iterator<true>(rcol);
545  if (nulls_are_equal == null_equality::UNEQUAL) {
546  if (thrust::any_of(
547  thrust::seq, lvalid, lvalid + lcol.size(), thrust::logical_not<bool>()) or
548  thrust::any_of(
549  thrust::seq, rvalid, rvalid + rcol.size(), thrust::logical_not<bool>())) {
550  return false;
551  }
552  } else {
553  if (not thrust::equal(thrust::seq, lvalid, lvalid + lcol.size(), rvalid)) {
554  return false;
555  }
556  }
557  }
558  if (lcol.type().id() == type_id::STRUCT) {
559  if (lcol.num_child_columns() == 0) { return true; }
560  lcol = detail::structs_column_device_view(lcol).sliced_child(0);
561  rcol = detail::structs_column_device_view(rcol).sliced_child(0);
562  } else if (lcol.type().id() == type_id::LIST) {
563  auto l_list_col = detail::lists_column_device_view(lcol);
564  auto r_list_col = detail::lists_column_device_view(rcol);
565 
566  auto lsizes = make_list_size_iterator(l_list_col);
567  auto rsizes = make_list_size_iterator(r_list_col);
568  if (not thrust::equal(thrust::seq, lsizes, lsizes + lcol.size(), rsizes)) {
569  return false;
570  }
571 
572  lcol = l_list_col.sliced_child();
573  rcol = r_list_col.sliced_child();
574  if (lcol.size() != rcol.size()) { return false; }
575  }
576  }
577 
578  auto comp =
579  column_comparator{element_comparator{nulls, lcol, rcol, nulls_are_equal}, lcol.size()};
580  return type_dispatcher<dispatch_void_if_nested>(lcol.type(), comp);
581  }

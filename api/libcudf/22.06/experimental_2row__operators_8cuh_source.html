231  template <typename Element,
232  CUDF_ENABLE_IF(cudf::is_relationally_comparable<Element, Element>())>
233  __device__ cuda::std::pair<weak_ordering, int> operator()(
234  size_type const lhs_element_index, size_type const rhs_element_index) const noexcept
235  {
236  if (_check_nulls) {
237  bool const lhs_is_null{_lhs.is_null(lhs_element_index)};
238  bool const rhs_is_null{_rhs.is_null(rhs_element_index)};

241  return cuda::std::pair(null_compare(lhs_is_null, rhs_is_null, _null_precedence), _depth);

259  template <typename Element, CUDF_ENABLE_IF(std::is_same_v<Element, cudf::struct_view>)>
260  __device__ cuda::std::pair<weak_ordering, int> operator()(
261  size_type const lhs_element_index, size_type const rhs_element_index) const noexcept
262  {
263  column_device_view lcol = _lhs;
264  column_device_view rcol = _rhs;
265  int depth = _depth;
266  while (lcol.type().id() == type_id::STRUCT) {
267  bool const lhs_is_null{lcol.is_null(lhs_element_index)};
268  bool const rhs_is_null{rcol.is_null(rhs_element_index)};
269 
270  if (lhs_is_null or rhs_is_null) { // at least one is null
271  weak_ordering state = null_compare(lhs_is_null, rhs_is_null, _null_precedence);
272  return cuda::std::pair(state, depth);
273  }
274 
275  if (lcol.num_child_columns() == 0) {
276  return cuda::std::pair(weak_ordering::EQUIVALENT, depth);
277  }
278 
279  // Non-empty structs have been modified to only have 1 child when using this.
280  lcol = detail::structs_column_device_view(lcol).get_sliced_child(0);
281  rcol = detail::structs_column_device_view(rcol).get_sliced_child(0);
282  ++depth;
283  }
284 
285  auto const comparator = element_comparator{_check_nulls, lcol, rcol, _null_precedence, depth};
286  return cudf::type_dispatcher<dispatch_void_if_nested>(
287  lcol.type(), comparator, lhs_element_index, rhs_element_index);
288  }

308  __device__ weak_ordering operator()(size_type const lhs_index,
309  size_type const rhs_index) const noexcept
310  {
311  int last_null_depth = std::numeric_limits<int>::max();
312  for (size_type i = 0; i < _lhs.num_columns(); ++i) {
313  int const depth = _depth.has_value() ? (*_depth)[i] : 0;
314  if (depth > last_null_depth) { continue; }
315 
316  bool const ascending =
317  _column_order.has_value() ? (*_column_order)[i] == order::ASCENDING : true;
318 
319  null_order const null_precedence =
320  _null_precedence.has_value() ? (*_null_precedence)[i] : null_order::BEFORE;
321 
322  auto const comparator =
323  element_comparator{_check_nulls, _lhs.column(i), _rhs.column(i), null_precedence, depth};
324  weak_ordering state;
325  cuda::std::tie(state, last_null_depth) =
326  cudf::type_dispatcher(_lhs.column(i).type(), comparator, lhs_index, rhs_index);
327 
328  if (state == weak_ordering::EQUIVALENT) { continue; }
329 
330  return ascending
331  ? state
332  : (state == weak_ordering::GREATER ? weak_ordering::LESS : weak_ordering::GREATER);
333  }
334  return weak_ordering::EQUIVALENT;
335  }

553  comparator(static_cast<cudf::size_type>(lhs_index), static_cast<cudf::size_type>(rhs_index));

557  return weak_ordering::GREATER;
558  } else if (left_right_ordering == weak_ordering::GREATER) {
559  return weak_ordering::LESS;
560  }
561  return weak_ordering::EQUIVALENT;
562  }

740  template <typename Element, CUDF_ENABLE_IF(cudf::is_equality_comparable<Element, Element>())>
741  __device__ bool operator()(size_type const lhs_element_index,
742  size_type const rhs_element_index) const noexcept
743  {
744  if (check_nulls) {
745  bool const lhs_is_null{lhs.is_null(lhs_element_index)};
746  bool const rhs_is_null{rhs.is_null(rhs_element_index)};
747  if (lhs_is_null and rhs_is_null) {

749  } else if (lhs_is_null != rhs_is_null) {

767  template <typename Element, CUDF_ENABLE_IF(cudf::is_nested<Element>())>
768  __device__ bool operator()(size_type const lhs_element_index,
769  size_type const rhs_element_index) const noexcept
770  {
771  column_device_view lcol = lhs.slice(lhs_element_index, 1);
772  column_device_view rcol = rhs.slice(rhs_element_index, 1);
773  while (is_nested(lcol.type())) {
774  if (check_nulls) {
775  auto lvalid = detail::make_validity_iterator<true>(lcol);
776  auto rvalid = detail::make_validity_iterator<true>(rcol);
777  if (nulls_are_equal == null_equality::UNEQUAL) {
778  if (thrust::any_of(
779  thrust::seq, lvalid, lvalid + lcol.size(), thrust::logical_not<bool>()) or
780  thrust::any_of(
781  thrust::seq, rvalid, rvalid + rcol.size(), thrust::logical_not<bool>())) {
782  return false;
783  }
784  } else {
785  if (not thrust::equal(thrust::seq, lvalid, lvalid + lcol.size(), rvalid)) {
786  return false;
787  }
788  }
789  }
790  if (lcol.type().id() == type_id::STRUCT) {
791  if (lcol.num_child_columns() == 0) { return true; }
792  // Non-empty structs are assumed to be decomposed and contain only one child
793  lcol = detail::structs_column_device_view(lcol).get_sliced_child(0);
794  rcol = detail::structs_column_device_view(rcol).get_sliced_child(0);
795  } else if (lcol.type().id() == type_id::LIST) {
796  auto l_list_col = detail::lists_column_device_view(lcol);
797  auto r_list_col = detail::lists_column_device_view(rcol);
798 
799  auto lsizes = make_list_size_iterator(l_list_col);
800  auto rsizes = make_list_size_iterator(r_list_col);
801  if (not thrust::equal(thrust::seq, lsizes, lsizes + lcol.size(), rsizes)) {
802  return false;
803  }
804 
805  lcol = l_list_col.get_sliced_child();
806  rcol = r_list_col.get_sliced_child();
807  if (lcol.size() != rcol.size()) { return false; }
808  }
809  }
810 
811  auto comp = column_comparator{element_comparator{check_nulls, lcol, rcol, nulls_are_equal},
812  lcol.size()};
813  return type_dispatcher<dispatch_void_if_nested>(lcol.type(), comp);
814  }

1087 template <template <typename> class hash_function, typename Nullate>
1088 class device_row_hasher {
1089  friend class row_hasher;
1090 
1091  public:
1092  device_row_hasher() = delete;
1093 
1094  __device__ auto operator()(size_type row_index) const noexcept
1095  {
1096  auto it = thrust::make_transform_iterator(_table.begin(), [=](auto const& column) {
1097  return cudf::type_dispatcher<dispatch_storage_type>(
1098  column.type(), element_hasher_adapter<hash_function>{_check_nulls}, column, row_index);
1099  });
1100 
1101  // Hash each element and combine all the hash values together
1102  return detail::accumulate(it, it + _table.num_columns(), _seed, [](auto hash, auto h) {
1103  return cudf::detail::hash_combine(hash, h);
1104  });
1105  }
1106 
1107  private:
1115  template <template <typename> class hash_fn>
1116  class element_hasher_adapter {
1117  static constexpr hash_value_type NULL_HASH = std::numeric_limits<hash_value_type>::max();
1118  static constexpr hash_value_type NON_NULL_HASH = 0;
1119 
1120  public:
1121  __device__ element_hasher_adapter(Nullate check_nulls) noexcept
1122  : _element_hasher(check_nulls), _check_nulls(check_nulls)
1123  {
1124  }
1125 
1126  template <typename T, CUDF_ENABLE_IF(not cudf::is_nested<T>())>
1127  __device__ hash_value_type operator()(column_device_view const& col,
1128  size_type row_index) const noexcept
1129  {
1130  return _element_hasher.template operator()<T>(col, row_index);
1131  }
1132 
1133  template <typename T, CUDF_ENABLE_IF(cudf::is_nested<T>())>
1134  __device__ hash_value_type operator()(column_device_view const& col,
1135  size_type row_index) const noexcept
1136  {
1137  auto hash = hash_value_type{0};
1138  column_device_view curr_col = col.slice(row_index, 1);
1139  while (is_nested(curr_col.type())) {
1140  if (_check_nulls) {
1141  auto validity_it = detail::make_validity_iterator<true>(curr_col);
1142  hash = detail::accumulate(
1143  validity_it, validity_it + curr_col.size(), hash, [](auto hash, auto is_valid) {
1144  return cudf::detail::hash_combine(hash, is_valid ? NON_NULL_HASH : NULL_HASH);
1145  });
1146  }
1147  if (curr_col.type().id() == type_id::STRUCT) {
1148  if (curr_col.num_child_columns() == 0) { return hash; }
1149  // Non-empty structs are assumed to be decomposed and contain only one child
1150  curr_col = detail::structs_column_device_view(curr_col).get_sliced_child(0);
1151  } else if (curr_col.type().id() == type_id::LIST) {
1152  auto list_col = detail::lists_column_device_view(curr_col);
1153  auto list_sizes = make_list_size_iterator(list_col);
1154  hash = detail::accumulate(
1155  list_sizes, list_sizes + list_col.size(), hash, [](auto hash, auto size) {
1156  return cudf::detail::hash_combine(hash, hash_fn<size_type>{}(size));
1157  });
1158  curr_col = list_col.get_sliced_child();
1159  }
1160  }
1161  for (int i = 0; i < curr_col.size(); ++i) {
1162  hash = cudf::detail::hash_combine(
1163  hash,
1164  type_dispatcher<dispatch_void_if_nested>(curr_col.type(), _element_hasher, curr_col, i));
1165  }
1166  return hash;
1167  }
1168 
1169  element_hasher<hash_fn, Nullate> const _element_hasher;
1170  Nullate const _check_nulls;
1171  };

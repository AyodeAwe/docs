202  bool const rhs_is_null{rcol.is_null(rhs_element_index)};

205  weak_ordering state = null_compare(lhs_is_null, rhs_is_null, _null_precedence);
206  return cuda::std::pair(state, depth);

210  return cuda::std::pair(weak_ordering::EQUIVALENT, depth);

242  __device__ weak_ordering operator()(size_type lhs_index, size_type rhs_index) const noexcept
243  {
244  int last_null_depth = std::numeric_limits<int>::max();
245  for (size_type i = 0; i < _lhs.num_columns(); ++i) {
246  int const depth = _depth.has_value() ? (*_depth)[i] : 0;
247  if (depth > last_null_depth) { continue; }
248 
249  bool const ascending =
250  _column_order.has_value() ? (*_column_order)[i] == order::ASCENDING : true;
251 
252  null_order const null_precedence =
253  _null_precedence.has_value() ? (*_null_precedence)[i] : null_order::BEFORE;
254 
255  auto const comparator =
256  element_comparator{_check_nulls, _lhs.column(i), _rhs.column(i), null_precedence, depth};
257  weak_ordering state;

259  cudf::type_dispatcher(_lhs.column(i).type(), comparator, lhs_index, rhs_index);

553  template <typename Element, CUDF_ENABLE_IF(cudf::is_equality_comparable<Element, Element>())>
554  __device__ bool operator()(size_type const lhs_element_index,
555  size_type const rhs_element_index) const noexcept
556  {
557  if (check_nulls) {
558  bool const lhs_is_null{lhs.is_null(lhs_element_index)};
559  bool const rhs_is_null{rhs.is_null(rhs_element_index)};
560  if (lhs_is_null and rhs_is_null) {

580  template <typename Element, CUDF_ENABLE_IF(cudf::is_nested<Element>())>
581  __device__ bool operator()(size_type const lhs_element_index,
582  size_type const rhs_element_index) const noexcept
583  {
584  column_device_view lcol = lhs.slice(lhs_element_index, 1);
585  column_device_view rcol = rhs.slice(rhs_element_index, 1);
586  while (is_nested(lcol.type())) {
587  if (check_nulls) {
588  auto lvalid = detail::make_validity_iterator<true>(lcol);
589  auto rvalid = detail::make_validity_iterator<true>(rcol);
590  if (nulls_are_equal == null_equality::UNEQUAL) {
591  if (thrust::any_of(
592  thrust::seq, lvalid, lvalid + lcol.size(), thrust::logical_not<bool>()) or
593  thrust::any_of(
594  thrust::seq, rvalid, rvalid + rcol.size(), thrust::logical_not<bool>())) {
595  return false;
596  }
597  } else {
598  if (not thrust::equal(thrust::seq, lvalid, lvalid + lcol.size(), rvalid)) {
599  return false;
600  }
601  }
602  }
603  if (lcol.type().id() == type_id::STRUCT) {
604  if (lcol.num_child_columns() == 0) { return true; }
605  // Non-empty structs are assumed to be decomposed and contain only one child
606  lcol = detail::structs_column_device_view(lcol).sliced_child(0);
607  rcol = detail::structs_column_device_view(rcol).sliced_child(0);
608  } else if (lcol.type().id() == type_id::LIST) {
609  auto l_list_col = detail::lists_column_device_view(lcol);
610  auto r_list_col = detail::lists_column_device_view(rcol);
611 
612  auto lsizes = make_list_size_iterator(l_list_col);
613  auto rsizes = make_list_size_iterator(r_list_col);
614  if (not thrust::equal(thrust::seq, lsizes, lsizes + lcol.size(), rsizes)) {
615  return false;
616  }
617 
618  lcol = l_list_col.sliced_child();
619  rcol = r_list_col.sliced_child();
620  if (lcol.size() != rcol.size()) { return false; }
621  }
622  }
623 
624  auto comp = column_comparator{element_comparator{check_nulls, lcol, rcol, nulls_are_equal},
625  lcol.size()};
626  return type_dispatcher<dispatch_void_if_nested>(lcol.type(), comp);
627  }

853  template <typename T, CUDF_ENABLE_IF(cudf::is_nested<T>())>
854  __device__ hash_value_type operator()(column_device_view const& col,
855  size_type row_index) const noexcept
856  {
857  auto hash = hash_value_type{0};
858  column_device_view curr_col = col.slice(row_index, 1);
859  while (is_nested(curr_col.type())) {
860  if (_check_nulls) {
861  auto validity_it = detail::make_validity_iterator<true>(curr_col);
862  hash = detail::accumulate(
863  validity_it, validity_it + curr_col.size(), hash, [](auto hash, auto is_valid) {
864  return cudf::detail::hash_combine(hash, is_valid ? NON_NULL_HASH : NULL_HASH);
865  });
866  }
867  if (curr_col.type().id() == type_id::STRUCT) {
868  if (curr_col.num_child_columns() == 0) { return hash; }
869  // Non-empty structs are assumed to be decomposed and contain only one child
870  curr_col = detail::structs_column_device_view(curr_col).sliced_child(0);
871  } else if (curr_col.type().id() == type_id::LIST) {

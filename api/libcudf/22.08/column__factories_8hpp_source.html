326 template <typename B>
327 std::unique_ptr<column> make_fixed_width_column(
328  data_type type,
329  size_type size,
330  B&& null_mask,
331  size_type null_count = cudf::UNKNOWN_NULL_COUNT,
332  rmm::cuda_stream_view stream = rmm::cuda_stream_default,
333  rmm::mr::device_memory_resource* mr = rmm::mr::get_current_device_resource())
334 {
335  CUDF_EXPECTS(is_fixed_width(type), "Invalid, non-fixed-width type.");
336  if (is_timestamp(type)) {
337  return make_timestamp_column(type, size, std::forward<B>(null_mask), null_count, stream, mr);
338  } else if (is_duration(type)) {
339  return make_duration_column(type, size, std::forward<B>(null_mask), null_count, stream, mr);
340  } else if (is_fixed_point(type)) {
341  return make_fixed_point_column(type, size, std::forward<B>(null_mask), null_count, stream, mr);
342  }
343  return make_numeric_column(type, size, std::forward<B>(null_mask), null_count, stream, mr);
344 }

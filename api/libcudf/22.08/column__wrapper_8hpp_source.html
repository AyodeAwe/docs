107  typename ToT = To,
108  std::enable_if_t<std::is_same_v<FromT, ToT>, void>* = nullptr>
109  constexpr ToT operator()(FromT element) const

126  std::is_constructible_v<ToT, FromT>),

243 {

245  auto const num_words = cudf::bitmask_allocation_size_bytes(size) / sizeof(bitmask_type);

290 template <typename StringsIterator, typename ValidityIterator>
291 auto make_chars_and_offsets(StringsIterator begin, StringsIterator end, ValidityIterator v)
292 {
293  std::vector<char> chars{};
294  std::vector<cudf::size_type> offsets(1, 0);
295  for (auto str = begin; str < end; ++str) {
296  std::string tmp = (*v++) ? std::string(*str) : std::string{};
297  chars.insert(chars.end(), std::cbegin(tmp), std::cend(tmp));
298  offsets.push_back(offsets.back() + tmp.length());
299  }
300  return std::pair(std::move(chars), std::move(offsets));
301 };

1528  void build_from_nested(std::initializer_list<lists_column_wrapper<T, SourceElementT>> elements,
1529  std::vector<bool> const& v)
1530  {

1532  0, [&v](auto i) { return v.empty() ? true : v[i]; });
1533 
1534  // compute the expected hierarchy and depth
1535  auto const hierarchy_and_depth =
1536  std::accumulate(elements.begin(),
1537  elements.end(),
1538  std::pair<column_view, int32_t>{{}, -1},
1539  [](auto acc, lists_column_wrapper const& lcw) {
1540  return lcw.depth > acc.second ? std::pair(lcw.get_view(), lcw.depth) : acc;
1541  });
1542  column_view expected_hierarchy = hierarchy_and_depth.first;
1543  int32_t const expected_depth = hierarchy_and_depth.second;
1544 
1545  // preprocess columns so that every column_view in 'cols' is an equivalent hierarchy
1546  auto [cols, stubs] = preprocess_columns(elements, expected_hierarchy, expected_depth);
1547 
1548  // generate offsets
1549  size_type count = 0;
1550  std::vector<size_type> offsetv;
1551  std::transform(cols.cbegin(),
1552  cols.cend(),
1553  valids,
1554  std::back_inserter(offsetv),
1555  [&](cudf::column_view const& col, bool valid) {
1556  // nulls are represented as a repeated offset
1557  size_type ret = count;
1558  if (valid) { count += col.size(); }
1559  return ret;
1560  });
1561  // add the final offset
1562  offsetv.push_back(count);
1563  auto offsets =
1564  cudf::test::fixed_width_column_wrapper<size_type>(offsetv.begin(), offsetv.end()).release();
1565 
1566  // concatenate them together, skipping children that are null.
1567  std::vector<column_view> children;
1568  thrust::copy_if(
1569  std::cbegin(cols), std::cend(cols), valids, std::back_inserter(children), thrust::identity{});
1570 
1571  auto data = children.empty() ? cudf::empty_like(expected_hierarchy) : concatenate(children);
1572 
1573  // increment depth
1574  depth = expected_depth + 1;
1575 
1576  // construct the list column
1577  wrapped =
1578  make_lists_column(cols.size(),
1579  std::move(offsets),
1580  std::move(data),
1581  v.size() <= 0 ? 0 : cudf::UNKNOWN_NULL_COUNT,
1582  v.size() <= 0 ? rmm::device_buffer{}
1583  : cudf::test::detail::make_null_mask(v.begin(), v.end()));

1669  std::pair<std::vector<column_view>, std::vector<std::unique_ptr<column>>> preprocess_columns(
1670  std::initializer_list<lists_column_wrapper<T, SourceElementT>> const& elements,
1671  column_view& expected_hierarchy,
1672  int expected_depth)
1673  {
1674  std::vector<std::unique_ptr<column>> stubs;
1675  std::vector<column_view> cols;
1676 
1677  // preprocess the incoming lists.
1678  // - unwrap any "root" lists
1679  // - handle incomplete hierarchies
1680  std::transform(elements.begin(),
1681  elements.end(),
1682  std::back_inserter(cols),
1683  [&](lists_column_wrapper const& l) -> column_view {
1684  // depth mismatch. attempt to normalize the short column.
1685  // this function will also catch if this is a legitimately broken
1686  // set of input
1687  if (l.depth < expected_depth) {
1688  if (l.root) {
1689  // this exception distinguishes between the following two cases:
1690  //
1691  // { {{{1, 2, 3}}}, {} }
1692  // In this case, row 0 is a List<List<List<int>>>, whereas row 1 is
1693  // just a List<> which is an apparent mismatch. However, because row 1
1694  // is empty we will allow that to semantically mean
1695  // "a List<List<List<int>>> that's empty at the top level"
1696  //
1697  // { {{{1, 2, 3}}}, {4, 5, 6} }
1698  // In this case, row 1 is a a concrete List<int> with actual values.
1699  // There is no way to rectify the differences so we will treat it as a
1700  // true column mismatch.
1701  CUDF_EXPECTS(l.wrapped->size() == 0, "Mismatch in column types!");
1702  stubs.push_back(empty_like(expected_hierarchy));
1703  } else {
1704  stubs.push_back(normalize_column(l.get_view(), expected_hierarchy));
1705  }
1706  return *(stubs.back());
1707  }
1708  // the empty hierarchy case
1709  return l.get_view();
1710  });
1711 
1712  return {std::move(cols), std::move(stubs)};
1713  }
1714 
1715  column_view get_view() const { return root ? lists_column_view(*wrapped).child() : *wrapped; }
1716 
1717  int depth = 0;
1718  bool root = false;
1719 };

206  [[nodiscard]] std::size_t get_byte_range_padding() const
207  {
208  auto const num_names = _names.size();
209  auto const num_dtypes = std::visit([](const auto& dtypes) { return dtypes.size(); }, _dtypes);
210  auto const num_columns = std::max(num_dtypes, num_names);
211 
212  auto const max_row_bytes = 16 * 1024; // 16KB
213  auto const column_bytes = 64;
214  auto const base_padding = 1024; // 1KB
215 
216  if (num_columns == 0) {
217  // Use flat size if the number of columns is not known
218  return max_row_bytes;
219  }
220 
221  // Expand the size based on the number of columns, if available
222  return base_padding + num_columns * column_bytes;
223  }

69 class istream_data_chunk_reader : public data_chunk_reader {
70  struct host_ticket {
71  cudaEvent_t event;
72  thrust::host_vector<char, thrust::system::cuda::experimental::pinned_allocator<char>> buffer;
73  };
74 
75  public:
76  istream_data_chunk_reader(std::unique_ptr<std::istream> datastream)
77  : _datastream(std::move(datastream)), _tickets(2)
78  {
79  // create an event to track the completion of the last device-to-host copy.
80  for (std::size_t i = 0; i < _tickets.size(); i++) {
81  CUDF_CUDA_TRY(cudaEventCreate(&(_tickets[i].event)));
82  }
83  }
84 
85  ~istream_data_chunk_reader()
86  {
87  for (std::size_t i = 0; i < _tickets.size(); i++) {
88  CUDF_CUDA_TRY(cudaEventDestroy(_tickets[i].event));
89  }
90  }
91 
92  void skip_bytes(std::size_t size) override { _datastream->ignore(size); };
93 
94  std::unique_ptr<device_data_chunk> get_next_chunk(std::size_t read_size,
95  rmm::cuda_stream_view stream) override
96  {
97  CUDF_FUNC_RANGE();
98 
99  auto& h_ticket = _tickets[_next_ticket_idx];
100 
101  _next_ticket_idx = (_next_ticket_idx + 1) % _tickets.size();
102 
103  // synchronize on the last host-to-device copy, so we don't clobber the host buffer.
104  CUDF_CUDA_TRY(cudaEventSynchronize(h_ticket.event));
105 
106  // resize the host buffer as necessary to contain the requested number of bytes
107  if (h_ticket.buffer.size() < read_size) { h_ticket.buffer.resize(read_size); }
108 
109  // read data from the host istream in to the pinned host memory buffer
110  _datastream->read(h_ticket.buffer.data(), read_size);
111 
112  // adjust the read size to reflect how many bytes were actually read from the data stream
113  read_size = _datastream->gcount();
114 
115  // get a view over some device memory we can use to buffer the read data on to device.
116  auto chunk = rmm::device_uvector<char>(read_size, stream);
117 
118  // copy the host-pinned data on to device
119  CUDF_CUDA_TRY(cudaMemcpyAsync( //
120  chunk.data(),
121  h_ticket.buffer.data(),
122  read_size,
123  cudaMemcpyHostToDevice,
124  stream.value()));
125 
126  // record the host-to-device copy.
127  CUDF_CUDA_TRY(cudaEventRecord(h_ticket.event, stream.value()));
128 
129  // return the view over device memory so it can be processed.
130  return std::make_unique<device_uvector_data_chunk>(std::move(chunk));
131  }
132 
133  private:
134  std::size_t _next_ticket_idx = 0;
135  std::unique_ptr<std::istream> _datastream;
136  std::vector<host_ticket> _tickets;
137 };

144 class device_span_data_chunk_reader : public data_chunk_reader {
145  public:
146  device_span_data_chunk_reader(device_span<char const> data) : _data(data) {}
147 
148  void skip_bytes(std::size_t read_size) override
149  {
150  if (read_size > _data.size() - _position) { read_size = _data.size() - _position; }
151  _position += read_size;
152  };
153 
154  std::unique_ptr<device_data_chunk> get_next_chunk(std::size_t read_size,
155  rmm::cuda_stream_view stream) override
156  {
157  // limit the read size to the number of bytes remaining in the device_span.
158  if (read_size > _data.size() - _position) { read_size = _data.size() - _position; }
159 
160  // create a view over the device span
161  auto chunk_span = _data.subspan(_position, read_size);
162 
163  // increment position
164  _position += read_size;
165 
166  // return the view over device memory so it can be processed.
167  return std::make_unique<device_span_data_chunk>(chunk_span);
168  }
169 
170  private:
171  device_span<char const> _data;
172  uint64_t _position = 0;
173 };

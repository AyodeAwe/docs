310  __device__ cuda::std::pair<weak_ordering, int> operator()(
311  size_type const lhs_element_index, size_type const rhs_element_index) const noexcept

315  bool const rhs_is_null{_rhs.is_null(rhs_element_index)};

336  template <typename Element, CUDF_ENABLE_IF(std::is_same_v<Element, cudf::struct_view>)>
337  __device__ cuda::std::pair<weak_ordering, int> operator()(
338  size_type const lhs_element_index, size_type const rhs_element_index) const noexcept
339  {
340  column_device_view lcol = _lhs;
341  column_device_view rcol = _rhs;
342  int depth = _depth;
343  while (lcol.type().id() == type_id::STRUCT) {
344  bool const lhs_is_null{lcol.is_null(lhs_element_index)};
345  bool const rhs_is_null{rcol.is_null(rhs_element_index)};
346 
347  if (lhs_is_null or rhs_is_null) { // at least one is null
348  weak_ordering state = null_compare(lhs_is_null, rhs_is_null, _null_precedence);
349  return cuda::std::pair(state, depth);
350  }
351 
352  if (lcol.num_child_columns() == 0) {
353  return cuda::std::pair(weak_ordering::EQUIVALENT, depth);
354  }
355 
356  // Non-empty structs have been modified to only have 1 child when using this.
357  lcol = detail::structs_column_device_view(lcol).get_sliced_child(0);
358  rcol = detail::structs_column_device_view(rcol).get_sliced_child(0);
359  ++depth;
360  }
361 
362  return cudf::type_dispatcher<dispatch_void_if_nested>(
363  lcol.type(),
364  element_comparator{_check_nulls, lcol, rcol, _null_precedence, depth, _comparator},
365  lhs_element_index,
366  rhs_element_index);
367  }

388  __device__ constexpr weak_ordering operator()(size_type const lhs_index,
389  size_type const rhs_index) const noexcept
390  {
391  int last_null_depth = std::numeric_limits<int>::max();
392  for (size_type i = 0; i < _lhs.num_columns(); ++i) {
393  int const depth = _depth.has_value() ? (*_depth)[i] : 0;
394  if (depth > last_null_depth) { continue; }
395 
396  bool const ascending =
397  _column_order.has_value() ? (*_column_order)[i] == order::ASCENDING : true;
398 
399  null_order const null_precedence =
400  _null_precedence.has_value() ? (*_null_precedence)[i] : null_order::BEFORE;

402  weak_ordering state;
403  cuda::std::tie(state, last_null_depth) = cudf::type_dispatcher(
404  _lhs.column(i).type(),
405  element_comparator{
406  _check_nulls, _lhs.column(i), _rhs.column(i), null_precedence, depth, _comparator},
407  lhs_index,
408  rhs_index);
409

411 
412  return ascending
413  ? state
414  : (state == weak_ordering::GREATER ? weak_ordering::LESS : weak_ordering::GREATER);
415  }
416  return weak_ordering::EQUIVALENT;
417  }

990  __device__ bool operator()(size_type const lhs_element_index,
991  size_type const rhs_element_index) const noexcept
992  {

995  bool const rhs_is_null{rhs.is_null(rhs_element_index)};

1016  template <typename Element, CUDF_ENABLE_IF(cudf::is_nested<Element>())>
1017  __device__ bool operator()(size_type const lhs_element_index,
1018  size_type const rhs_element_index) const noexcept
1019  {
1020  column_device_view lcol = lhs.slice(lhs_element_index, 1);
1021  column_device_view rcol = rhs.slice(rhs_element_index, 1);
1022  while (is_nested(lcol.type())) {
1023  if (check_nulls) {
1024  auto lvalid = detail::make_validity_iterator<true>(lcol);
1025  auto rvalid = detail::make_validity_iterator<true>(rcol);
1026  if (nulls_are_equal == null_equality::UNEQUAL) {
1027  if (thrust::any_of(
1028  thrust::seq, lvalid, lvalid + lcol.size(), thrust::logical_not<bool>()) or

1033  } else {

1035  return false;
1036  }
1037  }

1040  if (lcol.num_child_columns() == 0) { return true; }
1041  // Non-empty structs are assumed to be decomposed and contain only one child
1042  lcol = detail::structs_column_device_view(lcol).get_sliced_child(0);
1043  rcol = detail::structs_column_device_view(rcol).get_sliced_child(0);
1044  } else if (lcol.type().id() == type_id::LIST) {
1045  auto l_list_col = detail::lists_column_device_view(lcol);
1046  auto r_list_col = detail::lists_column_device_view(rcol);
1047

1049  auto rsizes = make_list_size_iterator(r_list_col);
1050  if (not thrust::equal(thrust::seq, lsizes, lsizes + lcol.size(), rsizes)) {
1051  return false;
1052  }

1055  rcol = r_list_col.get_sliced_child();
1056  if (lcol.size() != rcol.size()) { return false; }
1057  }

1061  element_comparator{check_nulls, lcol, rcol, nulls_are_equal, comparator}, lcol.size()};

1428  static constexpr hash_value_type NON_NULL_HASH = 0;
1429 
1430  public:

1438  size_type row_index) const noexcept
1439  {
1440  return _element_hasher.template operator()<T>(col, row_index);
1441  }

1444  __device__ hash_value_type operator()(column_device_view const& col,
1445  size_type row_index) const noexcept
1446  {
1447  auto hash = hash_value_type{0};
1448  column_device_view curr_col = col.slice(row_index, 1);
1449  while (is_nested(curr_col.type())) {
1450  if (_check_nulls) {
1451  auto validity_it = detail::make_validity_iterator<true>(curr_col);
1452  hash = detail::accumulate(
1453  validity_it, validity_it + curr_col.size(), hash, [](auto hash, auto is_valid) {
1454  return cudf::detail::hash_combine(hash, is_valid ? NON_NULL_HASH : NULL_HASH);
1455  });
1456  }
1457  if (curr_col.type().id() == type_id::STRUCT) {
1458  if (curr_col.num_child_columns() == 0) { return hash; }
1459  // Non-empty structs are assumed to be decomposed and contain only one child
1460  curr_col = detail::structs_column_device_view(curr_col).get_sliced_child(0);
1461  } else if (curr_col.type().id() == type_id::LIST) {
1462  auto list_col = detail::lists_column_device_view(curr_col);
1463  auto list_sizes = make_list_size_iterator(list_col);
1464  hash = detail::accumulate(
1465  list_sizes, list_sizes + list_col.size(), hash, [](auto hash, auto size) {
1466  return cudf::detail::hash_combine(hash, hash_fn<size_type>{}(size));
1467  });
1468  curr_col = list_col.get_sliced_child();
1469  }
1470  }
1471  for (int i = 0; i < curr_col.size(); ++i) {
1472  hash = cudf::detail::hash_combine(
1473  hash,
1474  type_dispatcher<dispatch_void_if_nested>(curr_col.type(), _element_hasher, curr_col, i));
1475  }
1476  return hash;
1477  }

1481  };

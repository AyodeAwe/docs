326 CUDA_HOST_DEVICE_CALLABLE constexpr decltype(auto) type_dispatcher(cudf::data_type dtype,
327  Functor f,
328  Ts&&... args)
329 {
330  switch (dtype.id()) {
331  case type_id::BOOL8:

333  std::forward<Ts>(args)...);

336  std::forward<Ts>(args)...);

339  std::forward<Ts>(args)...);

342  std::forward<Ts>(args)...);

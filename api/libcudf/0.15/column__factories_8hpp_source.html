254 template <typename B>
255 std::unique_ptr<column> make_fixed_width_column(
256  data_type type,
257  size_type size,
258  B&& null_mask,
259  size_type null_count = cudf::UNKNOWN_NULL_COUNT,
260  cudaStream_t stream = 0,
261  rmm::mr::device_memory_resource* mr = rmm::mr::get_default_resource())
262 {
263  CUDF_EXPECTS(is_fixed_width(type), "Invalid, non-fixed-width type.");
264  if (is_timestamp(type)) {
265  return make_timestamp_column(type, size, std::forward<B>(null_mask), null_count, stream, mr);
266  } else if (is_duration(type)) {
267  return make_duration_column(type, size, std::forward<B>(null_mask), null_count, stream, mr);
268  }
269  return make_numeric_column(type, size, std::forward<B>(null_mask), null_count, stream, mr);
270 }

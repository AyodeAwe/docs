87 template <typename Rep,
88  Radix Base,
89  typename T,
90  typename simt::std::enable_if_t<(simt::std::is_same<int32_t, T>::value &&
91  is_supported_representation_type<Rep>())>* = nullptr>
92 CUDA_HOST_DEVICE_CALLABLE Rep ipow(T exponent)
93 {
94  if (exponent == 0) return static_cast<Rep>(1);
95  auto extra = static_cast<Rep>(1);
96  auto square = static_cast<Rep>(Base);
97  while (exponent > 1) {
98  if (exponent & 1 /* odd */) {
99  extra *= square;
100  exponent -= 1;
101  }
102  exponent /= 2;
103  square *= square;
104  }
105  return square * extra;
106 }

147 template <typename Rep, Radix Rad, typename T>
148 CUDA_HOST_DEVICE_CALLABLE constexpr T right_shift_rounded(T const& val, scale_type const& scale)
149 {
150  Rep const factor = ipow<Rep, Rad>(scale._t);

185 template <typename Rep, Radix Rad, typename T>
186 CUDA_HOST_DEVICE_CALLABLE constexpr T shift(T const& val, scale_type const& scale)
187 {
188  if (scale == 0)

611 template <typename Rep, typename T>
612 CUDA_HOST_DEVICE_CALLABLE auto addition_overflow(T lhs, T rhs)
613 {
614  return rhs > 0 ? lhs > simt::std::numeric_limits<Rep>::max() - rhs
615  : lhs < simt::std::numeric_limits<Rep>::min() - rhs;
616 }
617 
626 template <typename Rep, typename T>
627 CUDA_HOST_DEVICE_CALLABLE auto subtraction_overflow(T lhs, T rhs)
628 {
629  return rhs > 0 ? lhs < simt::std::numeric_limits<Rep>::min() + rhs
630  : lhs > simt::std::numeric_limits<Rep>::max() + rhs;
631 }
632 
641 template <typename Rep, typename T>
642 CUDA_HOST_DEVICE_CALLABLE auto division_overflow(T lhs, T rhs)
643 {
644  return lhs == simt::std::numeric_limits<Rep>::min() && rhs == -1;
645 }

655 template <typename Rep, typename T>
656 CUDA_HOST_DEVICE_CALLABLE auto multiplication_overflow(T lhs, T rhs)
657 {
658  auto const min = simt::std::numeric_limits<Rep>::min();
659  auto const max = simt::std::numeric_limits<Rep>::max();
660  if (rhs > 0)
661  return lhs > max / rhs || lhs < min / rhs;
662  else if (rhs < -1)
663  return lhs > min / rhs || lhs < max / rhs;
664  else
665  return rhs == -1 && lhs == min;
666 }

669 template <typename Rep1, Radix Rad1>
670 CUDA_HOST_DEVICE_CALLABLE fixed_point<Rep1, Rad1> operator+(fixed_point<Rep1, Rad1> const& lhs,
671  fixed_point<Rep1, Rad1> const& rhs)
672 {
673  auto const scale = std::min(lhs._scale, rhs._scale);
674  auto const sum = lhs.rescaled(scale)._value + rhs.rescaled(scale)._value;
675 
676 #if defined(__CUDACC_DEBUG__)
677 
678  assert(("fixed_point overflow of underlying representation type " + print_rep<Rep1>(),
679  !addition_overflow<Rep1>(lhs.rescaled(scale)._value, rhs.rescaled(scale)._value)));
680 
681 #endif
682 
683  return fixed_point<Rep1, Rad1>{scaled_integer<Rep1>{sum, scale}};
684 }

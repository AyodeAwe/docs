78 template <typename Rep,
79  Radix Base,
80  typename T,
81  typename std::enable_if_t<(std::is_same<int32_t, T>::value &&
82  is_supported_representation_type<Rep>())>* = nullptr>
83 CUDA_HOST_DEVICE_CALLABLE Rep ipow(T exponent)
84 {
85  if (exponent == 0) return static_cast<Rep>(1);
86  auto extra = static_cast<Rep>(1);
87  auto square = static_cast<Rep>(Base);
88  while (exponent > 1) {
89  if (exponent & 1 /* odd */) {
90  extra *= square;
91  exponent -= 1;
92  }
93  exponent /= 2;
94  square *= square;
95  }
96  return square * extra;
97 }

138 template <typename Rep, Radix Rad, typename T>
139 CUDA_HOST_DEVICE_CALLABLE constexpr T right_shift_rounded(T const& val, scale_type const& scale)
140 {
141  Rep const factor = ipow<Rep, Rad>(scale._t);

176 template <typename Rep, Radix Rad, typename T>
177 CUDA_HOST_DEVICE_CALLABLE constexpr T shift(T const& val, scale_type const& scale)
178 {
179  if (scale == 0)
180  return val;
181  else if (scale > 0)
182  return right_shift<Rep, Rad>(val, scale);
183  else
184  return left_shift<Rep, Rad>(val, scale);

228 template <typename Rep,
229  Radix Rad,
230  typename T,
231  typename std::enable_if_t<std::is_floating_point<T>::value>* = nullptr>
232 CUDA_HOST_DEVICE_CALLABLE auto shift_with_precise_round(T const& value, scale_type const& scale)
233  -> Rep
234 {
235  if (scale == 0) return value;

581 template <typename Rep, typename T>
582 CUDA_HOST_DEVICE_CALLABLE auto addition_overflow(T lhs, T rhs)
583 {
584  return rhs > 0 ? lhs > std::numeric_limits<Rep>::max() - rhs
585  : lhs < std::numeric_limits<Rep>::min() - rhs;
586 }
587 
596 template <typename Rep, typename T>
597 CUDA_HOST_DEVICE_CALLABLE auto subtraction_overflow(T lhs, T rhs)
598 {
599  return rhs > 0 ? lhs < std::numeric_limits<Rep>::min() + rhs
600  : lhs > std::numeric_limits<Rep>::max() + rhs;
601 }
602 
611 template <typename Rep, typename T>
612 CUDA_HOST_DEVICE_CALLABLE auto division_overflow(T lhs, T rhs)
613 {
614  return lhs == std::numeric_limits<Rep>::min() && rhs == -1;
615 }
616 
625 template <typename Rep, typename T>
626 CUDA_HOST_DEVICE_CALLABLE auto multiplication_overflow(T lhs, T rhs)
627 {
628  auto const min = std::numeric_limits<Rep>::min();
629  auto const max = std::numeric_limits<Rep>::max();
630  if (rhs > 0)
631  return lhs > max / rhs || lhs < min / rhs;
632  else if (rhs < -1)
633  return lhs > min / rhs || lhs < max / rhs;
634  else
635  return rhs == -1 && lhs == min;
636 }

638 // PLUS Operation
639 template <typename Rep1, Radix Rad1>
640 CUDA_HOST_DEVICE_CALLABLE fixed_point<Rep1, Rad1> operator+(fixed_point<Rep1, Rad1> const& lhs,
641  fixed_point<Rep1, Rad1> const& rhs)
642 {
643  auto const scale = std::min(lhs._scale, rhs._scale);
644  auto const sum = lhs.rescaled(scale)._value + rhs.rescaled(scale)._value;
645 
646 #if defined(__CUDACC_DEBUG__)
647 
648  assert(("fixed_point overflow of underlying representation type " + print_rep<Rep1>(),
649  !addition_overflow<Rep1>(lhs.rescaled(scale)._value, rhs.rescaled(scale)._value)));
650 
651 #endif
652 
653  return fixed_point<Rep1, Rad1>{scaled_integer<Rep1>{sum, scale}};
654 }

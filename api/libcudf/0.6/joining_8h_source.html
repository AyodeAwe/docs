46 template <JoinType join_type,
47 typename output_index_type,
48 typename size_type>
49 gdf_error join_hash( gdf_table<size_type> const & left_table,
50 gdf_table<size_type> const & right_table,
51 gdf_column * const output_l,
52 gdf_column * const output_r,
53 bool flip_indices = false )
54 {
55
56 // Hash table is built on the right table.
57 // For inner joins, doesn't matter which table is build/probe, so we want
58 // to build the hash table on the smaller table.
59 if ((join_type == JoinType::INNER_JOIN) &&
60 (right_table.get_column_length() > left_table.get_column_length()))
61 {
62 return join_hash<join_type, output_index_type>(right_table,
63 left_table,
64 output_l,
65 output_r,
66 true );
67 }
68
69 return compute_hash_join<join_type, output_index_type>(output_l,
70 output_r,
71 left_table,
72 right_table,
73 flip_indices);
74 }

84 virtual void * alloc( size_t size, memory_space_t space) {
85 void *p = nullptr ;
86 if (size) {
87 if (memory_space_device == space) {
88 if (RMM_SUCCESS != RMM_ALLOC(&p, size, stream()))
89 throw cuda_exception_t(cudaPeekAtLastError());
90 }
91 else {
92 cudaError_t result = cudaMallocHost(&p, size);
93 if (cudaSuccess != result) throw cuda_exception_t(result);
94 }
95 }
96 return p;
97 }
98
99 virtual void free( void * p, memory_space_t space) {
100 if (p) {
101 if (memory_space_device == space) {
102 if (RMM_SUCCESS != RMM_FREE(p, stream()))
103 throw cuda_exception_t(cudaPeekAtLastError());
104 }
105 else {
106 cudaError_t result = cudaFreeHost(&p);
107 if (cudaSuccess != result) throw cuda_exception_t(result);
108 }
109 }
110 }
111 };

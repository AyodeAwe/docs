133 gdf_error create_bit_mask(bit_mask_t **mask, gdf_size_type number_of_records, int fill_value = -1, gdf_size_type padding_bytes = 64) {
134 //
135 // To handle padding, we will round the number_of_records up to the next padding boundary, then identify how many element
136 // that equates to. Then we can allocate the appropriate amount of storage.
137 //
138 gdf_size_type num_bytes = (number_of_records + 7) / 8;
139 gdf_size_type num_padding_blocks = (num_bytes + padding_bytes - 1) / padding_bytes;
140 gdf_size_type num_elements = bit_mask::num_elements(num_padding_blocks * 8 * padding_bytes);
141
142 RMM_TRY(RMM_ALLOC(mask, sizeof (bit_mask_t) * num_elements, 0));
143
144 if (fill_value == 0) {
145 CUDA_TRY(cudaMemset(*mask, 0, sizeof (bit_mask_t) * num_elements));
146 } else if (fill_value == 1) {
147 //
148 // Value outside range of [0, num_rows) is undefined, so we will
149 // initialize in the simplest manner... we'll initialize all
150 // elements to 1.
151 //
152 CUDA_TRY(cudaMemset(*mask, 0xff, sizeof (bit_mask_t) * num_elements));
153 }
154
155 return GDF_SUCCESS;
156 }

166 template < typename T>
167 CUDA_HOST_DEVICE_CALLABLE
168 bool is_valid( const bit_mask_t *valid, T record_idx) {
169 static_assert(std::is_integral<T>::value, "Record index must be of an integral type" );
170
171 const gdf_size_type rec{detail::which_element(record_idx)};
172 const gdf_size_type bit{detail::which_bit(record_idx)};
173
174 return ((valid[rec] & (bit_mask_t{1} << bit)) != 0);
175 }

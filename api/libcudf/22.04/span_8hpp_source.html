40 template <typename T, std::size_t Extent, typename Derived>
41 class span_base {
42  static_assert(Extent == dynamic_extent, "Only dynamic extent is supported");
43 
44  public:
45  using element_type = T;
46  using value_type = std::remove_cv<T>;
47  using size_type = std::size_t;
48  using difference_type = std::ptrdiff_t;
49  using pointer = T*;
50  using iterator = T*;
51  using const_pointer = T const*;
52  using reference = T&;
53  using const_reference = T const&;
54 
55  static constexpr std::size_t extent = Extent;
56 
57  constexpr span_base() noexcept {}
58  constexpr span_base(pointer data, size_type size) : _data(data), _size(size) {}
59  // constexpr span_base(pointer begin, pointer end) : _data(begin), _size(end - begin) {}
60  constexpr span_base(span_base const& other) noexcept = default;
61  constexpr span_base& operator=(span_base const& other) noexcept = default;
62 
63  // not noexcept due to undefined behavior when size = 0
64  constexpr reference front() const { return _data[0]; }
65  // not noexcept due to undefined behavior when size = 0
66  constexpr reference back() const { return _data[_size - 1]; }
67  // not noexcept due to undefined behavior when idx < 0 || idx >= size
68  constexpr reference operator[](size_type idx) const { return _data[idx]; }
69 
70  constexpr iterator begin() const noexcept { return _data; }
71  constexpr iterator end() const noexcept { return _data + _size; }
72  constexpr pointer data() const noexcept { return _data; }
73 
74  [[nodiscard]] constexpr size_type size() const noexcept { return _size; }
75  [[nodiscard]] constexpr size_type size_bytes() const noexcept { return sizeof(T) * _size; }
76  [[nodiscard]] constexpr bool empty() const noexcept { return _size == 0; }
77 
83  constexpr Derived first(size_type count) const noexcept { return Derived(_data, count); }
84 
90  constexpr Derived last(size_type count) const noexcept
91  {
92  return Derived(_data + _size - count, count);
93  }
94 
95  constexpr Derived subspan(size_type offset, size_type count) const noexcept
96  {
97  return Derived(_data + offset, count);
98  }
99 
100  private:
101  pointer _data{nullptr};
102  size_type _size{0};
103 };

239 template <typename T, template <typename, std::size_t> typename RowType>
240 class base_2dspan {
241  public:
242  using size_type = std::pair<size_t, size_t>;
243 
244  constexpr base_2dspan() noexcept = default;
245  constexpr base_2dspan(T* data, size_t rows, size_t columns) noexcept
246  : _data{data}, _size{rows, columns}
247  {
248  }
249  base_2dspan(T* data, size_type size) noexcept : _data{data}, _size{size} {}
250 
251  constexpr auto data() const noexcept { return _data; }
252  constexpr auto size() const noexcept { return _size; }
253  constexpr auto count() const noexcept { return size().first * size().second; }
254  [[nodiscard]] constexpr bool is_empty() const noexcept { return count() == 0; }
255 
256  static constexpr size_t flatten_index(size_t row, size_t column, size_type size) noexcept
257  {
258  return row * size.second + column;
259  }
260 
261  constexpr RowType<T, dynamic_extent> operator[](size_t row) const
262  {
263  return {this->data() + flatten_index(row, 0, this->size()), this->size().second};
264  }
265 
266  [[nodiscard]] constexpr RowType<T, dynamic_extent> front() const { return (*this)[0]; }
267  [[nodiscard]] constexpr RowType<T, dynamic_extent> back() const
268  {
269  return (*this)[size().first - 1];
270  }
271 
272  constexpr base_2dspan subspan(size_t first_row, size_t num_rows) const noexcept
273  {
274  return base_2dspan(
275  _data + flatten_index(first_row, 0, this->size()), num_rows, this->size().second);
276  }
277 
278  constexpr RowType<T, dynamic_extent> flat_view()
279  {
280  return {this->data(), this->size().first * this->size().second};
281  }

93 template <typename From, typename To>
94 struct fixed_width_type_converter {
95  // Are the types same - simply copy elements from [begin, end) to out
96  template <typename FromT = From,
97  typename ToT = To,
98  std::enable_if_t<std::is_same_v<FromT, ToT>, void>* = nullptr>
99  constexpr ToT operator()(FromT element) const
100  {
101  return element;
102  }
103 
104  // Are the types convertible or can target be constructed from source?
105  template <
106  typename FromT = From,
107  typename ToT = To,
108  std::enable_if_t<!std::is_same_v<FromT, ToT> && (cudf::is_convertible<FromT, ToT>::value ||
109  std::is_constructible_v<ToT, FromT>),
110  void>* = nullptr>
111  constexpr ToT operator()(FromT element) const
112  {
113  return static_cast<ToT>(element);
114  }
115 
116  // Convert integral values to timestamps
117  template <
118  typename FromT = From,
119  typename ToT = To,
120  std::enable_if_t<std::is_integral_v<FromT> && cudf::is_timestamp<ToT>(), void>* = nullptr>
121  constexpr ToT operator()(FromT element) const
122  {
123  return ToT{typename ToT::duration{element}};
124  }
125 };

138  typename ElementFrom,
139  typename InputIterator,
140  std::enable_if_t<not cudf::is_fixed_point<ElementTo>()>* = nullptr>
141 rmm::device_buffer make_elements(InputIterator begin, InputIterator end)
142 {
143  static_assert(cudf::is_fixed_width<ElementTo>(), "Unexpected non-fixed width type.");
144  auto transformer = fixed_width_type_converter<ElementFrom, ElementTo>{};
145  auto transform_begin = thrust::make_transform_iterator(begin, transformer);
146  auto const size = cudf::distance(begin, end);
147  auto const elements = thrust::host_vector<ElementTo>(transform_begin, transform_begin + size);
148  return rmm::device_buffer{elements.data(), size * sizeof(ElementTo), rmm::cuda_stream_default};
149 }

217 template <typename ValidityIterator>
218 std::vector<bitmask_type> make_null_mask_vector(ValidityIterator begin, ValidityIterator end)
219 {
220  auto const size = cudf::distance(begin, end);
221  auto const num_words = cudf::bitmask_allocation_size_bytes(size) / sizeof(bitmask_type);
222 
223  auto null_mask = std::vector<bitmask_type>(num_words, 0);
224  for (auto i = 0; i < size; ++i)
225  if (*(begin + i)) set_bit_unsafe(null_mask.data(), i);

266 template <typename StringsIterator, typename ValidityIterator>
267 auto make_chars_and_offsets(StringsIterator begin, StringsIterator end, ValidityIterator v)
268 {
269  std::vector<char> chars{};
270  std::vector<cudf::size_type> offsets(1, 0);
271  for (auto str = begin; str < end; ++str) {
272  std::string tmp = (*v++) ? std::string(*str) : std::string{};
273  chars.insert(chars.end(), std::cbegin(tmp), std::cend(tmp));
274  offsets.push_back(offsets.back() + tmp.length());
275  }
276  return std::make_pair(std::move(chars), std::move(offsets));
277 };

1480  std::vector<std::unique_ptr<column>> stubs;

1488  cols.cend(),

1490  std::back_inserter(offsetv),
1491  [&](cudf::column_view const& col, bool valid) {
1492  // nulls are represented as a repeated offset
1493  size_type ret = count;
1494  if (valid) { count += col.size(); }
1495  return ret;
1496  });

1498  offsetv.push_back(count);

1500  cudf::test::fixed_width_column_wrapper<size_type>(offsetv.begin(), offsetv.end()).release();

1504  thrust::copy_if(
1505  std::cbegin(cols), std::cend(cols), valids, std::back_inserter(children), thrust::identity{});

1508

1510  depth = expected_depth + 1;

1605  std::pair<std::vector<column_view>, std::vector<std::unique_ptr<column>>> preprocess_columns(
1606  std::initializer_list<lists_column_wrapper<T, SourceElementT>> const& elements,
1607  column_view& expected_hierarchy,
1608  int expected_depth)
1609  {
1610  std::vector<std::unique_ptr<column>> stubs;
1611  std::vector<column_view> cols;
1612 
1613  // preprocess the incoming lists.
1614  // - unwrap any "root" lists
1615  // - handle incomplete hierarchies
1616  std::transform(elements.begin(),
1617  elements.end(),
1618  std::back_inserter(cols),
1619  [&](lists_column_wrapper const& l) -> column_view {
1620  // depth mismatch. attempt to normalize the short column.
1621  // this function will also catch if this is a legitimately broken
1622  // set of input
1623  if (l.depth < expected_depth) {
1624  if (l.root) {
1625  // this exception distinguishes between the following two cases:
1626  //
1627  // { {{{1, 2, 3}}}, {} }
1628  // In this case, row 0 is a List<List<List<int>>>, whereas row 1 is
1629  // just a List<> which is an apparent mismatch. However, because row 1
1630  // is empty we will allow that to semantically mean
1631  // "a List<List<List<int>>> that's empty at the top level"
1632  //
1633  // { {{{1, 2, 3}}}, {4, 5, 6} }
1634  // In this case, row 1 is a a concrete List<int> with actual values.
1635  // There is no way to rectify the differences so we will treat it as a
1636  // true column mismatch.
1637  CUDF_EXPECTS(l.wrapped->size() == 0, "Mismatch in column types!");
1638  stubs.push_back(empty_like(expected_hierarchy));
1639  } else {
1640  stubs.push_back(normalize_column(l.get_view(), expected_hierarchy));
1641  }
1642  return *(stubs.back());
1643  }
1644  // the empty hierarchy case
1645  return l.get_view();
1646  });
1647 
1648  return {std::move(cols), std::move(stubs)};
1649  }
1650 
1651  column_view get_view() const { return root ? lists_column_view(*wrapped).child() : *wrapped; }
1652 
1653  int depth = 0;
1654  bool root = false;
1655 };

350  {
351  const std::string uri_start_delimiter = "//";
352  const std::string uri_end_delimiter = "?";
353 
354  arrow::Result<std::shared_ptr<arrow::fs::FileSystem>> result =
355  arrow::fs::FileSystemFromUri(static_cast<std::string>(arrow_uri));
356  CUDF_EXPECTS(result.ok(), "Failed to generate Arrow Filesystem instance from URI.");
357  filesystem = result.ValueOrDie();
358 
359  // Parse the path from the URI
360  size_t start = arrow_uri.find(uri_start_delimiter) == std::string::npos
361  ? 0
362  : arrow_uri.find(uri_start_delimiter) + uri_start_delimiter.size();
363  size_t end = arrow_uri.find(uri_end_delimiter) - start;
364  std::string_view path = arrow_uri.substr(start, end);
365 
366  arrow::Result<std::shared_ptr<arrow::io::RandomAccessFile>> in_stream =
367  filesystem->OpenInputFile(static_cast<std::string>(path).c_str());
368  CUDF_EXPECTS(in_stream.ok(), "Failed to open Arrow RandomAccessFile");
369  arrow_file = in_stream.ValueOrDie();
370  }

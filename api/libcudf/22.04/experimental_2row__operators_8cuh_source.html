160  bool const rhs_is_null{_rhs.is_null(rhs_element_index)};

186  size_type const rhs_element_index)

188  column_device_view lcol = _lhs;

228  __device__ bool operator()(size_type const lhs_index, size_type const rhs_index) const noexcept
229  {
230  int last_null_depth = std::numeric_limits<int>::max();
231  for (size_type i = 0; i < _lhs.num_columns(); ++i) {
232  int const depth = _depth.has_value() ? (*_depth)[i] : 0;
233  if (depth > last_null_depth) { continue; }
234 
235  bool const ascending =
236  _column_order.has_value() ? (*_column_order)[i] == order::ASCENDING : true;
237 
238  null_order const null_precedence =
239  _null_precedence.has_value() ? (*_null_precedence)[i] : null_order::BEFORE;
240 
241  auto const comparator =
242  element_comparator{_nulls, _lhs.column(i), _rhs.column(i), null_precedence, depth};
243 
244  weak_ordering state;
245  cuda::std::tie(state, last_null_depth) =
246  cudf::type_dispatcher(_lhs.column(i).type(), comparator, lhs_index, rhs_index);
247 
248  if (state == weak_ordering::EQUIVALENT) { continue; }
249 
250  return state == (ascending ? weak_ordering::LESS : weak_ordering::GREATER);
251  }

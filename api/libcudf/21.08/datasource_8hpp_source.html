315  {
316  const std::string uri_start_delimiter = "//";
317  const std::string uri_end_delimiter = "?";
318 
319  arrow::Result<std::shared_ptr<arrow::fs::FileSystem>> result =
320  arrow::fs::FileSystemFromUri(static_cast<std::string>(arrow_uri));
321  CUDF_EXPECTS(result.ok(), "Failed to generate Arrow Filesystem instance from URI.");
322  filesystem = result.ValueOrDie();
323 
324  // Parse the path from the URI
325  size_t start = arrow_uri.find(uri_start_delimiter) == std::string::npos
326  ? 0
327  : arrow_uri.find(uri_start_delimiter) + uri_start_delimiter.size();
328  size_t end = arrow_uri.find(uri_end_delimiter) - start;
329  std::string_view path = arrow_uri.substr(start, end);
330 
331  arrow::Result<std::shared_ptr<arrow::io::RandomAccessFile>> in_stream =
332  filesystem->OpenInputFile(static_cast<std::string>(path).c_str());
333  CUDF_EXPECTS(in_stream.ok(), "Failed to open Arrow RandomAccessFile");
334  arrow_file = in_stream.ValueOrDie();

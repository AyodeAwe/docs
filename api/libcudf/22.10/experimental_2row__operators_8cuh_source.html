317  __device__ cuda::std::pair<weak_ordering, int> operator()(
318  size_type const lhs_element_index, size_type const rhs_element_index) const noexcept
319  {
320  if (_check_nulls) {
321  bool const lhs_is_null{_lhs.is_null(lhs_element_index)};
322  bool const rhs_is_null{_rhs.is_null(rhs_element_index)};

324  if (lhs_is_null or rhs_is_null) { // at least one is null
325  return cuda::std::pair(null_compare(lhs_is_null, rhs_is_null, _null_precedence), _depth);

343  template <typename Element, CUDF_ENABLE_IF(std::is_same_v<Element, cudf::struct_view>)>
344  __device__ cuda::std::pair<weak_ordering, int> operator()(
345  size_type const lhs_element_index, size_type const rhs_element_index) const noexcept
346  {
347  column_device_view lcol = _lhs;
348  column_device_view rcol = _rhs;
349  int depth = _depth;
350  while (lcol.type().id() == type_id::STRUCT) {
351  bool const lhs_is_null{lcol.is_null(lhs_element_index)};
352  bool const rhs_is_null{rcol.is_null(rhs_element_index)};
353 
354  if (lhs_is_null or rhs_is_null) { // at least one is null
355  weak_ordering state = null_compare(lhs_is_null, rhs_is_null, _null_precedence);
356  return cuda::std::pair(state, depth);
357  }
358 
359  if (lcol.num_child_columns() == 0) {
360  return cuda::std::pair(weak_ordering::EQUIVALENT, depth);
361  }
362 
363  // Non-empty structs have been modified to only have 1 child when using this.
364  lcol = detail::structs_column_device_view(lcol).get_sliced_child(0);
365  rcol = detail::structs_column_device_view(rcol).get_sliced_child(0);
366  ++depth;
367  }
368 
369  return cudf::type_dispatcher<dispatch_void_if_nested>(
370  lcol.type(),
371  element_comparator{_check_nulls, lcol, rcol, _null_precedence, depth, _comparator},
372  lhs_element_index,
373  rhs_element_index);
374  }

395  __device__ constexpr weak_ordering operator()(size_type const lhs_index,
396  size_type const rhs_index) const noexcept
397  {
398  int last_null_depth = std::numeric_limits<int>::max();
399  for (size_type i = 0; i < _lhs.num_columns(); ++i) {
400  int const depth = _depth.has_value() ? (*_depth)[i] : 0;
401  if (depth > last_null_depth) { continue; }
402 
403  bool const ascending =
404  _column_order.has_value() ? (*_column_order)[i] == order::ASCENDING : true;
405 
406  null_order const null_precedence =
407  _null_precedence.has_value() ? (*_null_precedence)[i] : null_order::BEFORE;
408 
409  weak_ordering state;
410  cuda::std::tie(state, last_null_depth) = cudf::type_dispatcher(
411  _lhs.column(i).type(),
412  element_comparator{
413  _check_nulls, _lhs.column(i), _rhs.column(i), null_precedence, depth, _comparator},
414  lhs_index,
415  rhs_index);
416 
417  if (state == weak_ordering::EQUIVALENT) { continue; }
418 
419  return ascending
420  ? state
421  : (state == weak_ordering::GREATER ? weak_ordering::LESS : weak_ordering::GREATER);
422  }
423  return weak_ordering::EQUIVALENT;
424  }

1023  template <typename Element, CUDF_ENABLE_IF(cudf::is_nested<Element>())>
1024  __device__ bool operator()(size_type const lhs_element_index,

1028  column_device_view rcol = rhs.slice(rhs_element_index, 1);
1029  while (is_nested(lcol.type())) {
1030  if (check_nulls) {
1031  auto lvalid = detail::make_validity_iterator<true>(lcol);
1032  auto rvalid = detail::make_validity_iterator<true>(rcol);
1033  if (nulls_are_equal == null_equality::UNEQUAL) {
1034  if (thrust::any_of(
1035  thrust::seq, lvalid, lvalid + lcol.size(), thrust::logical_not<bool>()) or
1036  thrust::any_of(
1037  thrust::seq, rvalid, rvalid + rcol.size(), thrust::logical_not<bool>())) {
1038  return false;
1039  }
1040  } else {
1041  if (not thrust::equal(thrust::seq, lvalid, lvalid + lcol.size(), rvalid)) {
1042  return false;
1043  }
1044  }
1045  }
1046  if (lcol.type().id() == type_id::STRUCT) {
1047  if (lcol.num_child_columns() == 0) { return true; }
1048  // Non-empty structs are assumed to be decomposed and contain only one child
1049  lcol = detail::structs_column_device_view(lcol).get_sliced_child(0);
1050  rcol = detail::structs_column_device_view(rcol).get_sliced_child(0);
1051  } else if (lcol.type().id() == type_id::LIST) {
1052  auto l_list_col = detail::lists_column_device_view(lcol);
1053  auto r_list_col = detail::lists_column_device_view(rcol);
1054 
1055  auto lsizes = make_list_size_iterator(l_list_col);
1056  auto rsizes = make_list_size_iterator(r_list_col);
1057  if (not thrust::equal(thrust::seq, lsizes, lsizes + lcol.size(), rsizes)) {
1058  return false;
1059  }
1060 
1061  lcol = l_list_col.get_sliced_child();
1062  rcol = r_list_col.get_sliced_child();
1063  if (lcol.size() != rcol.size()) { return false; }
1064  }
1065  }
1066 
1067  auto comp = column_comparator{
1068  element_comparator{check_nulls, lcol, rcol, nulls_are_equal, comparator}, lcol.size()};
1069  return type_dispatcher<dispatch_void_if_nested>(lcol.type(), comp);
1070  }

1398 template <template <typename> class hash_function, typename Nullate>
1399 class device_row_hasher {
1400  friend class row_hasher;
1401 
1402  public:
1409  __device__ auto operator()(size_type row_index) const noexcept
1410  {
1411  auto it = thrust::make_transform_iterator(_table.begin(), [=](auto const& column) {
1412  return cudf::type_dispatcher<dispatch_storage_type>(
1413  column.type(), element_hasher_adapter<hash_function>{_check_nulls}, column, row_index);
1414  });
1415 
1416  // Hash each element and combine all the hash values together
1417  return detail::accumulate(it, it + _table.num_columns(), _seed, [](auto hash, auto h) {
1418  return cudf::detail::hash_combine(hash, h);
1419  });
1420  }
1421 
1422  private:
1430  template <template <typename> class hash_fn>
1431  class element_hasher_adapter {
1432  static constexpr hash_value_type NULL_HASH = std::numeric_limits<hash_value_type>::max();
1433  static constexpr hash_value_type NON_NULL_HASH = 0;
1434 
1435  public:
1436  __device__ element_hasher_adapter(Nullate check_nulls) noexcept
1437  : _element_hasher(check_nulls), _check_nulls(check_nulls)
1438  {
1439  }
1440 
1441  template <typename T, CUDF_ENABLE_IF(not cudf::is_nested<T>())>
1442  __device__ hash_value_type operator()(column_device_view const& col,
1443  size_type row_index) const noexcept
1444  {
1445  return _element_hasher.template operator()<T>(col, row_index);
1446  }
1447 
1448  template <typename T, CUDF_ENABLE_IF(cudf::is_nested<T>())>
1449  __device__ hash_value_type operator()(column_device_view const& col,
1450  size_type row_index) const noexcept
1451  {
1452  auto hash = hash_value_type{0};
1453  column_device_view curr_col = col.slice(row_index, 1);
1454  while (is_nested(curr_col.type())) {
1455  if (_check_nulls) {
1456  auto validity_it = detail::make_validity_iterator<true>(curr_col);
1457  hash = detail::accumulate(
1458  validity_it, validity_it + curr_col.size(), hash, [](auto hash, auto is_valid) {
1459  return cudf::detail::hash_combine(hash, is_valid ? NON_NULL_HASH : NULL_HASH);
1460  });
1461  }
1462  if (curr_col.type().id() == type_id::STRUCT) {
1463  if (curr_col.num_child_columns() == 0) { return hash; }
1464  // Non-empty structs are assumed to be decomposed and contain only one child
1465  curr_col = detail::structs_column_device_view(curr_col).get_sliced_child(0);
1466  } else if (curr_col.type().id() == type_id::LIST) {
1467  auto list_col = detail::lists_column_device_view(curr_col);
1468  auto list_sizes = make_list_size_iterator(list_col);
1469  hash = detail::accumulate(
1470  list_sizes, list_sizes + list_col.size(), hash, [](auto hash, auto size) {
1471  return cudf::detail::hash_combine(hash, hash_fn<size_type>{}(size));
1472  });
1473  curr_col = list_col.get_sliced_child();
1474  }
1475  }
1476  for (int i = 0; i < curr_col.size(); ++i) {
1477  hash = cudf::detail::hash_combine(
1478  hash,
1479  type_dispatcher<dispatch_void_if_nested>(curr_col.type(), _element_hasher, curr_col, i));
1480  }
1481  return hash;
1482  }
1483 
1484  element_hasher<hash_fn, Nullate> const _element_hasher;
1485  Nullate const _check_nulls;
1486  };

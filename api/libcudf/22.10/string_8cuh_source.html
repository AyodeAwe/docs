113 inline __device__ bool is_float(string_view const& d_str)
114 {
115  if (d_str.empty()) return false;
116  bool decimal_found = false;
117  bool exponent_found = false;
118  size_type bytes = d_str.size_bytes();
119  const char* data = d_str.data();
120  // sign character allowed at the beginning of the string
121  size_type ch_idx = (*data == '-' || *data == '+') ? 1 : 0;
122 
123  bool result = ch_idx < bytes;
124  // check for nan and infinity strings
125  if (result && data[ch_idx] > '9') {
126  auto const inf_nan = string_view(data + ch_idx, bytes - ch_idx);
127  if (is_nan_str(inf_nan) || is_inf_str(inf_nan)) return true;
128  }
129 
130  // check for float chars [0-9] and a single decimal '.'
131  // and scientific notation [eE][+-][0-9]
132  for (; ch_idx < bytes; ++ch_idx) {
133  auto chr = data[ch_idx];
134  if (chr >= '0' && chr <= '9') continue;
135  if (!decimal_found && chr == '.') {
136  decimal_found = true; // no more decimals
137  continue;
138  }
139  if (!exponent_found && (chr == 'e' || chr == 'E')) {
140  if (ch_idx + 1 < bytes) chr = data[ch_idx + 1];
141  if (chr == '-' || chr == '+') ++ch_idx;
142  decimal_found = true; // no decimal allowed in exponent
143  exponent_found = true; // no more exponents
144  continue;
145  }
146  return false;
147  }
148  return result;
149 }

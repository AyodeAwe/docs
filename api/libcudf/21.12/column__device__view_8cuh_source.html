1412 ColumnDeviceView* child_columns_to_device_array(ColumnViewIterator child_begin,
1413  ColumnViewIterator child_end,
1414  void* h_ptr,
1415  void* d_ptr)
1416 {
1417  ColumnDeviceView* d_children = detail::align_ptr_for_type<ColumnDeviceView>(d_ptr);
1418  auto num_children = std::distance(child_begin, child_end);
1419  if (num_children > 0) {
1420  // The beginning of the memory must be the fixed-sized ColumnDeviceView
1421  // struct objects in order for d_children to be used as an array.
1422  auto h_column = detail::align_ptr_for_type<ColumnDeviceView>(h_ptr);
1423  auto d_column = d_children;
1424 
1425  // Any child data is assigned past the end of this array: h_end and d_end.
1426  auto h_end = reinterpret_cast<int8_t*>(h_column + num_children);
1427  auto d_end = reinterpret_cast<int8_t*>(d_column + num_children);
1428  std::for_each(child_begin, child_end, [&](auto const& col) {
1429  // inplace-new each child into host memory
1430  new (h_column) ColumnDeviceView(col, h_end, d_end);
1431  h_column++; // advance to next child
1432  // update the pointers for holding this child column's child data
1433  auto col_child_data_size = ColumnDeviceView::extent(col) - sizeof(ColumnDeviceView);
1434  h_end += col_child_data_size;
1435  d_end += col_child_data_size;
1436  });
1437  }
1438  return d_children;
1439 }

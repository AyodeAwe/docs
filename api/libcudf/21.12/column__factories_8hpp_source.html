314 template <typename B>
315 std::unique_ptr<column> make_fixed_width_column(
316  data_type type,
317  size_type size,
318  B&& null_mask,
319  size_type null_count = cudf::UNKNOWN_NULL_COUNT,
320  rmm::cuda_stream_view stream = rmm::cuda_stream_default,
321  rmm::mr::device_memory_resource* mr = rmm::mr::get_current_device_resource())
322 {
323  CUDF_EXPECTS(is_fixed_width(type), "Invalid, non-fixed-width type.");
324  if (is_timestamp(type)) {
325  return make_timestamp_column(type, size, std::forward<B>(null_mask), null_count, stream, mr);
326  } else if (is_duration(type)) {
327  return make_duration_column(type, size, std::forward<B>(null_mask), null_count, stream, mr);
328  } else if (is_fixed_point(type)) {
329  return make_fixed_point_column(type, size, std::forward<B>(null_mask), null_count, stream, mr);
330  }
331  return make_numeric_column(type, size, std::forward<B>(null_mask), null_count, stream, mr);
332 }

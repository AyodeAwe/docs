113 template <typename From, typename To>
114 struct fixed_width_type_converter {
115  // Are the types same - simply copy elements from [begin, end) to out
116  template <typename FromT = From,
117  typename ToT = To,
118  typename std::enable_if<std::is_same<FromT, ToT>::value, void>::type* = nullptr>
119  ToT operator()(FromT element) const
120  {
121  return element;
122  }
123 
124  // Are the types convertible or can target be constructed from source?
125  template <typename FromT = From,
126  typename ToT = To,
127  typename std::enable_if<!std::is_same<FromT, ToT>::value &&
128  (cudf::is_convertible<FromT, ToT>::value ||
129  std::is_constructible<ToT, FromT>::value),
130  void>::type* = nullptr>
131  ToT operator()(FromT element) const
132  {
133  return static_cast<ToT>(element);
134  }
135 
136  // Convert integral values to timestamps
137  template <
138  typename FromT = From,
139  typename ToT = To,
140  typename std::enable_if<std::is_integral<FromT>::value && cudf::is_timestamp_t<ToT>::value,
141  void>::type* = nullptr>
142  ToT operator()(FromT element) const
143  {
144  return ToT{typename ToT::duration{element}};
145  }
146 };

286 auto make_chars_and_offsets(StringsIterator begin, StringsIterator end, ValidityIterator v)
287 {
288  std::vector<char> chars{};
289  std::vector<int32_t> offsets(1, 0);
290  for (auto str = begin; str < end; ++str) {
291  std::string tmp = (*v++) ? std::string(*str) : std::string{};

985  void build_from_nested(std::initializer_list<lists_column_wrapper<T, SourceElementT>> elements,
986  std::vector<bool> const& v)
987  {
988  auto valids = cudf::test::make_counting_transform_iterator(
989  0, [&v](auto i) { return v.empty() ? true : v[i]; });
990 
991  // compute the expected hierarchy and depth
992  auto const hierarchy_and_depth = std::accumulate(
993  elements.begin(),
994  elements.end(),
995  std::pair<column_view, int32_t>{{}, -1},
996  [](auto acc, lists_column_wrapper const& lcw) {
997  return lcw.depth > acc.second ? std::make_pair(lcw.get_view(), lcw.depth) : acc;
998  });
999  column_view expected_hierarchy = hierarchy_and_depth.first;
1000  int32_t const expected_depth = hierarchy_and_depth.second;
1001 
1002  // preprocess columns so that every column_view in 'cols' is an equivalent hierarchy
1003  std::vector<std::unique_ptr<column>> stubs;
1004  std::vector<column_view> cols;
1005  std::tie(cols, stubs) = preprocess_columns(elements, expected_hierarchy, expected_depth);
1006 
1007  // generate offsets
1008  size_type count = 0;
1009  std::vector<size_type> offsetv;
1010  std::transform(cols.cbegin(),
1011  cols.cend(),
1012  valids,
1013  std::back_inserter(offsetv),
1014  [&](cudf::column_view const& col, bool valid) {
1015  // nulls are represented as a repeated offset
1016  size_type ret = count;
1017  if (valid) { count += col.size(); }
1018  return ret;
1019  });
1020  // add the final offset
1021  offsetv.push_back(count);
1022  auto offsets =
1023  cudf::test::fixed_width_column_wrapper<size_type>(offsetv.begin(), offsetv.end()).release();
1024 
1025  // concatenate them together, skipping children that are null.
1026  std::vector<column_view> children;
1027  thrust::copy_if(std::cbegin(cols),
1028  std::cend(cols),
1029  valids, // stencil
1030  std::back_inserter(children),
1031  thrust::identity<bool>{});
1032 
1033  auto data = children.empty() ? cudf::empty_like(expected_hierarchy) : concatenate(children);
1034 
1035  // increment depth
1036  depth = expected_depth + 1;
1037 
1038  // construct the list column
1039  wrapped =
1040  make_lists_column(cols.size(),
1041  std::move(offsets),
1042  std::move(data),
1043  v.size() <= 0 ? 0 : cudf::UNKNOWN_NULL_COUNT,
1044  v.size() <= 0 ? rmm::device_buffer{0}
1045  : cudf::test::detail::make_null_mask(v.begin(), v.end()));
1046  }

1111  std::unique_ptr<column> normalize_column(column_view const& col,
1112  column_view const& expected_hierarchy)
1113  {
1114  // if are at the bottom of the short column, it must be empty
1115  if (col.type().id() != type_id::LIST) {
1116  CUDF_EXPECTS(col.size() == 0, "Encountered mismatched column!");
1117 
1118  auto remainder = empty_like(expected_hierarchy);
1119  return remainder;
1120  }
1121 
1122  lists_column_view lcv(col);
1123  return make_lists_column(col.size(),
1124  std::make_unique<column>(lcv.offsets()),
1125  normalize_column(lists_column_view(col).child(),
1126  lists_column_view(expected_hierarchy).child()),
1127  UNKNOWN_NULL_COUNT,
1128  copy_bitmask(col));

1131  std::pair<std::vector<column_view>, std::vector<std::unique_ptr<column>>> preprocess_columns(
1132  std::initializer_list<lists_column_wrapper<T, SourceElementT>> const& elements,
1133  column_view& expected_hierarchy,
1134  int expected_depth)
1135  {
1136  std::vector<std::unique_ptr<column>> stubs;
1137  std::vector<column_view> cols;
1138 
1139  // preprocess the incoming lists.
1140  // - unwrap any "root" lists
1141  // - handle incomplete hierarchies
1142  std::transform(elements.begin(),
1143  elements.end(),
1144  std::back_inserter(cols),
1145  [&](lists_column_wrapper const& l) -> column_view {
1146  // depth mismatch. attempt to normalize the short column.
1147  // this function will also catch if this is a legitmately broken
1148  // set of input
1149  if (l.depth < expected_depth) {
1150  if (l.root) {
1151  // this exception distinguishes between the following two cases:
1152  //
1153  // { {{{1, 2, 3}}}, {} }
1154  // In this case, row 0 is a List<List<List<int>>>, whereas row 1 is
1155  // just a List<> which is an apparent mismatch. However, because row 1
1156  // is empty we will allow that to semantically mean
1157  // "a List<List<List<int>>> that's empty at the top level"
1158  //
1159  // { {{{1, 2, 3}}}, {4, 5, 6} }
1160  // In this case, row 1 is a a concrete List<int> with actual values.
1161  // There is no way to rectify the differences so we will treat it as a
1162  // true column mismatch.
1163  CUDF_EXPECTS(l.wrapped->size() == 0, "Mismatch in column types!");
1164  stubs.push_back(empty_like(expected_hierarchy));
1165  } else {
1166  stubs.push_back(normalize_column(l.get_view(), expected_hierarchy));
1167  }
1168  return *(stubs.back());
1169  }
1170  // the empty hierarchy case
1171  return l.get_view();
1172  });
1173 
1174  return {std::move(cols), std::move(stubs)};
1175  }

114 template <typename From, typename To>
115 struct fixed_width_type_converter {
116  // Are the types same - simply copy elements from [begin, end) to out
117  template <typename FromT = From,
118  typename ToT = To,
119  typename std::enable_if<std::is_same<FromT, ToT>::value, void>::type* = nullptr>
120  ToT operator()(FromT element) const
121  {
122  return element;
123  }
124 
125  // Are the types convertible or can target be constructed from source?
126  template <typename FromT = From,
127  typename ToT = To,
128  typename std::enable_if<!std::is_same<FromT, ToT>::value &&
129  (cudf::is_convertible<FromT, ToT>::value ||
130  std::is_constructible<ToT, FromT>::value),
131  void>::type* = nullptr>
132  ToT operator()(FromT element) const
133  {
134  return static_cast<ToT>(element);
135  }
136 
137  // Convert integral values to timestamps
138  template <
139  typename FromT = From,
140  typename ToT = To,
141  typename std::enable_if<std::is_integral<FromT>::value && cudf::is_timestamp_t<ToT>::value,
142  void>::type* = nullptr>
143  ToT operator()(FromT element) const
144  {
145  return ToT{typename ToT::duration{element}};
146  }
147 };

288 {

290  std::vector<int32_t> offsets(1, 0);

1308  void build_from_nested(std::initializer_list<lists_column_wrapper<T, SourceElementT>> elements,
1309  std::vector<bool> const& v)
1310  {
1311  auto valids = cudf::test::make_counting_transform_iterator(
1312  0, [&v](auto i) { return v.empty() ? true : v[i]; });
1313 
1314  // compute the expected hierarchy and depth
1315  auto const hierarchy_and_depth = std::accumulate(
1316  elements.begin(),
1317  elements.end(),
1318  std::pair<column_view, int32_t>{{}, -1},
1319  [](auto acc, lists_column_wrapper const& lcw) {
1320  return lcw.depth > acc.second ? std::make_pair(lcw.get_view(), lcw.depth) : acc;
1321  });
1322  column_view expected_hierarchy = hierarchy_and_depth.first;
1323  int32_t const expected_depth = hierarchy_and_depth.second;
1324 
1325  // preprocess columns so that every column_view in 'cols' is an equivalent hierarchy
1326  std::vector<std::unique_ptr<column>> stubs;
1327  std::vector<column_view> cols;
1328  std::tie(cols, stubs) = preprocess_columns(elements, expected_hierarchy, expected_depth);
1329 
1330  // generate offsets
1331  size_type count = 0;
1332  std::vector<size_type> offsetv;
1333  std::transform(cols.cbegin(),
1334  cols.cend(),
1335  valids,
1336  std::back_inserter(offsetv),
1337  [&](cudf::column_view const& col, bool valid) {
1338  // nulls are represented as a repeated offset
1339  size_type ret = count;
1340  if (valid) { count += col.size(); }
1341  return ret;
1342  });
1343  // add the final offset
1344  offsetv.push_back(count);
1345  auto offsets =
1346  cudf::test::fixed_width_column_wrapper<size_type>(offsetv.begin(), offsetv.end()).release();
1347 
1348  // concatenate them together, skipping children that are null.
1349  std::vector<column_view> children;
1350  thrust::copy_if(std::cbegin(cols),
1351  std::cend(cols),
1352  valids, // stencil
1353  std::back_inserter(children),
1354  thrust::identity<bool>{});
1355 
1356  auto data = children.empty() ? cudf::empty_like(expected_hierarchy) : concatenate(children);
1357 
1358  // increment depth
1359  depth = expected_depth + 1;
1360 
1361  // construct the list column
1362  wrapped =
1363  make_lists_column(cols.size(),
1364  std::move(offsets),
1365  std::move(data),
1366  v.size() <= 0 ? 0 : cudf::UNKNOWN_NULL_COUNT,
1367  v.size() <= 0 ? rmm::device_buffer{0}
1368  : cudf::test::detail::make_null_mask(v.begin(), v.end()));
1369  }

1470  // this function will also catch if this is a legitmately broken

1475  //

1481  //

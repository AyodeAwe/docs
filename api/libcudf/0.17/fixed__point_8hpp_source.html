87 template <typename Rep,
88  Radix Base,
89  typename T,
90  typename simt::std::enable_if_t<(simt::std::is_same<int32_t, T>::value &&
91  is_supported_representation_type<Rep>())>* = nullptr>
92 CUDA_HOST_DEVICE_CALLABLE Rep ipow(T exponent)
93 {
94  if (exponent == 0) return static_cast<Rep>(1);
95  auto extra = static_cast<Rep>(1);
96  auto square = static_cast<Rep>(Base);
97  while (exponent > 1) {
98  if (exponent & 1 /* odd */) {
99  extra *= square;
100  exponent -= 1;
101  }
102  exponent /= 2;
103  square *= square;
104  }
105  return square * extra;
106 }

620 template <typename Rep, typename T>
621 CUDA_HOST_DEVICE_CALLABLE auto addition_overflow(T lhs, T rhs)
622 {
623  return rhs > 0 ? lhs > simt::std::numeric_limits<Rep>::max() - rhs
624  : lhs < simt::std::numeric_limits<Rep>::min() - rhs;
625 }

635 template <typename Rep, typename T>
636 CUDA_HOST_DEVICE_CALLABLE auto subtraction_overflow(T lhs, T rhs)
637 {
638  return rhs > 0 ? lhs < simt::std::numeric_limits<Rep>::min() + rhs
639  : lhs > simt::std::numeric_limits<Rep>::max() + rhs;
640 }

650 template <typename Rep, typename T>
651 CUDA_HOST_DEVICE_CALLABLE auto division_overflow(T lhs, T rhs)
652 {
653  return lhs == simt::std::numeric_limits<Rep>::min() && rhs == -1;
654 }

664 template <typename Rep, typename T>
665 CUDA_HOST_DEVICE_CALLABLE auto multiplication_overflow(T lhs, T rhs)
666 {
667  auto const min = simt::std::numeric_limits<Rep>::min();
668  auto const max = simt::std::numeric_limits<Rep>::max();
669  if (rhs > 0)
670  return lhs > max / rhs || lhs < min / rhs;
671  else if (rhs < -1)
672  return lhs > min / rhs || lhs < max / rhs;
673  else
674  return rhs == -1 && lhs == min;
675 }

165 template < typename T, typename FallbackFunc>
166 named_prog<T> getCached(std::string const & name, umap_str_shptr<T>& map, FallbackFunc func)
167 {
168 // Find memory cached T object
169 auto it = map.find(name);
170 if (it != map.end()) {
171 return std::make_pair(name, it->second);
172 } else { // Find file cached T object
173 bool successful_read = false ;
174 std::string serialized;
175 #if defined(JITIFY_USE_CACHE)
176 boost::filesystem::path cache_dir = getCacheDir();
177 if (not cache_dir.empty()) {
178 boost::filesystem::path file_name = cache_dir / name;
179 cacheFile file{file_name.string()};
180 serialized = file.read();
181 successful_read = file.is_read_successful();
182 }
183 #endif
184 if (not successful_read) {
185 // JIT compile and write to file if possible
186 serialized = func().serialize();
187 #if defined(JITIFY_USE_CACHE)
188 if (not cache_dir.empty()) {
189 boost::filesystem::path file_name = cache_dir / name;
190 cacheFile file{file_name.string()};
191 file.write(serialized);
192 }
193 #endif
194 }
195 // Add deserialized T to cache and return
196 auto program = std::make_shared<T>(T::deserialize(serialized));
197 map[name] = program;
198 return std::make_pair(name, program);
199 }
200 }

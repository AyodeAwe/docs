37 template < typename S>
38 inline S round_up_safe (S number_to_round, S modulus) {
39 auto remainder = number_to_round % modulus;
40 if (remainder == 0) { return number_to_round; }
41 auto rounded_up = number_to_round - remainder + modulus;
42 if (rounded_up < number_to_round) {
43 throw std::invalid_argument( "Attempt to round up beyond the type's maximum value" );
44 }
45 return rounded_up;
46 }

53 template < typename S>
54 inline S round_down_safe (S number_to_round, S modulus) {
55 auto remainder = number_to_round % modulus;
56 auto rounded_down = number_to_round - remainder;
57 return rounded_down;
58 }

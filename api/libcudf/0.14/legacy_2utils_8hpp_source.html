79 inline std::vector<gdf_dtype> target_dtypes(std::vector<gdf_dtype> const & source_dtypes,
80 std::vector<operators> const & ops)
81 {
82 std::vector<gdf_dtype> output_dtypes(source_dtypes.size());
83
84 std::transform(source_dtypes.begin(),
85 source_dtypes.end(),
86 ops.begin(),
87 output_dtypes.begin(),
88 [](gdf_dtype source_dtype, operators op) {
89 gdf_dtype t = cudf::type_dispatcher(source_dtype, target_type_mapper{}, op);
90 CUDF_EXPECTS(t != GDF_invalid,
91 "Invalid combination of input type and aggregation operation." );
92 return t;
93 });
94
95 return output_dtypes;
96 }

102 struct identity_initializer {
103 template < typename T>
104 T get_identity(operators op)
105 {
106 switch (op) {
107 case SUM: return corresponding_functor_t<SUM>::identity<T>();
108 case MIN: return corresponding_functor_t<MIN>::identity<T>();
109 case MAX: return corresponding_functor_t<MAX>::identity<T>();
110 case COUNT: return corresponding_functor_t<COUNT>::identity<T>();
111 default : CUDF_FAIL( "Invalid aggregation operation." );
112 }
113 }
114
115 template < typename T>
116 void operator()( gdf_column const & col, operators op, cudaStream_t stream = 0)
117 {
118 T* typed_data = static_cast< T* > (col. data );
119 thrust::fill(
120 rmm::exec_policy(stream)->on(stream), typed_data, typed_data + col. size , get_identity<T>(op));
121
122 // For COUNT operator, initialize column's bitmask to be all valid
123 if (( nullptr != col. valid ) and (COUNT == op)) {
124 CUDA_TRY(cudaMemsetAsync(
125 col. valid , 0xff, sizeof (cudf::valid_type) * gdf_valid_allocation_size(col. size ), stream));
126 }
127 }
128 };

170 static void update_nvcategories(table const & input_keys,
171 table& output_keys,
172 table const & input_values,
173 table& output_values)
174 {
175 gdf_error update_err = nvcategory_gather_table(input_keys, output_keys);
176 CUDF_EXPECTS(update_err == GDF_SUCCESS, "nvcategory_gather_table error for keys" );
177
178 // Filter out columns from (input/output)_values where the output datatype is
179 // not GDF_STRING_CATEGORY. This is possible in aggregations like `count`.
180 std::vector<gdf_column*> string_input_value_cols;
181 std::vector<gdf_column*> string_output_value_cols;
182
183 for (cudf::size_type i = 0; i < input_values.num_columns(); i++) {
184 if (output_values.get_column(i)->dtype == GDF_STRING_CATEGORY) {
185 string_input_value_cols.push_back( const_cast< gdf_column * > (input_values.get_column(i)));
186 string_output_value_cols.push_back( const_cast< gdf_column * > (output_values.get_column(i)));
187 }
188 }
189
190 cudf::table string_input_values(string_input_value_cols);
191 cudf::table string_output_values(string_output_value_cols);
192
193 update_err = nvcategory_gather_table(string_input_values, string_output_values);
194 CUDF_EXPECTS(update_err == GDF_SUCCESS, "nvcategory_gather_table error for values" );
195 }

37 namespace interpolate {
38 template < typename Result, typename T>
39 CUDA_HOST_DEVICE_CALLABLE Result linear(T lhs, T rhs, double frac)
40 {
41 // TODO: safe operation to avoid overflow/underflow
42 // double can fully represent int8-32 value range.
43 // Since the fraction part of double is 52 bits,
44 // double cannot fully represent int64.
45 // Underflow may occur when converting int64 to double
46 // detail: https://github.com/rapidsai/cudf/issues/1417
47
48 double dlhs = static_cast< double > (lhs);
49 double drhs = static_cast< double > (rhs);
50 double one_minus_frac = 1.0 - frac;
51 return static_cast< Result > (one_minus_frac * dlhs + frac * drhs);
52 }

54 template < typename Result, typename T>
55 CUDA_HOST_DEVICE_CALLABLE Result midpoint(T lhs, T rhs)
56 {
57 // TODO: try std::midpoint (C++20) if available
58 double dlhs = static_cast< double > (lhs);
59 double drhs = static_cast< double > (rhs);
60 return static_cast< Result > (dlhs / 2 + drhs / 2);
61 }
62
63 template < typename Result>
64 CUDA_HOST_DEVICE_CALLABLE Result midpoint(int64_t lhs, int64_t rhs)
65 {
66 // caring to avoid integer overflow and underflow between int64_t and Result( double )
67 int64_t half = lhs / 2 + rhs / 2;
68 int64_t rest = lhs % 2 + rhs % 2;
69 return static_cast< Result > ( static_cast< Result > (half) + static_cast< Result > (rest) * 0.5);
70 }
71
72 template <>
73 CUDA_HOST_DEVICE_CALLABLE int64_t midpoint(int64_t lhs, int64_t rhs)
74 {
75 // caring to avoid integer overflow
76 int64_t half = lhs / 2 + rhs / 2;
77 int64_t rest = lhs % 2 + rhs % 2;
78 int64_t result = half;
79
80 // rounding toward zero
81 result += (half >= 0 && rest != 0) ? rest / 2 : 0;
82 result += (half < 0 && rest != 0) ? 1 : 0;
83
84 return result;
85 }

89 struct quantile_index {
90 size_type lower;
91 size_type higher;
92 size_type nearest;
93 double fraction;
94
95 CUDA_HOST_DEVICE_CALLABLE
96 quantile_index (size_type count, double quantile )
97 {
98 quantile = std::min(std::max( quantile , 0.0), 1.0);
99
100 double val = quantile * (count - 1);
101 lower = std::floor(val);
102 higher = static_cast< size_type > (std::ceil(val));
103 nearest = static_cast< size_type > (std::nearbyint(val));
104 fraction = val - lower;
105 }
106 };

109 /* @brief computes a quantile value.
110 *
111 * Computes a value for a quantile by interpolating between two values on either
112 * side of the desired quantile.
113 *
114 * `get_value` must have signature: `T <T>(size_type)` where T can be
115 * `static_cast` to `Result`.
116 *
117 * @param[in] get_value Gets the value at a given index in range [0, size].
118 * @param[in] size Number of values indexed by `get_value`.
119 * @param[in] q Desired quantile in range [0, 1].
120 * @param[in] interp Strategy used to interpolate between the two values
121 * on either side of the desired quantile.
122 *
123 * @returns Value of the desired quantile.
124 */
125 template < typename Result, typename ValueAccessor>
126 CUDA_HOST_DEVICE_CALLABLE Result
127 select_quantile(ValueAccessor get_value, size_type size, double q, interpolation interp)
128 {
129 if (size < 2) { return get_value(0); }
130
131 quantile_index idx(size, q);
132
133 switch (interp) {
134 case interpolation::LINEAR :
135 return interpolate::linear<Result>(get_value(idx.lower), get_value(idx.higher), idx.fraction);
136
137 case interpolation::MIDPOINT :
138 return interpolate::midpoint<Result>(get_value(idx.lower), get_value(idx.higher));
139
140 case interpolation::LOWER : return static_cast< Result > (get_value(idx.lower));
141
142 case interpolation::HIGHER : return static_cast< Result > (get_value(idx.higher));
143
144 case interpolation::NEAREST : return static_cast< Result > (get_value(idx.nearest));
145
146 default :
147 #if defined(__CUDA_ARCH__)
148 release_assert( false && "Invalid interpolation operation for quantiles" );
149 return Result();
150 #else
151 CUDF_FAIL( "Invalid interpolation operation for quantiles." );
152 #endif
153 }
154 }

157 CUDA_HOST_DEVICE_CALLABLE Result
158 select_quantile_data(Iterator begin, size_type size, double q, interpolation interp)
159 {
160 quantile_index idx(size, q);
161
162 switch (interp) {
163 case interpolation::LOWER : return static_cast< Result > (*(begin + idx.lower));
164
165 case interpolation::HIGHER : return static_cast< Result > (*(begin + idx.higher));
166
167 case interpolation::NEAREST : return static_cast< Result > (*(begin + idx.nearest));
168
169 case interpolation::LINEAR :
170 return interpolate::linear<Result>(*(begin + idx.lower), *(begin + idx.higher), idx.fraction);
171
172 case interpolation::MIDPOINT :
173 return interpolate::midpoint<Result>(*(begin + idx.lower), *(begin + idx.higher));
174 }

177 release_assert( false && "Invalid interpolation operation for quantiles" );
178 return Result();
179 #else
180 CUDF_FAIL( "Invalid interpolation operation for quantiles." );
181 #endif

98 class container {
99 public :
100 container () { m_base = m_cur = m_end = nullptr ; }
101 container ( const uint8_t *base, size_t len) { init(base, len); }
102 void init( const uint8_t *base, size_t len) {
103 m_base = m_cur = base;
104 m_end = base + len;
105 }
106 ptrdiff_t bytecount() const { return m_cur - m_base; }
107 unsigned int getb() { return (m_cur < m_end) ? *m_cur++ : 0; }
108 uint64_t get_u64() {
109 uint64_t v = 0;
110 for (uint64_t l = 0;; l += 7) {
111 uint64_t c = getb();
112 v |= (c & 0x7f) << l;
113 if (c < 0x80) return v;
114 }
115 }
116 int64_t get_i64() {
117 uint64_t u = get_u64();
118 return (int64_t)((u >> 1u) ^ -(int64_t)(u & 1));
119 }
120 std::string get_str() {
121 const char *s;
122 size_t len = get_u64();
123 len = ((len & 1) || (m_cur >= m_end)) ? 0 : std::min(len >> 1, ( size_t )(m_end - m_cur));
124 s = reinterpret_cast< const char * > (m_cur);
125 m_cur += len;
126 return std::string(s, len);
127 }

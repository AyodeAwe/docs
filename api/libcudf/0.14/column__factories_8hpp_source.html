79 std::unique_ptr<column> make_numeric_column (
80 data_type type, size_type size, B&& null_mask,
81 size_type null_count = cudf::UNKNOWN_NULL_COUNT, cudaStream_t stream = 0,
82 rmm::mr::device_memory_resource* mr = rmm::mr::get_default_resource()) {

186 template < typename B>
187 std::unique_ptr<column> make_fixed_width_column (
188 data_type type, size_type size,
189 B&& null_mask,
190 size_type null_count = cudf::UNKNOWN_NULL_COUNT, cudaStream_t stream = 0,
191 rmm::mr::device_memory_resource* mr = rmm::mr::get_default_resource())
192 {
193 CUDF_EXPECTS( is_fixed_width (type), "Invalid, non-fixed-width type." );
194 if ( is_timestamp (type)){
195 return make_timestamp_column (type, size, std::forward<B>(null_mask), null_count, stream, mr);
196 }
197 return make_numeric_column (type, size, std::forward<B>(null_mask), null_count, stream, mr);
198 }

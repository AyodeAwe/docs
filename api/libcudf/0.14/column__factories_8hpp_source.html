203 template < typename B>
204 std::unique_ptr<column> make_fixed_width_column (
205 data_type type,
206 size_type size,
207 B&& null_mask,
208 size_type null_count = cudf::UNKNOWN_NULL_COUNT,
209 cudaStream_t stream = 0,
210 rmm::mr::device_memory_resource* mr = rmm::mr::get_default_resource())
211 {
212 CUDF_EXPECTS( is_fixed_width (type), "Invalid, non-fixed-width type." );
213 if ( is_timestamp (type)) {
214 return make_timestamp_column (type, size, std::forward<B>(null_mask), null_count, stream, mr);
215 }
216 return make_numeric_column (type, size, std::forward<B>(null_mask), null_count, stream, mr);
217 }

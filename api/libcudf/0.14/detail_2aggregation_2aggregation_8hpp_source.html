38 template < class Derived>
39 class derived_aggregation : public aggregation {
40 public :
41 derived_aggregation ( aggregation::Kind a) : aggregation (a) {}
42
43 bool is_equal( aggregation const & other) const override
44 {
45 if (this->aggregation::is_equal(other)) {
46 // Dispatch to operator== using static polymorphism
47 return static_cast< Derived const& > (*this) == static_cast< Derived const& > (other);
48 } else {
49 return false ;
50 }
51 }
52
53 size_t do_hash() const override
54 {
55 // Dispatch to hash_impl() using static polymorphism
56 return this->aggregation::do_hash() ^ static_cast< Derived const& > (*this).hash_impl();
57 }
58
59 std::unique_ptr<aggregation> clone() const override
60 {
61 // Dispatch to copy constructor using static polymorphism
62 return std::make_unique<Derived>( static_cast< Derived const& > (* this ));
63 }

69 struct quantile_aggregation final : derived_aggregation <quantile_aggregation> {
70 quantile_aggregation (std::vector<double> const & q, experimental::interpolation i)
71 : derived_aggregation { QUANTILE }, _quantiles {q}, _interpolation {i}
72 {
73 }
74 std::vector<double> _quantiles ;
75 experimental::interpolation _interpolation ;
76
77 protected :
78 friend class derived_aggregation < quantile_aggregation >;
79
80 bool operator== ( quantile_aggregation const & other) const
81 {
82 return _interpolation == other. _interpolation and
83 std::equal( _quantiles .begin(), _quantiles .end(), other. _quantiles .begin());
84 }
85
86 size_t hash_impl() const
87 {
88 return std::hash<int>{}( static_cast< int > ( _interpolation )) ^
89 std::accumulate(
90 _quantiles .cbegin(), _quantiles .cend(), size_t {0}, []( size_t a, double b) {
91 return a ^ std::hash<double>{}(b);
92 });
93 }
94 };

392 #pragma nv_exec_check_disable
393 template < typename F, typename ... Ts>
394 CUDA_HOST_DEVICE_CALLABLE decltype( auto ) aggregation_dispatcher( aggregation ::Kind k,
395 F&& f,
396 Ts&&... args)
397 {
398 switch (k) {
399 case aggregation::SUM :
400 return f.template operator()< aggregation::SUM >(std::forward<Ts>(args)...);
401 case aggregation::PRODUCT :
402 return f.template operator()< aggregation::PRODUCT >(std::forward<Ts>(args)...);
403 case aggregation::MIN :
404 return f.template operator()< aggregation::MIN >(std::forward<Ts>(args)...);
405 case aggregation::MAX :
406 return f.template operator()< aggregation::MAX >(std::forward<Ts>(args)...);
407 case aggregation::COUNT_VALID :
408 return f.template operator()< aggregation::COUNT_VALID >(std::forward<Ts>(args)...);
409 case aggregation::COUNT_ALL :
410 return f.template operator()< aggregation::COUNT_ALL >(std::forward<Ts>(args)...);
411 case aggregation::ANY :
412 return f.template operator()< aggregation::ANY >(std::forward<Ts>(args)...);
413 case aggregation::ALL :
414 return f.template operator()< aggregation::ALL >(std::forward<Ts>(args)...);
415 case aggregation::SUM_OF_SQUARES :
416 return f.template operator()< aggregation::SUM_OF_SQUARES >(std::forward<Ts>(args)...);
417 case aggregation::MEAN :
418 return f.template operator()< aggregation::MEAN >(std::forward<Ts>(args)...);
419 case aggregation::VARIANCE :
420 return f.template operator()< aggregation::VARIANCE >(std::forward<Ts>(args)...);
421 case aggregation::STD :
422 return f.template operator()< aggregation::STD >(std::forward<Ts>(args)...);
423 case aggregation::MEDIAN :
424 return f.template operator()< aggregation::MEDIAN >(std::forward<Ts>(args)...);
425 case aggregation::QUANTILE :
426 return f.template operator()< aggregation::QUANTILE >(std::forward<Ts>(args)...);
427 case aggregation::ARGMAX :
428 return f.template operator()< aggregation::ARGMAX >(std::forward<Ts>(args)...);
429 case aggregation::ARGMIN :
430 return f.template operator()< aggregation::ARGMIN >(std::forward<Ts>(args)...);
431 case aggregation::NUNIQUE :
432 return f.template operator()< aggregation::NUNIQUE >(std::forward<Ts>(args)...);
433 case aggregation::NTH_ELEMENT :
434 return f.template operator()< aggregation::NTH_ELEMENT >(std::forward<Ts>(args)...);
435 default : {
436 #ifndef __CUDA_ARCH__
437 CUDF_FAIL( "Unsupported aggregation." );
438 #else
439 release_assert( false && "Unsupported aggregation." );
440
441 // The following code will never be reached, but the compiler generates a
442 // warning if there isn't a return value.
443
444 // Need to find out what the return type is in order to have a default
445 // return value and solve the compiler warning for lack of a default
446 // return
447 using return_type =
448 decltype(f.template operator ()< aggregation::SUM >(std::forward<Ts>(args)...));
449 return return_type();
450 #endif
451 }
452 }

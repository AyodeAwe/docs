188 static inline int CountLeadingZeros32(uint32_t value) {
189 #if defined(__clang__) || defined(__GNUC__)
190 if (value == 0) return 32;
191 return static_cast< int > (__builtin_clz(value));
192 #elif defined(_MSC_VER)
193 unsigned long index;
194 return (_BitScanReverse(&index, static_cast< unsigned long > (value))) ? 31 - static_cast< int > (index)
195 : 32;
196 #else
197 int bitpos = 0;
198 while (value != 0) {
199 value >>= 1;
200 ++bitpos;
201 }
202 return 32 - bitpos;
203 #endif
204 }

309 class CompactProtocolWriter {

311 CompactProtocolWriter () { m_buf = nullptr ; }
312 CompactProtocolWriter (std::vector<uint8_t> *output) { m_buf = output; }
313 void putb(uint8_t v) { m_buf->push_back(v); }
314 void putb( const uint8_t *raw, uint32_t len) {
315 for (uint32_t i = 0; i < len; i++) m_buf->push_back(raw[i]);
316 }
317 uint32_t put_uint(uint64_t v) {
318 int l = 1;
319 while (v > 0x7f) {
320 putb( static_cast< uint8_t > (v | 0x80));
321 v >>= 7;
322 l++;
323 }
324 putb( static_cast< uint8_t > (v));
325 return l;
326 }
327 uint32_t put_int(int64_t v) {
328 int64_t s = (v < 0);
329 return put_uint(((v ^ -s) << 1) + s);
330 }
331 void put_fldh( int f, int cur, int t) {
332 if (f > cur && f <= cur + 15)
333 putb(((f - cur) << 4) | t);
334 else {
335 putb(t);
336 put_int(f);
337 }
338 }

27 // This is a wrapper struct that enforces "strong typing"
28 // at the construction site of the type. No implicit
29 // conversions will be allowed and you will need to use the
30 // name of the type alias (i.e. scale_type{0})
31 template < typename T>
32 struct strong_typedef {
33 T _t;
34 CUDA_HOST_DEVICE_CALLABLE explicit strong_typedef(T t) : _t(t) {}
35 CUDA_HOST_DEVICE_CALLABLE operator T() const { return _t; }
36 };
39 using scale_type = strong_typedef<int32_t>;

55 template < typename T>
56 constexpr inline auto is_supported_representation_type() {
57 return std::is_same<T, int32_t>::value
58 || std::is_same<T, int64_t>::value;

61 template < typename T>
62 constexpr inline auto is_supported_construction_value_type() {
63 return std::is_integral <T>::value

82 template < typename Rep, Radix Base, typename T,
83 typename std::enable_if_t<(std::is_same<int32_t, T>::value
84 && is_supported_representation_type<Rep>())>* = nullptr >
85 CUDA_HOST_DEVICE_CALLABLE
86 Rep ipow(T exponent) {
87 if (exponent == 0) return static_cast< Rep > (Base);
88 auto extra = static_cast< Rep > (1);
89 auto square = static_cast< Rep > (Base);
90 while (exponent > 1) {
91 if (exponent & 1 /* odd */ ) {
92 extra *= square;
93 exponent -= 1;
94 }
95 exponent /= 2;
96 square *= square;
97 }
98 return square * extra;
99 }

158 template < typename Rep, Radix Rad, typename T>
159 CUDA_HOST_DEVICE_CALLABLE
160 constexpr T shift (T const & val, scale_type const & scale) {
161 if (scale == 0) return val;
162 else if (scale > 0) return right_shift<Rep, Rad>(val, scale);
163 else return left_shift <Rep, Rad>(val, scale);
164 }

179 template < typename Rep, Radix Rad, typename T,
180 typename std::enable_if_t<is_supported_construction_value_type<T>()>* = nullptr >
181 CUDA_HOST_DEVICE_CALLABLE
182 auto shift_with_precise_round(T const & value, scale_type const & scale) -> int64_t {
183 if (scale == 0) return value;
184 int64_t const base = static_cast< int64_t > (Rad);
185 int64_t const factor = ipow<int64_t, Rad>(std::abs(scale));
186 int64_t const temp = scale <= 0 ? value * (factor * base) : value / (factor / base);
187 return std::roundf( static_cast< double > (temp) / base);
188 }

447 template < typename Rep, typename T>
448 CUDA_HOST_DEVICE_CALLABLE
449 auto addition_overflow (T lhs, T rhs) {
450 return rhs > 0 ? lhs > std::numeric_limits<Rep>::max() - rhs
451 : lhs < std::numeric_limits<Rep>::min() - rhs;
452 }
453
462 template < typename Rep, typename T>
463 CUDA_HOST_DEVICE_CALLABLE
464 auto subtraction_overflow (T lhs, T rhs) {
465 return rhs > 0 ? lhs < std::numeric_limits<Rep>::min() + rhs
466 : lhs > std::numeric_limits<Rep>::max() + rhs;
467 }

477 template < typename Rep, typename T>
478 CUDA_HOST_DEVICE_CALLABLE
479 auto division_overflow (T lhs, T rhs) {
480 return lhs == std::numeric_limits<Rep>::min() && rhs == -1;
481 }
482
491 template < typename Rep, typename T>
492 CUDA_HOST_DEVICE_CALLABLE
493 auto multiplication_overflow (T lhs, T rhs) {
494 auto const min = std::numeric_limits<Rep>::min();
495 auto const max = std::numeric_limits<Rep>::max();
496 if (rhs > 0) return lhs > max / rhs || lhs < min / rhs;
497 else if (rhs < -1) return lhs > min / rhs || lhs < max / rhs;
498 else return rhs == -1 && lhs == min;
499 }

501 // PLUS Operation
502 template < typename Rep1, Radix Rad1>
503 CUDA_HOST_DEVICE_CALLABLE
504 fixed_point<Rep1, Rad1> operator+ ( fixed_point<Rep1, Rad1> const & lhs,
505 fixed_point<Rep1, Rad1> const & rhs) {
506
507 auto const rhsv = lhs._scale > rhs._scale ? detail::shift_with_precise_round<Rep1, Rad1>(rhs._value, scale_type{lhs._scale - rhs._scale}) : rhs._value;
508 auto const lhsv = lhs._scale < rhs._scale ? detail::shift_with_precise_round<Rep1, Rad1>(lhs._value, scale_type{rhs._scale - lhs._scale}) : lhs._value;
509 auto const scale = lhs._scale > rhs._scale ? lhs._scale : rhs._scale;
510
511 #if defined(__CUDACC_DEBUG__)
512
513 assert(!addition_overflow<Rep1>(lhsv, rhsv) &&
514 "fixed_point overflow of underlying represenation type " + print_rep<Rep1>());
515
516 #endif
517
518 return fixed_point<Rep1, Rad1> { scaled_integer<Rep1> (lhsv + rhsv, scale)};
519 }

521 // MINUS Operation
522 template < typename Rep1, Radix Rad1>
523 CUDA_HOST_DEVICE_CALLABLE
524 fixed_point<Rep1, Rad1> operator- ( fixed_point<Rep1, Rad1> const & lhs,
525 fixed_point<Rep1, Rad1> const & rhs) {
526
527 auto const rhsv = lhs._scale > rhs._scale ? detail::shift_with_precise_round<Rep1, Rad1>(rhs._value, scale_type{lhs._scale - rhs._scale}) : rhs._value;
528 auto const lhsv = lhs._scale < rhs._scale ? detail::shift_with_precise_round<Rep1, Rad1>(lhs._value, scale_type{rhs._scale - lhs._scale}) : lhs._value;
529 auto const scale = lhs._scale > rhs._scale ? lhs._scale : rhs._scale;
530
531 #if defined(__CUDACC_DEBUG__)
532
533 assert(!subtraction_overflow<Rep1>(lhsv, rhsv) &&
534 "fixed_point overflow of underlying represenation type " + print_rep<Rep1>());
535
536 #endif
537
538 return fixed_point<Rep1, Rad1> { scaled_integer<Rep1> (lhsv - rhsv, scale)};
539 }

573 // EQUALITY COMPARISON Operation
574 template < typename Rep1, Radix Rad1>
575 CUDA_HOST_DEVICE_CALLABLE
576 bool operator== ( fixed_point<Rep1, Rad1> const & lhs,
577 fixed_point<Rep1, Rad1> const & rhs) {
578 auto const rhsv = lhs._scale > rhs._scale ? detail::shift_with_precise_round<Rep1, Rad1>(rhs._value, scale_type{lhs._scale - rhs._scale}) : rhs._value;
579 auto const lhsv = lhs._scale < rhs._scale ? detail::shift_with_precise_round<Rep1, Rad1>(lhs._value, scale_type{rhs._scale - lhs._scale}) : lhs._value;
580 return rhsv == lhsv;
581 }

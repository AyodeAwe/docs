26 // This is a wrapper struct that enforces "strong typing"
27 // at the construction site of the type. No implicit
28 // conversions will be allowed and you will need to use the
29 // name of the type alias (i.e. scale_type{0})
30 template < typename T>
31 struct strong_typedef {
32 T _t;
33 CUDA_HOST_DEVICE_CALLABLE explicit strong_typedef (T t) : _t(t) {}
34 CUDA_HOST_DEVICE_CALLABLE operator T() const { return _t; }
35 };

80 template < typename Rep, Radix Base, typename T,
81 typename std::enable_if_t<(std::is_same<int32_t, T>::value
82 && is_supported_representation_type<Rep>())>* = nullptr >
83 CUDA_HOST_DEVICE_CALLABLE
84 Rep ipow (T exponent) {
85 if (exponent == 0) return static_cast<Rep>(Base);
86 auto extra = static_cast<Rep>(1);
87 auto square = static_cast<Rep>(Base);
88 while (exponent > 1) {
89 if (exponent & 1 /* odd */ ) {
90 extra *= square;
91 exponent -= 1;
92 }
93 exponent /= 2;
94 square *= square;
95 }
96 return square * extra;
97 }

156 template < typename Rep, Radix Rad, typename T>
157 CUDA_HOST_DEVICE_CALLABLE
158 constexpr T shift (T const & val, scale_type const & scale) {
159 if (scale == 0) return val;
160 else if (scale > 0) return right_shift<Rep, Rad>(val, scale);
161 else return left_shift <Rep, Rad>(val, scale);
162 }

177 template < typename Rep, Radix Rad, typename T,
178 typename std::enable_if_t<is_supported_construction_value_type<T>()>* = nullptr >
179 CUDA_HOST_DEVICE_CALLABLE
180 auto shift_with_precise_round (T const & value, scale_type const & scale) -> int64_t {
181 if (scale == 0) return value;
182 int64_t const base = static_cast<int64_t>(Rad);
183 int64_t const factor = ipow<int64_t, Rad>(std::abs(scale));
184 int64_t const temp = scale <= 0 ? value * (factor * base) : value / (factor / base);
185 return std::roundf(static_cast<double>(temp) / base);
186 }

445 template < typename Rep, typename T>
446 CUDA_HOST_DEVICE_CALLABLE
447 auto addition_overflow (T lhs, T rhs) {
448 return rhs > 0 ? lhs > std::numeric_limits<Rep>::max() - rhs
449 : lhs < std::numeric_limits<Rep>::min() - rhs;
450 }
451
460 template < typename Rep, typename T>
461 CUDA_HOST_DEVICE_CALLABLE
462 auto subtraction_overflow (T lhs, T rhs) {
463 return rhs > 0 ? lhs < std::numeric_limits<Rep>::min() + rhs
464 : lhs > std::numeric_limits<Rep>::max() + rhs;
465 }
466
475 template < typename Rep, typename T>
476 CUDA_HOST_DEVICE_CALLABLE
477 auto division_overflow (T lhs, T rhs) {
478 return lhs == std::numeric_limits<Rep>::min() && rhs == -1;
479 }
480
489 template < typename Rep, typename T>
490 CUDA_HOST_DEVICE_CALLABLE
491 auto multiplication_overflow (T lhs, T rhs) {
492 auto const min = std::numeric_limits<Rep>::min();
493 auto const max = std::numeric_limits<Rep>::max();
494 if (rhs > 0) return lhs > max / rhs || lhs < min / rhs;
495 else if (rhs < -1) return lhs > min / rhs || lhs < max / rhs;
496 else return rhs == -1 && lhs == min;
497 }

499 // PLUS Operation
500 template < typename Rep1, Radix Rad1>
501 CUDA_HOST_DEVICE_CALLABLE
502 fixed_point<Rep1, Rad1> operator+ ( fixed_point<Rep1, Rad1> const & lhs,
503 fixed_point<Rep1, Rad1> const & rhs) {
504
505 auto const rhsv = lhs._scale > rhs._scale ? detail::shift_with_precise_round<Rep1, Rad1>(rhs._value, scale_type {lhs._scale - rhs._scale}) : rhs._value;
506 auto const lhsv = lhs._scale < rhs._scale ? detail::shift_with_precise_round<Rep1, Rad1>(lhs._value, scale_type {rhs._scale - lhs._scale}) : lhs._value;
507 auto const scale = lhs._scale > rhs._scale ? lhs._scale : rhs._scale;
508
509 #if defined(__CUDACC_DEBUG__)
510
511 assert(!addition_overflow<Rep1>(lhsv, rhsv) &&
512 "fixed_point overflow of underlying represenation type " + print_rep<Rep1>());
513
514 #endif
515
516 return fixed_point<Rep1, Rad1> { scaled_integer<Rep1> (lhsv + rhsv, scale)};
517 }

519 // MINUS Operation
520 template < typename Rep1, Radix Rad1>
521 CUDA_HOST_DEVICE_CALLABLE
522 fixed_point<Rep1, Rad1> operator- ( fixed_point<Rep1, Rad1> const & lhs,
523 fixed_point<Rep1, Rad1> const & rhs) {
524
525 auto const rhsv = lhs._scale > rhs._scale ? detail::shift_with_precise_round<Rep1, Rad1>(rhs._value, scale_type {lhs._scale - rhs._scale}) : rhs._value;
526 auto const lhsv = lhs._scale < rhs._scale ? detail::shift_with_precise_round<Rep1, Rad1>(lhs._value, scale_type {rhs._scale - lhs._scale}) : lhs._value;
527 auto const scale = lhs._scale > rhs._scale ? lhs._scale : rhs._scale;
528
529 #if defined(__CUDACC_DEBUG__)
530
531 assert(!subtraction_overflow<Rep1>(lhsv, rhsv) &&
532 "fixed_point overflow of underlying represenation type " + print_rep<Rep1>());
533
534 #endif
535
536 return fixed_point<Rep1, Rad1> { scaled_integer<Rep1> (lhsv - rhsv, scale)};
537 }

542 fixed_point<Rep1, Rad1> operator* ( fixed_point<Rep1, Rad1> const & lhs,

571 // EQUALITY COMPARISON Operation
572 template < typename Rep1, Radix Rad1>
573 CUDA_HOST_DEVICE_CALLABLE
574 bool operator== ( fixed_point<Rep1, Rad1> const & lhs,
575 fixed_point<Rep1, Rad1> const & rhs) {
576 auto const rhsv = lhs._scale > rhs._scale ? detail::shift_with_precise_round<Rep1, Rad1>(rhs._value, scale_type {lhs._scale - rhs._scale}) : rhs._value;
577 auto const lhsv = lhs._scale < rhs._scale ? detail::shift_with_precise_round<Rep1, Rad1>(lhs._value, scale_type {rhs._scale - lhs._scale}) : lhs._value;
578 return rhsv == lhsv;
579 }

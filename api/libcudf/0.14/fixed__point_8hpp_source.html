78 template <typename Rep,
79  Radix Base,
80  typename T,
81  typename std::enable_if_t<(std::is_same<int32_t, T>::value &&
82  is_supported_representation_type<Rep>())>* = nullptr>
83 CUDA_HOST_DEVICE_CALLABLE Rep ipow(T exponent)
84 {
85  if (exponent == 0) return static_cast<Rep>(Base);
86  auto extra = static_cast<Rep>(1);
87  auto square = static_cast<Rep>(Base);
88  while (exponent > 1) {
89  if (exponent & 1 /* odd */) {
90  extra *= square;
91  exponent -= 1;
92  }
93  exponent /= 2;
94  square *= square;
95  }
96  return square * extra;
97 }

154 template <typename Rep, Radix Rad, typename T>
155 CUDA_HOST_DEVICE_CALLABLE constexpr T shift(T const& val, scale_type const& scale)
156 {
157  if (scale == 0)
158  return val;
159  else if (scale > 0)
160  return right_shift<Rep, Rad>(val, scale);
161  else
162  return left_shift<Rep, Rad>(val, scale);
163 }

178 template <typename Rep,
179  Radix Rad,
180  typename T,
181  typename std::enable_if_t<is_supported_construction_value_type<T>()>* = nullptr>
182 CUDA_HOST_DEVICE_CALLABLE auto shift_with_precise_round(T const& value, scale_type const& scale)
183  -> int64_t
184 {
185  if (scale == 0) return value;
186  int64_t const base = static_cast<int64_t>(Rad);
187  int64_t const factor = ipow<int64_t, Rad>(std::abs(scale));
188  int64_t const temp = scale <= 0 ? value * (factor * base) : value / (factor / base);
189  return std::roundf(static_cast<double>(temp) / base);
190 }

472 template <typename Rep, typename T>
473 CUDA_HOST_DEVICE_CALLABLE auto division_overflow(T lhs, T rhs)
474 {
475  return lhs == std::numeric_limits<Rep>::min() && rhs == -1;
476 }

486 template <typename Rep, typename T>
487 CUDA_HOST_DEVICE_CALLABLE auto multiplication_overflow(T lhs, T rhs)
488 {
489  auto const min = std::numeric_limits<Rep>::min();
490  auto const max = std::numeric_limits<Rep>::max();
491  if (rhs > 0)
492  return lhs > max / rhs || lhs < min / rhs;
493  else if (rhs < -1)
494  return lhs > min / rhs || lhs < max / rhs;
495  else
496  return rhs == -1 && lhs == min;
497 }

499 // PLUS Operation
500 template <typename Rep1, Radix Rad1>
501 CUDA_HOST_DEVICE_CALLABLE fixed_point<Rep1, Rad1> operator+(fixed_point<Rep1, Rad1> const& lhs,
502  fixed_point<Rep1, Rad1> const& rhs)
503 {
504  auto const rhsv = lhs._scale > rhs._scale ? detail::shift_with_precise_round<Rep1, Rad1>(
505  rhs._value, scale_type{lhs._scale - rhs._scale})
506  : rhs._value;
507  auto const lhsv = lhs._scale < rhs._scale ? detail::shift_with_precise_round<Rep1, Rad1>(
508  lhs._value, scale_type{rhs._scale - lhs._scale})
509  : lhs._value;
510  auto const scale = lhs._scale > rhs._scale ? lhs._scale : rhs._scale;
511 
512 #if defined(__CUDACC_DEBUG__)
513 
514  assert(("fixed_point overflow of underlying representation type " + print_rep<Rep1>(),
515  !addition_overflow<Rep1>(lhsv, rhsv)));
516 
517 #endif
518 
519  return fixed_point<Rep1, Rad1>{scaled_integer<Rep1>(lhsv + rhsv, scale)};
520 }

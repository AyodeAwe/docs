26 // This is a wrapper struct that enforces "strong typing"
27 // at the construction site of the type. No implicit
28 // conversions will be allowed and you will need to use the
29 // name of the type alias (i.e. scale_type{0})
30 template < typename T>
31 struct strong_typedef {
32 T _t;
33 CUDA_HOST_DEVICE_CALLABLE explicit strong_typedef(T t) : _t(t) {}
34 CUDA_HOST_DEVICE_CALLABLE operator T() const { return _t; }
35 };

78 template < typename Rep,
79 Radix Base,
80 typename T,
81 typename std::enable_if_t<(std::is_same<int32_t, T>::value &&
82 is_supported_representation_type<Rep>())>* = nullptr >
83 CUDA_HOST_DEVICE_CALLABLE Rep ipow(T exponent)
84 {
85 if (exponent == 0) return static_cast< Rep > (Base);
86 auto extra = static_cast< Rep > (1);
87 auto square = static_cast< Rep > (Base);
88 while (exponent > 1) {
89 if (exponent & 1 /* odd */ ) {
90 extra *= square;
91 exponent -= 1;
92 }
93 exponent /= 2;
94 square *= square;
95 }
96 return square * extra;
97 }

154 template < typename Rep, Radix Rad, typename T>
155 CUDA_HOST_DEVICE_CALLABLE constexpr T shift (T const & val, scale_type const & scale)
156 {
157 if (scale == 0)
158 return val;
159 else if (scale > 0)
160 return right_shift<Rep, Rad>(val, scale);
161 else
162 return left_shift<Rep, Rad>(val, scale);
163 }

178 template < typename Rep,
179 Radix Rad,
180 typename T,
181 typename std::enable_if_t<is_supported_construction_value_type<T>()>* = nullptr >
182 CUDA_HOST_DEVICE_CALLABLE auto shift_with_precise_round(T const & value, scale_type const & scale)
183 -> int64_t
184 {
185 if (scale == 0) return value;
186 int64_t const base = static_cast< int64_t > (Rad);
187 int64_t const factor = ipow<int64_t, Rad>(std::abs(scale));
188 int64_t const temp = scale <= 0 ? value * (factor * base) : value / (factor / base);
189 return std::roundf( static_cast< double > (temp) / base);
190 }

437 template < typename Rep, typename T>
438 CUDA_HOST_DEVICE_CALLABLE auto addition_overflow (T lhs, T rhs)
439 {
440 return rhs > 0 ? lhs > std::numeric_limits<Rep>::max() - rhs
441 : lhs < std::numeric_limits<Rep>::min() - rhs;
442 }
443
452 template < typename Rep, typename T>
453 CUDA_HOST_DEVICE_CALLABLE auto subtraction_overflow (T lhs, T rhs)
454 {
455 return rhs > 0 ? lhs < std::numeric_limits<Rep>::min() + rhs
456 : lhs > std::numeric_limits<Rep>::max() + rhs;
457 }

467 template < typename Rep, typename T>
468 CUDA_HOST_DEVICE_CALLABLE auto division_overflow (T lhs, T rhs)
469 {
470 return lhs == std::numeric_limits<Rep>::min() && rhs == -1;
471 }

481 template < typename Rep, typename T>
482 CUDA_HOST_DEVICE_CALLABLE auto multiplication_overflow (T lhs, T rhs)
483 {
484 auto const min = std::numeric_limits<Rep>::min();
485 auto const max = std::numeric_limits<Rep>::max();
486 if (rhs > 0)
487 return lhs > max / rhs || lhs < min / rhs;
488 else if (rhs < -1)
489 return lhs > min / rhs || lhs < max / rhs;
490 else
491 return rhs == -1 && lhs == min;
492 }

494 // PLUS Operation
495 template < typename Rep1, Radix Rad1>
496 CUDA_HOST_DEVICE_CALLABLE fixed_point<Rep1, Rad1> operator+ ( fixed_point<Rep1, Rad1> const & lhs,
497 fixed_point<Rep1, Rad1> const & rhs)
498 {
499 auto const rhsv = lhs._scale > rhs._scale ? detail::shift_with_precise_round<Rep1, Rad1>(
500 rhs._value, scale_type{lhs._scale - rhs._scale})
501 : rhs._value;
502 auto const lhsv = lhs._scale < rhs._scale ? detail::shift_with_precise_round<Rep1, Rad1>(
503 lhs._value, scale_type{rhs._scale - lhs._scale})
504 : lhs._value;
505 auto const scale = lhs._scale > rhs._scale ? lhs._scale : rhs._scale;
506
507 #if defined(__CUDACC_DEBUG__)
508
509 assert(!addition_overflow<Rep1>(lhsv, rhsv) &&
510 "fixed_point overflow of underlying represenation type " + print_rep<Rep1>());
511
512 #endif
513
514 return fixed_point<Rep1, Rad1> { scaled_integer<Rep1> (lhsv + rhsv, scale)};
515 }

518 template < typename Rep1, Radix Rad1>
519 CUDA_HOST_DEVICE_CALLABLE fixed_point<Rep1, Rad1> operator- ( fixed_point<Rep1, Rad1> const & lhs,
520 fixed_point<Rep1, Rad1> const & rhs)
521 {
522 auto const rhsv = lhs._scale > rhs._scale ? detail::shift_with_precise_round<Rep1, Rad1>(
523 rhs._value, scale_type{lhs._scale - rhs._scale})
524 : rhs._value;
525 auto const lhsv = lhs._scale < rhs._scale ? detail::shift_with_precise_round<Rep1, Rad1>(
526 lhs._value, scale_type{rhs._scale - lhs._scale})
527 : lhs._value;
528 auto const scale = lhs._scale > rhs._scale ? lhs._scale : rhs._scale;
529
530 #if defined(__CUDACC_DEBUG__)
531
532 assert(!subtraction_overflow<Rep1>(lhsv, rhsv) &&
533 "fixed_point overflow of underlying represenation type " + print_rep<Rep1>());
534
535 #endif
536
537 return fixed_point<Rep1, Rad1> { scaled_integer<Rep1> (lhsv - rhsv, scale)};

543 fixed_point<Rep1, Rad1> const & rhs)

572 // EQUALITY COMPARISON Operation
573 template < typename Rep1, Radix Rad1>
574 CUDA_HOST_DEVICE_CALLABLE bool operator== ( fixed_point<Rep1, Rad1> const & lhs,
575 fixed_point<Rep1, Rad1> const & rhs)
576 {
577 auto const rhsv = lhs._scale > rhs._scale ? detail::shift_with_precise_round<Rep1, Rad1>(
578 rhs._value, scale_type{lhs._scale - rhs._scale})
579 : rhs._value;
580 auto const lhsv = lhs._scale < rhs._scale ? detail::shift_with_precise_round<Rep1, Rad1>(
581 lhs._value, scale_type{rhs._scale - lhs._scale})
582 : lhs._value;
583 return rhsv == lhsv;
584 }

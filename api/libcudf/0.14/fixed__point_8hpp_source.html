27 // This is a wrapper struct that enforces "strong typing"
28 // at the construction site of the type. No implicit
29 // conversions will be allowed and you will need to use the
30 // name of the type alias (i.e. scale_type{0})
31 template < typename T>
32 struct strong_typedef {
33 T _t;
34 CUDA_HOST_DEVICE_CALLABLE explicit strong_typedef(T t) : _t(t) {}
35 CUDA_HOST_DEVICE_CALLABLE operator T() const { return _t; }
36 };
39 using scale_type = strong_typedef<int32_t>;

50 enum class Radix : int32_t { BASE_2 = 2, BASE_10 = 10 };

52 template < typename T>
53 constexpr inline auto is_supported_representation_type() {
54 return std::is_same<T, int32_t>::value || std::is_same<T, int64_t>::value;
55 }

57 template < typename T>
58 constexpr inline auto is_supported_construction_value_type() {
59 return std::is_integral<T>::value || std::is_floating_point<T>::value;

77 template < typename Rep,
78 Radix Base,
79 typename T,
80 typename std::enable_if_t<(std::is_same<int32_t, T>::value &&
81 is_supported_representation_type<Rep>())>* = nullptr >
82 CUDA_HOST_DEVICE_CALLABLE Rep ipow(T exponent) {
83 if (exponent == 0) return static_cast< Rep > (Base);
84 auto extra = static_cast< Rep > (1);
85 auto square = static_cast< Rep > (Base);
86 while (exponent > 1) {
87 if (exponent & 1 /* odd */ ) {
88 extra *= square;
89 exponent -= 1;
90 }
91 exponent /= 2;
92 square *= square;
93 }
94 return square * extra;
95 }

116 template < typename Rep, Radix Rad, typename T>
117 CUDA_HOST_DEVICE_CALLABLE constexpr T right_shift(T const & val, scale_type const & scale) {
118 return val / ipow<Rep, Rad>(scale._t);
119 }

132 template < typename Rep, Radix Rad, typename T>
133 CUDA_HOST_DEVICE_CALLABLE constexpr T left_shift(T const & val, scale_type const & scale) {
134 return val * ipow<Rep, Rad>(-scale._t);
135 }

150 template < typename Rep, Radix Rad, typename T>
151 CUDA_HOST_DEVICE_CALLABLE constexpr T shift (T const & val, scale_type const & scale) {
152 if (scale == 0)
153 return val;
154 else if (scale > 0)
155 return right_shift<Rep, Rad>(val, scale);
156 else
157 return left_shift<Rep, Rad>(val, scale);
158 }

173 template < typename Rep,
174 Radix Rad,
175 typename T,
176 typename std::enable_if_t<is_supported_construction_value_type<T>()>* = nullptr >
177 CUDA_HOST_DEVICE_CALLABLE auto shift_with_precise_round(T const & value, scale_type const & scale)
178 -> int64_t {
179 if (scale == 0) return value;
180 int64_t const base = static_cast< int64_t > (Rad);
181 int64_t const factor = ipow<int64_t, Rad>(std::abs(scale));
182 int64_t const temp = scale <= 0 ? value * (factor * base) : value / (factor / base);
183 return std::roundf( static_cast< double > (temp) / base);
184 }

426 template < typename Rep, typename T>
427 CUDA_HOST_DEVICE_CALLABLE auto addition_overflow (T lhs, T rhs) {
428 return rhs > 0 ? lhs > std::numeric_limits<Rep>::max() - rhs
429 : lhs < std::numeric_limits<Rep>::min() - rhs;
430 }
431
440 template < typename Rep, typename T>
441 CUDA_HOST_DEVICE_CALLABLE auto subtraction_overflow (T lhs, T rhs) {
442 return rhs > 0 ? lhs < std::numeric_limits<Rep>::min() + rhs
443 : lhs > std::numeric_limits<Rep>::max() + rhs;
444 }
445
454 template < typename Rep, typename T>
455 CUDA_HOST_DEVICE_CALLABLE auto division_overflow (T lhs, T rhs) {
456 return lhs == std::numeric_limits<Rep>::min() && rhs == -1;
457 }

467 template < typename Rep, typename T>
468 CUDA_HOST_DEVICE_CALLABLE auto multiplication_overflow (T lhs, T rhs) {
469 auto const min = std::numeric_limits<Rep>::min();
470 auto const max = std::numeric_limits<Rep>::max();
471 if (rhs > 0)
472 return lhs > max / rhs || lhs < min / rhs;
473 else if (rhs < -1)
474 return lhs > min / rhs || lhs < max / rhs;
475 else
476 return rhs == -1 && lhs == min;
477 }

479 // PLUS Operation
480 template < typename Rep1, Radix Rad1>
481 CUDA_HOST_DEVICE_CALLABLE fixed_point<Rep1, Rad1> operator+ ( fixed_point<Rep1, Rad1> const & lhs,
482 fixed_point<Rep1, Rad1> const & rhs) {
483 auto const rhsv = lhs._scale > rhs._scale ? detail::shift_with_precise_round<Rep1, Rad1>(
484 rhs._value, scale_type{lhs._scale - rhs._scale})
485 : rhs._value;
486 auto const lhsv = lhs._scale < rhs._scale ? detail::shift_with_precise_round<Rep1, Rad1>(
487 lhs._value, scale_type{rhs._scale - lhs._scale})
488 : lhs._value;
489 auto const scale = lhs._scale > rhs._scale ? lhs._scale : rhs._scale;
490
491 #if defined(__CUDACC_DEBUG__)
492
493 assert(!addition_overflow<Rep1>(lhsv, rhsv) &&
494 "fixed_point overflow of underlying represenation type " + print_rep<Rep1>());
495
496 #endif
497
498 return fixed_point<Rep1, Rad1> { scaled_integer<Rep1> (lhsv + rhsv, scale)};
499 }

501 // MINUS Operation
502 template < typename Rep1, Radix Rad1>
503 CUDA_HOST_DEVICE_CALLABLE fixed_point<Rep1, Rad1> operator- ( fixed_point<Rep1, Rad1> const & lhs,
504 fixed_point<Rep1, Rad1> const & rhs) {
505 auto const rhsv = lhs._scale > rhs._scale ? detail::shift_with_precise_round<Rep1, Rad1>(
506 rhs._value, scale_type{lhs._scale - rhs._scale})
507 : rhs._value;
508 auto const lhsv = lhs._scale < rhs._scale ? detail::shift_with_precise_round<Rep1, Rad1>(
509 lhs._value, scale_type{rhs._scale - lhs._scale})
510 : lhs._value;
511 auto const scale = lhs._scale > rhs._scale ? lhs._scale : rhs._scale;
512
513 #if defined(__CUDACC_DEBUG__)
514
515 assert(!subtraction_overflow<Rep1>(lhsv, rhsv) &&
516 "fixed_point overflow of underlying represenation type " + print_rep<Rep1>());
517
518 #endif
519
520 return fixed_point<Rep1, Rad1> { scaled_integer<Rep1> (lhsv - rhsv, scale)};
521 }

523 // MULTIPLIES Operation
524 template < typename Rep1, Radix Rad1>
525 CUDA_HOST_DEVICE_CALLABLE fixed_point<Rep1, Rad1> operator* ( fixed_point<Rep1, Rad1> const & lhs,
526 fixed_point<Rep1, Rad1> const & rhs) {
527 #if defined(__CUDACC_DEBUG__)
528
529 assert(!multiplication_overflow<Rep1>(lhs._value, rhs._value) &&
530 "fixed_point overflow of underlying represenation type " + print_rep<Rep1>());
531
532 #endif
533
534 return fixed_point<Rep1, Rad1> {
535 scaled_integer<Rep1> (lhs._value * rhs._value, scale_type{lhs._scale + rhs._scale})};
536 }

541 fixed_point<Rep1, Rad1> const & rhs) {

553 // EQUALITY COMPARISON Operation
554 template < typename Rep1, Radix Rad1>
555 CUDA_HOST_DEVICE_CALLABLE bool operator== ( fixed_point<Rep1, Rad1> const & lhs,
556 fixed_point<Rep1, Rad1> const & rhs) {
557 auto const rhsv = lhs._scale > rhs._scale ? detail::shift_with_precise_round<Rep1, Rad1>(
558 rhs._value, scale_type{lhs._scale - rhs._scale})
559 : rhs._value;
560 auto const lhsv = lhs._scale < rhs._scale ? detail::shift_with_precise_round<Rep1, Rad1>(
561 lhs._value, scale_type{rhs._scale - lhs._scale})
562 : lhs._value;
563 return rhsv == lhsv;
564 }

25 // This is a wrapper struct that enforces "strong typing"
26 // at the construction site of the type. No implicit
27 // conversions will be allowed and you will need to use the
28 // name of the type alias (i.e. scale_type{0})
29 template < typename T>
30 struct strong_typedef {
31 T _t;
32 CUDA_HOST_DEVICE_CALLABLE explicit strong_typedef (T t) : _t(t) {}
33 CUDA_HOST_DEVICE_CALLABLE operator T() const { return _t; }
34 };

49 template < typename T>
50 constexpr inline auto is_supported_representation_type() {
51 return std::is_same<T, int32_t>::value

75 template < typename Rep, Radix Base, typename T,
76 typename std::enable_if_t<(std::is_same<int32_t, T>::value
77 && is_supported_representation_type<Rep>())>* = nullptr >
78 CUDA_HOST_DEVICE_CALLABLE
79 Rep ipow(T exponent) {
80 if (exponent == 0) return static_cast<Rep>(Base);
81 auto extra = static_cast<Rep>(1);
82 auto square = static_cast<Rep>(Base);
83 while (exponent > 1) {
84 if (exponent & 1 /* odd */ ) {
85 extra *= square;
86 exponent -= 1;
87 }
88 exponent /= 2;
89 square *= square;
90 }
91 return square * extra;
92 }

151 template < typename Rep, Radix Rad, typename T>
152 CUDA_HOST_DEVICE_CALLABLE
153 constexpr T shift (T const & val, scale_type const & scale) {
154 if (scale == 0) return val;
155 else if (scale > 0) return right_shift<Rep, Rad>(val, scale);
156 else return left_shift <Rep, Rad>(val, scale);
157 }

172 template < typename Rep, Radix Rad, typename T,
173 typename std::enable_if_t<is_supported_construction_value_type<T>()>* = nullptr >
174 CUDA_HOST_DEVICE_CALLABLE
175 auto shift_with_precise_round(T const & value, scale_type const & scale) -> int64_t {
176 if (scale == 0) return value;
177 int64_t const base = static_cast<int64_t>(Rad);
178 int64_t const factor = ipow<int64_t, Rad>(std::abs(scale));
179 int64_t const temp = scale <= 0 ? value * (factor * base) : value / (factor / base);
180 return std::roundf(static_cast<double>(temp) / base);
181 }

437 template < typename Rep, typename T>
438 CUDA_HOST_DEVICE_CALLABLE
439 auto addition_overflow(T lhs, T rhs) {
440 return rhs > 0 ? lhs > std::numeric_limits<Rep>::max() - rhs
441 : lhs < std::numeric_limits<Rep>::min() - rhs;
442 }

452 template < typename Rep, typename T>
453 CUDA_HOST_DEVICE_CALLABLE
454 auto subtraction_overflow(T lhs, T rhs) {
455 return rhs > 0 ? lhs < std::numeric_limits<Rep>::min() + rhs
456 : lhs > std::numeric_limits<Rep>::max() + rhs;
457 }

467 template < typename Rep, typename T>
468 CUDA_HOST_DEVICE_CALLABLE
469 auto division_overflow(T lhs, T rhs) {
470 return lhs == std::numeric_limits<Rep>::min() && rhs == -1;
471 }

481 template < typename Rep, typename T>
482 CUDA_HOST_DEVICE_CALLABLE
483 auto multiplication_overflow(T lhs, T rhs) {
484 auto const min = std::numeric_limits<Rep>::min();
485 auto const max = std::numeric_limits<Rep>::max();
486 if (rhs > 0) return lhs > max / rhs || lhs < min / rhs;
487 else if (rhs < -1) return lhs > min / rhs || lhs < max / rhs;
488 else return rhs == -1 && lhs == min;
489 }

491 // PLUS Operation
492 template < typename Rep1, Radix Rad1>
493 CUDA_HOST_DEVICE_CALLABLE
494 fixed_point<Rep1, Rad1> operator+( fixed_point<Rep1, Rad1> const & lhs,
495 fixed_point<Rep1, Rad1> const & rhs) {
496
497 auto const rhsv = lhs._scale > rhs._scale ? detail::shift_with_precise_round<Rep1, Rad1>(rhs._value, scale_type {lhs._scale - rhs._scale}) : rhs._value;
498 auto const lhsv = lhs._scale < rhs._scale ? detail::shift_with_precise_round<Rep1, Rad1>(lhs._value, scale_type {rhs._scale - lhs._scale}) : lhs._value;
499 auto const scale = lhs._scale > rhs._scale ? lhs._scale : rhs._scale;
500
501 #if defined(__CUDACC_DEBUG__)
502
503 assert(!addition_overflow<Rep1>(lhsv, rhsv) &&
504 "fixed_point overflow of underlying represenation type " + print_rep<Rep1>());
505
506 #endif
507
508 return fixed_point<Rep1, Rad1> { scaled_integer<Rep1> (lhsv + rhsv, scale)};
509 }

512 template < typename Rep1, Radix Rad1>
513 CUDA_HOST_DEVICE_CALLABLE
514 fixed_point<Rep1, Rad1> operator-( fixed_point<Rep1, Rad1> const & lhs,
515 fixed_point<Rep1, Rad1> const & rhs) {
516
517 auto const rhsv = lhs._scale > rhs._scale ? detail::shift_with_precise_round<Rep1, Rad1>(rhs._value, scale_type {lhs._scale - rhs._scale}) : rhs._value;
518 auto const lhsv = lhs._scale < rhs._scale ? detail::shift_with_precise_round<Rep1, Rad1>(lhs._value, scale_type {rhs._scale - lhs._scale}) : lhs._value;
519 auto const scale = lhs._scale > rhs._scale ? lhs._scale : rhs._scale;
520
521 #if defined(__CUDACC_DEBUG__)
522
523 assert(!subtraction_overflow<Rep1>(lhsv, rhsv) &&
524 "fixed_point overflow of underlying represenation type " + print_rep<Rep1>());
525
526 #endif
527
528 return fixed_point<Rep1, Rad1> { scaled_integer<Rep1> (lhsv - rhsv, scale)};
529 }

563 // EQUALITY COMPARISON Operation
564 template < typename Rep1, Radix Rad1>
565 CUDA_HOST_DEVICE_CALLABLE
566 bool operator== ( fixed_point<Rep1, Rad1> const & lhs,
567 fixed_point<Rep1, Rad1> const & rhs) {
568 auto const rhsv = lhs._scale > rhs._scale ? detail::shift_with_precise_round<Rep1, Rad1>(rhs._value, scale_type {lhs._scale - rhs._scale}) : rhs._value;
569 auto const lhsv = lhs._scale < rhs._scale ? detail::shift_with_precise_round<Rep1, Rad1>(lhs._value, scale_type {rhs._scale - lhs._scale}) : lhs._value;
570 return rhsv == lhsv;
571 }

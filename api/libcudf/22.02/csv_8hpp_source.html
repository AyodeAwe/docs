194  std::size_t get_byte_range_padding() const
195  {
196  auto const num_names = _names.size();
197  auto const num_dtypes = std::visit([](const auto& dtypes) { return dtypes.size(); }, _dtypes);
198  auto const num_columns = std::max(num_dtypes, num_names);
199 
200  auto const max_row_bytes = 16 * 1024; // 16KB
201  auto const column_bytes = 64;
202  auto const base_padding = 1024; // 1KB
203 
204  if (num_columns == 0) {
205  // Use flat size if the number of columns is not known
206  return max_row_bytes;
207  }
208 
209  // Expand the size based on the number of columns, if available
210  return base_padding + num_columns * column_bytes;
211  }

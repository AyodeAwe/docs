77 template <typename Rep,
78  Radix Base,
79  typename T,
80  typename cuda::std::enable_if_t<(cuda::std::is_same_v<int32_t, T> &&
81  is_supported_representation_type<Rep>())>* = nullptr>
82 CUDF_HOST_DEVICE inline Rep ipow(T exponent)
83 {
84  cudf_assert(exponent >= 0 && "integer exponentiation with negative exponent is not possible.");
85  if (exponent == 0) return static_cast<Rep>(1);
86  auto extra = static_cast<Rep>(1);
87  auto square = static_cast<Rep>(Base);
88  while (exponent > 1) {
89  if (exponent & 1 /* odd */) {
90  extra *= square;
91  exponent -= 1;
92  }
93  exponent /= 2;
94  square *= square;
95  }
96  return square * extra;
97 }

544  CUDF_HOST_DEVICE inline fixed_point<Rep, Rad> rescaled(scale_type scale) const
545  {
546  if (scale == _scale) return *this;
547  Rep const value = detail::shift<Rep, Rad>(_value, scale_type{scale - _scale});
548  return fixed_point<Rep, Rad>{scaled_integer<Rep>{value, scale}};

554  explicit operator std::string() const
555  {
556  if (_scale < 0) {
557  auto const av = detail::abs(_value);
558  Rep const n = detail::exp10<Rep>(-_scale);
559  Rep const f = av % n;
560  auto const num_zeros =
561  std::max(0, (-_scale - static_cast<int32_t>(detail::to_string(f).size())));
562  auto const zeros = std::string(num_zeros, '0');
563  auto const sign = _value < 0 ? std::string("-") : std::string();
564  return sign + detail::to_string(av / n) + std::string(".") + zeros +
565  detail::to_string(av % n);
566  } else {
567  auto const zeros = std::string(_scale, '0');
568  return detail::to_string(_value) + zeros;

626 template <typename Rep, typename T>
627 CUDF_HOST_DEVICE inline auto multiplication_overflow(T lhs, T rhs)
628 {
629  auto const min = cuda::std::numeric_limits<Rep>::min();
630  auto const max = cuda::std::numeric_limits<Rep>::max();
631  if (rhs > 0)
632  return lhs > max / rhs || lhs < min / rhs;
633  else if (rhs < -1)
634  return lhs > min / rhs || lhs < max / rhs;
635  else
636  return rhs == -1 && lhs == min;
637 }

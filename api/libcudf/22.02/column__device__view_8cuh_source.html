1298 template <typename ColumnDeviceView, typename ColumnViewIterator>
1299 ColumnDeviceView* child_columns_to_device_array(ColumnViewIterator child_begin,
1300  ColumnViewIterator child_end,
1301  void* h_ptr,
1302  void* d_ptr)
1303 {
1304  ColumnDeviceView* d_children = detail::align_ptr_for_type<ColumnDeviceView>(d_ptr);
1305  auto num_children = std::distance(child_begin, child_end);
1306  if (num_children > 0) {
1307  // The beginning of the memory must be the fixed-sized ColumnDeviceView
1308  // struct objects in order for d_children to be used as an array.
1309  auto h_column = detail::align_ptr_for_type<ColumnDeviceView>(h_ptr);
1310  auto d_column = d_children;
1311 
1312  // Any child data is assigned past the end of this array: h_end and d_end.
1313  auto h_end = reinterpret_cast<int8_t*>(h_column + num_children);
1314  auto d_end = reinterpret_cast<int8_t*>(d_column + num_children);
1315  std::for_each(child_begin, child_end, [&](auto const& col) {
1316  // inplace-new each child into host memory
1317  new (h_column) ColumnDeviceView(col, h_end, d_end);
1318  h_column++; // advance to next child
1319  // update the pointers for holding this child column's child data
1320  auto col_child_data_size = ColumnDeviceView::extent(col) - sizeof(ColumnDeviceView);
1321  h_end += col_child_data_size;
1322  d_end += col_child_data_size;
1323  });
1324  }
1325  return d_children;
1326 }

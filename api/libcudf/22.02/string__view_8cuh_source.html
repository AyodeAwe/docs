45 __device__ inline size_type characters_in_string(const char* str, size_type bytes)
46 {
47  if ((str == 0) || (bytes == 0)) return 0;
48  auto ptr = reinterpret_cast<uint8_t const*>(str);
49 #ifndef CUDF_JIT_UDF
50  return thrust::count_if(
51  thrust::seq, ptr, ptr + bytes, [](uint8_t chr) { return is_begin_utf8_char(chr); });
52 #else
53  size_type chars = 0;
54  auto const end = ptr + bytes;
55  while (ptr < end) {
56  chars += is_begin_utf8_char(*ptr++);
57  }
58  return chars;
59 #endif
60 }

253 __device__ inline size_type string_view::byte_offset(size_type pos) const
254 {
255  size_type offset = 0;
256  const char* sptr = _data;
257  const char* eptr = sptr + _bytes;
258  if (length() == size_bytes()) return pos;
259  while ((pos > 0) && (sptr < eptr)) {
260  size_type charbytes = strings::detail::bytes_in_utf8_byte(static_cast<uint8_t>(*sptr++));
261  if (charbytes) --pos;
262  offset += charbytes;
263  }
264  return offset;
265 }

272 __device__ inline int string_view::compare(const char* data, size_type bytes) const
273 {
274  size_type const len1 = size_bytes();
275  const unsigned char* ptr1 = reinterpret_cast<const unsigned char*>(this->data());
276  const unsigned char* ptr2 = reinterpret_cast<const unsigned char*>(data);
277  if ((ptr1 == ptr2) && (bytes == len1)) return 0;
278  size_type idx = 0;
279  for (; (idx < len1) && (idx < bytes); ++idx) {
280  if (*ptr1 != *ptr2) return static_cast<int32_t>(*ptr1) - static_cast<int32_t>(*ptr2);
281  ++ptr1;
282  ++ptr2;
283  }
284  if (idx < len1) return 1;
285  if (idx < bytes) return -1;
286  return 0;
287 }

328 __device__ inline size_type string_view::find(const char* str,
329  size_type bytes,
330  size_type pos,
331  size_type count) const
332 {
333  const char* sptr = data();
334  if (!str || !bytes) return -1;
335  size_type nchars = length();
336  if (count < 0) count = nchars;
337  size_type end = pos + count;
338  if (end < 0 || end > nchars) end = nchars;
339  size_type spos = byte_offset(pos);
340  size_type epos = byte_offset(end);
341 
342  size_type len2 = bytes;
343  size_type len1 = (epos - spos) - len2 + 1;
344 
345  const char* ptr1 = sptr + spos;
346  const char* ptr2 = str;
347  for (size_type idx = 0; idx < len1; ++idx) {
348  bool match = true;
349  for (size_type jdx = 0; match && (jdx < len2); ++jdx)
350  match = (ptr1[jdx] == ptr2[jdx]);
351  if (match) return character_offset(idx + spos);
352  ptr1++;
353  }
354  return -1;
355 }

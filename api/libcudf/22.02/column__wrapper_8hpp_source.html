93 template <typename From, typename To>
94 struct fixed_width_type_converter {
95  // Are the types same - simply copy elements from [begin, end) to out
96  template <typename FromT = From,
97  typename ToT = To,
98  typename std::enable_if<std::is_same_v<FromT, ToT>, void>::type* = nullptr>
99  constexpr ToT operator()(FromT element) const
100  {
101  return element;
102  }
103 
104  // Are the types convertible or can target be constructed from source?
105  template <typename FromT = From,
106  typename ToT = To,
107  typename std::enable_if<!std::is_same_v<FromT, ToT> &&
108  (cudf::is_convertible<FromT, ToT>::value ||
109  std::is_constructible<ToT, FromT>::value),
110  void>::type* = nullptr>
111  constexpr ToT operator()(FromT element) const
112  {
113  return static_cast<ToT>(element);
114  }
115 
116  // Convert integral values to timestamps
117  template <
118  typename FromT = From,
119  typename ToT = To,
120  typename std::enable_if<std::is_integral<FromT>::value && cudf::is_timestamp_t<ToT>::value,
121  void>::type* = nullptr>
122  constexpr ToT operator()(FromT element) const
123  {
124  return ToT{typename ToT::duration{element}};
125  }
126 };

148  auto const elements = thrust::host_vector<ElementTo>(transform_begin, transform_begin + size);

218 template <typename ValidityIterator>

220 {
221  auto const size = cudf::distance(begin, end);
222  auto const num_words = cudf::bitmask_allocation_size_bytes(size) / sizeof(bitmask_type);
223

226  if (*(begin + i)) set_bit_unsafe(null_mask.data(), i);

267 template <typename StringsIterator, typename ValidityIterator>
268 auto make_chars_and_offsets(StringsIterator begin, StringsIterator end, ValidityIterator v)
269 {
270  std::vector<char> chars{};
271  std::vector<cudf::size_type> offsets(1, 0);
272  for (auto str = begin; str < end; ++str) {
273  std::string tmp = (*v++) ? std::string(*str) : std::string{};
274  chars.insert(chars.end(), std::cbegin(tmp), std::cend(tmp));
275  offsets.push_back(offsets.back() + tmp.length());
276  }
277  return std::make_pair(std::move(chars), std::move(offsets));
278 };

1606  std::pair<std::vector<column_view>, std::vector<std::unique_ptr<column>>> preprocess_columns(
1607  std::initializer_list<lists_column_wrapper<T, SourceElementT>> const& elements,
1608  column_view& expected_hierarchy,
1609  int expected_depth)
1610  {
1611  std::vector<std::unique_ptr<column>> stubs;
1612  std::vector<column_view> cols;
1613 
1614  // preprocess the incoming lists.
1615  // - unwrap any "root" lists
1616  // - handle incomplete hierarchies
1617  std::transform(elements.begin(),
1618  elements.end(),
1619  std::back_inserter(cols),
1620  [&](lists_column_wrapper const& l) -> column_view {
1621  // depth mismatch. attempt to normalize the short column.
1622  // this function will also catch if this is a legitimately broken
1623  // set of input
1624  if (l.depth < expected_depth) {
1625  if (l.root) {
1626  // this exception distinguishes between the following two cases:
1627  //
1628  // { {{{1, 2, 3}}}, {} }
1629  // In this case, row 0 is a List<List<List<int>>>, whereas row 1 is
1630  // just a List<> which is an apparent mismatch. However, because row 1
1631  // is empty we will allow that to semantically mean
1632  // "a List<List<List<int>>> that's empty at the top level"
1633  //
1634  // { {{{1, 2, 3}}}, {4, 5, 6} }
1635  // In this case, row 1 is a a concrete List<int> with actual values.
1636  // There is no way to rectify the differences so we will treat it as a
1637  // true column mismatch.
1638  CUDF_EXPECTS(l.wrapped->size() == 0, "Mismatch in column types!");
1639  stubs.push_back(empty_like(expected_hierarchy));
1640  } else {
1641  stubs.push_back(normalize_column(l.get_view(), expected_hierarchy));
1642  }
1643  return *(stubs.back());
1644  }
1645  // the empty hierarchy case
1646  return l.get_view();
1647  });
1648 
1649  return {std::move(cols), std::move(stubs)};
1650  }
1651 
1652  column_view get_view() const { return root ? lists_column_view(*wrapped).child() : *wrapped; }
1653 
1654  int depth = 0;
1655  bool root = false;

91 template <typename From, typename To>
92 struct fixed_width_type_converter {
93  // Are the types same - simply copy elements from [begin, end) to out
94  template <typename FromT = From,
95  typename ToT = To,
96  typename std::enable_if<std::is_same<FromT, ToT>::value, void>::type* = nullptr>
97  constexpr ToT operator()(FromT element) const
98  {
99  return element;
100  }
101 
102  // Are the types convertible or can target be constructed from source?
103  template <typename FromT = From,
104  typename ToT = To,
105  typename std::enable_if<!std::is_same<FromT, ToT>::value &&
106  (cudf::is_convertible<FromT, ToT>::value ||
107  std::is_constructible<ToT, FromT>::value),
108  void>::type* = nullptr>
109  constexpr ToT operator()(FromT element) const
110  {
111  return static_cast<ToT>(element);
112  }
113 
114  // Convert integral values to timestamps
115  template <
116  typename FromT = From,
117  typename ToT = To,
118  typename std::enable_if<std::is_integral<FromT>::value && cudf::is_timestamp_t<ToT>::value,
119  void>::type* = nullptr>
120  constexpr ToT operator()(FromT element) const
121  {
122  return ToT{typename ToT::duration{element}};
123  }
124 };

218 {

220  auto const num_words = cudf::bitmask_allocation_size_bytes(size) / sizeof(bitmask_type);

264 template <typename StringsIterator, typename ValidityIterator>
265 auto make_chars_and_offsets(StringsIterator begin, StringsIterator end, ValidityIterator v)
266 {
267  std::vector<char> chars{};
268  std::vector<cudf::size_type> offsets(1, 0);
269  for (auto str = begin; str < end; ++str) {
270  std::string tmp = (*v++) ? std::string(*str) : std::string{};
271  chars.insert(chars.end(), std::cbegin(tmp), std::cend(tmp));
272  offsets.push_back(offsets.back() + tmp.length());
273  }
274  return std::make_pair(std::move(chars), std::move(offsets));
275 };

1598  std::pair<std::vector<column_view>, std::vector<std::unique_ptr<column>>> preprocess_columns(
1599  std::initializer_list<lists_column_wrapper<T, SourceElementT>> const& elements,
1600  column_view& expected_hierarchy,
1601  int expected_depth)
1602  {
1603  std::vector<std::unique_ptr<column>> stubs;
1604  std::vector<column_view> cols;
1605 
1606  // preprocess the incoming lists.
1607  // - unwrap any "root" lists
1608  // - handle incomplete hierarchies
1609  std::transform(elements.begin(),
1610  elements.end(),
1611  std::back_inserter(cols),
1612  [&](lists_column_wrapper const& l) -> column_view {
1613  // depth mismatch. attempt to normalize the short column.
1614  // this function will also catch if this is a legitmately broken
1615  // set of input
1616  if (l.depth < expected_depth) {
1617  if (l.root) {
1618  // this exception distinguishes between the following two cases:
1619  //
1620  // { {{{1, 2, 3}}}, {} }
1621  // In this case, row 0 is a List<List<List<int>>>, whereas row 1 is
1622  // just a List<> which is an apparent mismatch. However, because row 1
1623  // is empty we will allow that to semantically mean
1624  // "a List<List<List<int>>> that's empty at the top level"
1625  //
1626  // { {{{1, 2, 3}}}, {4, 5, 6} }
1627  // In this case, row 1 is a a concrete List<int> with actual values.
1628  // There is no way to rectify the differences so we will treat it as a
1629  // true column mismatch.
1630  CUDF_EXPECTS(l.wrapped->size() == 0, "Mismatch in column types!");
1631  stubs.push_back(empty_like(expected_hierarchy));
1632  } else {
1633  stubs.push_back(normalize_column(l.get_view(), expected_hierarchy));
1634  }
1635  return *(stubs.back());
1636  }
1637  // the empty hierarchy case
1638  return l.get_view();
1639  });
1640 
1641  return {std::move(cols), std::move(stubs)};
1642  }
1643 
1644  column_view get_view() const { return root ? lists_column_view(*wrapped).child() : *wrapped; }
1645 
1646  int depth = 0;
1647  bool root = false;

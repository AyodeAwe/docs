1050 template <typename ColumnDeviceView, typename ColumnViewIterator>
1051 ColumnDeviceView* child_columns_to_device_array(ColumnViewIterator child_begin,
1052  ColumnViewIterator child_end,
1053  void* h_ptr,
1054  void* d_ptr)
1055 {
1056  ColumnDeviceView* d_children = detail::align_ptr_for_type<ColumnDeviceView>(d_ptr);
1057  auto num_children = std::distance(child_begin, child_end);
1058  if (num_children > 0) {
1059  // The beginning of the memory must be the fixed-sized ColumnDeviceView
1060  // struct objects in order for d_children to be used as an array.
1061  auto h_column = detail::align_ptr_for_type<ColumnDeviceView>(h_ptr);
1062  auto d_column = d_children;
1063 
1064  // Any child data is assigned past the end of this array: h_end and d_end.
1065  auto h_end = reinterpret_cast<int8_t*>(h_column + num_children);
1066  auto d_end = reinterpret_cast<int8_t*>(d_column + num_children);
1067  std::for_each(child_begin, child_end, [&](auto const& col) {
1068  // inplace-new each child into host memory
1069  new (h_column) ColumnDeviceView(col, h_end, d_end);
1070  h_column++; // advance to next child
1071  // update the pointers for holding this child column's child data
1072  auto col_child_data_size = ColumnDeviceView::extent(col) - sizeof(ColumnDeviceView);
1073  h_end += col_child_data_size;
1074  d_end += col_child_data_size;
1075  });
1076  }
1077  return d_children;
1078 }

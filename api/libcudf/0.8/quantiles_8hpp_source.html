26 template < typename T_out, typename T_in>
27 void linear(T_out& result, T_in lhs, T_in rhs, double frac)
28 {
29 // TODO: safe operation to avoid overflow/underflow
30 // double can fully represent int8-32 value range.
31 // Since the fractoin part of double is 52 bits,
32 // double cannot fully represent int64.
33 // The underflow will be occurs at converting int64 to double
34 // detail: https://github.com/rapidsai/cudf/issues/1417
35
36 double dlhs = static_cast< double > (lhs);
37 double drhs = static_cast< double > (rhs);
38 double one_minus_frac = 1.0 - frac;
39
40 // result = static_cast<T_out>(static_cast<T_out>(lhs) + frac*static_cast<T_out>(rhs-lhs));
41 result = static_cast< T_out > (one_minus_frac * dlhs + frac * drhs);
42 }

45 template < typename T_out, typename T_in>
46 void midpoint(T_out& result, T_in lhs, T_in rhs)
47 {
48 // TODO: try std::midpoint (C++20) if available
49 double dlhs = static_cast< double > (lhs);
50 double drhs = static_cast< double > (rhs);
51
52 result = static_cast< T_out > ( dlhs /2 + drhs /2 );
53 }

58 template < typename T_out>
59 void midpoint(T_out& result, int64_t lhs, int64_t rhs)
60 {
61 // caring to avoid integer overflow and underflow between int64_t and T_out( double )
62 int64_t half = lhs/2 + rhs/2;
63 int64_t rest = (lhs%2 + rhs%2);
64
65 result = static_cast< T_out > ( static_cast< T_out > (half) + static_cast<T_out>(rest)*0.5);
66 }

69 template <>
70 void midpoint(int64_t& result, int64_t lhs, int64_t rhs)
71 {
72 // caring to avoid integer overflow
73 int64_t half = lhs/2 + rhs/2;
74 int64_t rest = (lhs%2 + rhs%2);
75 result = half;
76
77 // rounding toward zero
78 result += ( half >= 0 && rest != 0 )? rest/2 : 0;
79 result += ( half < 0 && rest != 0 )? 1 : 0;
80 }

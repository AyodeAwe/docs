165  block_type try_to_expand(std::size_t try_size, std::size_t min_size, cuda_stream_view stream)
166  {
167  while (try_size >= min_size) {
168  auto b = block_from_upstream(try_size, stream);
169  if (b.has_value()) {
170  current_pool_size_ += b.value().size();
171  return b.value();
172  }
173  if (try_size == min_size) break; // only try `size` once
174  try_size = std::max(min_size, try_size / 2);
175  }
176  RMM_LOG_ERROR("[A][Stream {}][Upstream {}B][FAILURE maximum pool size exceeded]",
177  fmt::ptr(stream.value()),
178  min_size);
179  RMM_FAIL("Maximum pool size exceeded", rmm::bad_alloc);
180  }

194  void initialize_pool(thrust::optional<std::size_t> initial_size,
195  thrust::optional<std::size_t> maximum_size)
196  {
197  auto const try_size = [&]() {
198  if (not initial_size.has_value()) {
199  std::size_t free{}, total{};
200  std::tie(free, total) = (get_upstream()->supports_get_mem_info())
201  ? get_upstream()->get_mem_info(cuda_stream_legacy)
202  : rmm::detail::available_device_memory();
203  return rmm::detail::align_up(std::min(free, total / 2),
204  rmm::detail::CUDA_ALLOCATION_ALIGNMENT);
205  } else {
206  return initial_size.value();
207  }
208  }();
209 
210  current_pool_size_ = 0; // try_to_expand will set this if it succeeds
211  maximum_pool_size_ = maximum_size;
212 
213  RMM_EXPECTS(try_size <= maximum_pool_size_.value_or(std::numeric_limits<std::size_t>::max()),
214  "Initial pool size exceeds the maximum pool size!");
215 
216  if (try_size > 0) {
217  auto const b = try_to_expand(try_size, try_size, cuda_stream_legacy);
218  this->insert_block(b, cuda_stream_legacy);
219  }
220  }

231  block_type expand_pool(std::size_t size, free_list& blocks, cuda_stream_view stream)
232  {
233  // Strategy: If maximum_pool_size_ is set, then grow geometrically, e.g. by halfway to the
234  // limit each time. If it is not set, grow exponentially, e.g. by doubling the pool size each
235  // time. Upon failure, attempt to back off exponentially, e.g. by half the attempted size,
236  // until either success or the attempt is less than the requested size.
237  return try_to_expand(size_to_grow(size), size, stream);
238  }

252  std::size_t size_to_grow(std::size_t size) const
253  {
254  if (maximum_pool_size_.has_value()) {
255  auto const unaligned_remaining = maximum_pool_size_.value() - pool_size();
256  auto const remaining =
257  rmm::detail::align_up(unaligned_remaining, rmm::detail::CUDA_ALLOCATION_ALIGNMENT);
258  auto const aligned_size = rmm::detail::align_up(size, rmm::detail::CUDA_ALLOCATION_ALIGNMENT);
259  return (aligned_size <= remaining) ? std::max(aligned_size, remaining / 2) : 0;
260  } else
261  return std::max(size, pool_size());
262  };

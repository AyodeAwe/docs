205  void set_element_async(std::size_t element_index, value_type const& v, cuda_stream_view s)
206  {
207  RMM_EXPECTS(
208  element_index < size(), rmm::out_of_range, "Attempt to access out of bounds element.");
209  if constexpr (std::is_fundamental<value_type>::value) {
210  if constexpr (std::is_same<value_type, bool>::value) {
211  RMM_CUDA_TRY(cudaMemsetAsync(element_ptr(element_index), v, sizeof(v), s.value()));
212  } else {
213  if (v == value_type{0}) {
214  set_element_to_zero_async(element_index, s);
215  } else {
216  RMM_CUDA_TRY(cudaMemcpyAsync(
217  element_ptr(element_index), &v, sizeof(v), cudaMemcpyDefault, s.value()));
218  }
219  }
220  } else {
221  RMM_CUDA_TRY(
222  cudaMemcpyAsync(element_ptr(element_index), &v, sizeof(v), cudaMemcpyDefault, s.value()));
223  }
224  }

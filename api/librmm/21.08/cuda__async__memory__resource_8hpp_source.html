56  cuda_async_memory_resource(thrust::optional<std::size_t> initial_pool_size = {},
57  thrust::optional<std::size_t> release_threshold = {})
58  {
59 #ifdef RMM_CUDA_MALLOC_ASYNC_SUPPORT
60  // Check if cudaMallocAsync Memory pool supported
61  auto const device = rmm::detail::current_device();
62  int cuda_pool_supported{};
63  auto e =
64  cudaDeviceGetAttribute(&cuda_pool_supported, cudaDevAttrMemoryPoolsSupported, device.value());
65  RMM_EXPECTS(e == cudaSuccess && cuda_pool_supported,
66  "cudaMallocAsync not supported with this CUDA driver/runtime version");
67 
68  // Construct explicit pool
69  cudaMemPoolProps pool_props{};
70  pool_props.allocType = cudaMemAllocationTypePinned;
71  pool_props.handleTypes = cudaMemHandleTypePosixFileDescriptor;
72  pool_props.location.type = cudaMemLocationTypeDevice;
73  pool_props.location.id = device.value();
74  RMM_CUDA_TRY(cudaMemPoolCreate(&cuda_pool_handle_, &pool_props));
75 
76  auto const [free, total] = rmm::detail::available_device_memory();
77 
78  // Need an l-value to take address to pass to cudaMemPoolSetAttribute
79  uint64_t threshold = release_threshold.value_or(total);
80  RMM_CUDA_TRY(
81  cudaMemPoolSetAttribute(cuda_pool_handle_, cudaMemPoolAttrReleaseThreshold, &threshold));
82 
83  // Allocate and immediately deallocate the initial_pool_size to prime the pool with the
84  // specified size
85  auto const pool_size = initial_pool_size.value_or(free * 0.5);
86  auto p = do_allocate(pool_size, cuda_stream_default);
87  do_deallocate(p, pool_size, cuda_stream_default);
88 
89 #else
90  RMM_FAIL(
91  "cudaMallocAsync not supported by the version of the CUDA Toolkit used for this build");
92 #endif
93  }

121  void* do_allocate(std::size_t bytes, cuda_stream_view stream) override
122  {
123  if (allocation_alignment_ == rmm::detail::CUDA_ALLOCATION_ALIGNMENT ||
124  bytes < alignment_threshold_) {
125  return upstream_->allocate(bytes, stream);
126  } else {
127  auto const size = upstream_allocation_size(bytes);
128  void* pointer = upstream_->allocate(size, stream);
129  auto const address = reinterpret_cast<std::size_t>(pointer);
130  auto const aligned_address = rmm::detail::align_up(address, allocation_alignment_);
131  void* aligned_pointer = reinterpret_cast<void*>(aligned_address);
132  if (pointer != aligned_pointer) {
133  lock_guard lock(mtx_);
134  pointers_.emplace(aligned_pointer, pointer);
135  }
136  return aligned_pointer;
137  }
138  }

149  void do_deallocate(void* p, std::size_t bytes, cuda_stream_view stream) override
150  {

152  bytes < alignment_threshold_) {
153  upstream_->deallocate(p, bytes, stream);
154  } else {

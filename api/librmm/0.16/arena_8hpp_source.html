160 inline block first_fit(std::set<block>& free_blocks, std::size_t size)
161 {
162  auto const iter = std::find_if(
163  free_blocks.cbegin(), free_blocks.cend(), [size](auto const& b) { return b.fits(size); });
164 
165  if (iter == free_blocks.cend()) {
166  return {};
167  } else {
168  // Remove the block from the free_list.
169  auto const b = *iter;
170  auto const i = free_blocks.erase(iter);
171 
172  // Split the block and put the remainder back.
173  auto const split = b.split(size);
174  auto const remainder = split.second;
175  if (remainder.is_valid()) { free_blocks.insert(i, remainder); }
176  return split.first;
177  }
178 }

190 inline block split_block(std::set<block>& free_blocks, block const& b, std::size_t size)
191 {
192  auto const split = b.split(size);
193  auto const remainder = split.second;
194  if (remainder.is_valid()) free_blocks.emplace(remainder);
195  return split.first;
196 }

205 inline block coalesce_block(std::set<block>& free_blocks, block const& b)
206 {
207  if (!b.is_valid()) return b;
208 
209  // Find the right place (in ascending address order) to insert the block.
210  auto const next = free_blocks.lower_bound(b);
211  auto const previous = next == free_blocks.cbegin() ? next : std::prev(next);
212 
213  // Coalesce with neighboring blocks.
214  bool const merge_prev = previous->is_contiguous_before(b);
215  bool const merge_next = next != free_blocks.cend() && b.is_contiguous_before(*next);
216 
217  block merged{};
218  if (merge_prev && merge_next) {
219  merged = previous->merge(b).merge(*next);
220  free_blocks.erase(previous);
221  auto const i = free_blocks.erase(next);
222  free_blocks.insert(i, merged);
223  } else if (merge_prev) {
224  merged = previous->merge(b);
225  auto const i = free_blocks.erase(previous);
226  free_blocks.insert(i, merged);
227  } else if (merge_next) {
228  merged = b.merge(*next);
229  auto const i = free_blocks.erase(next);
230  free_blocks.insert(i, merged);
231  } else {
232  free_blocks.emplace(b);
233  merged = b;
234  }
235  return merged;
236 }

273  global_arena(Upstream* upstream_mr, std::size_t initial_size, std::size_t maximum_size)

275  {
276  RMM_EXPECTS(nullptr != upstream_mr_, "Unexpected null upstream pointer.");
277  RMM_EXPECTS(initial_size == default_initial_size || initial_size == align_up(initial_size),
278  "Error, Initial arena size required to be a multiple of 256 bytes");
279  RMM_EXPECTS(maximum_size_ == default_maximum_size || maximum_size_ == align_up(maximum_size_),
280  "Error, Maximum arena size required to be a multiple of 256 bytes");
281 
282  std::size_t free{}, total{};
283  RMM_CUDA_TRY(cudaMemGetInfo(&free, &total));
284  if (initial_size == default_initial_size) {
285  initial_size = align_up(std::min(free, total / 2));
286  }
287  if (maximum_size_ == default_maximum_size) { maximum_size_ = align_down(free) - reserved_size; }
288

292  }

619  {

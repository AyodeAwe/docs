60  cuda_async_memory_resource(thrust::optional<std::size_t> initial_pool_size = {},
61  thrust::optional<std::size_t> release_threshold = {})
62  {
63 #ifdef RMM_CUDA_MALLOC_ASYNC_SUPPORT
64  // Check if cudaMallocAsync Memory pool supported
65  RMM_EXPECTS(is_supported(),
66  "cudaMallocAsync not supported with this CUDA driver/runtime version");
67 
68  // Construct explicit pool
69  cudaMemPoolProps pool_props{};
70  pool_props.allocType = cudaMemAllocationTypePinned;
71  pool_props.handleTypes = cudaMemHandleTypePosixFileDescriptor;
72  pool_props.location.type = cudaMemLocationTypeDevice;
73  pool_props.location.id = rmm::detail::current_device().value();
74  RMM_CUDA_TRY(cudaMemPoolCreate(&cuda_pool_handle_, &pool_props));
75 
76  // CUDA drivers before 11.5 have known incompatibilities with the async allocator.
77  // We'll disable `cudaMemPoolReuseAllowOpportunistic` if cuda driver < 11.5.
78  // See https://github.com/NVIDIA/spark-rapids/issues/4710.
79  int driver_version{};
80  RMM_CUDA_TRY(cudaDriverGetVersion(&driver_version));
81  constexpr auto min_async_version{11050};
82  if (driver_version < min_async_version) {
83  int disabled{0};
84  RMM_CUDA_TRY(
85  cudaMemPoolSetAttribute(cuda_pool_handle_, cudaMemPoolReuseAllowOpportunistic, &disabled));
86  }
87 
88  auto const [free, total] = rmm::detail::available_device_memory();
89 
90  // Need an l-value to take address to pass to cudaMemPoolSetAttribute
91  uint64_t threshold = release_threshold.value_or(total);
92  RMM_CUDA_TRY(
93  cudaMemPoolSetAttribute(cuda_pool_handle_, cudaMemPoolAttrReleaseThreshold, &threshold));
94 
95  // Allocate and immediately deallocate the initial_pool_size to prime the pool with the
96  // specified size
97  auto const pool_size = initial_pool_size.value_or(free / 2);
98  auto* ptr = do_allocate(pool_size, cuda_stream_default);
99  do_deallocate(ptr, pool_size, cuda_stream_default);
100 
101 #else
102  RMM_FAIL(
103  "cudaMallocAsync not supported by the version of the CUDA Toolkit used for this build");
104 #endif
105  }

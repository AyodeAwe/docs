62  cuda_async_memory_resource(thrust::optional<std::size_t> initial_pool_size = {},
63  thrust::optional<std::size_t> release_threshold = {})
64  {
65 #ifdef RMM_CUDA_MALLOC_ASYNC_SUPPORT
66  // Check if cudaMallocAsync Memory pool supported
67  RMM_EXPECTS(rmm::detail::async_alloc::is_supported(),
68  "cudaMallocAsync not supported with this CUDA driver/runtime version");
69 
70  // Construct explicit pool
71  cudaMemPoolProps pool_props{};
72  pool_props.allocType = cudaMemAllocationTypePinned;
73  pool_props.handleTypes = cudaMemHandleTypePosixFileDescriptor;
74  pool_props.location.type = cudaMemLocationTypeDevice;
75  pool_props.location.id = rmm::detail::current_device().value();
76  cudaMemPool_t cuda_pool_handle{};
77  RMM_CUDA_TRY(rmm::detail::async_alloc::cudaMemPoolCreate(&cuda_pool_handle, &pool_props));
78  pool_ = cuda_async_view_memory_resource{cuda_pool_handle};
79 
80  // CUDA drivers before 11.5 have known incompatibilities with the async allocator.
81  // We'll disable `cudaMemPoolReuseAllowOpportunistic` if cuda driver < 11.5.
82  // See https://github.com/NVIDIA/spark-rapids/issues/4710.
83  int driver_version{};
84  RMM_CUDA_TRY(cudaDriverGetVersion(&driver_version));
85  constexpr auto min_async_version{11050};
86  if (driver_version < min_async_version) {
87  int disabled{0};
88  RMM_CUDA_TRY(rmm::detail::async_alloc::cudaMemPoolSetAttribute(
89  pool_handle(), cudaMemPoolReuseAllowOpportunistic, &disabled));
90  }
91 
92  auto const [free, total] = rmm::detail::available_device_memory();
93 
94  // Need an l-value to take address to pass to cudaMemPoolSetAttribute
95  uint64_t threshold = release_threshold.value_or(total);
96  RMM_CUDA_TRY(rmm::detail::async_alloc::cudaMemPoolSetAttribute(
97  pool_handle(), cudaMemPoolAttrReleaseThreshold, &threshold));
98 
99  // Allocate and immediately deallocate the initial_pool_size to prime the pool with the
100  // specified size
101  auto const pool_size = initial_pool_size.value_or(free / 2);
102  auto* ptr = do_allocate(pool_size, cuda_stream_default);
103  do_deallocate(ptr, pool_size, cuda_stream_default);
104 
105 #else
106  RMM_FAIL(
107  "cudaMallocAsync not supported by the version of the CUDA Toolkit used for this build");
108 #endif
109  }

183  void insert(block_type const& b)
184  {
185  if (is_empty()) {
186  free_list::insert(cend(), b);
187  return;
188  }
189 
190  // Find the right place (in ascending ptr order) to insert the block
191  // Can't use binary_search because it's a linked list and will be quadratic
192  auto const next = std::find_if(begin(), end(), [b](block_type const& i) { return b < i; });
193  auto const previous = (next == cbegin()) ? next : std::prev(next);
194 
195  // Coalesce with neighboring blocks or insert the new block if it can't be coalesced
196  bool const merge_prev = previous->is_contiguous_before(b);
197  bool const merge_next = (next != cend()) && b.is_contiguous_before(*next);
198 
199  if (merge_prev && merge_next) {
200  *previous = previous->merge(b).merge(*next);
201  erase(next);
202  } else if (merge_prev) {
203  *previous = previous->merge(b);
204  } else if (merge_next) {
205  *next = b.merge(*next);
206  } else {
207  free_list::insert(next, b); // cannot be coalesced, just insert
208  }
209  }
210 
219  void insert(free_list&& other)
220  {
221  std::for_each(std::make_move_iterator(other.begin()),
222  std::make_move_iterator(other.end()),
223  [this](block_type&& b) { this->insert(std::move(b)); });
224  }

234  block_type get_block(std::size_t size)
235  {
236  // find best fit block
237  auto const iter =
238  std::min_element(cbegin(), cend(), [size](block_type const& lhs, block_type const& rhs) {
239  return lhs.is_better_fit(size, rhs);
240  });
241 
242  if (iter != cend() && iter->fits(size)) {
243  // Remove the block from the free_list and return it.
244  block_type const found = *iter;
245  erase(iter);
246  return found;
247  }
248 
249  return block_type{}; // not found
250  }

206  void set_element_async(std::size_t element_index, value_type const& v, cuda_stream_view s)
207  {
208  RMM_EXPECTS(
209  element_index < size(), rmm::out_of_range, "Attempt to access out of bounds element.");
210  if constexpr (std::is_fundamental<value_type>::value) {
211  if constexpr (std::is_same<value_type, bool>::value) {
212  RMM_CUDA_TRY(cudaMemsetAsync(element_ptr(element_index), v, sizeof(v), s.value()));
213  } else {
214  if (v == value_type{0}) {
215  set_element_to_zero_async(element_index, s);
216  } else {
217  RMM_CUDA_TRY(cudaMemcpyAsync(
218  element_ptr(element_index), &v, sizeof(v), cudaMemcpyDefault, s.value()));
219  }
220  }
221  } else {
222  RMM_CUDA_TRY(
223  cudaMemcpyAsync(element_ptr(element_index), &v, sizeof(v), cudaMemcpyDefault, s.value()));
224  }
225  }

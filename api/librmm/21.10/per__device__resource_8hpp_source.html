129 inline device_memory_resource* get_per_device_resource(cuda_device_id id)
130 {
131  std::lock_guard<std::mutex> lock{detail::map_lock()};
132  auto& map = detail::get_map();
133  // If a resource was never set for `id`, set to the initial resource
134  auto const found = map.find(id.value());
135  return (found == map.end()) ? (map[id.value()] = detail::initial_resource()) : found->second;

165 inline device_memory_resource* set_per_device_resource(cuda_device_id id,
166  device_memory_resource* new_mr)
167 {
168  std::lock_guard<std::mutex> lock{detail::map_lock()};
169  auto& map = detail::get_map();
170  auto const old_itr = map.find(id.value());
171  // If a resource didn't previously exist for `id`, return pointer to initial_resource
172  auto old_mr = (old_itr == map.end()) ? detail::initial_resource() : old_itr->second;
173  map[id.value()] = (new_mr == nullptr) ? detail::initial_resource() : new_mr;
174  return old_mr;

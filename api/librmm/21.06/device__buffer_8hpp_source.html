209  device_buffer& operator=(device_buffer const& other)
210  {
211  if (&other != this) {
212  // If the current capacity is large enough and the resources are
213  // compatible, just reuse the existing memory
214  if ((capacity() > other.size()) and _mr->is_equal(*other._mr)) {
215  resize(other.size(), other.stream());
216  copy(other.data(), other.size());
217  } else {
218  // Otherwise, need to deallocate and allocate new memory
219  deallocate();
220  set_stream(other.stream());
221  _mr = other._mr;
222  allocate(other.size());
223  copy(other.data(), other.size());
224  }
225  }
226  return *this;
227  }

299  void resize(std::size_t new_size, cuda_stream_view stream = cuda_stream_view{})
300  {
301  set_stream(stream);
302  // If the requested size is smaller than the current capacity, just update
303  // the size without any allocations
304  if (new_size <= capacity()) {
305  _size = new_size;
306  } else {
307  void* const new_data = _mr->allocate(new_size, this->stream());
308  RMM_CUDA_TRY(
309  cudaMemcpyAsync(new_data, data(), size(), cudaMemcpyDefault, this->stream().value()));
310  deallocate();
311  _data = new_data;
312  _size = new_size;
313  _capacity = new_size;
314  }
315  }

330  void shrink_to_fit(cuda_stream_view stream = cuda_stream_view{})
331  {
332  set_stream(stream);
333  if (size() != capacity()) {
334  // Invoke copy ctor on self which only copies `[0, size())` and swap it
335  // with self. The temporary `device_buffer` will hold the old contents
336  // which will then be destroyed
337  auto tmp = device_buffer{*this, stream, _mr};
338  std::swap(tmp, *this);
339  }

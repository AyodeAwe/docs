211  arena& get_thread_arena()
212  {
213  auto const thread_id = std::this_thread::get_id();
214  {
215  read_lock lock(mtx_);
216  auto const iter = thread_arenas_.find(thread_id);
217  if (iter != thread_arenas_.end()) { return *iter->second; }
218  }
219  {
220  write_lock lock(mtx_);
221  auto thread_arena = std::make_shared<arena>(global_arena_);
222  thread_arenas_.emplace(thread_id, thread_arena);
223  thread_local detail::arena::arena_cleaner<Upstream> cleaner{thread_arena};
224  return *thread_arena;

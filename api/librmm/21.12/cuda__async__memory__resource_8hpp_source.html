60  cuda_async_memory_resource(thrust::optional<std::size_t> initial_pool_size = {},
61  thrust::optional<std::size_t> release_threshold = {})
62  {
63 #ifdef RMM_CUDA_MALLOC_ASYNC_SUPPORT
64  // Check if cudaMallocAsync Memory pool supported
65  auto const device = rmm::detail::current_device();
66  int cuda_pool_supported{};
67  auto result =
68  cudaDeviceGetAttribute(&cuda_pool_supported, cudaDevAttrMemoryPoolsSupported, device.value());
69  RMM_EXPECTS(result == cudaSuccess && cuda_pool_supported,
70  "cudaMallocAsync not supported with this CUDA driver/runtime version");
71 
72  // Construct explicit pool
73  cudaMemPoolProps pool_props{};
74  pool_props.allocType = cudaMemAllocationTypePinned;
75  pool_props.handleTypes = cudaMemHandleTypePosixFileDescriptor;
76  pool_props.location.type = cudaMemLocationTypeDevice;
77  pool_props.location.id = device.value();
78  RMM_CUDA_TRY(cudaMemPoolCreate(&cuda_pool_handle_, &pool_props));
79 
80  auto const [free, total] = rmm::detail::available_device_memory();
81 
82  // Need an l-value to take address to pass to cudaMemPoolSetAttribute
83  uint64_t threshold = release_threshold.value_or(total);
84  RMM_CUDA_TRY(
85  cudaMemPoolSetAttribute(cuda_pool_handle_, cudaMemPoolAttrReleaseThreshold, &threshold));
86 
87  // Allocate and immediately deallocate the initial_pool_size to prime the pool with the
88  // specified size
89  auto const pool_size = initial_pool_size.value_or(free / 2);
90  auto* ptr = do_allocate(pool_size, cuda_stream_default);
91  do_deallocate(ptr, pool_size, cuda_stream_default);
92 
93 #else
94  RMM_FAIL(
95  "cudaMallocAsync not supported by the version of the CUDA Toolkit used for this build");
96 #endif

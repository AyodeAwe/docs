178 inline block first_fit(std::set<block>& free_blocks, std::size_t size)
179 {
180  auto const iter = std::find_if(
181  free_blocks.cbegin(), free_blocks.cend(), [size](auto const& blk) { return blk.fits(size); });
182 
183  if (iter == free_blocks.cend()) { return {}; }
184  // Remove the block from the free_list.
185  auto const blk = *iter;
186  auto const next = free_blocks.erase(iter);
187 
188  if (blk.size() > size) {
189  // Split the block and put the remainder back.
190  auto const split = blk.split(size);
191  free_blocks.insert(next, split.second);
192  return split.first;
193  }
194  return blk;
195 }

204 inline block coalesce_block(std::set<block>& free_blocks, block const& blk)
205 {
206  if (!blk.is_valid()) { return blk; }
207 
208  // Find the right place (in ascending address order) to insert the block.
209  auto const next = free_blocks.lower_bound(blk);
210  auto const previous = next == free_blocks.cbegin() ? next : std::prev(next);
211 
212  // Coalesce with neighboring blocks.
213  bool const merge_prev = previous->is_contiguous_before(blk);
214  bool const merge_next = next != free_blocks.cend() && blk.is_contiguous_before(*next);
215 
216  block merged{};
217  if (merge_prev && merge_next) {
218  merged = previous->merge(blk).merge(*next);
219  free_blocks.erase(previous);
220  auto const iter = free_blocks.erase(next);
221  free_blocks.insert(iter, merged);
222  } else if (merge_prev) {
223  merged = previous->merge(blk);
224  auto const iter = free_blocks.erase(previous);
225  free_blocks.insert(iter, merged);
226  } else if (merge_next) {
227  merged = blk.merge(*next);
228  auto const iter = free_blocks.erase(next);
229  free_blocks.insert(iter, merged);
230  } else {
231  free_blocks.emplace(blk);
232  merged = blk;
233  }
234  return merged;
235 }

278  global_arena(Upstream* upstream_mr, std::size_t initial_size, std::size_t maximum_size)
279  : upstream_mr_{upstream_mr}, maximum_size_{maximum_size}
280  {
281  RMM_EXPECTS(nullptr != upstream_mr_, "Unexpected null upstream pointer.");
282  RMM_EXPECTS(initial_size == default_initial_size || initial_size == align_up(initial_size),
283  "Error, Initial arena size required to be a multiple of 256 bytes");
284  RMM_EXPECTS(maximum_size_ == default_maximum_size || maximum_size_ == align_up(maximum_size_),
285  "Error, Maximum arena size required to be a multiple of 256 bytes");
286 
287  if (initial_size == default_initial_size || maximum_size == default_maximum_size) {
288  std::size_t free{};
289  std::size_t total{};
290  RMM_CUDA_TRY(cudaMemGetInfo(&free, &total));
291  if (initial_size == default_initial_size) {
292  initial_size = align_up(std::min(free, total / 2));
293  }
294  if (maximum_size_ == default_maximum_size) {
295  maximum_size_ = align_down(free) - reserved_size;
296  }
297  }
298  RMM_EXPECTS(initial_size <= maximum_size_, "Initial arena size exceeds the maximum pool size!");
299 
300  free_blocks_.emplace(expand_arena(initial_size));
301  }

50 uniform_nbr_sample_impl(
51  raft::handle_t const& handle,
52  graph_view_t const& graph_view,
53  rmm::device_uvector<typename graph_view_t::vertex_type>& d_in,
54  raft::host_span<const int> h_fan_out,
55  rmm::device_uvector<typename graph_view_t::edge_type> const& global_out_degrees,
56  rmm::device_uvector<typename graph_view_t::edge_type> const& global_degree_offsets,
57  bool with_replacement,
58  uint64_t seed)
59 {
60  using vertex_t = typename graph_view_t::vertex_type;
61  using edge_t = typename graph_view_t::edge_type;
62  using weight_t = typename graph_view_t::weight_type;
63 
64 #ifdef NO_CUGRAPH_OPS
65  CUGRAPH_FAIL(
66  "uniform_nbr_sampl_impl not supported in this configuration, built with NO_CUGRAPH_OPS");
67 #else
68 
69  namespace cugraph_ops = cugraph::ops::gnn::graph;
70 
71  CUGRAPH_EXPECTS(h_fan_out.size() > 0,
72  "Invalid input argument: number of levels must be non-zero.");
73 
74  rmm::device_uvector<vertex_t> d_result_src(0, handle.get_stream());
75  rmm::device_uvector<vertex_t> d_result_dst(0, handle.get_stream());
76  auto d_result_indices =
77  thrust::make_optional(rmm::device_uvector<weight_t>(0, handle.get_stream()));
78 
79  size_t level{0};
80  size_t row_comm_size{1};
81 
82  if constexpr (graph_view_t::is_multi_gpu) {
83  auto& row_comm = handle.get_subcomm(cugraph::partition_2d::key_naming_t().row_name());
84  seed += row_comm.get_rank();
85  row_comm_size = row_comm.get_size();
86  }
87 
88  for (auto&& k_level : h_fan_out) {
89  // prep step for extracting out-degs(sources):
90  if constexpr (graph_view_t::is_multi_gpu) {
91  d_in = shuffle_int_vertices_by_gpu_id(
92  handle, std::move(d_in), graph_view.vertex_partition_range_lasts());
93  d_in = allgather_active_majors(handle, std::move(d_in));
94  }
95 
96  rmm::device_uvector<vertex_t> d_out_src(0, handle.get_stream());
97  rmm::device_uvector<vertex_t> d_out_dst(0, handle.get_stream());
98  auto d_out_indices = std::make_optional(rmm::device_uvector<weight_t>(0, handle.get_stream()));
99 
100  if (k_level > 0) {
101  // extract out-degs(sources):
102  auto&& d_out_degs =
103  get_active_major_global_degrees(handle, graph_view, d_in, global_out_degrees);
104 
105  // eliminate 0 degree vertices
106  std::tie(d_in, d_out_degs) =
107  cugraph::detail::filter_degree_0_vertices(handle, std::move(d_in), std::move(d_out_degs));
108 
109  // segmented-random-generation of indices:
110  rmm::device_uvector<edge_t> d_rnd_indices(d_in.size() * k_level, handle.get_stream());
111 
112  raft::random::RngState rng_state(seed);
113  seed += d_rnd_indices.size() * row_comm_size;
114 
115  if (d_rnd_indices.size() > 0) {
116  // FIXME: This cugraph_ops function does not handle 0 inputs properly
117  cugraph_ops::get_sampling_index(d_rnd_indices.data(),
118  rng_state,
119  d_out_degs.data(),
120  static_cast<edge_t>(d_out_degs.size()),
121  static_cast<int32_t>(k_level),
122  with_replacement,
123  handle.get_stream());
124  }
125 
126  std::tie(d_out_src, d_out_dst, d_out_indices) =
127  gather_local_edges(handle,
128  graph_view,
129  d_in,
130  std::move(d_rnd_indices),
131  static_cast<edge_t>(k_level),
132  global_degree_offsets);
133  } else {
134  std::tie(d_out_src, d_out_dst, d_out_indices) =
135  gather_one_hop_edgelist(handle, graph_view, d_in);
136  }
137 
138  // resize accumulators:
139  auto old_sz = d_result_dst.size();
140  auto add_sz = d_out_dst.size();
141  auto new_sz = old_sz + add_sz;
142 
143  d_result_src.resize(new_sz, handle.get_stream());
144  d_result_dst.resize(new_sz, handle.get_stream());
145  d_result_indices->resize(new_sz, handle.get_stream());
146 
147  raft::copy(
148  d_result_src.begin() + old_sz, d_out_src.begin(), d_out_src.size(), handle.get_stream());
149  raft::copy(
150  d_result_dst.begin() + old_sz, d_out_dst.begin(), d_out_dst.size(), handle.get_stream());
151  raft::copy(d_result_indices->begin() + old_sz,
152  d_out_indices->begin(),
153  d_out_indices->size(),
154  handle.get_stream());
155 
156  d_in = std::move(d_out_dst);
157 
158  ++level;
159  }
160 
161  return count_and_remove_duplicates<vertex_t, edge_t, weight_t>(
162  handle, std::move(d_result_src), std::move(d_result_dst), std::move(*d_result_indices));
163 #endif
164 }

35 template <typename vertex_t, typename edge_t, typename weight_t>
36 void exact_fa2(raft::handle_t const &handle,
37  GraphCOOView<vertex_t, edge_t, weight_t> &graph,
38  float *pos,
39  const int max_iter = 500,
40  float *x_start = nullptr,
41  float *y_start = nullptr,
42  bool outbound_attraction_distribution = true,
43  bool lin_log_mode = false,
44  bool prevent_overlapping = false,
45  const float edge_weight_influence = 1.0,
46  const float jitter_tolerance = 1.0,
47  const float scaling_ratio = 2.0,
48  bool strong_gravity_mode = false,
49  const float gravity = 1.0,
50  bool verbose = false,
51  internals::GraphBasedDimRedCallback *callback = nullptr)
52 {
53  rmm::cuda_stream_view stream(handle.get_stream());
54  const edge_t e = graph.number_of_edges;
55  const vertex_t n = graph.number_of_vertices;
56 
57  float *d_repel{nullptr};
58  float *d_attract{nullptr};
59  float *d_old_forces{nullptr};
60  int *d_mass{nullptr};
61  float *d_swinging{nullptr};
62  float *d_traction{nullptr};
63 
64  rmm::device_uvector<float> repel(n * 2, stream);
65  rmm::device_uvector<float> attract(n * 2, stream);
66  rmm::device_uvector<float> old_forces(n * 2, stream);
67  thrust::fill(rmm::exec_policy(stream), old_forces.begin(), old_forces.end(), 0.f);
68  // FA2 requires degree + 1.
69  rmm::device_uvector<int> mass(n, stream);
70  thrust::fill(rmm::exec_policy(stream), mass.begin(), mass.end(), 1);
71  rmm::device_uvector<float> swinging(n, stream);
72  rmm::device_uvector<float> traction(n, stream);
73 
74  d_repel = repel.data();
75  d_attract = attract.data();
76  d_old_forces = old_forces.data();
77  d_mass = mass.data();
78  d_swinging = swinging.data();
79  d_traction = traction.data();
80 
81  int random_state = 0;
82  random_vector(pos, n * 2, random_state, stream.value());
83 
84  if (x_start && y_start) {
85  raft::copy(pos, x_start, n, stream.value());
86  raft::copy(pos + n, y_start, n, stream.value());
87  }
88 
89  // Sort COO for coalesced memory access.
90  sort(graph, stream.value());
91  CHECK_CUDA(stream.value());
92 
93  graph.degree(d_mass, cugraph::DegreeDirection::OUT);
94  CHECK_CUDA(stream.value());
95 
96  const vertex_t *row = graph.src_indices;
97  const vertex_t *col = graph.dst_indices;
98  const weight_t *v = graph.edge_data;
99 
100  float speed = 1.f;
101  float speed_efficiency = 1.f;
102  float outbound_att_compensation = 1.f;
103  float jt = 0.f;
104 
105  if (outbound_attraction_distribution) {
106  int sum = thrust::reduce(rmm::exec_policy(stream), mass.begin(), mass.end());
107  outbound_att_compensation = sum / (float)n;
108  }
109 
110  if (callback) {
111  callback->setup<float>(n, 2);
112  callback->on_preprocess_end(pos);
113  }
114 
115  for (int iter = 0; iter < max_iter; ++iter) {
116  // Reset force arrays
117  thrust::fill(rmm::exec_policy(stream), repel.begin(), repel.end(), 0.f);
118  thrust::fill(rmm::exec_policy(stream), attract.begin(), attract.end(), 0.f);
119  thrust::fill(rmm::exec_policy(stream), swinging.begin(), swinging.end(), 0.f);
120  thrust::fill(rmm::exec_policy(stream), traction.begin(), traction.end(), 0.f);
121 
122  // Exact repulsion
123  apply_repulsion<vertex_t>(
124  pos, pos + n, d_repel, d_repel + n, d_mass, scaling_ratio, n, stream.value());
125 
126  apply_gravity<vertex_t>(pos,
127  pos + n,
128  d_attract,
129  d_attract + n,
130  d_mass,
131  gravity,
132  strong_gravity_mode,
133  scaling_ratio,
134  n,
135  stream.value());
136 
137  apply_attraction<vertex_t, edge_t, weight_t>(row,
138  col,
139  v,
140  e,
141  pos,
142  pos + n,
143  d_attract,
144  d_attract + n,
145  d_mass,
146  outbound_attraction_distribution,
147  lin_log_mode,
148  edge_weight_influence,
149  outbound_att_compensation,
150  stream.value());
151 
152  compute_local_speed(d_repel,
153  d_repel + n,
154  d_attract,
155  d_attract + n,
156  d_old_forces,
157  d_old_forces + n,
158  d_mass,
159  d_swinging,
160  d_traction,
161  n,
162  stream.value());
163 
164  // Compute global swinging and traction values.
165  const float s = thrust::reduce(rmm::exec_policy(stream), swinging.begin(), swinging.end());
166  const float t = thrust::reduce(rmm::exec_policy(stream), traction.begin(), traction.end());
167 
168  adapt_speed<vertex_t>(jitter_tolerance, &jt, &speed, &speed_efficiency, s, t, n);
169 
170  apply_forces<vertex_t>(pos,
171  pos + n,
172  d_repel,
173  d_repel + n,
174  d_attract,
175  d_attract + n,
176  d_old_forces,
177  d_old_forces + n,
178  d_swinging,
179  speed,
180  n,
181  stream.value());
182 
183  if (callback) callback->on_epoch_end(pos);
184 
185  if (verbose) {
186  std::cout << "iteration: " << iter + 1 << ", speed: " << speed
187  << ", speed_efficiency: " << speed_efficiency << ", jt: " << jt
188  << ", swinging: " << s << ", traction: " << t << "\n";
189  }
190  }
191 
192  if (callback) callback->on_train_end(pos);
193 }

404  VT *dst_indices(void) { return static_cast<VT *>(dst_indices_.data()); }
405  WT *edge_data(void) { return static_cast<WT *>(edge_data_.data()); }
406 
407  GraphCOOContents<VT, ET, WT> release() noexcept
408  {
409  VT number_of_vertices = number_of_vertices_;
410  ET number_of_edges = number_of_edges_;
411  number_of_vertices_ = 0;

413  return GraphCOOContents<VT, ET, WT>{
414  number_of_vertices,
415  number_of_edges,
416  std::make_unique<rmm::device_buffer>(std::move(src_indices_)),
417  std::make_unique<rmm::device_buffer>(std::move(dst_indices_)),
418  std::make_unique<rmm::device_buffer>(std::move(edge_data_))};

420

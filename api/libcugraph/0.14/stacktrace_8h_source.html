40 static inline void printStackTrace(std::ostream &eout = std::cerr, unsigned int max_frames = 63)
41 {
42 #if defined(_WIN32) || defined(__ANDROID__) || defined(ANDROID) || defined(__QNX__) || \
43  defined(__QNXNTO__)
44  // TODO add code for windows stack trace and android stack trace
45 #else
46  std::stringstream out;
47 
48  // storage array for stack trace address data
49  void *addrlist[max_frames + 1];
50 
51  // retrieve current stack addresses
52  int addrlen = backtrace(addrlist, sizeof(addrlist) / sizeof(void *));
53  if (addrlen == 0) {
54  out << " <empty, possibly corrupt>\n";
55  return;
56  }
57 
58  // resolve addresses into strings containing "filename(function+address)",
59  // this array must be free()-ed
60  std::unique_ptr<char *, decltype(&::free)> symbollist(backtrace_symbols(addrlist, addrlen),
61  &::free);
62  // char** symbollist = backtrace_symbols(addrlist, addrlen);
63 
64  // allocate string which will be filled with the demangled function name
65  size_t funcnamesize = 256;
66  std::vector<char> funcname_v(funcnamesize);
67  char *funcname = funcname_v.data();
68 
69  // iterate over the returned symbol lines. skip the first, it is the
70  // address of this function.
71  for (int i = 1; i < addrlen; i++) {
72  char *begin_name = 0, *begin_offset = 0, *end_offset = 0;
73 
74  // find parentheses and +address offset surrounding the mangled name:
75  // ./module(function+0x15c) [0x8048a6d]
76  for (char *p = symbollist.get()[i]; *p; ++p) {
77  if (*p == '(')
78  begin_name = p;
79  else if (*p == '+')
80  begin_offset = p;
81  else if (*p == ')' && begin_offset) {
82  end_offset = p;
83  break;
84  }
85  }
86 
87  if (begin_name && begin_offset && end_offset && begin_name < begin_offset) {
88  *begin_name++ = '\0';
89  *begin_offset++ = '\0';
90  *end_offset = '\0';
91 
92  // mangled name is now in [begin_name, begin_offset) and caller
93  // offset in [begin_offset, end_offset). now apply
94  // __cxa_demangle():
95 
96  int status;
97  char *ret = abi::__cxa_demangle(begin_name, funcname, &funcnamesize, &status);
98  if (status == 0) {
99  funcname = ret; // use possibly realloc()-ed string
100  out << " " << symbollist.get()[i] << " : " << funcname << "+" << begin_offset << "\n";
101  } else {
102  // demangling failed. Output function name as a C function with
103  // no arguments.
104  out << " " << symbollist.get()[i] << " : " << begin_name << "()+" << begin_offset << "\n";
105  }
106  } else {
107  // couldn't parse the line? print the whole line.
108  out << " " << symbollist.get()[i] << "\n";
109  }
110  }
111  eout << out.str();

115  // while(1);
116 #endif
117 }

342  : number_of_vertices_(graph.number_of_vertices),
343  number_of_edges_(graph.number_of_edges),
344  src_indices_(graph.src_indices, graph.number_of_edges * sizeof(VT), stream, mr),
345  dst_indices_(graph.dst_indices, graph.number_of_edges * sizeof(VT), stream, mr)

355  VT *src_indices(void) { return static_cast<VT *>(src_indices_.data()); }
356  VT *dst_indices(void) { return static_cast<VT *>(dst_indices_.data()); }
357  WT *edge_data(void) { return static_cast<WT *>(edge_data_.data()); }
358 
359  GraphCOOContents<VT, ET, WT> release() noexcept
360  {
361  VT number_of_vertices = number_of_vertices_;
362  ET number_of_edges = number_of_edges_;
363  number_of_vertices_ = 0;
364  number_of_edges_ = 0;
365  return GraphCOOContents<VT, ET, WT>{
366  number_of_vertices,
367  number_of_edges,
368  std::make_unique<rmm::device_buffer>(std::move(src_indices_)),
369  std::make_unique<rmm::device_buffer>(std::move(dst_indices_)),
370  std::make_unique<rmm::device_buffer>(std::move(edge_data_))};
371  }
372 
373  GraphCOOView<VT, ET, WT> view(void) noexcept
374  {
375  return GraphCOOView<VT, ET, WT>(
376  src_indices(), dst_indices(), edge_data(), number_of_vertices_, number_of_edges_);

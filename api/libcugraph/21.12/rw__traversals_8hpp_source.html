108 struct uniform_selector_t {
109  using vertex_t = typename graph_type::vertex_type;
110  using edge_t = typename graph_type::edge_type;

112 
113  struct sampler_t {
114  sampler_t(edge_t const* ro,
115  vertex_t const* ci,
116  weight_t const* w,
117  edge_t const* ptr_d_cache_out_degs)
118  : row_offsets_(ro), col_indices_(ci), values_(w), ptr_d_cache_out_degs_(ptr_d_cache_out_degs)
119  {
120  }
121 
122  __device__ thrust::optional<thrust::tuple<vertex_t, weight_t>> operator()(vertex_t src_v,
123  real_t rnd_val) const
124  {
125  auto crt_out_deg = ptr_d_cache_out_degs_[src_v];
126  if (crt_out_deg == 0) return thrust::nullopt; // src_v is a sink
127 
128  real_t max_ub = static_cast<real_t>(crt_out_deg - 1);
129  auto interp_vindx = rnd_val * max_ub;
130  vertex_t v_indx = static_cast<vertex_t>(interp_vindx);
131 
132  auto col_indx = v_indx >= crt_out_deg ? crt_out_deg - 1 : v_indx;
133  auto start_row = row_offsets_[src_v];
134 
135  auto weight_value =
136  (values_ == nullptr ? weight_t{1}
137  : values_[start_row + col_indx]); // account for un-weighted graphs
138  return thrust::optional{thrust::make_tuple(col_indices_[start_row + col_indx], weight_value)};
139  }
140 
141  private:
142  edge_t const* row_offsets_;
143  vertex_t const* col_indices_;
144  weight_t const* values_;
145 
146  edge_t const* ptr_d_cache_out_degs_;
147  };

268 struct biased_selector_t {
269  using vertex_t = typename graph_type::vertex_type;
270  using edge_t = typename graph_type::edge_type;
271  using weight_t = typename graph_type::weight_type;
272 
273  struct sampler_t {
274  sampler_t(edge_t const* ro,
275  vertex_t const* ci,
276  weight_t const* w,
277  weight_t const* ptr_d_sum_weights)
278  : row_offsets_(ro), col_indices_(ci), values_(w), ptr_d_sum_weights_(ptr_d_sum_weights)
279  {
280  }
281 
282  // pre-condition:
283  //
284  // Sum(weights(neighborhood(src_v))) are pre-computed and
285  // stored in ptr_d_sum_weights_ (too expensive to check, here);
286  //
287  __device__ thrust::optional<thrust::tuple<vertex_t, weight_t>> operator()(vertex_t src_v,
288  real_t rnd_val) const
289  {
290  weight_t run_sum_w{0};
291  auto rnd_sum_weights = rnd_val * ptr_d_sum_weights_[src_v];
292 
293  auto col_indx_begin = row_offsets_[src_v];
294  auto col_indx_end = row_offsets_[src_v + 1];
295  if (col_indx_begin == col_indx_end) return thrust::nullopt; // src_v is a sink
296 
297  auto col_indx = col_indx_begin;
298  auto prev_col_indx = col_indx;
299 
300  for (; col_indx < col_indx_end; ++col_indx) {
301  if (rnd_sum_weights < run_sum_w) break;
302 
303  run_sum_w += values_[col_indx];
304  prev_col_indx = col_indx;
305  }
306  return thrust::optional{
307  thrust::make_tuple(col_indices_[prev_col_indx], values_[prev_col_indx])};
308  }
309 
310  private:
311  edge_t const* row_offsets_;
312  vertex_t const* col_indices_;
313  weight_t const* values_;
314 
315  weight_t const* ptr_d_sum_weights_;
316  };

356  template <typename graph_t,
357  typename random_walker_t,
358  typename selector_t,
359  typename index_t,
360  typename real_t,
361  typename seed_t>
362  void operator()(
363  graph_t const& graph, // graph being traversed
364  random_walker_t const& rand_walker, // random walker object for which traversal is driven
365  selector_t const& selector, // sampling type (uniform, biased, etc.)
366  seed_t seed0, // initial seed value
367  device_vec_t<typename graph_t::vertex_type>& d_coalesced_v, // crt coalesced vertex set
368  device_vec_t<typename graph_t::weight_type>& d_coalesced_w, // crt coalesced weight set
369  device_vec_t<index_t>& d_paths_sz, // crt paths sizes
370  device_vec_t<typename graph_t::edge_type>&
371  d_crt_out_degs, // crt out-degs for current set of vertices
372  device_vec_t<real_t>& d_random, // crt set of random real values
373  device_vec_t<typename graph_t::vertex_type>&
374  d_col_indx) // crt col col indices to be used for retrieving next step
375  const
376  {
377  auto const& handle = rand_walker.get_handle();
378 
379  // start from 1, as 0-th was initialized above:
380  //
381  for (decltype(max_depth_) step_indx = 1; step_indx < max_depth_; ++step_indx) {
382  // take one-step in-sync for each path in parallel:
383  //
384  rand_walker.step(graph,
385  selector,
386  seed0 + static_cast<seed_t>(step_indx),
387  d_coalesced_v,
388  d_coalesced_w,
389  d_paths_sz,
390  d_crt_out_degs,
391  d_random,
392  d_col_indx);
393 
394  // early exit: all paths have reached sinks:
395  //
396  if (rand_walker.all_paths_stopped(d_crt_out_degs)) break;
397  }
398  }
399 
400  size_t get_random_buff_sz(void) const { return num_paths_; }
401  size_t get_tmp_buff_sz(void) const { return num_paths_; }
402 
403  private:
404  size_t num_paths_;
405  size_t max_depth_;
406 };

421  template <typename graph_t,
422  typename random_walker_t,
423  typename selector_t,
424  typename index_t,
425  typename real_t,
426  typename seed_t>
427  void operator()(
428  graph_t const& graph, // graph being traversed
429  random_walker_t const& rand_walker, // random walker object for which traversal is driven
430  selector_t const& selector, // sampling type (uniform, biased, etc.)
431  seed_t seed0, // initial seed value
432  device_vec_t<typename graph_t::vertex_type>& d_coalesced_v, // crt coalesced vertex set
433  device_vec_t<typename graph_t::weight_type>& d_coalesced_w, // crt coalesced weight set
434  device_vec_t<index_t>& d_paths_sz, // crt paths sizes
435  device_vec_t<typename graph_t::edge_type>&
436  d_crt_out_degs, // ignored: out-degs for the current set of vertices
437  device_vec_t<real_t>& d_random, // _entire_ set of random real values
438  device_vec_t<typename graph_t::vertex_type>&
439  d_col_indx) // ignored: crt col indices to be used for retrieving next step
440  // (Note: coalesced set updated on-the-go)
441  const
442  {
443  using vertex_t = typename graph_t::vertex_type;
444  using edge_t = typename graph_t::edge_type;
445  using weight_t = typename graph_t::weight_type;
446  using random_engine_t = typename random_walker_t::rnd_engine_t;
447  using sampler_t = typename selector_t::sampler_type;
448 
449  auto const& handle = rand_walker.get_handle();
450  auto* ptr_d_random = raw_ptr(d_random);
451 
452  random_engine_t::generate_random(handle, ptr_d_random, d_random.size(), seed0);
453 
454  auto* ptr_d_sizes = raw_ptr(d_paths_sz);
455 
456  // next step sampler functor:
457  //
458  sampler_t const& sampler = selector.get_strategy();
459 
460  // start from 1, as 0-th was initialized above:
461  //
462  thrust::for_each(handle.get_thrust_policy(),
463  thrust::make_counting_iterator<index_t>(0),
464  thrust::make_counting_iterator<index_t>(num_paths_),
465  [max_depth = max_depth_,
466  ptr_coalesced_v = raw_ptr(d_coalesced_v),
467  ptr_coalesced_w = raw_ptr(d_coalesced_w),
468  ptr_d_random,
469  ptr_d_sizes,
470  sampler] __device__(auto path_index) {
471  auto chunk_offset = path_index * max_depth;
472  vertex_t src_vertex = ptr_coalesced_v[chunk_offset];
473 
474  for (index_t step_indx = 1; step_indx < max_depth; ++step_indx) {
475  // indexing into coalesced arrays of size num_paths x (max_depth -1):
476  // (d_random, d_coalesced_w)
477  //
478  auto stepping_index = chunk_offset - path_index + step_indx - 1;
479 
480  auto real_rnd_indx = ptr_d_random[stepping_index];
481 
482  auto opt_tpl_vn_wn = sampler(src_vertex, real_rnd_indx);
483  if (!opt_tpl_vn_wn.has_value()) break;
484 
485  src_vertex = thrust::get<0>(*opt_tpl_vn_wn);
486  auto crt_weight = thrust::get<1>(*opt_tpl_vn_wn);
487 
488  ptr_coalesced_v[chunk_offset + step_indx] = src_vertex;
489  ptr_coalesced_w[stepping_index] = crt_weight;
490  ptr_d_sizes[path_index]++;
491  }
492  });
493  }

496  size_t get_tmp_buff_sz(void) const
497  {
498  return 0;
499  } // no need for tmp buffers

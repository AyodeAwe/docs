74  partition_t(std::vector<vertex_t> const& vertex_partition_offsets,
75  int row_comm_size,
76  int col_comm_size,
77  int row_comm_rank,
78  int col_comm_rank)
79  : vertex_partition_offsets_(vertex_partition_offsets),
80  comm_rank_(col_comm_rank * row_comm_size + row_comm_rank),
81  row_comm_size_(row_comm_size),
82  col_comm_size_(col_comm_size),
83  row_comm_rank_(row_comm_rank),
84  col_comm_rank_(col_comm_rank)
85  {
86  CUGRAPH_EXPECTS(
87  vertex_partition_offsets.size() == static_cast<size_t>(row_comm_size * col_comm_size + 1),
88  "Invalid API parameter: erroneous vertex_partition_offsets.size().");
89 
90  CUGRAPH_EXPECTS(
91  std::is_sorted(vertex_partition_offsets_.begin(), vertex_partition_offsets_.end()),
92  "Invalid API parameter: partition.vertex_partition_offsets values should be non-descending.");
93  CUGRAPH_EXPECTS(vertex_partition_offsets_[0] == vertex_t{0},
94  "Invalid API parameter: partition.vertex_partition_offsets[0] should be 0.");
95 
96  vertex_t start_offset{0};
97  matrix_partition_major_value_start_offsets_.assign(get_number_of_matrix_partitions(), 0);
98  for (size_t i = 0; i < matrix_partition_major_value_start_offsets_.size(); ++i) {
99  matrix_partition_major_value_start_offsets_[i] = start_offset;
100  start_offset += get_matrix_partition_major_last(i) - get_matrix_partition_major_first(i);
101  }
102  }

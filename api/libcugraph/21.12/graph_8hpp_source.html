154  std::vector<edge_t const*> offsets(adj_matrix_partition_offsets_.size(), nullptr);
155  std::vector<vertex_t const*> indices(adj_matrix_partition_indices_.size(), nullptr);
156  auto weights = adj_matrix_partition_weights_
157  ? std::make_optional<std::vector<weight_t const*>>(
158  (*adj_matrix_partition_weights_).size(), nullptr)
159  : std::nullopt;
160  auto dcs_nzd_vertices = adj_matrix_partition_dcs_nzd_vertices_
161  ? std::make_optional<std::vector<vertex_t const*>>(
162  (*adj_matrix_partition_dcs_nzd_vertices_).size(), nullptr)
163  : std::nullopt;
164  auto dcs_nzd_vertex_counts =
165  adj_matrix_partition_dcs_nzd_vertex_counts_
166  ? std::make_optional<std::vector<vertex_t>>(
167  (*adj_matrix_partition_dcs_nzd_vertex_counts_).size(), vertex_t{0})
168  : std::nullopt;
169  for (size_t i = 0; i < offsets.size(); ++i) {
170  offsets[i] = adj_matrix_partition_offsets_[i].data();
171  indices[i] = adj_matrix_partition_indices_[i].data();
172  if (weights) { (*weights)[i] = (*adj_matrix_partition_weights_)[i].data(); }
173  if (dcs_nzd_vertices) {
174  (*dcs_nzd_vertices)[i] = (*adj_matrix_partition_dcs_nzd_vertices_)[i].data();
175  (*dcs_nzd_vertex_counts)[i] = (*adj_matrix_partition_dcs_nzd_vertex_counts_)[i];
176  }
177  }
178 
179  return graph_view_t<vertex_t, edge_t, weight_t, store_transposed, multi_gpu>(
180  *(this->get_handle_ptr()),
181  offsets,
182  indices,
183  weights,
184  dcs_nzd_vertices,
185  dcs_nzd_vertex_counts,
186  graph_view_meta_t<vertex_t, edge_t, multi_gpu>{
187  this->get_number_of_vertices(),
188  this->get_number_of_edges(),
189  this->get_graph_properties(),
190  partition_,
191  adj_matrix_partition_segment_offsets_,
192  local_sorted_unique_edge_rows_
193  ? std::optional<vertex_t const*>{(*local_sorted_unique_edge_rows_).data()}
194  : std::nullopt,
195  local_sorted_unique_edge_rows_

38 template <typename vertex_t, typename edge_t, typename weight_t>
39 void barnes_hut(raft::handle_t const &handle,
40  GraphCOOView<vertex_t, edge_t, weight_t> &graph,
41  float *pos,
42  const int max_iter = 500,
43  float *x_start = nullptr,
44  float *y_start = nullptr,
45  bool outbound_attraction_distribution = true,
46  bool lin_log_mode = false,
47  bool prevent_overlapping = false,
48  const float edge_weight_influence = 1.0,
49  const float jitter_tolerance = 1.0,
50  const float theta = 0.5,
51  const float scaling_ratio = 2.0,
52  bool strong_gravity_mode = false,
53  const float gravity = 1.0,
54  bool verbose = false,
55  internals::GraphBasedDimRedCallback *callback = nullptr)
56 {
57  rmm::cuda_stream_view stream_view(handle.get_stream_view());
58  const edge_t e = graph.number_of_edges;
59  const vertex_t n = graph.number_of_vertices;
60 
61  const int blocks = getMultiProcessorCount();
62  // A tiny jitter to promote numerical stability/
63  const float epssq = 0.0025;
64  // We use the same array for nodes and cells.
65  int nnodes = n * 2;
66  if (nnodes < 1024 * blocks) nnodes = 1024 * blocks;
67  while ((nnodes & (32 - 1)) != 0) nnodes++;
68  nnodes--;
69 
70  // Allocate more space
71  //---------------------------------------------------
72  rmm::device_uvector<unsigned> d_limiter(1, stream_view);
73  rmm::device_uvector<int> d_maxdepthd(1, stream_view);
74  rmm::device_uvector<int> d_bottomd(1, stream_view);
75  rmm::device_uvector<float> d_radiusd(1, stream_view);
76 
77  unsigned *limiter = d_limiter.data();
78  int *maxdepthd = d_maxdepthd.data();
79  int *bottomd = d_bottomd.data();
80  float *radiusd = d_radiusd.data();
81 
82  InitializationKernel<<<1, 1, 0, stream_view.value()>>>(limiter, maxdepthd, radiusd);
83  CHECK_CUDA(stream_view.value());
84 
85  const int FOUR_NNODES = 4 * nnodes;
86  const int FOUR_N = 4 * n;
87  const float theta_squared = theta * theta;
88  const int NNODES = nnodes;
89 
90  rmm::device_uvector<int> d_startl(nnodes + 1, stream_view);
91  rmm::device_uvector<int> d_childl((nnodes + 1) * 4, stream_view);
92  // FA2 requires degree + 1
93  rmm::device_uvector<int> d_massl(nnodes + 1, stream_view);
94  thrust::fill(rmm::exec_policy(stream_view), d_massl.begin(), d_massl.end(), 1);
95 
96  rmm::device_uvector<float> d_maxxl(blocks * FACTOR1, stream_view);
97  rmm::device_uvector<float> d_maxyl(blocks * FACTOR1, stream_view);
98  rmm::device_uvector<float> d_minxl(blocks * FACTOR1, stream_view);
99  rmm::device_uvector<float> d_minyl(blocks * FACTOR1, stream_view);
100 
101  // Actual mallocs
102  int *startl = d_startl.data();
103  int *childl = d_childl.data();
104  int *massl = d_massl.data();
105 
106  float *maxxl = d_maxxl.data();
107  float *maxyl = d_maxyl.data();
108  float *minxl = d_minxl.data();
109  float *minyl = d_minyl.data();
110 
111  // SummarizationKernel
112  rmm::device_uvector<int> d_countl(nnodes + 1, stream_view);
113  int *countl = d_countl.data();
114 
115  // SortKernel
116  rmm::device_uvector<int> d_sortl(nnodes + 1, stream_view);
117  int *sortl = d_sortl.data();
118 
119  // RepulsionKernel
120  rmm::device_uvector<float> d_rep_forces((nnodes + 1) * 2, stream_view);
121  float *rep_forces = d_rep_forces.data();
122 
123  rmm::device_uvector<float> d_radius_squared(1, stream_view);
124  float *radiusd_squared = d_radius_squared.data();
125 
126  rmm::device_uvector<float> d_nodes_pos((nnodes + 1) * 2, stream_view);
127  float *nodes_pos = d_nodes_pos.data();
128 
129  // Initialize positions with random values
130  int random_state = 0;
131 
132  // Copy start x and y positions.
133  if (x_start && y_start) {
134  raft::copy(nodes_pos, x_start, n, stream_view.value());
135  raft::copy(nodes_pos + nnodes + 1, y_start, n, stream_view.value());
136  } else {
137  raft::random::Rng rng(random_state);
138  rng.uniform<float, size_t>(nodes_pos, (nnodes + 1) * 2, -100.0f, 100.0f, stream_view.value());
139  }
140 
141  // Allocate arrays for force computation
142  float *attract{nullptr};
143  float *old_forces{nullptr};
144  float *swinging{nullptr};
145  float *traction{nullptr};
146 
147  rmm::device_uvector<float> d_attract(n * 2, stream_view);
148  rmm::device_uvector<float> d_old_forces(n * 2, stream_view);
149  rmm::device_uvector<float> d_swinging(n, stream_view);
150  rmm::device_uvector<float> d_traction(n, stream_view);
151 
152  attract = d_attract.data();
153  old_forces = d_old_forces.data();
154  swinging = d_swinging.data();
155  traction = d_traction.data();
156 
157  thrust::fill(rmm::exec_policy(stream_view), d_old_forces.begin(), d_old_forces.end(), 0.f);
158 
159  // Sort COO for coalesced memory access.
160  sort(graph, stream_view.value());
161  CHECK_CUDA(stream_view.value());
162 
163  graph.degree(massl, cugraph::DegreeDirection::OUT);
164  CHECK_CUDA(stream_view.value());
165 
166  const vertex_t *row = graph.src_indices;
167  const vertex_t *col = graph.dst_indices;
168  const weight_t *v = graph.edge_data;
169 
170  // Scalars used to adapt global speed.
171  float speed = 1.f;
172  float speed_efficiency = 1.f;
173  float outbound_att_compensation = 1.f;
174  float jt = 0.f;
175 
176  // If outboundAttractionDistribution active, compensate.
177  if (outbound_attraction_distribution) {
178  int sum = thrust::reduce(rmm::exec_policy(stream_view), d_massl.begin(), d_massl.begin() + n);
179  outbound_att_compensation = sum / (float)n;
180  }
181 
182  //
183  // Set cache levels for faster algorithm execution
184  //---------------------------------------------------
185  cudaFuncSetCacheConfig(BoundingBoxKernel, cudaFuncCachePreferShared);
186  cudaFuncSetCacheConfig(TreeBuildingKernel, cudaFuncCachePreferL1);
187  cudaFuncSetCacheConfig(ClearKernel1, cudaFuncCachePreferL1);
188  cudaFuncSetCacheConfig(ClearKernel2, cudaFuncCachePreferL1);
189  cudaFuncSetCacheConfig(SummarizationKernel, cudaFuncCachePreferShared);
190  cudaFuncSetCacheConfig(SortKernel, cudaFuncCachePreferL1);
191  cudaFuncSetCacheConfig(RepulsionKernel, cudaFuncCachePreferL1);
192  cudaFuncSetCacheConfig(apply_forces_bh, cudaFuncCachePreferL1);
193 
194  if (callback) {
195  callback->setup<float>(nnodes + 1, 2);
196  callback->on_preprocess_end(nodes_pos);
197  }
198 
199  for (int iter = 0; iter < max_iter; ++iter) {
200  // Reset force values
201  thrust::fill(rmm::exec_policy(stream_view), d_rep_forces.begin(), d_rep_forces.end(), 0.f);
202  thrust::fill(rmm::exec_policy(stream_view), d_attract.begin(), d_attract.end(), 0.f);
203  thrust::fill(rmm::exec_policy(stream_view), d_swinging.begin(), d_swinging.end(), 0.f);
204  thrust::fill(rmm::exec_policy(stream_view), d_traction.begin(), d_traction.end(), 0.f);
205 
206  ResetKernel<<<1, 1, 0, stream_view.value()>>>(radiusd_squared, bottomd, NNODES, radiusd);
207  CHECK_CUDA(stream_view.value());
208 
209  // Compute bounding box arround all bodies
210  BoundingBoxKernel<<<blocks * FACTOR1, THREADS1, 0, stream_view.value()>>>(
211  startl,
212  childl,
213  massl,
214  nodes_pos,
215  nodes_pos + nnodes + 1,
216  maxxl,
217  maxyl,
218  minxl,
219  minyl,
220  FOUR_NNODES,
221  NNODES,
222  n,
223  limiter,
224  radiusd);
225  CHECK_CUDA(stream_view.value());
226 
227  ClearKernel1<<<blocks, 1024, 0, stream_view.value()>>>(childl, FOUR_NNODES, FOUR_N);
228  CHECK_CUDA(stream_view.value());
229 
230  // Build quadtree
231  TreeBuildingKernel<<<blocks * FACTOR2, THREADS2, 0, stream_view.value()>>>(
232  childl, nodes_pos, nodes_pos + nnodes + 1, NNODES, n, maxdepthd, bottomd, radiusd);
233  CHECK_CUDA(stream_view.value());
234 
235  ClearKernel2<<<blocks, 1024, 0, stream_view.value()>>>(startl, massl, NNODES, bottomd);
236  CHECK_CUDA(stream_view.value());
237 
238  // Summarizes mass and position for each cell, bottom up approach
239  SummarizationKernel<<<blocks * FACTOR3, THREADS3, 0, stream_view.value()>>>(
240  countl, childl, massl, nodes_pos, nodes_pos + nnodes + 1, NNODES, n, bottomd);
241  CHECK_CUDA(stream_view.value());
242 
243  // Group closed bodies together, used to speed up Repulsion kernel
244  SortKernel<<<blocks * FACTOR4, THREADS4, 0, stream_view.value()>>>(
245  sortl, countl, startl, childl, NNODES, n, bottomd);
246  CHECK_CUDA(stream_view.value());
247 
248  // Force computation O(n . log(n))
249  RepulsionKernel<<<blocks * FACTOR5, THREADS5, 0, stream_view.value()>>>(scaling_ratio,
250  theta,
251  epssq,
252  sortl,
253  childl,
254  massl,
255  nodes_pos,
256  nodes_pos + nnodes + 1,
257  rep_forces,
258  rep_forces + nnodes + 1,
259  theta_squared,
260  NNODES,
261  FOUR_NNODES,
262  n,
263  radiusd_squared,
264  maxdepthd);
265  CHECK_CUDA(stream_view.value());
266 
267  apply_gravity<vertex_t>(nodes_pos,
268  nodes_pos + nnodes + 1,
269  attract,
270  attract + n,
271  massl,
272  gravity,
273  strong_gravity_mode,
274  scaling_ratio,
275  n,
276  stream_view.value());
277 
278  apply_attraction<vertex_t, edge_t, weight_t>(row,
279  col,
280  v,
281  e,
282  nodes_pos,
283  nodes_pos + nnodes + 1,
284  attract,
285  attract + n,
286  massl,
287  outbound_attraction_distribution,
288  lin_log_mode,
289  edge_weight_influence,
290  outbound_att_compensation,
291  stream_view.value());
292 
293  compute_local_speed(rep_forces,
294  rep_forces + nnodes + 1,
295  attract,
296  attract + n,
297  old_forces,
298  old_forces + n,
299  massl,
300  swinging,
301  traction,
302  n,
303  stream_view.value());
304 
305  // Compute global swinging and traction values
306  const float s =
307  thrust::reduce(rmm::exec_policy(stream_view), d_swinging.begin(), d_swinging.end());
308 
309  const float t =
310  thrust::reduce(rmm::exec_policy(stream_view), d_traction.begin(), d_traction.end());
311 
312  // Compute global speed based on gloab and local swinging and traction.
313  adapt_speed<vertex_t>(jitter_tolerance, &jt, &speed, &speed_efficiency, s, t, n);
314 
315  // Update positions
316  apply_forces_bh<<<blocks * FACTOR6, THREADS6, 0, stream_view.value()>>>(nodes_pos,
317  nodes_pos + nnodes + 1,
318  attract,
319  attract + n,
320  rep_forces,
321  rep_forces + nnodes + 1,
322  old_forces,
323  old_forces + n,
324  swinging,
325  speed,
326  n);
327 
328  if (callback) callback->on_epoch_end(nodes_pos);
329 
330  if (verbose) {
331  std::cout << "iteration: " << iter + 1 << ", speed: " << speed
332  << ", speed_efficiency: " << speed_efficiency << ", jt: " << jt
333  << ", swinging: " << s << ", traction: " << t << "\n";
334  }
335  }
336 
337  // Copy nodes positions into final output pos
338  raft::copy(pos, nodes_pos, n, stream_view.value());
339  raft::copy(pos + n, nodes_pos + nnodes + 1, n, stream_view.value());
340 
341  if (callback) callback->on_train_end(nodes_pos);
342 }

25 template <typename vertex_t, typename edge_t, typename weight_t>
26 __global__ void attraction_kernel(const vertex_t *restrict row,
27  const vertex_t *restrict col,
28  const weight_t *restrict v,
29  const edge_t e,
30  const float *restrict x_pos,
31  const float *restrict y_pos,
32  float *restrict attract_x,
33  float *restrict attract_y,
34  const int *restrict mass,
35  bool outbound_attraction_distribution,
36  bool lin_log_mode,
37  const float edge_weight_influence,
38  const float coef)
39 {
40  vertex_t i, src, dst;
41  weight_t weight = 1;
42  // For every edge
43  for (i = threadIdx.x + blockIdx.x * blockDim.x; i < e; i += gridDim.x * blockDim.x) {
44  src = row[i];
45  dst = col[i];
46 
47  // We only need the lower triangular part
48  if (dst <= src) return;
49 
50  if (v) { weight = v[i]; }
51  weight = pow(weight, edge_weight_influence);
52 
53  float x_dist = x_pos[src] - x_pos[dst];
54  float y_dist = y_pos[src] - y_pos[dst];
55  float factor = -coef * weight;
56 
57  if (lin_log_mode) {
58  float distance = pow(x_dist, 2) + pow(y_dist, 2);
59  distance += FLT_EPSILON;
60  distance = sqrt(distance);
61  factor *= log(1 + distance) / distance;
62  }
63  if (outbound_attraction_distribution) factor /= mass[src];
64 
65  // Force computation
66  atomicAdd(&attract_x[src], x_dist * factor);
67  atomicAdd(&attract_y[src], y_dist * factor);
68  atomicAdd(&attract_x[dst], -x_dist * factor);
69  atomicAdd(&attract_y[dst], -y_dist * factor);
70  }
71 }

118 template <typename vertex_t>
119 __global__ void linear_gravity_kernel(const float *restrict x_pos,
120  const float *restrict y_pos,
121  float *restrict attract_x,
122  float *restrict attract_y,
123  const int *restrict mass,
124  const float gravity,
125  const vertex_t n)
126 {
127  // For every node.
128  for (int i = threadIdx.x + blockIdx.x * blockDim.x; i < n; i += gridDim.x * blockDim.x) {
129  float x_dist = x_pos[i];
130  float y_dist = y_pos[i];
131  float distance = sqrt(x_dist * x_dist + y_dist * y_dist + FLT_EPSILON);
132  float factor = mass[i] * gravity / distance;
133  attract_x[i] -= x_dist * factor;
134  attract_y[i] -= y_dist * factor;
135  }
136 }

159 template <typename vertex_t>
160 void apply_gravity(const float *restrict x_pos,
161  const float *restrict y_pos,
162  float *restrict attract_x,
163  float *restrict attract_y,
164  const int *restrict mass,
165  const float gravity,

167  const float scaling_ratio,

173  nthreads.y = 1;

180  strong_gravity_kernel<vertex_t><<<nblocks, nthreads, 0, stream>>>(

189 template <typename vertex_t>
190 __global__ void local_speed_kernel(const float *restrict repel_x,
191  const float *restrict repel_y,
192  const float *restrict attract_x,
193  const float *restrict attract_y,
194  const float *restrict old_dx,
195  const float *restrict old_dy,
196  const int *restrict mass,
197  float *restrict swinging,
198  float *restrict traction,
199  const vertex_t n)
200 {
201  // For every node.
202  for (int i = threadIdx.x + blockIdx.x * blockDim.x; i < n; i += gridDim.x * blockDim.x) {
203  const float dx = repel_x[i] + attract_x[i];
204  const float dy = repel_y[i] + attract_y[i];
205  float node_swinging = mass[i] * sqrt(pow(old_dx[i] - dx, 2) + pow(old_dy[i] - dy, 2));
206  float node_traction = 0.5 * mass[i] * sqrt(pow(old_dx[i] + dx, 2) + pow(old_dy[i] + dy, 2));
207  swinging[i] = node_swinging;
208  traction[i] = node_traction;
209  }
210 }

238 template <typename vertex_t>
239 void adapt_speed(const float jitter_tolerance,
240  float *restrict jt,
241  float *restrict speed,
242  float *restrict speed_efficiency,
243  const float s,
244  const float t,
245  const vertex_t n)
246 {
247  float estimated_jt = 0.05 * sqrt(n);
248  float min_jt = sqrt(estimated_jt);
249  float max_jt = 10;
250  float target_speed;
251  float min_speed_efficiency = 0.05;
252  const float max_rise = 0.5;
253 
254  *jt = jitter_tolerance * max(min_jt, min(max_jt, estimated_jt * t / (n * n)));
255 
256  if (s / t > 2.0) {
257  if (*speed_efficiency > min_speed_efficiency) { *speed_efficiency *= 0.5; }
258  *jt = max(*jt, jitter_tolerance);
259  }
260 
261  if (s == 0)
262  target_speed = FLT_MAX;
263  else
264  target_speed = (*jt * *speed_efficiency * t) / s;
265 
266  if (s > *jt * t) {
267  if (*speed_efficiency > min_speed_efficiency) *speed_efficiency *= .7;
268  } else if (*speed < 1000)
269  *speed_efficiency *= 1.3;
270 
271  *speed = *speed + min(target_speed - *speed, max_rise * *speed);
272 }

274 template <typename vertex_t>
275 __global__ void update_positions_kernel(float *restrict x_pos,
276  float *restrict y_pos,
277  const float *restrict repel_x,
278  const float *restrict repel_y,
279  const float *restrict attract_x,
280  const float *restrict attract_y,
281  float *restrict old_dx,
282  float *restrict old_dy,
283  const float *restrict swinging,
284  const float speed,
285  const vertex_t n)
286 {
287  // For every node.
288  for (int i = threadIdx.x + blockIdx.x * blockDim.x; i < n; i += gridDim.x * blockDim.x) {
289  const float factor = speed / (1.0 + sqrt(speed * swinging[i]));
290  const float dx = (repel_x[i] + attract_x[i]);
291  const float dy = (repel_y[i] + attract_y[i]);
292 
293  x_pos[i] += dx * factor;
294  y_pos[i] += dy * factor;
295  old_dx[i] = dx;
296  old_dy[i] = dy;
297  }
298 }

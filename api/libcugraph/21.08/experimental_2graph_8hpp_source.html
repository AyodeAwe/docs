78  graph_view_t<vertex_t, edge_t, weight_t, store_transposed, multi_gpu> view() const
79  {
80  std::vector<edge_t const*> offsets(adj_matrix_partition_offsets_.size(), nullptr);
81  std::vector<vertex_t const*> indices(adj_matrix_partition_indices_.size(), nullptr);
82  auto weights = adj_matrix_partition_weights_
83  ? std::make_optional<std::vector<weight_t const*>>(
84  (*adj_matrix_partition_weights_).size(), nullptr)
85  : std::nullopt;
86  auto dcs_nzd_vertices = adj_matrix_partition_dcs_nzd_vertices_
87  ? std::make_optional<std::vector<vertex_t const*>>(
88  (*adj_matrix_partition_dcs_nzd_vertices_).size(), nullptr)
89  : std::nullopt;
90  auto dcs_nzd_vertex_counts =
91  adj_matrix_partition_dcs_nzd_vertex_counts_
92  ? std::make_optional<std::vector<vertex_t>>(
93  (*adj_matrix_partition_dcs_nzd_vertex_counts_).size(), vertex_t{0})
94  : std::nullopt;
95  for (size_t i = 0; i < offsets.size(); ++i) {
96  offsets[i] = adj_matrix_partition_offsets_[i].data();
97  indices[i] = adj_matrix_partition_indices_[i].data();
98  if (weights) { (*weights)[i] = (*adj_matrix_partition_weights_)[i].data(); }
99  if (dcs_nzd_vertices) {
100  (*dcs_nzd_vertices)[i] = (*adj_matrix_partition_dcs_nzd_vertices_)[i].data();
101  (*dcs_nzd_vertex_counts)[i] = (*adj_matrix_partition_dcs_nzd_vertex_counts_)[i];
102  }
103  }
104 
105  return graph_view_t<vertex_t, edge_t, weight_t, store_transposed, multi_gpu>(
106  *(this->get_handle_ptr()),
107  offsets,
108  indices,
109  weights,
110  dcs_nzd_vertices,
111  dcs_nzd_vertex_counts,
112  partition_,
113  this->get_number_of_vertices(),
114  this->get_number_of_edges(),
115  this->get_graph_properties(),
116  adj_matrix_partition_segment_offsets_,
117  false);
118  }

115  template <typename graph_t,
116  typename random_walker_t,
117  typename index_t,
118  typename real_t,
119  typename seed_t>
120  void operator()(
121  graph_t const& graph, // graph being traversed
122  random_walker_t const& rand_walker, // random walker object for which traversal is driven
123  seed_t seed0, // initial seed value
124  device_vec_t<typename graph_t::vertex_type>& d_coalesced_v, // crt coalesced vertex set
125  device_vec_t<typename graph_t::weight_type>& d_coalesced_w, // crt coalesced weight set
126  device_vec_t<index_t>& d_paths_sz, // crt paths sizes
127  device_vec_t<typename graph_t::edge_type>&
128  d_crt_out_degs, // crt out-degs for current set of vertices
129  device_vec_t<real_t>& d_random, // crt set of random real values
130  device_vec_t<typename graph_t::vertex_type>&
131  d_col_indx, // crt col col indices to be used for retrieving next step
132  device_vec_t<typename graph_t::vertex_type>&
133  d_next_v, // crt set of destination vertices, for next step
134  device_vec_t<typename graph_t::weight_type>&
135  d_next_w) // set of weights between src and destination vertices, for next step
136  const
137  {
138  // start from 1, as 0-th was initialized above:
139  //
140  for (decltype(max_depth_) step_indx = 1; step_indx < max_depth_; ++step_indx) {
141  // take one-step in-sync for each path in parallel:
142  //
143  rand_walker.step(graph,
144  seed0 + static_cast<seed_t>(step_indx),
145  d_coalesced_v,
146  d_coalesced_w,
147  d_paths_sz,
148  d_crt_out_degs,
149  d_random,
150  d_col_indx,
151  d_next_v,
152  d_next_w);
153 
154  // early exit: all paths have reached sinks:
155  //
156  if (rand_walker.all_paths_stopped(d_crt_out_degs)) break;
157  }
158  }

161  size_t get_tmp_buff_sz(void) const { return num_paths_; }

165  size_t max_depth_;

181  template <typename graph_t,
182  typename random_walker_t,
183  typename index_t,
184  typename real_t,
185  typename seed_t>
186  void operator()(
187  graph_t const& graph, // graph being traversed
188  random_walker_t const& rand_walker, // random walker object for which traversal is driven
189  seed_t seed0, // initial seed value
190  device_vec_t<typename graph_t::vertex_type>& d_coalesced_v, // crt coalesced vertex set
191  device_vec_t<typename graph_t::weight_type>& d_coalesced_w, // crt coalesced weight set
192  device_vec_t<index_t>& d_paths_sz, // crt paths sizes
193  device_vec_t<typename graph_t::edge_type>&
194  d_crt_out_degs, // ignored: out-degs for the current set of vertices
195  device_vec_t<real_t>& d_random, // _entire_ set of random real values
196  device_vec_t<typename graph_t::vertex_type>&
197  d_col_indx, // ignored: crt col indices to be used for retrieving next step
198  device_vec_t<typename graph_t::vertex_type>&
199  d_next_v, // ignored: crt set of destination vertices, for next step (coalesced set
200  // updated directly, instead)
201  device_vec_t<typename graph_t::weight_type>&
202  d_next_w) // ignored: set of weights between src and destination vertices, for next step
203  // (coalesced set updated directly, instead)
204  const
205  {
206  using vertex_t = typename graph_t::vertex_type;
207  using edge_t = typename graph_t::edge_type;
208  using weight_t = typename graph_t::weight_type;
209  using random_engine_t = typename random_walker_t::rnd_engine_t;
210 
211  auto const& handle = rand_walker.get_handle();
212  auto* ptr_d_random = raw_ptr(d_random);
213 
214  random_engine_t::generate_random(handle, ptr_d_random, d_random.size(), seed0);
215 
216  auto const* col_indices = graph.get_matrix_partition_view().get_indices();
217  auto const* row_offsets = graph.get_matrix_partition_view().get_offsets();
218  auto const* values = graph.get_matrix_partition_view().get_weights()
219  ? *(graph.get_matrix_partition_view().get_weights())
220  : static_cast<weight_t*>(nullptr);
221  auto* ptr_d_sizes = raw_ptr(d_paths_sz);
222  auto const& d_cached_out_degs = rand_walker.get_out_degs();
223 
224  auto rnd_to_indx_convertor = [] __device__(real_t rnd_vindx, edge_t crt_out_deg) {
225  real_t max_ub = static_cast<real_t>(crt_out_deg - 1);
226  auto interp_vindx = rnd_vindx * max_ub + real_t{.5};
227  vertex_t v_indx = static_cast<vertex_t>(interp_vindx);
228  return (v_indx >= crt_out_deg ? crt_out_deg - 1 : v_indx);
229  };
230 
231  auto next_vw =
232  [row_offsets,
233  col_indices,
234  values] __device__(auto v_indx, // src vertex to find dst from
235  auto col_indx) { // column index, in {0,...,out_deg(v_indx)-1},
236  // extracted from random value in [0..1]
237  auto start_row = row_offsets[v_indx];
238 
239  auto weight_value =
240  (values == nullptr ? weight_t{1}
241  : values[start_row + col_indx]); // account for un-weighted graphs
242  return thrust::make_tuple(col_indices[start_row + col_indx], weight_value);
243  };
244 
245  // start from 1, as 0-th was initialized above:
246  //
247  thrust::for_each(rmm::exec_policy(handle.get_stream_view()),
248  thrust::make_counting_iterator<index_t>(0),
249  thrust::make_counting_iterator<index_t>(num_paths_),
250  [max_depth = max_depth_,
251  ptr_d_cache_out_degs = raw_const_ptr(d_cached_out_degs),
252  ptr_coalesced_v = raw_ptr(d_coalesced_v),
253  ptr_coalesced_w = raw_ptr(d_coalesced_w),
254  ptr_d_random,
255  ptr_d_sizes,
256  rnd_to_indx_convertor,
257  next_vw] __device__(auto path_index) {
258  auto chunk_offset = path_index * max_depth;
259  vertex_t src_vertex = ptr_coalesced_v[chunk_offset];
260 
261  for (index_t step_indx = 1; step_indx < max_depth; ++step_indx) {
262  auto crt_out_deg = ptr_d_cache_out_degs[src_vertex];
263  if (crt_out_deg == 0) break;
264 
265  // indexing into coalesced arrays of size num_paths x (max_depth -1):
266  // (d_random, d_coalesced_w)
267  //
268  auto stepping_index = chunk_offset - path_index + step_indx - 1;
269 
270  auto real_rnd_indx = ptr_d_random[stepping_index];
271 
272  auto col_indx = rnd_to_indx_convertor(real_rnd_indx, crt_out_deg);
273  auto pair_vw = next_vw(src_vertex, col_indx);
274 
275  src_vertex = thrust::get<0>(pair_vw);
276  auto crt_weight = thrust::get<1>(pair_vw);
277 
278  ptr_coalesced_v[chunk_offset + step_indx] = src_vertex;
279  ptr_coalesced_w[stepping_index] = crt_weight;
280  ptr_d_sizes[path_index]++;
281  }
282  });
283  }

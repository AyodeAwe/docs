1758 /*
1759  * @brief Compute triangle counts.
1760  *
1761  * Compute triangle counts for the entire set of vertices (if @p vertices is std::nullopt) or the
1762  * given vertices (@p vertices.has_value() is true).
1763  *
1764  * @tparam vertex_t Type of vertex identifiers. Needs to be an integral type.
1765  * @tparam edge_t Type of edge identifiers. Needs to be an integral type.
1766  * @tparam weight_t Type of edge weights. Needs to be a floating point type.
1767  * @tparam multi_gpu Flag indicating whether template instantiation should target single-GPU (false)
1768  * @param handle RAFT handle object to encapsulate resources (e.g. CUDA stream, communicator, and
1769  * handles to various CUDA libraries) to run graph algorithms.
1770  * @param graph_view Graph view object.
1771  * @param vertices Vertices to compute triangle counts. If @p vertices.has_value() is false, compute
1772  * triangle counts for the entire set of vertices.
1773  * @param counts Output triangle count array. The size of the array should be the local vertex
1774  * partition range size (if @p vertices is std::nullopt) or the size of @p vertices (if @p
1775  * vertices.has_value() is true).
1776  * @param do_expensive_check A flag to run expensive checks for input arguments (if set to `true`).
1777  */
1778 template <typename vertex_t, typename edge_t, typename weight_t, bool multi_gpu>
1779 void triangle_count(raft::handle_t const& handle,
1780  graph_view_t<vertex_t, edge_t, weight_t, false, multi_gpu> const& graph_view,

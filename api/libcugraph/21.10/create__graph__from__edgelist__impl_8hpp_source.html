33 template <typename vertex_t,
34  typename edge_t,
35  typename weight_t,
36  bool store_transposed,
37  bool multi_gpu>
38 std::enable_if_t<
39  multi_gpu,
40  std::tuple<cugraph::graph_t<vertex_t, edge_t, weight_t, store_transposed, multi_gpu>,
41  std::optional<rmm::device_uvector<vertex_t>>>>
42 create_graph_from_edgelist_impl(raft::handle_t const& handle,
43  std::optional<rmm::device_uvector<vertex_t>>&& local_vertex_span,
44  rmm::device_uvector<vertex_t>&& edgelist_rows,
45  rmm::device_uvector<vertex_t>&& edgelist_cols,
46  std::optional<rmm::device_uvector<weight_t>>&& edgelist_weights,
47  graph_properties_t graph_properties,
48  bool renumber)
49 {
50  CUGRAPH_EXPECTS(renumber, "renumber should be true if multi_gpu is true.");
51 
52  auto& comm = handle.get_comms();
53  auto const comm_size = comm.get_size();
54  auto const comm_rank = comm.get_rank();
55  auto& row_comm = handle.get_subcomm(cugraph::partition_2d::key_naming_t().row_name());
56  auto const row_comm_size = row_comm.get_size();
57  auto& col_comm = handle.get_subcomm(cugraph::partition_2d::key_naming_t().col_name());
58  auto const col_comm_size = col_comm.get_size();
59 
60  // 1. groupby edges to their target local adjacency matrix partition (and further groupby within
61  // the local partition by applying the compute_gpu_id_from_vertex_t to minor vertex IDs).
62 
63  auto edge_counts = cugraph::detail::groupby_and_count_edgelist_by_local_partition_id(
64  handle,
65  store_transposed ? edgelist_cols : edgelist_rows,
66  store_transposed ? edgelist_rows : edgelist_cols,
67  edgelist_weights,
68  true);
69 
70  std::vector<size_t> h_edge_counts(edge_counts.size());
71  raft::update_host(
72  h_edge_counts.data(), edge_counts.data(), edge_counts.size(), handle.get_stream());
73  handle.get_stream_view().synchronize();
74 
75  std::vector<edge_t> edgelist_edge_counts(col_comm_size, edge_t{0});
76  auto edgelist_intra_partition_segment_offsets =
77  std::make_optional<std::vector<std::vector<edge_t>>>(
78  col_comm_size, std::vector<edge_t>(row_comm_size + 1, edge_t{0}));
79  for (int i = 0; i < col_comm_size; ++i) {
80  edgelist_edge_counts[i] = std::accumulate(h_edge_counts.begin() + row_comm_size * i,
81  h_edge_counts.begin() + row_comm_size * (i + 1),
82  edge_t{0});
83  std::partial_sum(h_edge_counts.begin() + row_comm_size * i,
84  h_edge_counts.begin() + row_comm_size * (i + 1),
85  (*edgelist_intra_partition_segment_offsets)[i].begin() + 1);
86  }
87  std::vector<edge_t> edgelist_displacements(col_comm_size, edge_t{0});
88  std::partial_sum(edgelist_edge_counts.begin(),
89  edgelist_edge_counts.end() - 1,
90  edgelist_displacements.begin() + 1);
91 
92  // 2. renumber
93 
94  std::vector<vertex_t*> major_ptrs(col_comm_size);
95  std::vector<vertex_t*> minor_ptrs(major_ptrs.size());
96  for (int i = 0; i < col_comm_size; ++i) {
97  major_ptrs[i] = (store_transposed ? edgelist_cols.begin() : edgelist_rows.begin()) +
98  edgelist_displacements[i];
99  minor_ptrs[i] = (store_transposed ? edgelist_rows.begin() : edgelist_cols.begin()) +
100  edgelist_displacements[i];
101  }
102  auto [renumber_map_labels, meta] = cugraph::renumber_edgelist<vertex_t, edge_t, multi_gpu>(
103  handle,
104  local_vertex_span
105  ? std::optional<std::tuple<vertex_t const*, vertex_t>>{std::make_tuple(
106  (*local_vertex_span).data(), static_cast<vertex_t>((*local_vertex_span).size()))}
107  : std::nullopt,
108  major_ptrs,
109  minor_ptrs,
110  edgelist_edge_counts,
111  edgelist_intra_partition_segment_offsets);
112 
113  // 3. create a graph
114 
115  std::vector<cugraph::edgelist_t<vertex_t, edge_t, weight_t>> edgelists(col_comm_size);
116  for (int i = 0; i < col_comm_size; ++i) {
117  edgelists[i] = cugraph::edgelist_t<vertex_t, edge_t, weight_t>{
118  edgelist_rows.data() + edgelist_displacements[i],
119  edgelist_cols.data() + edgelist_displacements[i],
120  edgelist_weights
121  ? std::optional<weight_t const*>{(*edgelist_weights).data() + edgelist_displacements[i]}
122  : std::nullopt,
123  static_cast<edge_t>(edgelist_edge_counts[i])};
124  }
125 
126  return std::make_tuple(
127  cugraph::graph_t<vertex_t, edge_t, weight_t, store_transposed, multi_gpu>(
128  handle,
129  edgelists,
130  cugraph::graph_meta_t<vertex_t, edge_t, multi_gpu>{
131  meta.number_of_vertices,
132  meta.number_of_edges,
133  graph_properties,
134  meta.partition,
135  meta.segment_offsets,
136  store_transposed ? meta.num_local_unique_edge_minors : meta.num_local_unique_edge_majors,
137  store_transposed ? meta.num_local_unique_edge_majors : meta.num_local_unique_edge_minors}),
138  std::optional<rmm::device_uvector<vertex_t>>{std::move(renumber_map_labels)});
139 }

141 template <typename vertex_t,
142  typename edge_t,
143  typename weight_t,
144  bool store_transposed,
145  bool multi_gpu>
146 std::enable_if_t<
147  !multi_gpu,
148  std::tuple<cugraph::graph_t<vertex_t, edge_t, weight_t, store_transposed, multi_gpu>,
149  std::optional<rmm::device_uvector<vertex_t>>>>
150 create_graph_from_edgelist_impl(raft::handle_t const& handle,
151  std::optional<rmm::device_uvector<vertex_t>>&& vertex_span,
152  rmm::device_uvector<vertex_t>&& edgelist_rows,
153  rmm::device_uvector<vertex_t>&& edgelist_cols,
154  std::optional<rmm::device_uvector<weight_t>>&& edgelist_weights,
155  graph_properties_t graph_properties,
156  bool renumber)
157 {
158  auto renumber_map_labels =
159  renumber ? std::make_optional<rmm::device_uvector<vertex_t>>(0, handle.get_stream())
160  : std::nullopt;
161  std::optional<std::vector<vertex_t>> segment_offsets{std::nullopt};
162  renumber_meta_t<vertex_t, edge_t, multi_gpu> meta{};
163  if (renumber) {
164  std::tie(*renumber_map_labels, meta) = cugraph::renumber_edgelist<vertex_t, edge_t, multi_gpu>(
165  handle,
166  vertex_span ? std::optional<std::tuple<vertex_t const*, vertex_t>>{std::make_tuple(
167  (*vertex_span).data(), static_cast<vertex_t>((*vertex_span).size()))}
168  : std::nullopt,
169  store_transposed ? edgelist_cols.data() : edgelist_rows.data(),
170  store_transposed ? edgelist_rows.data() : edgelist_cols.data(),
171  static_cast<edge_t>(edgelist_rows.size()));
172  }
173 
174  vertex_t num_vertices{};
175  if (renumber) {
176  num_vertices = static_cast<vertex_t>((*renumber_map_labels).size());
177  } else {
178  if (vertex_span) {
179  num_vertices = static_cast<vertex_t>((*vertex_span).size());
180  } else {
181  num_vertices = 1 + cugraph::detail::compute_maximum_vertex_id(
182  handle.get_stream_view(), edgelist_rows, edgelist_cols);
183  }
184  }
185 
186  return std::make_tuple(
187  cugraph::graph_t<vertex_t, edge_t, weight_t, store_transposed, multi_gpu>(
188  handle,
189  cugraph::edgelist_t<vertex_t, edge_t, weight_t>{
190  edgelist_rows.data(),
191  edgelist_cols.data(),
192  edgelist_weights ? std::optional<weight_t const*>{(*edgelist_weights).data()}
193  : std::nullopt,
194  static_cast<edge_t>(edgelist_rows.size())},
195  cugraph::graph_meta_t<vertex_t, edge_t, multi_gpu>{
196  num_vertices,
197  graph_properties,
198  renumber ? std::optional<std::vector<vertex_t>>{meta.segment_offsets} : std::nullopt}),
199  std::move(renumber_map_labels));
200 }
